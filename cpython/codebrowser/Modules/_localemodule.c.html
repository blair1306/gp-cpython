50/* the grouping is terminated by either 0 or CHAR_MAX */
51static PyObject*
52copy_grouping(const char* s)
53{
54 int i;
55 PyObject *result, *val = NULL;
56
57 if (s[0] == '\0') {
58 /* empty string: no grouping at all */
59 return PyList_New(0);
60 }
61
62 for (i = 0; s[i] != '\0' && s[i] != CHAR_MAX; i++)
63 ; /* nothing */
64
65 result = PyList_New(i+1);
66 if (!result)
67 return NULL;
68
69 i = -1;
70 do {
71 i++;
72 val = PyLong_FromLong(s[i]);
73 if (!val)
74 break;
75 if (PyList_SetItem(result, i, val)) {
76 Py_DECREF(val);
77 val = NULL;
78 break;
79 }
80 } while (s[i] != '\0' && s[i] != CHAR_MAX);
81
82 if (!val) {
83 Py_DECREF(result);
84 return NULL;
85 }
86
87 return result;
88}

90static PyObject*
91PyLocale_setlocale(PyObject* self, PyObject* args)
92{
93 int category;
94 char *locale = NULL, *result;
95 PyObject *result_object;
96
97 if (!PyArg_ParseTuple(args, "i|z:setlocale", &category, &locale))
98 return NULL;
99
100#if defined(MS_WINDOWS)
101 if (category < LC_MIN || category > LC_MAX)
102 {
103 PyErr_SetString(Error, "invalid locale category");
104 return NULL;
105 }
106#endif
107
108 if (locale) {
109 /* set locale */
110 result = setlocale(category, locale);
111 if (!result) {
112 /* operation failed, no setting was changed */
113 PyErr_SetString(Error, "unsupported locale setting");
114 return NULL;
115 }
116 result_object = PyUnicode_DecodeLocale(result, NULL);
117 if (!result_object)
118 return NULL;
119 } else {
120 /* get locale */
121 result = setlocale(category, NULL);
122 if (!result) {
123 PyErr_SetString(Error, "locale query failed");
124 return NULL;
125 }
126 result_object = PyUnicode_DecodeLocale(result, NULL);
127 }
128 return result_object;
129}

131PyDoc_STRVAR(localeconv__doc__,
132"() -> dict. Returns numeric and monetary locale-specific parameters.");
133
134static PyObject*
135PyLocale_localeconv(PyObject* self)
136{
137 PyObject* result;
138 struct lconv *l;
139 PyObject *x;
140
141 result = PyDict_New();
142 if (!result)
143 return NULL;
144
145 /* if LC_NUMERIC is different in the C library, use saved value */
146 l = localeconv();
147
148 /* hopefully, the localeconv result survives the C library calls
149 involved herein */
150
151#define RESULT(key, obj)\
152 do { \
153 if (obj == NULL) \
154 goto failed; \
155 if (PyDict_SetItemString(result, key, obj) < 0) { \
156 Py_DECREF(obj); \
157 goto failed; \
158 } \
159 Py_DECREF(obj); \
160 } while (0)
161
162#define RESULT_STRING(s)\
163 do { \
164 x = PyUnicode_DecodeLocale(l->s, NULL); \
165 RESULT(#s, x); \
166 } while (0)
167
168#define RESULT_INT(i)\
169 do { \
170 x = PyLong_FromLong(l->i); \
171 RESULT(#i, x); \
172 } while (0)
173
174 /* Numeric information */
175 RESULT_STRING(decimal_point);
176 RESULT_STRING(thousands_sep);
177 x = copy_grouping(l->grouping);
178 RESULT("grouping", x);
179
180 /* Monetary information */
181 RESULT_STRING(int_curr_symbol);
182 RESULT_STRING(currency_symbol);
183 RESULT_STRING(mon_decimal_point);
184 RESULT_STRING(mon_thousands_sep);
185 x = copy_grouping(l->mon_grouping);
186 RESULT("mon_grouping", x);
187
188 RESULT_STRING(positive_sign);
189 RESULT_STRING(negative_sign);
190 RESULT_INT(int_frac_digits);
191 RESULT_INT(frac_digits);
192 RESULT_INT(p_cs_precedes);
193 RESULT_INT(p_sep_by_space);
194 RESULT_INT(n_cs_precedes);
195 RESULT_INT(n_sep_by_space);
196 RESULT_INT(p_sign_posn);
197 RESULT_INT(n_sign_posn);
198 return result;
199
200 failed:
201 Py_XDECREF(result);
202 return NULL;
203}

206PyDoc_STRVAR(strcoll__doc__,
207"string,string -> int. Compares two strings according to the locale.");

209static PyObject*
210PyLocale_strcoll(PyObject* self, PyObject* args)
211{
212 PyObject *os1, *os2, *result = NULL;
213 wchar_t *ws1 = NULL, *ws2 = NULL;
214
215 if (!PyArg_ParseTuple(args, "UU:strcoll", &os1, &os2))
216 return NULL;
217 /* Convert the unicode strings to wchar[]. */
218 ws1 = PyUnicode_AsWideCharString(os1, NULL);
219 if (ws1 == NULL)
220 goto done;
221 ws2 = PyUnicode_AsWideCharString(os2, NULL);
222 if (ws2 == NULL)
223 goto done;
224 /* Collate the strings. */
225 result = PyLong_FromLong(wcscoll(ws1, ws2));
226 done:
227 /* Deallocate everything. */
228 if (ws1) PyMem_FREE(ws1);
229 if (ws2) PyMem_FREE(ws2);
230 return result;
231}
232#endif

234#ifdef HAVE_WCSXFRM
235PyDoc_STRVAR(strxfrm__doc__,
236"strxfrm(string) -> string.\n\
237\n\
238Return a string that can be used as a key for locale-aware comparisons.");
239
240static PyObject*
241PyLocale_strxfrm(PyObject* self, PyObject* args)
242{
243 PyObject *str;
244 Py_ssize_t n1;
245 wchar_t *s = NULL, *buf = NULL;
246 size_t n2;
247 PyObject *result = NULL;
248
249 if (!PyArg_ParseTuple(args, "U:strxfrm", &str))
250 return NULL;
251
252 s = PyUnicode_AsWideCharString(str, &n1);
253 if (s == NULL)
254 goto exit;
255 if (wcslen(s) != (size_t)n1) {
256 PyErr_SetString(PyExc_ValueError,
257 "embedded null character");
258 goto exit;
259 }
260
261 /* assume no change in size, first */
262 n1 = n1 + 1;
263 buf = PyMem_New(wchar_t, n1);
264 if (!buf) {
265 PyErr_NoMemory();
266 goto exit;
267 }
268 errno = 0;
269 n2 = wcsxfrm(buf, s, n1);
270 if (errno && errno != ERANGE) {
271 PyErr_SetFromErrno(PyExc_OSError);
272 goto exit;
273 }
274 if (n2 >= (size_t)n1) {
275 /* more space needed */
276 wchar_t * new_buf = PyMem_Realloc(buf, (n2+1)*sizeof(wchar_t));
277 if (!new_buf) {
278 PyErr_NoMemory();
279 goto exit;
280 }
281 buf = new_buf;
282 errno = 0;
283 n2 = wcsxfrm(buf, s, n2+1);
284 if (errno) {
285 PyErr_SetFromErrno(PyExc_OSError);
286 goto exit;
287 }
288 }
289 result = PyUnicode_FromWideChar(buf, n2);
290exit:
291 PyMem_Free(buf);
292 PyMem_Free(s);
293 return result;
294}
295#endif

298static PyObject*
299PyLocale_getdefaultlocale(PyObject* self)
300{
301 char encoding[100];
302 char locale[100];
303
304 PyOS_snprintf(encoding, sizeof(encoding), "cp%d", GetACP());
305
306 if (GetLocaleInfo(LOCALE_USER_DEFAULT,
307 LOCALE_SISO639LANGNAME,
308 locale, sizeof(locale))) {
309 Py_ssize_t i = strlen(locale);
310 locale[i++] = '_';
311 if (GetLocaleInfo(LOCALE_USER_DEFAULT,
312 LOCALE_SISO3166CTRYNAME,
313 locale+i, (int)(sizeof(locale)-i)))
314 return Py_BuildValue("ss", locale, encoding);
315 }
316
317 /* If we end up here, this windows version didn't know about
318 ISO639/ISO3166 names (it's probably Windows 95). Return the
319 Windows language identifier instead (a hexadecimal number) */
320
321 locale[0] = '0';
322 locale[1] = 'x';
323 if (GetLocaleInfo(LOCALE_USER_DEFAULT, LOCALE_IDEFAULTLANGUAGE,
324 locale+2, sizeof(locale)-2)) {
325 return Py_BuildValue("ss", locale, encoding);
326 }
327
328 /* cannot determine the language code (very unlikely) */
329 Py_INCREF(Py_None);
330 return Py_BuildValue("Os", Py_None, encoding);
331}

443PyDoc_STRVAR(nl_langinfo__doc__,
444"nl_langinfo(key) -> string\n"
445"Return the value for the locale information associated with key.");
446
447static PyObject*
448PyLocale_nl_langinfo(PyObject* self, PyObject* args)
449{
450 int item, i;
451 if (!PyArg_ParseTuple(args, "i:nl_langinfo", &item))
452 return NULL;
453 /* Check whether this is a supported constant. GNU libc sometimes
454 returns numeric values in the char* return value, which would
455 crash PyUnicode_FromString. */
456 for (i = 0; langinfo_constants[i].name; i++)
457 if (langinfo_constants[i].value == item) {
458 /* Check NULL as a workaround for GNU libc's returning NULL
459 instead of an empty string for nl_langinfo(ERA). */
460 const char *result = nl_langinfo(item);
461 result = result != NULL ? result : "";
462 return PyUnicode_DecodeLocale(result, NULL);
463 }
464 PyErr_SetString(PyExc_ValueError, "unsupported langinfo constant");
465 return NULL;
466}
467#endif /* HAVE_LANGINFO_H */

529PyDoc_STRVAR(bindtextdomain__doc__,
530"bindtextdomain(domain, dir) -> string\n"
531"Bind the C library's domain to dir.");
532
533static PyObject*
534PyIntl_bindtextdomain(PyObject* self,PyObject*args)
535{
536 char *domain, *dirname, *current_dirname;
537 PyObject *dirname_obj, *dirname_bytes = NULL, *result;
538 if (!PyArg_ParseTuple(args, "sO", &domain, &dirname_obj))
539 return 0;
540 if (!strlen(domain)) {
541 PyErr_SetString(Error, "domain must be a non-empty string");
542 return 0;
543 }
544 if (dirname_obj != Py_None) {
545 if (!PyUnicode_FSConverter(dirname_obj, &dirname_bytes))
546 return NULL;
547 dirname = PyBytes_AsString(dirname_bytes);
548 } else {
549 dirname_bytes = NULL;
550 dirname = NULL;
551 }
552 current_dirname = bindtextdomain(domain, dirname);
553 if (current_dirname == NULL) {
554 Py_XDECREF(dirname_bytes);
555 PyErr_SetFromErrno(PyExc_OSError);
556 return NULL;
557 }
558 result = PyUnicode_DecodeLocale(current_dirname, NULL);
559 Py_XDECREF(dirname_bytes);
560 return result;
561}

635PyMODINIT_FUNC
636PyInit__locale(void)
637{
638 PyObject *m;
639#ifdef HAVE_LANGINFO_H
640 int i;
641#endif
642
643 m = PyModule_Create(&_localemodule);
644 if (m == NULL)
645 return NULL;
646
647 PyModule_AddIntMacro(m, LC_CTYPE);
648 PyModule_AddIntMacro(m, LC_TIME);
649 PyModule_AddIntMacro(m, LC_COLLATE);
650 PyModule_AddIntMacro(m, LC_MONETARY);
651
652#ifdef LC_MESSAGES
653 PyModule_AddIntMacro(m, LC_MESSAGES);
654#endif /* LC_MESSAGES */
655
656 PyModule_AddIntMacro(m, LC_NUMERIC);
657 PyModule_AddIntMacro(m, LC_ALL);
658 PyModule_AddIntMacro(m, CHAR_MAX);
659
660 Error = PyErr_NewException("locale.Error", NULL, NULL);
661 if (Error == NULL) {
662 Py_DECREF(m);
663 return NULL;
664 }
665 PyModule_AddObject(m, "Error", Error);
666
667#ifdef HAVE_LANGINFO_H
668 for (i = 0; langinfo_constants[i].name; i++) {
669 PyModule_AddIntConstant(m, langinfo_constants[i].name,
670 langinfo_constants[i].value);
671 }
672#endif
673
674 if (PyErr_Occurred()) {
675 Py_DECREF(m);
676 return NULL;
677 }
678 return m;
679}
