120static Py_ssize_t
121ascii_escape_unichar(Py_UCS4 c, unsigned char *output, Py_ssize_t chars)
122{
123 /* Escape unicode code point c to ASCII escape sequences
124 in char *output. output must have at least 12 bytes unused to
125 accommodate an escaped surrogate pair "\uXXXX\uXXXX" */
126 output[chars++] = '\\';
127 switch (c) {
128 case '\\': output[chars++] = c; break;
129 case '"': output[chars++] = c; break;
130 case '\b': output[chars++] = 'b'; break;
131 case '\f': output[chars++] = 'f'; break;
132 case '\n': output[chars++] = 'n'; break;
133 case '\r': output[chars++] = 'r'; break;
134 case '\t': output[chars++] = 't'; break;
135 default:
136 if (c >= 0x10000) {
137 /* UTF-16 surrogate pair */
138 Py_UCS4 v = Py_UNICODE_HIGH_SURROGATE(c);
139 output[chars++] = 'u';
140 output[chars++] = Py_hexdigits[(v >> 12) & 0xf];
141 output[chars++] = Py_hexdigits[(v >> 8) & 0xf];
142 output[chars++] = Py_hexdigits[(v >> 4) & 0xf];
143 output[chars++] = Py_hexdigits[(v ) & 0xf];
144 c = Py_UNICODE_LOW_SURROGATE(c);
145 output[chars++] = '\\';
146 }
147 output[chars++] = 'u';
148 output[chars++] = Py_hexdigits[(c >> 12) & 0xf];
149 output[chars++] = Py_hexdigits[(c >> 8) & 0xf];
150 output[chars++] = Py_hexdigits[(c >> 4) & 0xf];
151 output[chars++] = Py_hexdigits[(c ) & 0xf];
152 }
153 return chars;
154}

156static PyObject *
157ascii_escape_unicode(PyObject *pystr)
158{
159 /* Take a PyUnicode pystr and return a new ASCII-only escaped PyUnicode */
160 Py_ssize_t i;
161 Py_ssize_t input_chars;
162 Py_ssize_t output_size;
163 Py_ssize_t chars;
164 PyObject *rval;
165 void *input;
166 unsigned char *output;
167 int kind;
168
169 if (PyUnicode_READY(pystr) == -1)
170 return NULL;
171
172 input_chars = PyUnicode_GET_LENGTH(pystr);
173 input = PyUnicode_DATA(pystr);
174 kind = PyUnicode_KIND(pystr);
175
176 /* Compute the output size */
177 for (i = 0, output_size = 2; i < input_chars; i++) {
178 Py_UCS4 c = PyUnicode_READ(kind, input, i);
179 Py_ssize_t d;
180 if (S_CHAR(c)) {
181 d = 1;
182 }
183 else {
184 switch(c) {
185 case '\\': case '"': case '\b': case '\f':
186 case '\n': case '\r': case '\t':
187 d = 2; break;
188 default:
189 d = c >= 0x10000 ? 12 : 6;
190 }
191 }
192 if (output_size > PY_SSIZE_T_MAX - d) {
193 PyErr_SetString(PyExc_OverflowError, "string is too long to escape");
194 return NULL;
195 }
196 output_size += d;
197 }
198
199 rval = PyUnicode_New(output_size, 127);
200 if (rval == NULL) {
201 return NULL;
202 }
203 output = PyUnicode_1BYTE_DATA(rval);
204 chars = 0;
205 output[chars++] = '"';
206 for (i = 0; i < input_chars; i++) {
207 Py_UCS4 c = PyUnicode_READ(kind, input, i);
208 if (S_CHAR(c)) {
209 output[chars++] = c;
210 }
211 else {
212 chars = ascii_escape_unichar(c, output, chars);
213 }
214 }
215 output[chars++] = '"';
216#ifdef Py_DEBUG
217 assert(_PyUnicode_CheckConsistency(rval, 1));
218#endif
219 return rval;
220}

222static PyObject *
223escape_unicode(PyObject *pystr)
224{
225 /* Take a PyUnicode pystr and return a new escaped PyUnicode */
226 Py_ssize_t i;
227 Py_ssize_t input_chars;
228 Py_ssize_t output_size;
229 Py_ssize_t chars;
230 PyObject *rval;
231 void *input;
232 int kind;
233 Py_UCS4 maxchar;
234
235 if (PyUnicode_READY(pystr) == -1)
236 return NULL;
237
238 maxchar = PyUnicode_MAX_CHAR_VALUE(pystr);
239 input_chars = PyUnicode_GET_LENGTH(pystr);
240 input = PyUnicode_DATA(pystr);
241 kind = PyUnicode_KIND(pystr);
242
243 /* Compute the output size */
244 for (i = 0, output_size = 2; i < input_chars; i++) {
245 Py_UCS4 c = PyUnicode_READ(kind, input, i);
246 Py_ssize_t d;
247 switch (c) {
248 case '\\': case '"': case '\b': case '\f':
249 case '\n': case '\r': case '\t':
250 d = 2;
251 break;
252 default:
253 if (c <= 0x1f)
254 d = 6;
255 else
256 d = 1;
257 }
258 if (output_size > PY_SSIZE_T_MAX - d) {
259 PyErr_SetString(PyExc_OverflowError, "string is too long to escape");
260 return NULL;
261 }
262 output_size += d;
263 }
264
265 rval = PyUnicode_New(output_size, maxchar);
266 if (rval == NULL)
267 return NULL;
268
269 kind = PyUnicode_KIND(rval);
270
271#define ENCODE_OUTPUT do { \
272 chars = 0; \
273 output[chars++] = '"'; \
274 for (i = 0; i < input_chars; i++) { \
275 Py_UCS4 c = PyUnicode_READ(kind, input, i); \
276 switch (c) { \
277 case '\\': output[chars++] = '\\'; output[chars++] = c; break; \
278 case '"': output[chars++] = '\\'; output[chars++] = c; break; \
279 case '\b': output[chars++] = '\\'; output[chars++] = 'b'; break; \
280 case '\f': output[chars++] = '\\'; output[chars++] = 'f'; break; \
281 case '\n': output[chars++] = '\\'; output[chars++] = 'n'; break; \
282 case '\r': output[chars++] = '\\'; output[chars++] = 'r'; break; \
283 case '\t': output[chars++] = '\\'; output[chars++] = 't'; break; \
284 default: \
285 if (c <= 0x1f) { \
286 output[chars++] = '\\'; \
287 output[chars++] = 'u'; \
288 output[chars++] = '0'; \
289 output[chars++] = '0'; \
290 output[chars++] = Py_hexdigits[(c >> 4) & 0xf]; \
291 output[chars++] = Py_hexdigits[(c ) & 0xf]; \
292 } else { \
293 output[chars++] = c; \
294 } \
295 } \
296 } \
297 output[chars++] = '"'; \
298 } while (0)
299
300 if (kind == PyUnicode_1BYTE_KIND) {
301 Py_UCS1 *output = PyUnicode_1BYTE_DATA(rval);
302 ENCODE_OUTPUT;
303 } else if (kind == PyUnicode_2BYTE_KIND) {
304 Py_UCS2 *output = PyUnicode_2BYTE_DATA(rval);
305 ENCODE_OUTPUT;
306 } else {
307 Py_UCS4 *output = PyUnicode_4BYTE_DATA(rval);
308 assert(kind == PyUnicode_4BYTE_KIND);
309 ENCODE_OUTPUT;
310 }
311#undef ENCODE_OUTPUT
312
313#ifdef Py_DEBUG
314 assert(_PyUnicode_CheckConsistency(rval, 1));
315#endif
316 return rval;
317}

393static PyObject *
394scanstring_unicode(PyObject *pystr, Py_ssize_t end, int strict, Py_ssize_t *next_end_ptr)
395{
396 /* Read the JSON string from PyUnicode pystr.
397 end is the index of the first character after the quote.
398 if strict is zero then literal control characters are allowed
399 *next_end_ptr is a return-by-reference index of the character
400 after the end quote
401
402 Return value is a new PyUnicode
403 */
404 PyObject *rval = NULL;
405 Py_ssize_t len;
406 Py_ssize_t begin = end - 1;
407 Py_ssize_t next /* = begin */;
408 const void *buf;
409 int kind;
410 PyObject *chunks = NULL;
411 PyObject *chunk = NULL;
412
413 if (PyUnicode_READY(pystr) == -1)
414 return 0;
415
416 len = PyUnicode_GET_LENGTH(pystr);
417 buf = PyUnicode_DATA(pystr);
418 kind = PyUnicode_KIND(pystr);
419
420 if (end < 0 || len < end) {
421 PyErr_SetString(PyExc_ValueError, "end is out of bounds");
422 goto bail;
423 }
424 while (1) {
425 /* Find the end of the string or the next escape */
426 Py_UCS4 c = 0;
427 for (next = end; next < len; next++) {
428 c = PyUnicode_READ(kind, buf, next);
429 if (c == '"' || c == '\\') {
430 break;
431 }
432 else if (strict && c <= 0x1f) {
433 raise_errmsg("Invalid control character at", pystr, next);
434 goto bail;
435 }
436 }
437 if (!(c == '"' || c == '\\')) {
438 raise_errmsg("Unterminated string starting at", pystr, begin);
439 goto bail;
440 }
441 /* Pick up this chunk if it's not zero length */
442 if (next != end) {
443 APPEND_OLD_CHUNK
444 chunk = PyUnicode_FromKindAndData(
445 kind,
446 (char*)buf + kind * end,
447 next - end);
448 if (chunk == NULL) {
449 goto bail;
450 }
451 }
452 next++;
453 if (c == '"') {
454 end = next;
455 break;
456 }
457 if (next == len) {
458 raise_errmsg("Unterminated string starting at", pystr, begin);
459 goto bail;
460 }
461 c = PyUnicode_READ(kind, buf, next);
462 if (c != 'u') {
463 /* Non-unicode backslash escapes */
464 end = next + 1;
465 switch (c) {
466 case '"': break;
467 case '\\': break;
468 case '/': break;
469 case 'b': c = '\b'; break;
470 case 'f': c = '\f'; break;
471 case 'n': c = '\n'; break;
472 case 'r': c = '\r'; break;
473 case 't': c = '\t'; break;
474 default: c = 0;
475 }
476 if (c == 0) {
477 raise_errmsg("Invalid \\escape", pystr, end - 2);
478 goto bail;
479 }
480 }
481 else {
482 c = 0;
483 next++;
484 end = next + 4;
485 if (end >= len) {
486 raise_errmsg("Invalid \\uXXXX escape", pystr, next - 1);
487 goto bail;
488 }
489 /* Decode 4 hex digits */
490 for (; next < end; next++) {
491 Py_UCS4 digit = PyUnicode_READ(kind, buf, next);
492 c <<= 4;
493 switch (digit) {
494 case '0': case '1': case '2': case '3': case '4':
495 case '5': case '6': case '7': case '8': case '9':
496 c |= (digit - '0'); break;
497 case 'a': case 'b': case 'c': case 'd': case 'e':
498 case 'f':
499 c |= (digit - 'a' + 10); break;
500 case 'A': case 'B': case 'C': case 'D': case 'E':
501 case 'F':
502 c |= (digit - 'A' + 10); break;
503 default:
504 raise_errmsg("Invalid \\uXXXX escape", pystr, end - 5);
505 goto bail;
506 }
507 }
508 /* Surrogate pair */
509 if (Py_UNICODE_IS_HIGH_SURROGATE(c) && end + 6 < len &&
510 PyUnicode_READ(kind, buf, next++) == '\\' &&
511 PyUnicode_READ(kind, buf, next++) == 'u') {
512 Py_UCS4 c2 = 0;
513 end += 6;
514 /* Decode 4 hex digits */
515 for (; next < end; next++) {
516 Py_UCS4 digit = PyUnicode_READ(kind, buf, next);
517 c2 <<= 4;
518 switch (digit) {
519 case '0': case '1': case '2': case '3': case '4':
520 case '5': case '6': case '7': case '8': case '9':
521 c2 |= (digit - '0'); break;
522 case 'a': case 'b': case 'c': case 'd': case 'e':
523 case 'f':
524 c2 |= (digit - 'a' + 10); break;
525 case 'A': case 'B': case 'C': case 'D': case 'E':
526 case 'F':
527 c2 |= (digit - 'A' + 10); break;
528 default:
529 raise_errmsg("Invalid \\uXXXX escape", pystr, end - 5);
530 goto bail;
531 }
532 }
533 if (Py_UNICODE_IS_LOW_SURROGATE(c2))
534 c = Py_UNICODE_JOIN_SURROGATES(c, c2);
535 else
536 end -= 6;
537 }
538 }
539 APPEND_OLD_CHUNK
540 chunk = PyUnicode_FromKindAndData(PyUnicode_4BYTE_KIND, &c, 1);
541 if (chunk == NULL) {
542 goto bail;
543 }
544 }
545
546 if (chunks == NULL) {
547 if (chunk != NULL)
548 rval = chunk;
549 else
550 rval = PyUnicode_FromStringAndSize("", 0);
551 }
552 else {
553 APPEND_OLD_CHUNK
554 rval = join_list_unicode(chunks);
555 if (rval == NULL) {
556 goto bail;
557 }
558 Py_CLEAR(chunks);
559 }
560
561 *next_end_ptr = end;
562 return rval;
563bail:
564 *next_end_ptr = -1;
565 Py_XDECREF(chunks);
566 Py_XDECREF(chunk);
567 return NULL;
568}

705 Py_ssize_t end_idx;
706 PyObject *val = NULL;
707 PyObject *rval = NULL;
708 PyObject *key = NULL;
709 int has_pairs_hook = (s->object_pairs_hook != Py_None);
710 Py_ssize_t next_idx;
711
712 if (PyUnicode_READY(pystr) == -1)
713 return NULL;
714
715 str = PyUnicode_DATA(pystr);
716 kind = PyUnicode_KIND(pystr);
717 end_idx = PyUnicode_GET_LENGTH(pystr) - 1;
718
719 if (has_pairs_hook)
720 rval = PyList_New(0);
721 else
722 rval = PyDict_New();
723 if (rval == NULL)
724 return NULL;
725
726 /* skip whitespace after { */
727 while (idx <= end_idx && IS_WHITESPACE(PyUnicode_READ(kind,str, idx))) idx++;
728
729 /* only loop if the object is non-empty */
730 if (idx > end_idx || PyUnicode_READ(kind, str, idx) != '}') {
731 while (1) {
732 PyObject *memokey;
733
734 /* read key */
735 if (idx > end_idx || PyUnicode_READ(kind, str, idx) != '"') {
736 raise_errmsg("Expecting property name enclosed in double quotes", pystr, idx);
737 goto bail;
738 }
739 key = scanstring_unicode(pystr, idx + 1, s->strict, &next_idx);
740 if (key == NULL)
741 goto bail;
742 memokey = PyDict_GetItem(s->memo, key);
743 if (memokey != NULL) {
744 Py_INCREF(memokey);
745 Py_DECREF(key);
746 key = memokey;
747 }
748 else {
749 if (PyDict_SetItem(s->memo, key, key) < 0)
750 goto bail;
751 }
752 idx = next_idx;
753
754 /* skip whitespace between key and : delimiter, read :, skip whitespace */
755 while (idx <= end_idx && IS_WHITESPACE(PyUnicode_READ(kind, str, idx))) idx++;
756 if (idx > end_idx || PyUnicode_READ(kind, str, idx) != ':') {
757 raise_errmsg("Expecting ':' delimiter", pystr, idx);
758 goto bail;
759 }
760 idx++;
761 while (idx <= end_idx && IS_WHITESPACE(PyUnicode_READ(kind, str, idx))) idx++;
762
763 /* read any JSON term */
764 val = scan_once_unicode(s, pystr, idx, &next_idx);
765 if (val == NULL)
766 goto bail;
767
768 if (has_pairs_hook) {
769 PyObject *item = PyTuple_Pack(2, key, val);
770 if (item == NULL)
771 goto bail;
772 Py_CLEAR(key);
773 Py_CLEAR(val);
774 if (PyList_Append(rval, item) == -1) {
775 Py_DECREF(item);
776 goto bail;
777 }
778 Py_DECREF(item);
779 }
780 else {
781 if (PyDict_SetItem(rval, key, val) < 0)
782 goto bail;
783 Py_CLEAR(key);
784 Py_CLEAR(val);
785 }
786 idx = next_idx;
787
788 /* skip whitespace before } or , */
789 while (idx <= end_idx && IS_WHITESPACE(PyUnicode_READ(kind, str, idx))) idx++;
790
791 /* bail if the object is closed or we didn't get the , delimiter */
792 if (idx <= end_idx && PyUnicode_READ(kind, str, idx) == '}')
793 break;
794 if (idx > end_idx || PyUnicode_READ(kind, str, idx) != ',') {
795 raise_errmsg("Expecting ',' delimiter", pystr, idx);
796 goto bail;
797 }
798 idx++;
799
800 /* skip whitespace after , delimiter */
801 while (idx <= end_idx && IS_WHITESPACE(PyUnicode_READ(kind, str, idx))) idx++;
802 }
803 }
804
805 *next_idx_ptr = idx + 1;
806
807 if (has_pairs_hook) {
808 val = PyObject_CallFunctionObjArgs(s->object_pairs_hook, rval, NULL);
809 Py_DECREF(rval);
810 return val;
811 }
812
813 /* if object_hook is not None: rval = object_hook(rval) */
814 if (s->object_hook != Py_None) {
815 val = PyObject_CallFunctionObjArgs(s->object_hook, rval, NULL);
816 Py_DECREF(rval);
817 return val;
818 }
819 return rval;
820bail:
821 Py_XDECREF(key);
822 Py_XDECREF(val);
823 Py_XDECREF(rval);
824 return NULL;
825}

827static PyObject *
828_parse_array_unicode(PyScannerObject *s, PyObject *pystr, Py_ssize_t idx, Py_ssize_t *next_idx_ptr) {
829 /* Read a JSON array from PyUnicode pystr.
830 idx is the index of the first character after the opening brace.
831 *next_idx_ptr is a return-by-reference index to the first character after
832 the closing brace.
833
834 Returns a new PyList
835 */
836 void *str;
837 int kind;
838 Py_ssize_t end_idx;
839 PyObject *val = NULL;
840 PyObject *rval;
841 Py_ssize_t next_idx;
842
843 if (PyUnicode_READY(pystr) == -1)
844 return NULL;
845
846 rval = PyList_New(0);
847 if (rval == NULL)
848 return NULL;
849
850 str = PyUnicode_DATA(pystr);
851 kind = PyUnicode_KIND(pystr);
852 end_idx = PyUnicode_GET_LENGTH(pystr) - 1;
853
854 /* skip whitespace after [ */
855 while (idx <= end_idx && IS_WHITESPACE(PyUnicode_READ(kind, str, idx))) idx++;
856
857 /* only loop if the array is non-empty */
858 if (idx > end_idx || PyUnicode_READ(kind, str, idx) != ']') {
859 while (1) {
860
861 /* read any JSON term */
862 val = scan_once_unicode(s, pystr, idx, &next_idx);
863 if (val == NULL)
864 goto bail;
865
866 if (PyList_Append(rval, val) == -1)
867 goto bail;
868
869 Py_CLEAR(val);
870 idx = next_idx;
871
872 /* skip whitespace between term and , */
873 while (idx <= end_idx && IS_WHITESPACE(PyUnicode_READ(kind, str, idx))) idx++;
874
875 /* bail if the array is closed or we didn't get the , delimiter */
876 if (idx <= end_idx && PyUnicode_READ(kind, str, idx) == ']')
877 break;
878 if (idx > end_idx || PyUnicode_READ(kind, str, idx) != ',') {
879 raise_errmsg("Expecting ',' delimiter", pystr, idx);
880 goto bail;
881 }
882 idx++;
883
884 /* skip whitespace after , */
885 while (idx <= end_idx && IS_WHITESPACE(PyUnicode_READ(kind, str, idx))) idx++;
886 }
887 }
888
889 /* verify that idx < end_idx, PyUnicode_READ(kind, str, idx) should be ']' */
890 if (idx > end_idx || PyUnicode_READ(kind, str, idx) != ']') {
891 raise_errmsg("Expecting value", pystr, end_idx);
892 goto bail;
893 }
894 *next_idx_ptr = idx + 1;
895 return rval;
896bail:
897 Py_XDECREF(val);
898 Py_DECREF(rval);
899 return NULL;
900}

928static PyObject *
929_match_number_unicode(PyScannerObject *s, PyObject *pystr, Py_ssize_t start, Py_ssize_t *next_idx_ptr) {
930 /* Read a JSON number from PyUnicode pystr.
931 idx is the index of the first character of the number
932 *next_idx_ptr is a return-by-reference index to the first character after
933 the number.
934
935 Returns a new PyObject representation of that number:
936 PyLong, or PyFloat.
937 May return other types if parse_int or parse_float are set
938 */
939 void *str;
940 int kind;
941 Py_ssize_t end_idx;
942 Py_ssize_t idx = start;
943 int is_float = 0;
944 PyObject *rval;
945 PyObject *numstr = NULL;
946 PyObject *custom_func;
947
948 if (PyUnicode_READY(pystr) == -1)
949 return NULL;
950
951 str = PyUnicode_DATA(pystr);
952 kind = PyUnicode_KIND(pystr);
953 end_idx = PyUnicode_GET_LENGTH(pystr) - 1;
954
955 /* read a sign if it's there, make sure it's not the end of the string */
956 if (PyUnicode_READ(kind, str, idx) == '-') {
957 idx++;
958 if (idx > end_idx) {
959 raise_stop_iteration(start);
960 return NULL;
961 }
962 }
963
964 /* read as many integer digits as we find as long as it doesn't start with 0 */
965 if (PyUnicode_READ(kind, str, idx) >= '1' && PyUnicode_READ(kind, str, idx) <= '9') {
966 idx++;
967 while (idx <= end_idx && PyUnicode_READ(kind, str, idx) >= '0' && PyUnicode_READ(kind, str, idx) <= '9') idx++;
968 }
969 /* if it starts with 0 we only expect one integer digit */
970 else if (PyUnicode_READ(kind, str, idx) == '0') {
971 idx++;
972 }
973 /* no integer digits, error */
974 else {
975 raise_stop_iteration(start);
976 return NULL;
977 }
978
979 /* if the next char is '.' followed by a digit then read all float digits */
980 if (idx < end_idx && PyUnicode_READ(kind, str, idx) == '.' && PyUnicode_READ(kind, str, idx + 1) >= '0' && PyUnicode_READ(kind, str, idx + 1) <= '9') {
981 is_float = 1;
982 idx += 2;
983 while (idx <= end_idx && PyUnicode_READ(kind, str, idx) >= '0' && PyUnicode_READ(kind, str, idx) <= '9') idx++;
984 }
985
986 /* if the next char is 'e' or 'E' then maybe read the exponent (or backtrack) */
987 if (idx < end_idx && (PyUnicode_READ(kind, str, idx) == 'e' || PyUnicode_READ(kind, str, idx) == 'E')) {
988 Py_ssize_t e_start = idx;
989 idx++;
990
991 /* read an exponent sign if present */
992 if (idx < end_idx && (PyUnicode_READ(kind, str, idx) == '-' || PyUnicode_READ(kind, str, idx) == '+')) idx++;
993
994 /* read all digits */
995 while (idx <= end_idx && PyUnicode_READ(kind, str, idx) >= '0' && PyUnicode_READ(kind, str, idx) <= '9') idx++;
996
997 /* if we got a digit, then parse as float. if not, backtrack */
998 if (PyUnicode_READ(kind, str, idx - 1) >= '0' && PyUnicode_READ(kind, str, idx - 1) <= '9') {
999 is_float = 1;
1000 }
1001 else {
1002 idx = e_start;
1003 }
1004 }
1005
1006 if (is_float && s->parse_float != (PyObject *)&PyFloat_Type)
1007 custom_func = s->parse_float;
1008 else if (!is_float && s->parse_int != (PyObject *) &PyLong_Type)
1009 custom_func = s->parse_int;
1010 else
1011 custom_func = NULL;
1012
1013 if (custom_func) {
1014 /* copy the section we determined to be a number */
1015 numstr = PyUnicode_FromKindAndData(kind,
1016 (char*)str + kind * start,
1017 idx - start);
1018 if (numstr == NULL)
1019 return NULL;
1020 rval = PyObject_CallFunctionObjArgs(custom_func, numstr, NULL);
1021 }
1022 else {
1023 Py_ssize_t i, n;
1024 char *buf;
1025 /* Straight conversion to ASCII, to avoid costly conversion of
1026 decimal unicode digits (which cannot appear here) */
1027 n = idx - start;
1028 numstr = PyBytes_FromStringAndSize(NULL, n);
1029 if (numstr == NULL)
1030 return NULL;
1031 buf = PyBytes_AS_STRING(numstr);
1032 for (i = 0; i < n; i++) {
1033 buf[i] = (char) PyUnicode_READ(kind, str, i + start);
1034 }
1035 if (is_float)
1036 rval = PyFloat_FromString(numstr);
1037 else
1038 rval = PyLong_FromString(buf, NULL, 10);
1039 }
1040 Py_DECREF(numstr);
1041 *next_idx_ptr = idx;
1042 return rval;
1043}

1045static PyObject *
1046scan_once_unicode(PyScannerObject *s, PyObject *pystr, Py_ssize_t idx, Py_ssize_t *next_idx_ptr)
1047{
1048 /* Read one JSON term (of any kind) from PyUnicode pystr.
1049 idx is the index of the first character of the term
1050 *next_idx_ptr is a return-by-reference index to the first character after
1051 the number.
1052
1053 Returns a new PyObject representation of the term.
1054 */
1055 PyObject *res;
1056 void *str;
1057 int kind;
1058 Py_ssize_t length;
1059
1060 if (PyUnicode_READY(pystr) == -1)
1061 return NULL;
1062
1063 str = PyUnicode_DATA(pystr);
1064 kind = PyUnicode_KIND(pystr);
1065 length = PyUnicode_GET_LENGTH(pystr);
1066
1067 if (idx < 0) {
1068 PyErr_SetString(PyExc_ValueError, "idx cannot be negative");
1069 return NULL;
1070 }
1071 if (idx >= length) {
1072 raise_stop_iteration(idx);
1073 return NULL;
1074 }
1075
1076 switch (PyUnicode_READ(kind, str, idx)) {
1077 case '"':
1078 /* string */
1079 return scanstring_unicode(pystr, idx + 1, s->strict, next_idx_ptr);
1080 case '{':
1081 /* object */
1082 if (Py_EnterRecursiveCall(" while decoding a JSON object "
1083 "from a unicode string"))
1084 return NULL;
1085 res = _parse_object_unicode(s, pystr, idx + 1, next_idx_ptr);
1086 Py_LeaveRecursiveCall();
1087 return res;
1088 case '[':
1089 /* array */
1090 if (Py_EnterRecursiveCall(" while decoding a JSON array "
1091 "from a unicode string"))
1092 return NULL;
1093 res = _parse_array_unicode(s, pystr, idx + 1, next_idx_ptr);
1094 Py_LeaveRecursiveCall();
1095 return res;
1096 case 'n':
1097 /* null */
1098 if ((idx + 3 < length) && PyUnicode_READ(kind, str, idx + 1) == 'u' && PyUnicode_READ(kind, str, idx + 2) == 'l' && PyUnicode_READ(kind, str, idx + 3) == 'l') {
1099 *next_idx_ptr = idx + 4;
1100 Py_RETURN_NONE;
1101 }
1102 break;
1103 case 't':
1104 /* true */
1105 if ((idx + 3 < length) && PyUnicode_READ(kind, str, idx + 1) == 'r' && PyUnicode_READ(kind, str, idx + 2) == 'u' && PyUnicode_READ(kind, str, idx + 3) == 'e') {
1106 *next_idx_ptr = idx + 4;
1107 Py_RETURN_TRUE;
1108 }
1109 break;
1110 case 'f':
1111 /* false */
1112 if ((idx + 4 < length) && PyUnicode_READ(kind, str, idx + 1) == 'a' &&
1113 PyUnicode_READ(kind, str, idx + 2) == 'l' &&
1114 PyUnicode_READ(kind, str, idx + 3) == 's' &&
1115 PyUnicode_READ(kind, str, idx + 4) == 'e') {
1116 *next_idx_ptr = idx + 5;
1117 Py_RETURN_FALSE;
1118 }
1119 break;
1120 case 'N':
1121 /* NaN */
1122 if ((idx + 2 < length) && PyUnicode_READ(kind, str, idx + 1) == 'a' &&
1123 PyUnicode_READ(kind, str, idx + 2) == 'N') {
1124 return _parse_constant(s, "NaN", idx, next_idx_ptr);
1125 }
1126 break;
1127 case 'I':
1128 /* Infinity */
1129 if ((idx + 7 < length) && PyUnicode_READ(kind, str, idx + 1) == 'n' &&
1130 PyUnicode_READ(kind, str, idx + 2) == 'f' &&
1131 PyUnicode_READ(kind, str, idx + 3) == 'i' &&
1132 PyUnicode_READ(kind, str, idx + 4) == 'n' &&
1133 PyUnicode_READ(kind, str, idx + 5) == 'i' &&
1134 PyUnicode_READ(kind, str, idx + 6) == 't' &&
1135 PyUnicode_READ(kind, str, idx + 7) == 'y') {
1136 return _parse_constant(s, "Infinity", idx, next_idx_ptr);
1137 }
1138 break;
1139 case '-':
1140 /* -Infinity */
1141 if ((idx + 8 < length) && PyUnicode_READ(kind, str, idx + 1) == 'I' &&
1142 PyUnicode_READ(kind, str, idx + 2) == 'n' &&
1143 PyUnicode_READ(kind, str, idx + 3) == 'f' &&
1144 PyUnicode_READ(kind, str, idx + 4) == 'i' &&
1145 PyUnicode_READ(kind, str, idx + 5) == 'n' &&
1146 PyUnicode_READ(kind, str, idx + 6) == 'i' &&
1147 PyUnicode_READ(kind, str, idx + 7) == 't' &&
1148 PyUnicode_READ(kind, str, idx + 8) == 'y') {
1149 return _parse_constant(s, "-Infinity", idx, next_idx_ptr);
1150 }
1151 break;
1152 }
1153 /* Didn't find a string, object, array, or named constant. Look for a number. */
1154 return _match_number_unicode(s, pystr, idx, next_idx_ptr);
1155}

1440static int
1441encoder_listencode_obj(PyEncoderObject *s, _PyAccu *acc,
1442 PyObject *obj, Py_ssize_t indent_level)
1443{
1444 /* Encode Python object obj to a JSON term */
1445 PyObject *newobj;
1446 int rv;
1447
1448 if (obj == Py_None || obj == Py_True || obj == Py_False) {
1449 PyObject *cstr = _encoded_const(obj);
1450 if (cstr == NULL)
1451 return -1;
1452 return _steal_accumulate(acc, cstr);
1453 }
1454 else if (PyUnicode_Check(obj))
1455 {
1456 PyObject *encoded = encoder_encode_string(s, obj);
1457 if (encoded == NULL)
1458 return -1;
1459 return _steal_accumulate(acc, encoded);
1460 }
1461 else if (PyLong_Check(obj)) {
1462 PyObject *encoded = PyLong_Type.tp_str(obj);
1463 if (encoded == NULL)
1464 return -1;
1465 return _steal_accumulate(acc, encoded);
1466 }
1467 else if (PyFloat_Check(obj)) {
1468 PyObject *encoded = encoder_encode_float(s, obj);
1469 if (encoded == NULL)
1470 return -1;
1471 return _steal_accumulate(acc, encoded);
1472 }
1473 else if (PyList_Check(obj) || PyTuple_Check(obj)) {
1474 if (Py_EnterRecursiveCall(" while encoding a JSON object"))
1475 return -1;
1476 rv = encoder_listencode_list(s, acc, obj, indent_level);
1477 Py_LeaveRecursiveCall();
1478 return rv;
1479 }
1480 else if (PyDict_Check(obj)) {
1481 if (Py_EnterRecursiveCall(" while encoding a JSON object"))
1482 return -1;
1483 rv = encoder_listencode_dict(s, acc, obj, indent_level);
1484 Py_LeaveRecursiveCall();
1485 return rv;
1486 }
1487 else {
1488 PyObject *ident = NULL;
1489 if (s->markers != Py_None) {
1490 int has_key;
1491 ident = PyLong_FromVoidPtr(obj);
1492 if (ident == NULL)
1493 return -1;
1494 has_key = PyDict_Contains(s->markers, ident);
1495 if (has_key) {
1496 if (has_key != -1)
1497 PyErr_SetString(PyExc_ValueError, "Circular reference detected");
1498 Py_DECREF(ident);
1499 return -1;
1500 }
1501 if (PyDict_SetItem(s->markers, ident, obj)) {
1502 Py_DECREF(ident);
1503 return -1;
1504 }
1505 }
1506 newobj = PyObject_CallFunctionObjArgs(s->defaultfn, obj, NULL);
1507 if (newobj == NULL) {
1508 Py_XDECREF(ident);
1509 return -1;
1510 }
1511
1512 if (Py_EnterRecursiveCall(" while encoding a JSON object")) {
1513 Py_DECREF(newobj);
1514 Py_XDECREF(ident);
1515 return -1;
1516 }
1517 rv = encoder_listencode_obj(s, acc, newobj, indent_level);
1518 Py_LeaveRecursiveCall();
1519
1520 Py_DECREF(newobj);
1521 if (rv) {
1522 Py_XDECREF(ident);
1523 return -1;
1524 }
1525 if (ident != NULL) {
1526 if (PyDict_DelItem(s->markers, ident)) {
1527 Py_XDECREF(ident);
1528 return -1;
1529 }
1530 Py_XDECREF(ident);
1531 }
1532 return rv;
1533 }
1534}

1537encoder_listencode_dict(PyEncoderObject *s, _PyAccu *acc,
1538 PyObject *dct, Py_ssize_t indent_level)
1539{

1542 static PyObject *close_dict = NULL;
1543 static PyObject *empty_dict = NULL;
1544 PyObject *kstr = NULL;
1545 PyObject *ident = NULL;
1546 PyObject *it = NULL;
1547 PyObject *items;
1548 PyObject *item = NULL;
1549 Py_ssize_t idx;
1550
1551 if (open_dict == NULL || close_dict == NULL || empty_dict == NULL) {
1552 open_dict = PyUnicode_InternFromString("{");
1553 close_dict = PyUnicode_InternFromString("}");
1554 empty_dict = PyUnicode_InternFromString("{}");
1555 if (open_dict == NULL || close_dict == NULL || empty_dict == NULL)
1556 return -1;
1557 }
1558 if (PyDict_GET_SIZE(dct) == 0) /* Fast path */
1559 return _PyAccu_Accumulate(acc, empty_dict);
1560
1561 if (s->markers != Py_None) {
1562 int has_key;
1563 ident = PyLong_FromVoidPtr(dct);
1564 if (ident == NULL)
1565 goto bail;
1566 has_key = PyDict_Contains(s->markers, ident);
1567 if (has_key) {
1568 if (has_key != -1)
1569 PyErr_SetString(PyExc_ValueError, "Circular reference detected");
1570 goto bail;
1571 }
1572 if (PyDict_SetItem(s->markers, ident, dct)) {
1573 goto bail;
1574 }
1575 }
1576
1577 if (_PyAccu_Accumulate(acc, open_dict))
1578 goto bail;
1579
1580 if (s->indent != Py_None) {
1581 /* TODO: DOES NOT RUN */
1582 indent_level += 1;
1583 /*
1584 newline_indent = '\n' + (' ' * (_indent * _current_indent_level))
1585 separator = _item_separator + newline_indent
1586 buf += newline_indent
1587 */
1588 }
1589
1590 items = PyMapping_Items(dct);
1591 if (items == NULL)
1592 goto bail;
1593 if (s->sort_keys && PyList_Sort(items) < 0) {
1594 Py_DECREF(items);
1595 goto bail;
1596 }
1597 it = PyObject_GetIter(items);
1598 Py_DECREF(items);
1599 if (it == NULL)
1600 goto bail;

1602 while ((item = PyIter_Next(it)) != NULL) {
1603 PyObject *encoded, *key, *value;

1610 Py_INCREF(key);

1612 }

1617 }

1620 True and False are also 1 and 0.*/
1621 kstr = _encoded_const(key);
1622 if (kstr == NULL)
1623 goto bail;
1624 }
1625 else if (PyLong_Check(key)) {
1626 kstr = PyLong_Type.tp_str(key);
1627 if (kstr == NULL) {
1628 goto bail;
1629 }
1630 }
1631 else if (s->skipkeys) {
1632 Py_DECREF(item);
1633 continue;
1634 }
1635 else {
1636 /* TODO: include repr of key */
1637 PyErr_SetString(PyExc_TypeError, "keys must be a string");
1638 goto bail;
1639 }
1640
1641 if (idx) {
1642 if (_PyAccu_Accumulate(acc, s->item_separator))
1643 goto bail;
1644 }
1645
1646 encoded = encoder_encode_string(s, kstr);
1647 Py_CLEAR(kstr);
1648 if (encoded == NULL)
1649 goto bail;
1650 if (_PyAccu_Accumulate(acc, encoded)) {
1651 Py_DECREF(encoded);
1652 goto bail;
1653 }
1654 Py_DECREF(encoded);
1655 if (_PyAccu_Accumulate(acc, s->key_separator))
1656 goto bail;
1657
1658 value = PyTuple_GET_ITEM(item, 1);
1659 if (encoder_listencode_obj(s, acc, value, indent_level))
1660 goto bail;
1661 idx += 1;
1662 Py_DECREF(item);
1663 }
1664 if (PyErr_Occurred())
1665 goto bail;
1666 Py_CLEAR(it);
1667
1668 if (ident != NULL) {
1669 if (PyDict_DelItem(s->markers, ident))
1670 goto bail;
1671 Py_CLEAR(ident);
1672 }
1673 /* TODO DOES NOT RUN; dead code
1674 if (s->indent != Py_None) {
1675 indent_level -= 1;
1676
1677 yield '\n' + (' ' * (_indent * _current_indent_level))
1678 }*/
1679 if (_PyAccu_Accumulate(acc, close_dict))
1680 goto bail;
1681 return 0;
1682
1683bail:
1684 Py_XDECREF(it);
1685 Py_XDECREF(item);
1686 Py_XDECREF(kstr);
1687 Py_XDECREF(ident);
1688 return -1;
