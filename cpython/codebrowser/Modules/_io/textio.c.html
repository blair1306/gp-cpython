298#define SEEN_CR 1
299#define SEEN_LF 2
300#define SEEN_CRLF 4
301#define SEEN_ALL (SEEN_CR | SEEN_LF | SEEN_CRLF)
302
303PyObject *
304_PyIncrementalNewlineDecoder_decode(PyObject *myself,
305 PyObject *input, int final)
306{
307 PyObject *output;
308 Py_ssize_t output_len;
309 nldecoder_object *self = (nldecoder_object *) myself;
310
311 if (self->decoder == NULL) {
312 PyErr_SetString(PyExc_ValueError,
313 "IncrementalNewlineDecoder.__init__ not called");
314 return NULL;
315 }
316
317 /* decode input (with the eventual \r from a previous pass) */
318 if (self->decoder != Py_None) {
319 output = PyObject_CallMethodObjArgs(self->decoder,
320 _PyIO_str_decode, input, final ? Py_True : Py_False, NULL);
321 }
322 else {
323 output = input;
324 Py_INCREF(output);
325 }
326
327 if (check_decoded(output) < 0)
328 return NULL;
329
330 output_len = PyUnicode_GET_LENGTH(output);
331 if (self->pendingcr && (final || output_len > 0)) {
332 /* Prefix output with CR */
333 int kind;
334 PyObject *modified;
335 char *out;
336
337 modified = PyUnicode_New(output_len + 1,
338 PyUnicode_MAX_CHAR_VALUE(output));
339 if (modified == NULL)
340 goto error;
341 kind = PyUnicode_KIND(modified);
342 out = PyUnicode_DATA(modified);
343 PyUnicode_WRITE(kind, PyUnicode_DATA(modified), 0, '\r');
344 memcpy(out + kind, PyUnicode_DATA(output), kind * output_len);
345 Py_DECREF(output);
346 output = modified; /* output remains ready */
347 self->pendingcr = 0;
348 output_len++;
349 }
350
351 /* retain last \r even when not translating data:
352 * then readline() is sure to get \r\n in one pass
353 */
354 if (!final) {
355 if (output_len > 0
356 && PyUnicode_READ_CHAR(output, output_len - 1) == '\r')
357 {
358 PyObject *modified = PyUnicode_Substring(output, 0, output_len -1);
359 if (modified == NULL)
360 goto error;
361 Py_DECREF(output);
362 output = modified;
363 self->pendingcr = 1;
364 }
365 }
366
367 /* Record which newlines are read and do newline translation if desired,
368 all in one pass. */
369 {
370 void *in_str;
371 Py_ssize_t len;
372 int seennl = self->seennl;
373 int only_lf = 0;
374 int kind;
375
376 in_str = PyUnicode_DATA(output);
377 len = PyUnicode_GET_LENGTH(output);
378 kind = PyUnicode_KIND(output);
379
380 if (len == 0)
381 return output;
382
383 /* If, up to now, newlines are consistently \n, do a quick check
384 for the \r *byte* with the libc's optimized memchr.
385 */
386 if (seennl == SEEN_LF || seennl == 0) {
387 only_lf = (memchr(in_str, '\r', kind * len) == NULL);
388 }
389
390 if (only_lf) {
391 /* If not already seen, quick scan for a possible "\n" character.
392 (there's nothing else to be done, even when in translation mode)
393 */
394 if (seennl == 0 &&
395 memchr(in_str, '\n', kind * len) != NULL) {
396 if (kind == PyUnicode_1BYTE_KIND)
397 seennl |= SEEN_LF;
398 else {
399 Py_ssize_t i = 0;
400 for (;;) {
401 Py_UCS4 c;
402 /* Fast loop for non-control characters */
403 while (PyUnicode_READ(kind, in_str, i) > '\n')
404 i++;
405 c = PyUnicode_READ(kind, in_str, i++);
406 if (c == '\n') {
407 seennl |= SEEN_LF;
408 break;
409 }
410 if (i >= len)
411 break;
412 }
413 }
414 }
415 /* Finished: we have scanned for newlines, and none of them
416 need translating */
417 }
418 else if (!self->translate) {
419 Py_ssize_t i = 0;
420 /* We have already seen all newline types, no need to scan again */
421 if (seennl == SEEN_ALL)
422 goto endscan;
423 for (;;) {
424 Py_UCS4 c;
425 /* Fast loop for non-control characters */
426 while (PyUnicode_READ(kind, in_str, i) > '\r')
427 i++;
428 c = PyUnicode_READ(kind, in_str, i++);
429 if (c == '\n')
430 seennl |= SEEN_LF;
431 else if (c == '\r') {
432 if (PyUnicode_READ(kind, in_str, i) == '\n') {
433 seennl |= SEEN_CRLF;
434 i++;
435 }
436 else
437 seennl |= SEEN_CR;
438 }
439 if (i >= len)
440 break;
441 if (seennl == SEEN_ALL)
442 break;
443 }
444 endscan:
445 ;
446 }
447 else {
448 void *translated;
449 int kind = PyUnicode_KIND(output);
450 void *in_str = PyUnicode_DATA(output);
451 Py_ssize_t in, out;
452 /* XXX: Previous in-place translation here is disabled as
453 resizing is not possible anymore */
454 /* We could try to optimize this so that we only do a copy
455 when there is something to translate. On the other hand,
456 we already know there is a \r byte, so chances are high
457 that something needs to be done. */
458 translated = PyMem_Malloc(kind * len);
459 if (translated == NULL) {
460 PyErr_NoMemory();
461 goto error;
462 }
463 in = out = 0;
464 for (;;) {
465 Py_UCS4 c;
466 /* Fast loop for non-control characters */
467 while ((c = PyUnicode_READ(kind, in_str, in++)) > '\r')
468 PyUnicode_WRITE(kind, translated, out++, c);
469 if (c == '\n') {
470 PyUnicode_WRITE(kind, translated, out++, c);
471 seennl |= SEEN_LF;
472 continue;
473 }
474 if (c == '\r') {
475 if (PyUnicode_READ(kind, in_str, in) == '\n') {
476 in++;
477 seennl |= SEEN_CRLF;
478 }
479 else
480 seennl |= SEEN_CR;
481 PyUnicode_WRITE(kind, translated, out++, '\n');
482 continue;
483 }
484 if (in > len)
485 break;
486 PyUnicode_WRITE(kind, translated, out++, c);
487 }
488 Py_DECREF(output);
489 output = PyUnicode_FromKindAndData(kind, translated, out);
490 PyMem_Free(translated);
491 if (!output)
492 return NULL;
493 }
494 self->seennl |= seennl;
495 }
496
497 return output;
498
499 error:
500 Py_DECREF(output);
501 return NULL;
502}

1473 Py_INCREF(chars);
1474 }
1475
1476 self->decoded_chars_used += n;
1477 return chars;
1478}

1480/* Read and decode the next chunk of data from the BufferedReader.
1481 */
1482static int
1483textiowrapper_read_chunk(textio *self, Py_ssize_t size_hint)
1484{
1485 PyObject *dec_buffer = NULL;
1486 PyObject *dec_flags = NULL;
1487 PyObject *input_chunk = NULL;
1488 Py_buffer input_chunk_buf;
1489 PyObject *decoded_chars, *chunk_size;
1490 Py_ssize_t nbytes, nchars;
1491 int eof;
1492
1493 /* The return value is True unless EOF was reached. The decoded string is
1494 * placed in self._decoded_chars (replacing its previous value). The
1495 * entire input chunk is sent to the decoder, though some of it may remain
1496 * buffered in the decoder, yet to be converted.
1497 */
1498
1499 if (self->decoder == NULL) {
1500 _unsupported("not readable");
1501 return -1;
1502 }
1503
1504 if (self->telling) {
1505 /* To prepare for tell(), we need to snapshot a point in the file
1506 * where the decoder's input buffer is empty.
1507 */
1508
1509 PyObject *state = PyObject_CallMethodObjArgs(self->decoder,
1510 _PyIO_str_getstate, NULL);
1511 if (state == NULL)
1512 return -1;
1513 /* Given this, we know there was a valid snapshot point
1514 * len(dec_buffer) bytes ago with decoder state (b'', dec_flags).
1515 */
1516 if (PyArg_ParseTuple(state, "OO", &dec_buffer, &dec_flags) < 0) {
1517 Py_DECREF(state);
1518 return -1;
1519 }
1520
1521 if (!PyBytes_Check(dec_buffer)) {
1522 PyErr_Format(PyExc_TypeError,
1523 "decoder getstate() should have returned a bytes "
1524 "object, not '%.200s'",
1525 Py_TYPE(dec_buffer)->tp_name);
1526 Py_DECREF(state);
1527 return -1;
1528 }
1529 Py_INCREF(dec_buffer);
1530 Py_INCREF(dec_flags);
1531 Py_DECREF(state);
1532 }
1533
1534 /* Read a chunk, decode it, and put the result in self._decoded_chars. */
1535 if (size_hint > 0) {
1536 size_hint = (Py_ssize_t)(Py_MAX(self->b2cratio, 1.0) * size_hint);
1537 }
1538 chunk_size = PyLong_FromSsize_t(Py_MAX(self->chunk_size, size_hint));
1539 if (chunk_size == NULL)
1540 goto fail;
1541
1542 input_chunk = PyObject_CallMethodObjArgs(self->buffer,
1543 (self->has_read1 ? _PyIO_str_read1: _PyIO_str_read),
1544 chunk_size, NULL);
1545 Py_DECREF(chunk_size);
1546 if (input_chunk == NULL)
1547 goto fail;
1548
1549 if (PyObject_GetBuffer(input_chunk, &input_chunk_buf, 0) != 0) {
1550 PyErr_Format(PyExc_TypeError,
1551 "underlying %s() should have returned a bytes-like object, "
1552 "not '%.200s'", (self->has_read1 ? "read1": "read"),
1553 Py_TYPE(input_chunk)->tp_name);
1554 goto fail;
1555 }
1556
1557 nbytes = input_chunk_buf.len;
1558 eof = (nbytes == 0);
1559 if (Py_TYPE(self->decoder) == &PyIncrementalNewlineDecoder_Type) {
1560 decoded_chars = _PyIncrementalNewlineDecoder_decode(
1561 self->decoder, input_chunk, eof);
1562 }
1563 else {
1564 decoded_chars = PyObject_CallMethodObjArgs(self->decoder,
1565 _PyIO_str_decode, input_chunk, eof ? Py_True : Py_False, NULL);
1566 }
1567 PyBuffer_Release(&input_chunk_buf);
1568
1569 if (check_decoded(decoded_chars) < 0)
1570 goto fail;
1571 textiowrapper_set_decoded_chars(self, decoded_chars);
1572 nchars = PyUnicode_GET_LENGTH(decoded_chars);
1573 if (nchars > 0)
1574 self->b2cratio = (double) nbytes / nchars;
1575 else
1576 self->b2cratio = 0.0;
1577 if (nchars > 0)
1578 eof = 0;
1579
1580 if (self->telling) {
1581 /* At the snapshot point, len(dec_buffer) bytes before the read, the
1582 * next input to be decoded is dec_buffer + input_chunk.
1583 */
1584 PyObject *next_input = dec_buffer;
1585 PyBytes_Concat(&next_input, input_chunk);
1586 if (next_input == NULL) {
1587 dec_buffer = NULL; /* Reference lost to PyBytes_Concat */
1588 goto fail;
1589 }
1590 Py_XSETREF(self->snapshot, Py_BuildValue("NN", dec_flags, next_input));
1591 }
1592 Py_DECREF(input_chunk);
1593
1594 return (eof == 0);
1595
1596 fail:
1597 Py_XDECREF(dec_buffer);
1598 Py_XDECREF(dec_flags);
1599 Py_XDECREF(input_chunk);
1600 return -1;
1601}

1710/* NOTE: `end` must point to the real end of the Py_UCS4 storage,
1711 that is to the NUL character. Otherwise the function will produce
1712 incorrect results. */
1713static const char *
1714find_control_char(int kind, const char *s, const char *end, Py_UCS4 ch)
1715{
1716 if (kind == PyUnicode_1BYTE_KIND) {
1717 assert(ch < 256);
1718 return (char *) memchr((void *) s, (char) ch, end - s);
1719 }
1720 for (;;) {
1721 while (PyUnicode_READ(kind, s, 0) > ch)
1722 s += kind;
1723 if (PyUnicode_READ(kind, s, 0) == ch)
1724 return s;
1725 if (s == end)
1726 return NULL;
1727 s += kind;
1728 }
1729}

1731Py_ssize_t
1732_PyIO_find_line_ending(
1733 int translated, int universal, PyObject *readnl,
1734 int kind, const char *start, const char *end, Py_ssize_t *consumed)
1735{
1736 Py_ssize_t len = ((char*)end - (char*)start)/kind;
1737
1738 if (translated) {
1739 /* Newlines are already translated, only search for \n */
1740 const char *pos = find_control_char(kind, start, end, '\n');
1741 if (pos != NULL)
1742 return (pos - start)/kind + 1;
1743 else {
1744 *consumed = len;
1745 return -1;
1746 }
1747 }
1748 else if (universal) {
1749 /* Universal newline search. Find any of \r, \r\n, \n
1750 * The decoder ensures that \r\n are not split in two pieces
1751 */
1752 const char *s = start;
1753 for (;;) {
1754 Py_UCS4 ch;
1755 /* Fast path for non-control chars. The loop always ends
1756 since the Unicode string is NUL-terminated. */
1757 while (PyUnicode_READ(kind, s, 0) > '\r')
1758 s += kind;
1759 if (s >= end) {
1760 *consumed = len;
1761 return -1;
1762 }
1763 ch = PyUnicode_READ(kind, s, 0);
1764 s += kind;
1765 if (ch == '\n')
1766 return (s - start)/kind;
1767 if (ch == '\r') {
1768 if (PyUnicode_READ(kind, s, 0) == '\n')
1769 return (s - start)/kind + 1;
1770 else
1771 return (s - start)/kind;
1772 }
1773 }
1774 }
1775 else {
1776 /* Non-universal mode. */
1777 Py_ssize_t readnl_len = PyUnicode_GET_LENGTH(readnl);
1778 Py_UCS1 *nl = PyUnicode_1BYTE_DATA(readnl);
1779 /* Assume that readnl is an ASCII character. */
1780 assert(PyUnicode_KIND(readnl) == PyUnicode_1BYTE_KIND);
1781 if (readnl_len == 1) {
1782 const char *pos = find_control_char(kind, start, end, nl[0]);
1783 if (pos != NULL)
1784 return (pos - start)/kind + 1;
1785 *consumed = len;
1786 return -1;
1787 }
1788 else {
1789 const char *s = start;
1790 const char *e = end - (readnl_len - 1)*kind;
1791 const char *pos;
1792 if (e < s)
1793 e = s;
1794 while (s < e) {
1795 Py_ssize_t i;
1796 const char *pos = find_control_char(kind, s, end, nl[0]);
1797 if (pos == NULL || pos >= e)
1798 break;
1799 for (i = 1; i < readnl_len; i++) {
1800 if (PyUnicode_READ(kind, pos, i) != nl[i])
1801 break;
1802 }
1803 if (i == readnl_len)
1804 return (pos - start)/kind + readnl_len;
1805 s = pos + kind;
1806 }
1807 pos = find_control_char(kind, e, end, nl[0]);
1808 if (pos == NULL)
1809 *consumed = len;
1810 else
1811 *consumed = (pos - start)/kind;
1812 return -1;
1813 }
1814 }
1815}

1817static PyObject *
1818_textiowrapper_readline(textio *self, Py_ssize_t limit)
1819{
1820 PyObject *line = NULL, *chunks = NULL, *remaining = NULL;
1821 Py_ssize_t start, endpos, chunked, offset_to_buffer;
1822 int res;
1823
1824 CHECK_CLOSED(self);
1825
1826 if (_textiowrapper_writeflush(self) < 0)
1827 return NULL;
1828
1829 chunked = 0;
1830
1831 while (1) {
1832 char *ptr;
1833 Py_ssize_t line_len;
1834 int kind;
1835 Py_ssize_t consumed = 0;
1836
1837 /* First, get some data if necessary */
1838 res = 1;
1839 while (!self->decoded_chars ||
1840 !PyUnicode_GET_LENGTH(self->decoded_chars)) {
1841 res = textiowrapper_read_chunk(self, 0);
1842 if (res < 0) {
1843 /* NOTE: PyErr_SetFromErrno() calls PyErr_CheckSignals()
1844 when EINTR occurs so we needn't do it ourselves. */
1845 if (_PyIO_trap_eintr()) {
1846 continue;
1847 }
1848 goto error;
1849 }
1850 if (res == 0)
1851 break;
1852 }
1853 if (res == 0) {
1854 /* end of file */
1855 textiowrapper_set_decoded_chars(self, NULL);
1856 Py_CLEAR(self->snapshot);
1857 start = endpos = offset_to_buffer = 0;
1858 break;
1859 }
1860
1861 if (remaining == NULL) {
1862 line = self->decoded_chars;
1863 start = self->decoded_chars_used;
1864 offset_to_buffer = 0;
1865 Py_INCREF(line);
1866 }
1867 else {
1868 assert(self->decoded_chars_used == 0);
1869 line = PyUnicode_Concat(remaining, self->decoded_chars);
1870 start = 0;
1871 offset_to_buffer = PyUnicode_GET_LENGTH(remaining);
1872 Py_CLEAR(remaining);
1873 if (line == NULL)
1874 goto error;
1875 if (PyUnicode_READY(line) == -1)
1876 goto error;
1877 }
1878
1879 ptr = PyUnicode_DATA(line);
1880 line_len = PyUnicode_GET_LENGTH(line);
1881 kind = PyUnicode_KIND(line);
1882
1883 endpos = _PyIO_find_line_ending(
1884 self->readtranslate, self->readuniversal, self->readnl,
1885 kind,
1886 ptr + kind * start,
1887 ptr + kind * line_len,
1888 &consumed);
1889 if (endpos >= 0) {
1890 endpos += start;
1891 if (limit >= 0 && (endpos - start) + chunked >= limit)
1892 endpos = start + limit - chunked;
1893 break;
1894 }
1895
1896 /* We can put aside up to `endpos` */
1897 endpos = consumed + start;
1898 if (limit >= 0 && (endpos - start) + chunked >= limit) {
1899 /* Didn't find line ending, but reached length limit */
1900 endpos = start + limit - chunked;
1901 break;
1902 }
1903
1904 if (endpos > start) {
1905 /* No line ending seen yet - put aside current data */
1906 PyObject *s;
1907 if (chunks == NULL) {
1908 chunks = PyList_New(0);
1909 if (chunks == NULL)
1910 goto error;
1911 }
1912 s = PyUnicode_Substring(line, start, endpos);
1913 if (s == NULL)
1914 goto error;
1915 if (PyList_Append(chunks, s) < 0) {
1916 Py_DECREF(s);
1917 goto error;
1918 }
1919 chunked += PyUnicode_GET_LENGTH(s);
1920 Py_DECREF(s);
1921 }
1922 /* There may be some remaining bytes we'll have to prepend to the
1923 next chunk of data */
1924 if (endpos < line_len) {
1925 remaining = PyUnicode_Substring(line, endpos, line_len);
1926 if (remaining == NULL)
1927 goto error;
1928 }
1929 Py_CLEAR(line);
1930 /* We have consumed the buffer */
1931 textiowrapper_set_decoded_chars(self, NULL);
1932 }
1933
1934 if (line != NULL) {
1935 /* Our line ends in the current buffer */
1936 self->decoded_chars_used = endpos - offset_to_buffer;
1937 if (start > 0 || endpos < PyUnicode_GET_LENGTH(line)) {
1938 PyObject *s = PyUnicode_Substring(line, start, endpos);
1939 Py_CLEAR(line);
1940 if (s == NULL)
1941 goto error;
1942 line = s;
1943 }
1944 }
1945 if (remaining != NULL) {
1946 if (chunks == NULL) {
1947 chunks = PyList_New(0);
1948 if (chunks == NULL)
1949 goto error;
1950 }
1951 if (PyList_Append(chunks, remaining) < 0)
1952 goto error;
1953 Py_CLEAR(remaining);
1954 }
1955 if (chunks != NULL) {
1956 if (line != NULL) {
1957 if (PyList_Append(chunks, line) < 0)
1958 goto error;
1959 Py_DECREF(line);
1960 }
1961 line = PyUnicode_Join(_PyIO_empty_str, chunks);
1962 if (line == NULL)
1963 goto error;
1964 Py_CLEAR(chunks);
1965 }
1966 if (line == NULL) {
1967 Py_INCREF(_PyIO_empty_str);
1968 line = _PyIO_empty_str;
1969 }
1970
1971 return line;
1972
1973 error:
1974 Py_XDECREF(chunks);
1975 Py_XDECREF(remaining);
1976 Py_XDECREF(line);
1977 return NULL;
1978}

1994/* Seek and Tell */
1995
1996typedef struct {
1997 Py_off_t start_pos;
1998 int dec_flags;
1999 int bytes_to_feed;
2000 int chars_to_skip;
2001 char need_eof;
2002} cookie_type;
2003
2004/*
2005 To speed up cookie packing/unpacking, we store the fields in a temporary
2006 string and call _PyLong_FromByteArray() or _PyLong_AsByteArray (resp.).
2007 The following macros define at which offsets in the intermediary byte
2008 string the various CookieStruct fields will be stored.
2009 */
2010
2011#define COOKIE_BUF_LEN (sizeof(Py_off_t) + 3 * sizeof(int) + sizeof(char))
2012
2013#if PY_BIG_ENDIAN
2014/* We want the least significant byte of start_pos to also be the least
2015 significant byte of the cookie, which means that in big-endian mode we
2016 must copy the fields in reverse order. */
2017
2018# define OFF_START_POS (sizeof(char) + 3 * sizeof(int))
2019# define OFF_DEC_FLAGS (sizeof(char) + 2 * sizeof(int))
2020# define OFF_BYTES_TO_FEED (sizeof(char) + sizeof(int))
2021# define OFF_CHARS_TO_SKIP (sizeof(char))
2022# define OFF_NEED_EOF 0
2023
2024#else
2025/* Little-endian mode: the least significant byte of start_pos will
2026 naturally end up the least significant byte of the cookie. */
2027
2028# define OFF_START_POS 0
2029# define OFF_DEC_FLAGS (sizeof(Py_off_t))
2030# define OFF_BYTES_TO_FEED (sizeof(Py_off_t) + sizeof(int))
2031# define OFF_CHARS_TO_SKIP (sizeof(Py_off_t) + 2 * sizeof(int))
2032# define OFF_NEED_EOF (sizeof(Py_off_t) + 3 * sizeof(int))
2033
2034#endif
2035
2036static int
2037textiowrapper_parse_cookie(cookie_type *cookie, PyObject *cookieObj)
2038{
2039 unsigned char buffer[COOKIE_BUF_LEN];
2040 PyLongObject *cookieLong = (PyLongObject *)PyNumber_Long(cookieObj);
2041 if (cookieLong == NULL)
2042 return -1;
2043
2044 if (_PyLong_AsByteArray(cookieLong, buffer, sizeof(buffer),
2045 PY_LITTLE_ENDIAN, 0) < 0) {
2046 Py_DECREF(cookieLong);
2047 return -1;
2048 }
2049 Py_DECREF(cookieLong);
2050
2051 memcpy(&cookie->start_pos, buffer + OFF_START_POS, sizeof(cookie->start_pos));
2052 memcpy(&cookie->dec_flags, buffer + OFF_DEC_FLAGS, sizeof(cookie->dec_flags));
2053 memcpy(&cookie->bytes_to_feed, buffer + OFF_BYTES_TO_FEED, sizeof(cookie->bytes_to_feed));
2054 memcpy(&cookie->chars_to_skip, buffer + OFF_CHARS_TO_SKIP, sizeof(cookie->chars_to_skip));
2055 memcpy(&cookie->need_eof, buffer + OFF_NEED_EOF, sizeof(cookie->need_eof));
2056
2057 return 0;
2058}

2475 * can safely start from there and advance to this location).

2493 chars_to_skip -= chars_decoded;
2494 cookie.dec_flags = dec_flags;
2495 cookie.bytes_to_feed = 0;

2497 }

2501 }

2504 PyObject *decoded = _PyObject_CallMethodId(
2505 self->decoder, &PyId_decode, "yi", "", /* final = */ 1);
2506 if (check_decoded(decoded) < 0)
2507 goto fail;

2509 Py_DECREF(decoded);
2510 cookie.need_eof = 1;
2511
2512 if (chars_decoded < chars_to_skip) {
2513 PyErr_SetString(PyExc_OSError,

2515 goto fail;
2516 }
2517 }
2518
2519finally:

2521 Py_DECREF(saved_state);

2524 Py_DECREF(res);
2525
2526 /* The returned cookie corresponds to the last safe start point. */
2527 cookie.chars_to_skip = Py_SAFE_DOWNCAST(chars_to_skip, Py_ssize_t, int);

2535 _PyErr_ChainExceptions(type, value, traceback);
2536 Py_DECREF(saved_state);
2537 Py_XDECREF(res);
2538 }
2539 return NULL;
2540}
