22BLAKE2_LOCAL_INLINE(uint32_t) load32( const void *src )
23{
24#if defined(NATIVE_LITTLE_ENDIAN)
25 uint32_t w;
26 memcpy(&w, src, sizeof w);
27 return w;
28#else
29 const uint8_t *p = ( const uint8_t * )src;
30 uint32_t w = *p++;
31 w |= ( uint32_t )( *p++ ) << 8;
32 w |= ( uint32_t )( *p++ ) << 16;
33 w |= ( uint32_t )( *p++ ) << 24;
34 return w;
35#endif
36}

39{
40#if defined(NATIVE_LITTLE_ENDIAN)

43 return w;
44#else
45 const uint8_t *p = ( const uint8_t * )src;
46 uint64_t w = *p++;
47 w |= ( uint64_t )( *p++ ) << 8;
48 w |= ( uint64_t )( *p++ ) << 16;
49 w |= ( uint64_t )( *p++ ) << 24;
50 w |= ( uint64_t )( *p++ ) << 32;
51 w |= ( uint64_t )( *p++ ) << 40;
52 w |= ( uint64_t )( *p++ ) << 48;
53 w |= ( uint64_t )( *p++ ) << 56;
54 return w;
55#endif
56}

58BLAKE2_LOCAL_INLINE(void) store32( void *dst, uint32_t w )
59{
60#if defined(NATIVE_LITTLE_ENDIAN)
61 memcpy(dst, &w, sizeof w);
62#else
63 uint8_t *p = ( uint8_t * )dst;
64 *p++ = ( uint8_t )w; w >>= 8;
65 *p++ = ( uint8_t )w; w >>= 8;
66 *p++ = ( uint8_t )w; w >>= 8;
67 *p++ = ( uint8_t )w;
68#endif
69}

111BLAKE2_LOCAL_INLINE(uint32_t) rotl32( const uint32_t w, const unsigned c )
112{
113 return ( w << c ) | ( w >> ( 32 - c ) );
114}
115
116BLAKE2_LOCAL_INLINE(uint64_t) rotl64( const uint64_t w, const unsigned c )
117{
118 return ( w << c ) | ( w >> ( 64 - c ) );
119}

121BLAKE2_LOCAL_INLINE(uint32_t) rotr32( const uint32_t w, const unsigned c )
122{
123 return ( w >> c ) | ( w << ( 32 - c ) );
124}
125
126BLAKE2_LOCAL_INLINE(uint64_t) rotr64( const uint64_t w, const unsigned c )
127{
128 return ( w >> c ) | ( w << ( 64 - c ) );
129}

132BLAKE2_LOCAL_INLINE(void) secure_zero_memory(void *v, size_t n)
133{
134 static void *(*const volatile memset_v)(void *, int, size_t) = &memset;
