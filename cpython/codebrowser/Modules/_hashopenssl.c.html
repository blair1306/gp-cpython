78static PyObject *
79_setException(PyObject *exc)
80{
81 unsigned long errcode;
82 const char *lib, *func, *reason;
83
84 errcode = ERR_peek_last_error();
85 if (!errcode) {
86 PyErr_SetString(exc, "unknown reasons");
87 return NULL;
88 }
89 ERR_clear_error();
90
91 lib = ERR_lib_error_string(errcode);
92 func = ERR_func_error_string(errcode);
93 reason = ERR_reason_error_string(errcode);
94
95 if (lib && func) {
96 PyErr_Format(exc, "[%s: %s] %s", lib, func, reason);
97 }
98 else if (lib) {
99 PyErr_Format(exc, "[%s] %s", lib, reason);
100 }
101 else {
102 PyErr_SetString(exc, reason);
103 }
104 return NULL;
105}

108static EVPobject *
109newEVPobject(PyObject *name)
110{
111 EVPobject *retval = (EVPobject *)PyObject_New(EVPobject, &EVPtype);
112 if (retval == NULL) {
113 return NULL;
114 }
115
116 retval->ctx = EVP_MD_CTX_new();
117 if (retval->ctx == NULL) {
118 PyErr_NoMemory();
119 return NULL;
120 }
121
122 /* save the name for .name to return */
123 Py_INCREF(name);
124 retval->name = name;
125#ifdef WITH_THREAD
126 retval->lock = NULL;
127#endif
128
129 return retval;
130}

132static void
133EVP_hash(EVPobject *self, const void *vp, Py_ssize_t len)
134{
135 unsigned int process;
136 const unsigned char *cp = (const unsigned char *)vp;
137 while (0 < len) {
138 if (len > (Py_ssize_t)MUNCH_SIZE)
139 process = MUNCH_SIZE;
140 else
141 process = Py_SAFE_DOWNCAST(len, Py_ssize_t, unsigned int);
142 if (!EVP_DigestUpdate(self->ctx, (const void*)cp, process)) {
143 _setException(PyExc_ValueError);
144 break;
145 }
146 len -= process;
147 cp += process;
148 }
149}

165static int
166locked_EVP_MD_CTX_copy(EVP_MD_CTX *new_ctx_p, EVPobject *self)
167{
168 int result;
169 ENTER_HASHLIB(self);
170 result = EVP_MD_CTX_copy(new_ctx_p, self->ctx);
171 LEAVE_HASHLIB(self);
172 return result;
173}

197static PyObject *
198EVP_digest(EVPobject *self, PyObject *unused)
199{
200 unsigned char digest[EVP_MAX_MD_SIZE];
201 EVP_MD_CTX *temp_ctx;
202 PyObject *retval;
203 unsigned int digest_size;
204
205 temp_ctx = EVP_MD_CTX_new();
206 if (temp_ctx == NULL) {
207 PyErr_NoMemory();
208 return NULL;
209 }
210
211 if (!locked_EVP_MD_CTX_copy(temp_ctx, self)) {
212 return _setException(PyExc_ValueError);
213 }
214 digest_size = EVP_MD_CTX_size(temp_ctx);
215 if (!EVP_DigestFinal(temp_ctx, digest, NULL)) {
216 _setException(PyExc_ValueError);
217 return NULL;
218 }
219
220 retval = PyBytes_FromStringAndSize((const char *)digest, digest_size);
221 EVP_MD_CTX_free(temp_ctx);
222 return retval;
223}

228static PyObject *
229EVP_hexdigest(EVPobject *self, PyObject *unused)
230{
231 unsigned char digest[EVP_MAX_MD_SIZE];
232 EVP_MD_CTX *temp_ctx;
233 unsigned int digest_size;
234
235 temp_ctx = EVP_MD_CTX_new();
236 if (temp_ctx == NULL) {
237 PyErr_NoMemory();
238 return NULL;
239 }
240
241 /* Get the raw (binary) digest value */
242 if (!locked_EVP_MD_CTX_copy(temp_ctx, self)) {
243 return _setException(PyExc_ValueError);
244 }
245 digest_size = EVP_MD_CTX_size(temp_ctx);
246 if (!EVP_DigestFinal(temp_ctx, digest, NULL)) {
247 _setException(PyExc_ValueError);
248 return NULL;
249 }
250
251 EVP_MD_CTX_free(temp_ctx);
252
253 return _Py_strhex((const char *)digest, digest_size);
254}

259static PyObject *
260EVP_update(EVPobject *self, PyObject *args)
261{
262 PyObject *obj;
263 Py_buffer view;
264
265 if (!PyArg_ParseTuple(args, "O:update", &obj))
266 return NULL;
267
268 GET_BUFFER_VIEW_OR_ERROUT(obj, &view);
269
270#ifdef WITH_THREAD
271 if (self->lock == NULL && view.len >= HASHLIB_GIL_MINSIZE) {
272 self->lock = PyThread_allocate_lock();
273 /* fail? lock = NULL and we fail over to non-threaded code. */
274 }
275
276 if (self->lock != NULL) {
277 Py_BEGIN_ALLOW_THREADS
278 PyThread_acquire_lock(self->lock, 1);
279 EVP_hash(self, view.buf, view.len);
280 PyThread_release_lock(self->lock);
281 Py_END_ALLOW_THREADS
282 } else {
283 EVP_hash(self, view.buf, view.len);
284 }
285#else
286 EVP_hash(self, view.buf, view.len);
287#endif
288
289 PyBuffer_Release(&view);
290 Py_RETURN_NONE;
291}

460static PyObject *
461EVPnew(PyObject *name_obj,
462 const EVP_MD *digest, const EVP_MD_CTX *initial_ctx,
463 const unsigned char *cp, Py_ssize_t len)
464{
465 EVPobject *self;
466
467 if (!digest && !initial_ctx) {
468 PyErr_SetString(PyExc_ValueError, "unsupported hash type");
469 return NULL;
470 }
471
472 if ((self = newEVPobject(name_obj)) == NULL)
473 return NULL;
474
475 if (initial_ctx) {
476 EVP_MD_CTX_copy(self->ctx, initial_ctx);
477 } else {
478 if (!EVP_DigestInit(self->ctx, digest)) {
479 _setException(PyExc_ValueError);
480 Py_DECREF(self);
481 return NULL;
482 }
483 }
484
485 if (cp && len) {
486 if (len >= HASHLIB_GIL_MINSIZE) {
487 Py_BEGIN_ALLOW_THREADS
488 EVP_hash(self, cp, len);
489 Py_END_ALLOW_THREADS
490 } else {
491 EVP_hash(self, cp, len);
492 }
493 }
494
495 return (PyObject *)self;
496}

548#if !HAS_FAST_PKCS5_PBKDF2_HMAC
549/* Improved implementation of PKCS5_PBKDF2_HMAC()
550 *
551 * PKCS5_PBKDF2_HMAC_fast() hashes the password exactly one time instead of
552 * `iter` times. Today (2013) the iteration count is typically 100,000 or
553 * more. The improved algorithm is not subject to a Denial-of-Service
554 * vulnerability with overly large passwords.
555 *
556 * Also OpenSSL < 1.0 don't provide PKCS5_PBKDF2_HMAC(), only
557 * PKCS5_PBKDF2_SHA1.
558 */
559static int
560PKCS5_PBKDF2_HMAC_fast(const char *pass, int passlen,
561 const unsigned char *salt, int saltlen,
562 int iter, const EVP_MD *digest,
563 int keylen, unsigned char *out)
564{
565 unsigned char digtmp[EVP_MAX_MD_SIZE], *p, itmp[4];
566 int cplen, j, k, tkeylen, mdlen;
567 unsigned long i = 1;
568 HMAC_CTX hctx_tpl, hctx;
569
570 mdlen = EVP_MD_size(digest);
571 if (mdlen < 0)
572 return 0;
573
574 HMAC_CTX_init(&hctx_tpl);
575 HMAC_CTX_init(&hctx);
576 p = out;
577 tkeylen = keylen;
578 if (!HMAC_Init_ex(&hctx_tpl, pass, passlen, digest, NULL)) {
579 HMAC_CTX_cleanup(&hctx_tpl);
580 return 0;
581 }
582 while (tkeylen) {
583 if (tkeylen > mdlen)
584 cplen = mdlen;
585 else
586 cplen = tkeylen;
587 /* We are unlikely to ever use more than 256 blocks (5120 bits!)
588 * but just in case...
589 */
590 itmp[0] = (unsigned char)((i >> 24) & 0xff);
591 itmp[1] = (unsigned char)((i >> 16) & 0xff);
592 itmp[2] = (unsigned char)((i >> 8) & 0xff);
593 itmp[3] = (unsigned char)(i & 0xff);
594 if (!HMAC_CTX_copy(&hctx, &hctx_tpl)) {
595 HMAC_CTX_cleanup(&hctx_tpl);
596 return 0;
597 }
598 if (!HMAC_Update(&hctx, salt, saltlen)
599 || !HMAC_Update(&hctx, itmp, 4)
600 || !HMAC_Final(&hctx, digtmp, NULL)) {
601 HMAC_CTX_cleanup(&hctx_tpl);
602 HMAC_CTX_cleanup(&hctx);
603 return 0;
604 }
605 HMAC_CTX_cleanup(&hctx);
606 memcpy(p, digtmp, cplen);
607 for (j = 1; j < iter; j++) {
608 if (!HMAC_CTX_copy(&hctx, &hctx_tpl)) {
609 HMAC_CTX_cleanup(&hctx_tpl);
610 return 0;
611 }
612 if (!HMAC_Update(&hctx, digtmp, mdlen)
613 || !HMAC_Final(&hctx, digtmp, NULL)) {
614 HMAC_CTX_cleanup(&hctx_tpl);
615 HMAC_CTX_cleanup(&hctx);
616 return 0;
617 }
618 HMAC_CTX_cleanup(&hctx);
619 for (k = 0; k < cplen; k++) {
620 p[k] ^= digtmp[k];
621 }
622 }
623 tkeylen-= cplen;
624 i++;
625 p+= cplen;
626 }
627 HMAC_CTX_cleanup(&hctx_tpl);
628 return 1;
629}
630#endif

633PyDoc_STRVAR(pbkdf2_hmac__doc__,
634"pbkdf2_hmac(hash_name, password, salt, iterations, dklen=None) -> key\n\
635\n\
636Password based key derivation function 2 (PKCS #5 v2.0) with HMAC as\n\
637pseudorandom function.");
638
639static PyObject *
640pbkdf2_hmac(PyObject *self, PyObject *args, PyObject *kwdict)
641{
642 static char *kwlist[] = {"hash_name", "password", "salt", "iterations",
643 "dklen", NULL};
644 PyObject *key_obj = NULL, *dklen_obj = Py_None;
645 char *name, *key;
646 Py_buffer password, salt;
647 long iterations, dklen;
648 int retval;
649 const EVP_MD *digest;
650
651 if (!PyArg_ParseTupleAndKeywords(args, kwdict, "sy*y*l|O:pbkdf2_hmac",
652 kwlist, &name, &password, &salt,
653 &iterations, &dklen_obj)) {
654 return NULL;
655 }
656
657 digest = EVP_get_digestbyname(name);
658 if (digest == NULL) {
659 PyErr_SetString(PyExc_ValueError, "unsupported hash type");
660 goto end;
661 }
662
663 if (password.len > INT_MAX) {
664 PyErr_SetString(PyExc_OverflowError,
665 "password is too long.");
666 goto end;
667 }
668
669 if (salt.len > INT_MAX) {
670 PyErr_SetString(PyExc_OverflowError,
671 "salt is too long.");
672 goto end;
673 }
674
675 if (iterations < 1) {
676 PyErr_SetString(PyExc_ValueError,
677 "iteration value must be greater than 0.");
678 goto end;
679 }
680 if (iterations > INT_MAX) {
681 PyErr_SetString(PyExc_OverflowError,
682 "iteration value is too great.");
683 goto end;
684 }
685
686 if (dklen_obj == Py_None) {
687 dklen = EVP_MD_size(digest);
688 } else {
689 dklen = PyLong_AsLong(dklen_obj);
690 if ((dklen == -1) && PyErr_Occurred()) {
691 goto end;
692 }
693 }
694 if (dklen < 1) {
695 PyErr_SetString(PyExc_ValueError,
696 "key length must be greater than 0.");
697 goto end;
698 }
699 if (dklen > INT_MAX) {
700 /* INT_MAX is always smaller than dkLen max (2^32 - 1) * hLen */
701 PyErr_SetString(PyExc_OverflowError,
702 "key length is too great.");
703 goto end;
704 }
705
706 key_obj = PyBytes_FromStringAndSize(NULL, dklen);
707 if (key_obj == NULL) {
708 goto end;
709 }
710 key = PyBytes_AS_STRING(key_obj);
711
712 Py_BEGIN_ALLOW_THREADS
713#if HAS_FAST_PKCS5_PBKDF2_HMAC
714 retval = PKCS5_PBKDF2_HMAC((char*)password.buf, (int)password.len,
715 (unsigned char *)salt.buf, (int)salt.len,
716 iterations, digest, dklen,
717 (unsigned char *)key);
718#else
719 retval = PKCS5_PBKDF2_HMAC_fast((char*)password.buf, (int)password.len,
720 (unsigned char *)salt.buf, (int)salt.len,
721 iterations, digest, dklen,
722 (unsigned char *)key);
723#endif
724 Py_END_ALLOW_THREADS
725
726 if (!retval) {
727 Py_CLEAR(key_obj);
728 _setException(PyExc_ValueError);
729 goto end;
730 }
731
732 end:
733 PyBuffer_Release(&password);
734 PyBuffer_Release(&salt);
735 return key_obj;
736}

862/* State for our callback function so that it can accumulate a result. */
863typedef struct _internal_name_mapper_state {
864 PyObject *set;
865 int error;
866} _InternalNameMapperState;
867
868
869/* A callback function to pass to OpenSSL's OBJ_NAME_do_all(...) */
870static void
871_openssl_hash_name_mapper(const OBJ_NAME *openssl_obj_name, void *arg)
872{
873 _InternalNameMapperState *state = (_InternalNameMapperState *)arg;
874 PyObject *py_name;
875
876 assert(state != NULL);
877 if (openssl_obj_name == NULL)
878 return;
879 /* Ignore aliased names, they pollute the list and OpenSSL appears to
880 * have its own definition of alias as the resulting list still
881 * contains duplicate and alternate names for several algorithms. */
882 if (openssl_obj_name->alias)
883 return;
884
885 py_name = PyUnicode_FromString(openssl_obj_name->name);
886 if (py_name == NULL) {
887 state->error = 1;
888 } else {
889 if (PySet_Add(state->set, py_name) != 0) {
890 state->error = 1;
891 }
892 Py_DECREF(py_name);
893 }
894}

897/* Ask OpenSSL for a list of supported ciphers, filling in a Python set. */
898static PyObject*
899generate_hash_name_list(void)
900{
901 _InternalNameMapperState state;
902 state.set = PyFrozenSet_New(NULL);
903 if (state.set == NULL)
904 return NULL;
905 state.error = 0;
906
907 OBJ_NAME_do_all(OBJ_NAME_TYPE_MD_METH, &_openssl_hash_name_mapper, &state);
908
909 if (state.error) {
910 Py_DECREF(state.set);
911 return NULL;
912 }
913 return state.set;
914}

917/*
918 * This macro generates constructor function definitions for specific
919 * hash algorithms. These constructors are much faster than calling
920 * the generic one passing it a python string and are noticeably
921 * faster than calling a python new() wrapper. Thats important for
922 * code that wants to make hashes of a bunch of small strings.
923 * The first call will lazy-initialize, which reports an exception
924 * if initialization fails.
925 */
926#define GEN_CONSTRUCTOR(NAME) \
927 static PyObject * \
928 EVP_new_ ## NAME (PyObject *self, PyObject **args, Py_ssize_t nargs) \
929 { \
930 PyObject *data_obj = NULL; \
931 Py_buffer view = { 0 }; \
932 PyObject *ret_obj; \
933 \
934 if (!_PyArg_ParseStack(args, nargs, "|O:" #NAME , &data_obj)) { \
935 return NULL; \
936 } \
937 \
938 if (CONST_new_ ## NAME ## _ctx_p == NULL) { \
939 EVP_MD_CTX *ctx_p = EVP_MD_CTX_new(); \
940 if (!EVP_get_digestbyname(#NAME) || \
941 !EVP_DigestInit(ctx_p, EVP_get_digestbyname(#NAME))) { \
942 _setException(PyExc_ValueError); \
943 EVP_MD_CTX_free(ctx_p); \
944 return NULL; \
945 } \
946 CONST_new_ ## NAME ## _ctx_p = ctx_p; \
947 } \
948 \
949 if (data_obj) \
950 GET_BUFFER_VIEW_OR_ERROUT(data_obj, &view); \
951 \
952 ret_obj = EVPnew( \
953 CONST_ ## NAME ## _name_obj, \
954 NULL, \
955 CONST_new_ ## NAME ## _ctx_p, \
956 (unsigned char*)view.buf, \
957 view.len); \
958 \
959 if (data_obj) \
960 PyBuffer_Release(&view); \
961 return ret_obj; \
962 }
963
964/* a PyMethodDef structure for the constructor */
965#define CONSTRUCTOR_METH_DEF(NAME) \
966 {"openssl_" #NAME, (PyCFunction)EVP_new_ ## NAME, METH_FASTCALL, \
967 PyDoc_STR("Returns a " #NAME \
968 " hash object; optionally initialized with a string") \
969 }

972 constructor constants if they haven't been initialized already. */
973#define INIT_CONSTRUCTOR_CONSTANTS(NAME) do { \
974 if (CONST_ ## NAME ## _name_obj == NULL) { \
975 CONST_ ## NAME ## _name_obj = PyUnicode_FromString(#NAME); \
976 } \
977} while (0);
