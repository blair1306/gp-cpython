84static PyObject*
85node2tuple(node *n, /* node to convert */
86 SeqMaker mkseq, /* create sequence */
87 SeqInserter addelem, /* func. to add elem. in seq. */
88 int lineno, /* include line numbers? */
89 int col_offset) /* include column offsets? */
90{
91 PyObject *result = NULL, *w;
92
93 if (n == NULL) {
94 Py_RETURN_NONE;
95 }
96
97 if (ISNONTERMINAL(TYPE(n))) {
98 int i;
99
100 result = mkseq(1 + NCH(n) + (TYPE(n) == encoding_decl));
101 if (result == NULL)
102 goto error;
103
104 w = PyLong_FromLong(TYPE(n));
105 if (w == NULL)
106 goto error;
107 (void) addelem(result, 0, w);
108
109 for (i = 0; i < NCH(n); i++) {
110 w = node2tuple(CHILD(n, i), mkseq, addelem, lineno, col_offset);
111 if (w == NULL)
112 goto error;
113 (void) addelem(result, i+1, w);
114 }
115
116 if (TYPE(n) == encoding_decl) {
117 w = PyUnicode_FromString(STR(n));
118 if (w == NULL)
119 goto error;
120 (void) addelem(result, i+1, w);
121 }
122 }
123 else if (ISTERMINAL(TYPE(n))) {
124 result = mkseq(2 + lineno + col_offset);
125 if (result == NULL)
126 goto error;
127
128 w = PyLong_FromLong(TYPE(n));
129 if (w == NULL)
130 goto error;
131 (void) addelem(result, 0, w);
132
133 w = PyUnicode_FromString(STR(n));
134 if (w == NULL)
135 goto error;
136 (void) addelem(result, 1, w);
137
138 if (lineno == 1) {
139 w = PyLong_FromLong(n->n_lineno);
140 if (w == NULL)
141 goto error;
142 (void) addelem(result, 2, w);
143 }
144
145 if (col_offset == 1) {
146 w = PyLong_FromLong(n->n_col_offset);
147 if (w == NULL)
148 goto error;
149 (void) addelem(result, 3, w);
150 }
151 }
152 else {
153 PyErr_SetString(PyExc_SystemError,
154 "unrecognized parse tree node type");
155 return ((PyObject*) NULL);
156 }
157 return result;
158
159error:
160 Py_XDECREF(result);
161 return NULL;
162}

265static int
266parser_compare_nodes(node *left, node *right)
267{
268 int j;
269
270 if (TYPE(left) < TYPE(right))
271 return (-1);
272
273 if (TYPE(right) < TYPE(left))
274 return (1);
275
276 if (ISTERMINAL(TYPE(left)))
277 return (strcmp(STR(left), STR(right)));
278
279 if (NCH(left) < NCH(right))
280 return (-1);
281
282 if (NCH(right) < NCH(left))
283 return (1);
284
285 for (j = 0; j < NCH(left); ++j) {
286 int v = parser_compare_nodes(CHILD(left, j), CHILD(right, j));
287
288 if (v != 0)
289 return (v);
290 }
291 return (0);
292}

294/* parser_richcompare(PyObject* left, PyObject* right, int op)
295 *
296 * Comparison function used by the Python operators ==, !=, <, >, <=, >=
297 * This really just wraps a call to parser_compare_nodes() with some easy
298 * checks and protection code.
299 *
300 */
301
302#define TEST_COND(cond) ((cond) ? Py_True : Py_False)
303
304static PyObject *
305parser_richcompare(PyObject *left, PyObject *right, int op)
306{
307 int result;
308 PyObject *v;
309
310 /* neither argument should be NULL, unless something's gone wrong */
311 if (left == NULL || right == NULL) {
312 PyErr_BadInternalCall();
313 return NULL;
314 }
315
316 /* both arguments should be instances of PyST_Object */
317 if (!PyST_Object_Check(left) || !PyST_Object_Check(right)) {
318 v = Py_NotImplemented;
319 goto finished;
320 }
321
322 if (left == right)
323 /* if arguments are identical, they're equal */
324 result = 0;
325 else
326 result = parser_compare_nodes(((PyST_Object *)left)->st_node,
327 ((PyST_Object *)right)->st_node);
328
329 /* Convert return value to a Boolean */
330 switch (op) {
331 case Py_EQ:
332 v = TEST_COND(result == 0);
333 break;
334 case Py_NE:
335 v = TEST_COND(result != 0);
336 break;
337 case Py_LE:
338 v = TEST_COND(result <= 0);
339 break;
340 case Py_GE:
341 v = TEST_COND(result >= 0);
342 break;
343 case Py_LT:
344 v = TEST_COND(result < 0);
345 break;
346 case Py_GT:
347 v = TEST_COND(result > 0);
348 break;
349 default:
350 PyErr_BadArgument();
351 return NULL;
352 }
353 finished:
354 Py_INCREF(v);
355 return v;
356}

475/* parser_compilest(PyObject* self, PyObject* args)
476 *
477 * This function creates code objects from the parse tree represented by
478 * the passed-in data object. An optional file name is passed in as well.
479 *
480 */
481static PyObject*
482parser_compilest(PyST_Object *self, PyObject *args, PyObject *kw)
483{
484 PyObject* res = NULL;
485 PyArena* arena = NULL;
486 mod_ty mod;
487 PyObject* filename = NULL;
488 int ok;
489
490 static char *keywords[] = {"st", "filename", NULL};
491
492 if (self == NULL || PyModule_Check(self))
493 ok = PyArg_ParseTupleAndKeywords(args, kw, "O!|O&:compilest", keywords,
494 &PyST_Type, &self,
495 PyUnicode_FSDecoder, &filename);
496 else
497 ok = PyArg_ParseTupleAndKeywords(args, kw, "|O&:compile", &keywords[1],
498 PyUnicode_FSDecoder, &filename);
499 if (!ok)
500 goto error;
501
502 if (filename == NULL) {
503 filename = PyUnicode_FromString("<syntax-tree>");
504 if (filename == NULL)
505 goto error;
506 }
507
508 arena = PyArena_New();
509 if (!arena)
510 goto error;
511
512 mod = PyAST_FromNodeObject(self->st_node, &self->st_flags,
513 filename, arena);
514 if (!mod)
515 goto error;
516
517 res = (PyObject *)PyAST_CompileObject(mod, filename,
518 &self->st_flags, -1, arena);
519error:
520 Py_XDECREF(filename);
521 if (arena != NULL)
522 PyArena_Free(arena);
523 return res;
524}

592/* PyObject* parser_do_parse(PyObject* args, int type)
593 *
594 * Internal function to actually execute the parse and return the result if
595 * successful or set an exception if not.
596 *
597 */
598static PyObject*
599parser_do_parse(PyObject *args, PyObject *kw, const char *argspec, int type)
600{
601 char* string = 0;
602 PyObject* res = 0;
603 int flags = 0;
604 perrdetail err;
605
606 static char *keywords[] = {"source", NULL};
607
608 if (PyArg_ParseTupleAndKeywords(args, kw, argspec, keywords, &string)) {
609 node* n = PyParser_ParseStringFlagsFilenameEx(string, NULL,
610 &_PyParser_Grammar,
611 (type == PyST_EXPR)
612 ? eval_input : file_input,
613 &err, &flags);
614
615 if (n) {
616 res = parser_newstobject(n, type);
617 if (res)
618 ((PyST_Object *)res)->st_flags.cf_flags = flags & PyCF_MASK;
619 }
620 else {
621 PyParser_SetError(&err);
622 }
623 PyParser_ClearError(&err);
624 }
625 return (res);
626}

654/* This is the messy part of the code. Conversion from a tuple to an ST
655 * object requires that the input tuple be valid without having to rely on
656 * catching an exception from the compiler. This is done to allow the
657 * compiler itself to remain fast, since most of its input will come from
658 * the parser directly, and therefore be known to be syntactically correct.
659 * This validation is done to ensure that we don't core dump the compile
660 * phase, returning an exception instead.
661 *
662 * Two aspects can be broken out in this code: creating a node tree from
663 * the tuple passed in, and verifying that it is indeed valid. It may be
664 * advantageous to expand the number of ST types to include funcdefs and
665 * lambdadefs to take advantage of the optimizer, recognizing those STs
666 * here. They are not necessary, and not quite as useful in a raw form.
667 * For now, let's get expressions and suites working reliably.
668 */
669
670
671static node* build_node_tree(PyObject *tuple);
672
673static int
674validate_node(node *tree)
675{
676 int type = TYPE(tree);
677 int nch = NCH(tree);
678 dfa *nt_dfa;
679 state *dfa_state;
680 int pos, arc;
681
682 assert(ISNONTERMINAL(type));
683 type -= NT_OFFSET;
684 if (type >= _PyParser_Grammar.g_ndfas) {
685 PyErr_Format(parser_error, "Unrecognized node type %d.", TYPE(tree));
686 return 0;
687 }
688 nt_dfa = &_PyParser_Grammar.g_dfa[type];
689 REQ(tree, nt_dfa->d_type);
690
691 /* Run the DFA for this nonterminal. */
692 dfa_state = &nt_dfa->d_state[nt_dfa->d_initial];
693 for (pos = 0; pos < nch; ++pos) {
694 node *ch = CHILD(tree, pos);
695 int ch_type = TYPE(ch);
696 for (arc = 0; arc < dfa_state->s_narcs; ++arc) {
697 short a_label = dfa_state->s_arc[arc].a_lbl;
698 assert(a_label < _PyParser_Grammar.g_ll.ll_nlabels);
699 if (_PyParser_Grammar.g_ll.ll_label[a_label].lb_type == ch_type) {
700 /* The child is acceptable; if non-terminal, validate it recursively. */
701 if (ISNONTERMINAL(ch_type) && !validate_node(ch))
702 return 0;
703
704 /* Update the state, and move on to the next child. */
705 dfa_state = &nt_dfa->d_state[dfa_state->s_arc[arc].a_arrow];
706 goto arc_found;
707 }
708 }
709 /* What would this state have accepted? */
710 {
711 short a_label = dfa_state->s_arc->a_lbl;
712 int next_type;
713 if (!a_label) /* Wouldn't accept any more children */
714 goto illegal_num_children;
715
716 next_type = _PyParser_Grammar.g_ll.ll_label[a_label].lb_type;
717 if (ISNONTERMINAL(next_type))
718 PyErr_Format(parser_error, "Expected node type %d, got %d.",
719 next_type, ch_type);
720 else
721 PyErr_Format(parser_error, "Illegal terminal: expected %s.",
722 _PyParser_TokenNames[next_type]);
723 return 0;
724 }
725
726arc_found:
727 continue;
728 }
729 /* Are we in a final state? If so, return 1 for successful validation. */
730 for (arc = 0; arc < dfa_state->s_narcs; ++arc) {
731 if (!dfa_state->s_arc[arc].a_lbl) {
732 return 1;
733 }
734 }
735
736illegal_num_children:
737 PyErr_Format(parser_error,
738 "Illegal number of children for %s node.", nt_dfa->d_name);
739 return 0;
740}

742/* PyObject* parser_tuple2st(PyObject* self, PyObject* args)
743 *
744 * This is the public function, called from the Python code. It receives a
745 * single tuple object from the caller, and creates an ST object if the
746 * tuple can be validated. It does this by checking the first code of the
747 * tuple, and, if acceptable, builds the internal representation. If this
748 * step succeeds, the internal representation is validated as fully as
749 * possible with the recursive validate_node() routine defined above.
750 *
751 * This function must be changed if support is to be added for PyST_FRAGMENT
752 * ST objects.
753 *
754 */
755static PyObject*
756parser_tuple2st(PyST_Object *self, PyObject *args, PyObject *kw)
757{
758 NOTE(ARGUNUSED(self))
759 PyObject *st = 0;
760 PyObject *tuple;
761 node *tree;
762
763 static char *keywords[] = {"sequence", NULL};
764
765 if (!PyArg_ParseTupleAndKeywords(args, kw, "O:sequence2st", keywords,
766 &tuple))
767 return (0);
768 if (!PySequence_Check(tuple)) {
769 PyErr_SetString(PyExc_ValueError,
770 "sequence2st() requires a single sequence argument");
771 return (0);
772 }
773 /*
774 * Convert the tree to the internal form before checking it.
775 */
776 tree = build_node_tree(tuple);
777 if (tree != 0) {
778 node *validation_root = NULL;
779 int tree_type = 0;
780 switch (TYPE(tree)) {
781 case eval_input:
782 /* Might be an eval form. */
783 tree_type = PyST_EXPR;
784 validation_root = tree;
785 break;
786 case encoding_decl:
787 /* This looks like an encoding_decl so far. */
788 if (NCH(tree) == 1) {
789 tree_type = PyST_SUITE;
790 validation_root = CHILD(tree, 0);
791 }
792 else {
793 err_string("Error Parsing encoding_decl");
794 }
795 break;
796 case file_input:
797 /* This looks like an exec form so far. */
798 tree_type = PyST_SUITE;
799 validation_root = tree;
800 break;
801 default:
802 /* This is a fragment, at best. */
803 err_string("parse tree does not use a valid start symbol");
804 }
805
806 if (validation_root != NULL && validate_node(validation_root))
807 st = parser_newstobject(tree, tree_type);
808 else
809 PyNode_Free(tree);
810 }
811 /* Make sure we raise an exception on all errors. We should never
812 * get this, but we'd do well to be sure something is done.
813 */
814 if (st == NULL && !PyErr_Occurred())
815 err_string("unspecified ST error occurred");
816
817 return st;
818}

821/* node* build_node_children()
822 *
823 * Iterate across the children of the current non-terminal node and build
824 * their structures. If successful, return the root of this portion of
825 * the tree, otherwise, 0. Any required exception will be specified already,
826 * and no memory will have been deallocated.
827 *
828 */
829static node*
830build_node_children(PyObject *tuple, node *root, int *line_num)
831{
832 Py_ssize_t len = PyObject_Size(tuple);
833 Py_ssize_t i;
834 int err;
835
836 if (len < 0) {
837 return NULL;
838 }
839 for (i = 1; i < len; ++i) {
840 /* elem must always be a sequence, however simple */
841 PyObject* elem = PySequence_GetItem(tuple, i);
842 int ok = elem != NULL;
843 int type = 0;
844 char *strn = 0;
845
846 if (ok)
847 ok = PySequence_Check(elem);
848 if (ok) {
849 PyObject *temp = PySequence_GetItem(elem, 0);
850 if (temp == NULL)
851 ok = 0;
852 else {
853 ok = PyLong_Check(temp);
854 if (ok) {
855 type = _PyLong_AsInt(temp);
856 if (type == -1 && PyErr_Occurred()) {
857 Py_DECREF(temp);
858 Py_DECREF(elem);
859 return NULL;
860 }
861 }
862 Py_DECREF(temp);
863 }
864 }
865 if (!ok) {
866 PyObject *err = Py_BuildValue("Os", elem,
867 "Illegal node construct.");
868 PyErr_SetObject(parser_error, err);
869 Py_XDECREF(err);
870 Py_XDECREF(elem);
871 return NULL;
872 }
873 if (ISTERMINAL(type)) {
874 Py_ssize_t len = PyObject_Size(elem);
875 PyObject *temp;
876 const char *temp_str;
877
878 if ((len != 2) && (len != 3)) {
879 err_string("terminal nodes must have 2 or 3 entries");
880 Py_DECREF(elem);
881 return NULL;
882 }
883 temp = PySequence_GetItem(elem, 1);
884 if (temp == NULL) {
885 Py_DECREF(elem);
886 return NULL;
887 }
888 if (!PyUnicode_Check(temp)) {
889 PyErr_Format(parser_error,
890 "second item in terminal node must be a string,"
891 " found %s",
892 Py_TYPE(temp)->tp_name);
893 Py_DECREF(temp);
894 Py_DECREF(elem);
895 return NULL;
896 }
897 if (len == 3) {
898 PyObject *o = PySequence_GetItem(elem, 2);
899 if (o == NULL) {
900 Py_DECREF(temp);
901 Py_DECREF(elem);
902 return NULL;
903 }
904 if (PyLong_Check(o)) {
905 int num = _PyLong_AsInt(o);
906 if (num == -1 && PyErr_Occurred()) {
907 Py_DECREF(o);
908 Py_DECREF(temp);
909 Py_DECREF(elem);
910 return NULL;
911 }
912 *line_num = num;
913 }
914 else {
915 PyErr_Format(parser_error,
916 "third item in terminal node must be an"
917 " integer, found %s",
918 Py_TYPE(temp)->tp_name);
919 Py_DECREF(o);
920 Py_DECREF(temp);
921 Py_DECREF(elem);
922 return NULL;
923 }
924 Py_DECREF(o);
925 }
926 temp_str = PyUnicode_AsUTF8AndSize(temp, &len);
927 if (temp_str == NULL) {
928 Py_DECREF(temp);
929 Py_DECREF(elem);
930 return NULL;
931 }
932 strn = (char *)PyObject_MALLOC(len + 1);
933 if (strn == NULL) {
934 Py_DECREF(temp);
935 Py_DECREF(elem);
936 PyErr_NoMemory();
937 return NULL;
938 }
939 (void) memcpy(strn, temp_str, len + 1);
940 Py_DECREF(temp);
941 }
942 else if (!ISNONTERMINAL(type)) {
943 /*
944 * It has to be one or the other; this is an error.
945 * Raise an exception.
946 */
947 PyObject *err = Py_BuildValue("Os", elem, "unknown node type.");
948 PyErr_SetObject(parser_error, err);
949 Py_XDECREF(err);
950 Py_DECREF(elem);
951 return NULL;
952 }
953 err = PyNode_AddChild(root, type, strn, *line_num, 0);
954 if (err == E_NOMEM) {
955 Py_DECREF(elem);
956 PyObject_FREE(strn);
957 PyErr_NoMemory();
958 return NULL;
959 }
960 if (err == E_OVERFLOW) {
961 Py_DECREF(elem);
962 PyObject_FREE(strn);
963 PyErr_SetString(PyExc_ValueError,
964 "unsupported number of child nodes");
965 return NULL;
966 }
967
968 if (ISNONTERMINAL(type)) {
969 node* new_child = CHILD(root, i - 1);
970
971 if (new_child != build_node_children(elem, new_child, line_num)) {
972 Py_DECREF(elem);
973 return NULL;
974 }
975 }
976 else if (type == NEWLINE) { /* It's true: we increment the */
977 ++(*line_num); /* line number *after* the newline! */
978 }
979 Py_DECREF(elem);
980 }
981 return root;
982}

984
985static node*
986build_node_tree(PyObject *tuple)
987{
988 node* res = 0;
989 PyObject *temp = PySequence_GetItem(tuple, 0);
990 long num = -1;
991
992 if (temp != NULL)
993 num = PyLong_AsLong(temp);
994 Py_XDECREF(temp);
995 if (ISTERMINAL(num)) {
996 /*
997 * The tuple is simple, but it doesn't start with a start symbol.
998 * Raise an exception now and be done with it.
999 */
1000 tuple = Py_BuildValue("Os", tuple,
1001 "Illegal syntax-tree; cannot start with terminal symbol.");
1002 PyErr_SetObject(parser_error, tuple);
1003 Py_XDECREF(tuple);
1004 }
1005 else if (ISNONTERMINAL(num)) {
1006 /*
1007 * Not efficient, but that can be handled later.
1008 */
1009 int line_num = 0;
1010 PyObject *encoding = NULL;
1011
1012 if (num == encoding_decl) {
1013 encoding = PySequence_GetItem(tuple, 2);
1014 if (encoding == NULL) {
1015 PyErr_SetString(parser_error, "missed encoding");
1016 return NULL;
1017 }
1018 if (!PyUnicode_Check(encoding)) {
1019 PyErr_Format(parser_error,
1020 "encoding must be a string, found %.200s",
1021 Py_TYPE(encoding)->tp_name);
1022 Py_DECREF(encoding);
1023 return NULL;
1024 }
1025 /* tuple isn't borrowed anymore here, need to DECREF */
1026 tuple = PySequence_GetSlice(tuple, 0, 2);
1027 if (tuple == NULL) {
1028 Py_DECREF(encoding);
1029 return NULL;
1030 }
1031 }
1032 res = PyNode_New(num);
1033 if (res != NULL) {
1034 if (res != build_node_children(tuple, res, &line_num)) {
1035 PyNode_Free(res);
1036 res = NULL;
1037 }
1038 if (res && encoding) {
1039 Py_ssize_t len;
1040 const char *temp;
1041 temp = PyUnicode_AsUTF8AndSize(encoding, &len);
1042 if (temp == NULL) {
1043 PyNode_Free(res);
1044 Py_DECREF(encoding);
1045 Py_DECREF(tuple);
1046 return NULL;
1047 }
1048 res->n_str = (char *)PyObject_MALLOC(len + 1);
1049 if (res->n_str == NULL) {
1050 PyNode_Free(res);
1051 Py_DECREF(encoding);
1052 Py_DECREF(tuple);
1053 PyErr_NoMemory();
1054 return NULL;
1055 }
1056 (void) memcpy(res->n_str, temp, len + 1);
1057 }
1058 }
1059 if (encoding != NULL) {
1060 Py_DECREF(encoding);
1061 Py_DECREF(tuple);
1062 }
1063 }
1064 else {
1065 /* The tuple is illegal -- if the number is neither TERMINAL nor
1066 * NONTERMINAL, we can't use it. Not sure the implementation
1067 * allows this condition, but the API doesn't preclude it.
1068 */
1069 PyObject *err = Py_BuildValue("Os", tuple,
1070 "Illegal component tuple.");
1071 PyErr_SetObject(parser_error, err);
1072 Py_XDECREF(err);
1073 }
1074
1075 return (res);
1076}

1083static PyObject*
1084parser__pickler(PyObject *self, PyObject *args)
1085{
1086 NOTE(ARGUNUSED(self))
1087 PyObject *result = NULL;
1088 PyObject *st = NULL;
1089 PyObject *empty_dict = NULL;
1090
1091 if (PyArg_ParseTuple(args, "O!:_pickler", &PyST_Type, &st)) {
1092 PyObject *newargs;
1093 PyObject *tuple;
1094
1095 if ((empty_dict = PyDict_New()) == NULL)
1096 goto finally;
1097 if ((newargs = Py_BuildValue("Oi", st, 1)) == NULL)
1098 goto finally;
1099 tuple = parser_st2tuple((PyST_Object*)NULL, newargs, empty_dict);
1100 if (tuple != NULL) {
1101 result = Py_BuildValue("O(O)", pickle_constructor, tuple);
1102 Py_DECREF(tuple);
1103 }
1104 Py_DECREF(newargs);
1105 }
1106 finally:
1107 Py_XDECREF(empty_dict);
1108
1109 return (result);
1110}

1160PyMODINIT_FUNC PyInit_parser(void); /* supply a prototype */
1161
1162PyMODINIT_FUNC
1163PyInit_parser(void)
1164{
1165 PyObject *module, *copyreg;
1166
1167 if (PyType_Ready(&PyST_Type) < 0)
1168 return NULL;
1169 module = PyModule_Create(&parsermodule);
1170 if (module == NULL)
1171 return NULL;
1172
1173 if (parser_error == 0)
1174 parser_error = PyErr_NewException("parser.ParserError", NULL, NULL);
1175
1176 if (parser_error == 0)
1177 return NULL;
1178 /* CAUTION: The code next used to skip bumping the refcount on
1179 * parser_error. That's a disaster if PyInit_parser() gets called more
1180 * than once. By incref'ing, we ensure that each module dict that
1181 * gets created owns its reference to the shared parser_error object,
1182 * and the file static parser_error vrbl owns a reference too.
1183 */
1184 Py_INCREF(parser_error);
1185 if (PyModule_AddObject(module, "ParserError", parser_error) != 0)
1186 return NULL;
1187
1188 Py_INCREF(&PyST_Type);
1189 PyModule_AddObject(module, "STType", (PyObject*)&PyST_Type);
1190
1191 PyModule_AddStringConstant(module, "__copyright__",
1192 parser_copyright_string);
1193 PyModule_AddStringConstant(module, "__doc__",
1194 parser_doc_string);
1195 PyModule_AddStringConstant(module, "__version__",
1196 parser_version_string);
1197
1198 /* Register to support pickling.
1199 * If this fails, the import of this module will fail because an
1200 * exception will be raised here; should we clear the exception?
1201 */
1202 copyreg = PyImport_ImportModuleNoBlock("copyreg");
1203 if (copyreg != NULL) {
1204 PyObject *func, *pickler;
1205 _Py_IDENTIFIER(pickle);
1206 _Py_IDENTIFIER(sequence2st);
1207 _Py_IDENTIFIER(_pickler);
1208
1209 func = _PyObject_GetAttrId(copyreg, &PyId_pickle);
1210 pickle_constructor = _PyObject_GetAttrId(module, &PyId_sequence2st);
1211 pickler = _PyObject_GetAttrId(module, &PyId__pickler);
1212 Py_XINCREF(pickle_constructor);
1213 if ((func != NULL) && (pickle_constructor != NULL)
1214 && (pickler != NULL)) {
1215 PyObject *res;
1216
1217 res = PyObject_CallFunctionObjArgs(func, &PyST_Type, pickler,
1218 pickle_constructor, NULL);
1219 Py_XDECREF(res);
1220 }
1221 Py_XDECREF(func);
1222 Py_XDECREF(pickle_constructor);
1223 Py_XDECREF(pickler);
1224 Py_DECREF(copyreg);
1225 }
1226 return module;
