19static int
20PyCStgDict_init(StgDictObject *self, PyObject *args, PyObject *kwds)
21{
22 if (PyDict_Type.tp_init((PyObject *)self, args, kwds) < 0)
23 return -1;
24 self->format = NULL;
25 self->ndim = 0;
26 self->shape = NULL;
27 return 0;
28}

41static void

51static PyObject *
52PyCStgDict_sizeof(StgDictObject *self, void *unused)
53{
54 Py_ssize_t res;
55
56 res = _PyDict_SizeOf((PyDictObject *)self);
57 res += sizeof(StgDictObject) - sizeof(PyDictObject);
58 if (self->format)
59 res += strlen(self->format) + 1;
60 res += self->ndim * sizeof(Py_ssize_t);
61 if (self->ffi_type_pointer.elements)
62 res += (self->length + 1) * sizeof(ffi_type *);
63 return PyLong_FromSsize_t(res);
64}

66int
67PyCStgDict_clone(StgDictObject *dst, StgDictObject *src)
68{
69 char *d, *s;
70 Py_ssize_t size;
71
72 PyCStgDict_clear(dst);
73 PyMem_Free(dst->ffi_type_pointer.elements);
74 PyMem_Free(dst->format);
75 dst->format = NULL;
76 PyMem_Free(dst->shape);
77 dst->shape = NULL;
78 dst->ffi_type_pointer.elements = NULL;
79
80 d = (char *)dst;
81 s = (char *)src;
82 memcpy(d + sizeof(PyDictObject),
83 s + sizeof(PyDictObject),
84 sizeof(StgDictObject) - sizeof(PyDictObject));
85
86 Py_XINCREF(dst->proto);
87 Py_XINCREF(dst->argtypes);
88 Py_XINCREF(dst->converters);
89 Py_XINCREF(dst->restype);
90 Py_XINCREF(dst->checker);
91
92 if (src->format) {
93 dst->format = PyMem_Malloc(strlen(src->format) + 1);
94 if (dst->format == NULL) {
95 PyErr_NoMemory();
96 return -1;
97 }
98 strcpy(dst->format, src->format);
99 }
100 if (src->shape) {
101 dst->shape = PyMem_Malloc(sizeof(Py_ssize_t) * src->ndim);
102 if (dst->shape == NULL) {
103 PyErr_NoMemory();
104 return -1;
105 }
106 memcpy(dst->shape, src->shape,
107 sizeof(Py_ssize_t) * src->ndim);
108 }
109
110 if (src->ffi_type_pointer.elements == NULL)
111 return 0;
112 size = sizeof(ffi_type *) * (src->length + 1);
113 dst->ffi_type_pointer.elements = PyMem_Malloc(size);
114 if (dst->ffi_type_pointer.elements == NULL) {
115 PyErr_NoMemory();
116 return -1;
117 }
118 memcpy(dst->ffi_type_pointer.elements,
119 src->ffi_type_pointer.elements,
120 size);
121 return 0;
122}

171/* May return NULL, but does not set an exception! */
172StgDictObject *
173PyType_stgdict(PyObject *obj)
174{
175 PyTypeObject *type;
176
177 if (!PyType_Check(obj))
178 return NULL;
179 type = (PyTypeObject *)obj;
180 if (!type->tp_dict || !PyCStgDict_CheckExact(type->tp_dict))
181 return NULL;
182 return (StgDictObject *)type->tp_dict;
183}
184
185/* May return NULL, but does not set an exception! */
186/*
187 This function should be as fast as possible, so we don't call PyType_stgdict
188 above but inline the code, and avoid the PyType_Check().
189*/
190StgDictObject *
191PyObject_stgdict(PyObject *self)
192{
193 PyTypeObject *type = self->ob_type;
194 if (!type->tp_dict || !PyCStgDict_CheckExact(type->tp_dict))
195 return NULL;
196 return (StgDictObject *)type->tp_dict;
197}

199/* descr is the descriptor for a field marked as anonymous. Get all the
200 _fields_ descriptors from descr->proto, create new descriptors with offset
201 and index adjusted, and stuff them into type.
202 */
203static int
204MakeFields(PyObject *type, CFieldObject *descr,
205 Py_ssize_t index, Py_ssize_t offset)
206{
207 Py_ssize_t i;
208 PyObject *fields;
209 PyObject *fieldlist;
210
211 fields = PyObject_GetAttrString(descr->proto, "_fields_");
212 if (fields == NULL)
213 return -1;
214 fieldlist = PySequence_Fast(fields, "_fields_ must be a sequence");
215 Py_DECREF(fields);
216 if (fieldlist == NULL)
217 return -1;
218
219 for (i = 0; i < PySequence_Fast_GET_SIZE(fieldlist); ++i) {
220 PyObject *pair = PySequence_Fast_GET_ITEM(fieldlist, i); /* borrowed */
221 PyObject *fname, *ftype, *bits;
222 CFieldObject *fdescr;
223 CFieldObject *new_descr;
224 /* Convert to PyArg_UnpackTuple... */
225 if (!PyArg_ParseTuple(pair, "OO|O", &fname, &ftype, &bits)) {
226 Py_DECREF(fieldlist);
227 return -1;
228 }
229 fdescr = (CFieldObject *)PyObject_GetAttr(descr->proto, fname);
230 if (fdescr == NULL) {
231 Py_DECREF(fieldlist);
232 return -1;
233 }
234 if (Py_TYPE(fdescr) != &PyCField_Type) {
235 PyErr_SetString(PyExc_TypeError, "unexpected type");
236 Py_DECREF(fdescr);
237 Py_DECREF(fieldlist);
238 return -1;
239 }
240 if (fdescr->anonymous) {
241 int rc = MakeFields(type, fdescr,
242 index + fdescr->index,
243 offset + fdescr->offset);
244 Py_DECREF(fdescr);
245 if (rc == -1) {
246 Py_DECREF(fieldlist);
247 return -1;
248 }
249 continue;
250 }
251 new_descr = (CFieldObject *)_PyObject_CallNoArg((PyObject *)&PyCField_Type);
252 if (new_descr == NULL) {
253 Py_DECREF(fdescr);
254 Py_DECREF(fieldlist);
255 return -1;
256 }
257 assert(Py_TYPE(new_descr) == &PyCField_Type);
258 new_descr->size = fdescr->size;
259 new_descr->offset = fdescr->offset + offset;
260 new_descr->index = fdescr->index + index;
261 new_descr->proto = fdescr->proto;
262 Py_XINCREF(new_descr->proto);
263 new_descr->getfunc = fdescr->getfunc;
264 new_descr->setfunc = fdescr->setfunc;
265
266 Py_DECREF(fdescr);
267
268 if (-1 == PyObject_SetAttr(type, fname, (PyObject *)new_descr)) {
269 Py_DECREF(fieldlist);
270 Py_DECREF(new_descr);
271 return -1;
272 }
273 Py_DECREF(new_descr);
274 }
275 Py_DECREF(fieldlist);
276 return 0;
277}

279/* Iterate over the names in the type's _anonymous_ attribute, if present,
280 */
281static int
282MakeAnonFields(PyObject *type)
283{
284 PyObject *anon;
285 PyObject *anon_names;
286 Py_ssize_t i;
287
288 anon = PyObject_GetAttrString(type, "_anonymous_");
289 if (anon == NULL) {
290 PyErr_Clear();
291 return 0;
292 }
293 anon_names = PySequence_Fast(anon, "_anonymous_ must be a sequence");
294 Py_DECREF(anon);
295 if (anon_names == NULL)
296 return -1;
297
298 for (i = 0; i < PySequence_Fast_GET_SIZE(anon_names); ++i) {
299 PyObject *fname = PySequence_Fast_GET_ITEM(anon_names, i); /* borrowed */
300 CFieldObject *descr = (CFieldObject *)PyObject_GetAttr(type, fname);
301 if (descr == NULL) {
302 Py_DECREF(anon_names);
303 return -1;
304 }
305 assert(Py_TYPE(descr) == &PyCField_Type);
306 descr->anonymous = 1;
307
308 /* descr is in the field descriptor. */
309 if (-1 == MakeFields(type, (CFieldObject *)descr,
310 ((CFieldObject *)descr)->index,
311 ((CFieldObject *)descr)->offset)) {
312 Py_DECREF(descr);
313 Py_DECREF(anon_names);
314 return -1;
315 }
316 Py_DECREF(descr);
317 }
318
319 Py_DECREF(anon_names);
320 return 0;
321}

323/*
324 Retrive the (optional) _pack_ attribute from a type, the _fields_ attribute,
325 and create an StgDictObject. Used for Structure and Union subclasses.
326*/
327int
328PyCStructUnionType_update_stgdict(PyObject *type, PyObject *fields, int isStruct)
329{
330 StgDictObject *stgdict, *basedict;
331 Py_ssize_t len, offset, size, align, i;
332 Py_ssize_t union_size, total_align;
333 Py_ssize_t field_size = 0;
334 int bitofs;
335 PyObject *isPacked;
336 int pack = 0;
337 Py_ssize_t ffi_ofs;
338 int big_endian;
339
340 /* HACK Alert: I cannot be bothered to fix ctypes.com, so there has to
341 be a way to use the old, broken sematics: _fields_ are not extended
342 but replaced in subclasses.
343
344 XXX Remove this in ctypes 1.0!
345 */
346 int use_broken_old_ctypes_semantics;
347
348 if (fields == NULL)
349 return 0;
350
351#ifdef WORDS_BIGENDIAN
352 big_endian = PyObject_HasAttrString(type, "_swappedbytes_") ? 0 : 1;
353#else
354 big_endian = PyObject_HasAttrString(type, "_swappedbytes_") ? 1 : 0;
355#endif
356
357 use_broken_old_ctypes_semantics = \
358 PyObject_HasAttrString(type, "_use_broken_old_ctypes_structure_semantics_");
359
360 isPacked = PyObject_GetAttrString(type, "_pack_");
361 if (isPacked) {
362 pack = _PyLong_AsInt(isPacked);
363 if (pack < 0 || PyErr_Occurred()) {
364 Py_XDECREF(isPacked);
365 PyErr_SetString(PyExc_ValueError,
366 "_pack_ must be a non-negative integer");
367 return -1;
368 }
369 Py_DECREF(isPacked);
370 } else
371 PyErr_Clear();
372
373 len = PySequence_Length(fields);
374 if (len == -1) {
375 PyErr_SetString(PyExc_TypeError,
376 "'_fields_' must be a sequence of pairs");
377 return -1;
378 }
379
380 stgdict = PyType_stgdict(type);
381 if (!stgdict)
382 return -1;
383 /* If this structure/union is already marked final we cannot assign
384 _fields_ anymore. */
385
386 if (stgdict->flags & DICTFLAG_FINAL) {/* is final ? */
387 PyErr_SetString(PyExc_AttributeError,
388 "_fields_ is final");
389 return -1;
390 }
391
392 if (stgdict->format) {
393 PyMem_Free(stgdict->format);
394 stgdict->format = NULL;
395 }
396
397 if (stgdict->ffi_type_pointer.elements)
398 PyMem_Free(stgdict->ffi_type_pointer.elements);
399
400 basedict = PyType_stgdict((PyObject *)((PyTypeObject *)type)->tp_base);
401 if (basedict && !use_broken_old_ctypes_semantics) {
402 size = offset = basedict->size;
403 align = basedict->align;
404 union_size = 0;
405 total_align = align ? align : 1;
406 stgdict->ffi_type_pointer.type = FFI_TYPE_STRUCT;
407 stgdict->ffi_type_pointer.elements = PyMem_New(ffi_type *, basedict->length + len + 1);
408 if (stgdict->ffi_type_pointer.elements == NULL) {
409 PyErr_NoMemory();
410 return -1;
411 }
412 memset(stgdict->ffi_type_pointer.elements, 0,
413 sizeof(ffi_type *) * (basedict->length + len + 1));
414 if (basedict->length > 0) {
415 memcpy(stgdict->ffi_type_pointer.elements,
416 basedict->ffi_type_pointer.elements,
417 sizeof(ffi_type *) * (basedict->length));
418 }
419 ffi_ofs = basedict->length;
420 } else {
421 offset = 0;
422 size = 0;
423 align = 0;
424 union_size = 0;
425 total_align = 1;
426 stgdict->ffi_type_pointer.type = FFI_TYPE_STRUCT;
427 stgdict->ffi_type_pointer.elements = PyMem_New(ffi_type *, len + 1);
428 if (stgdict->ffi_type_pointer.elements == NULL) {
429 PyErr_NoMemory();
430 return -1;
431 }
432 memset(stgdict->ffi_type_pointer.elements, 0,
433 sizeof(ffi_type *) * (len + 1));
434 ffi_ofs = 0;
435 }
436
437 assert(stgdict->format == NULL);
438 if (isStruct && !isPacked) {
439 stgdict->format = _ctypes_alloc_format_string(NULL, "T{");
440 } else {
441 /* PEP3118 doesn't support union, or packed structures (well,
442 only standard packing, but we dont support the pep for
443 that). Use 'B' for bytes. */
444 stgdict->format = _ctypes_alloc_format_string(NULL, "B");
445 }
446 if (stgdict->format == NULL)
447 return -1;
448
449#define realdict ((PyObject *)&stgdict->dict)
450 for (i = 0; i < len; ++i) {
451 PyObject *name = NULL, *desc = NULL;
452 PyObject *pair = PySequence_GetItem(fields, i);
453 PyObject *prop;
454 StgDictObject *dict;
455 int bitsize = 0;
456
457 if (!pair || !PyArg_ParseTuple(pair, "UO|i", &name, &desc, &bitsize)) {
458 PyErr_SetString(PyExc_TypeError,
459 "'_fields_' must be a sequence of (name, C type) pairs");
460 Py_XDECREF(pair);
461 return -1;
462 }
463 dict = PyType_stgdict(desc);
464 if (dict == NULL) {
465 Py_DECREF(pair);
466 PyErr_Format(PyExc_TypeError,
467 "second item in _fields_ tuple (index %zd) must be a C type",
468 i);
469 return -1;
470 }
471 stgdict->ffi_type_pointer.elements[ffi_ofs + i] = &dict->ffi_type_pointer;
472 if (dict->flags & (TYPEFLAG_ISPOINTER | TYPEFLAG_HASPOINTER))
473 stgdict->flags |= TYPEFLAG_HASPOINTER;
474 dict->flags |= DICTFLAG_FINAL; /* mark field type final */
475 if (PyTuple_Size(pair) == 3) { /* bits specified */
476 switch(dict->ffi_type_pointer.type) {
477 case FFI_TYPE_UINT8:
478 case FFI_TYPE_UINT16:
479 case FFI_TYPE_UINT32:
480 case FFI_TYPE_SINT64:
481 case FFI_TYPE_UINT64:
482 break;
483
484 case FFI_TYPE_SINT8:
485 case FFI_TYPE_SINT16:
486 case FFI_TYPE_SINT32:
487 if (dict->getfunc != _ctypes_get_fielddesc("c")->getfunc
488#ifdef CTYPES_UNICODE
489 && dict->getfunc != _ctypes_get_fielddesc("u")->getfunc
490#endif
491 )
492 break;
493 /* else fall through */
494 default:
495 PyErr_Format(PyExc_TypeError,
496 "bit fields not allowed for type %s",
497 ((PyTypeObject *)desc)->tp_name);
498 Py_DECREF(pair);
499 return -1;
500 }
501 if (bitsize <= 0 || bitsize > dict->size * 8) {
502 PyErr_SetString(PyExc_ValueError,
503 "number of bits invalid for bit field");
504 Py_DECREF(pair);
505 return -1;
506 }
507 } else
508 bitsize = 0;
509
510 if (isStruct && !isPacked) {
511 const char *fieldfmt = dict->format ? dict->format : "B";
512 const char *fieldname = PyUnicode_AsUTF8(name);
513 char *ptr;
514 Py_ssize_t len;
515 char *buf;
516
517 if (fieldname == NULL)
518 {
519 Py_DECREF(pair);
520 return -1;
521 }
522
523 len = strlen(fieldname) + strlen(fieldfmt);
524
525 buf = PyMem_Malloc(len + 2 + 1);
526 if (buf == NULL) {
527 Py_DECREF(pair);
528 PyErr_NoMemory();
529 return -1;
530 }
531 sprintf(buf, "%s:%s:", fieldfmt, fieldname);
532
533 ptr = stgdict->format;
534 if (dict->shape != NULL) {
535 stgdict->format = _ctypes_alloc_format_string_with_shape(
536 dict->ndim, dict->shape, stgdict->format, buf);
537 } else {
538 stgdict->format = _ctypes_alloc_format_string(stgdict->format, buf);
539 }
540 PyMem_Free(ptr);
541 PyMem_Free(buf);
542
543 if (stgdict->format == NULL) {
544 Py_DECREF(pair);
545 return -1;
546 }
547 }
548
549 if (isStruct) {
550 prop = PyCField_FromDesc(desc, i,
551 &field_size, bitsize, &bitofs,
552 &size, &offset, &align,
553 pack, big_endian);
554 } else /* union */ {
555 size = 0;
556 offset = 0;
557 align = 0;
558 prop = PyCField_FromDesc(desc, i,
559 &field_size, bitsize, &bitofs,
560 &size, &offset, &align,
561 pack, big_endian);
562 union_size = max(size, union_size);
563 }
564 total_align = max(align, total_align);
565
566 if (!prop) {
567 Py_DECREF(pair);
568 return -1;
569 }
570 if (-1 == PyObject_SetAttr(type, name, prop)) {
571 Py_DECREF(prop);
572 Py_DECREF(pair);
573 return -1;
574 }
575 Py_DECREF(pair);
576 Py_DECREF(prop);
577 }
578#undef realdict
579
580 if (isStruct && !isPacked) {
581 char *ptr = stgdict->format;
582 stgdict->format = _ctypes_alloc_format_string(stgdict->format, "}");
583 PyMem_Free(ptr);
584 if (stgdict->format == NULL)
585 return -1;
586 }
587
588 if (!isStruct)
589 size = union_size;
590
591 /* Adjust the size according to the alignment requirements */
592 size = ((size + total_align - 1) / total_align) * total_align;
593
594 stgdict->ffi_type_pointer.alignment = Py_SAFE_DOWNCAST(total_align,
595 Py_ssize_t,
596 unsigned short);
597 stgdict->ffi_type_pointer.size = size;
598
599 stgdict->size = size;
600 stgdict->align = total_align;
601 stgdict->length = len; /* ADD ffi_ofs? */
602
603 /* We did check that this flag was NOT set above, it must not
604 have been set until now. */
605 if (stgdict->flags & DICTFLAG_FINAL) {
606 PyErr_SetString(PyExc_AttributeError,
607 "Structure or union cannot contain itself");
608 return -1;
609 }
610 stgdict->flags |= DICTFLAG_FINAL;
611
612 return MakeAnonFields(type);
613}
