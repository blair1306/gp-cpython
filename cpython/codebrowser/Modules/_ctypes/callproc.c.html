15/*
16 How are functions called, and how are parameters converted to C ?
17
18 1. _ctypes.c::PyCFuncPtr_call receives an argument tuple 'inargs' and a
19 keyword dictionary 'kwds'.
20
21 2. After several checks, _build_callargs() is called which returns another
22 tuple 'callargs'. This may be the same tuple as 'inargs', a slice of
23 'inargs', or a completely fresh tuple, depending on several things (is it a
24 COM method?, are 'paramflags' available?).
25
26 3. _build_callargs also calculates bitarrays containing indexes into
27 the callargs tuple, specifying how to build the return value(s) of
28 the function.
29
30 4. _ctypes_callproc is then called with the 'callargs' tuple. _ctypes_callproc first
31 allocates two arrays. The first is an array of 'struct argument' items, the
32 second array has 'void *' entries.
33
34 5. If 'converters' are present (converters is a sequence of argtypes'
35 from_param methods), for each item in 'callargs' converter is called and the
36 result passed to ConvParam. If 'converters' are not present, each argument
37 is directly passed to ConvParm.
38
39 6. For each arg, ConvParam stores the contained C data (or a pointer to it,
40 for structures) into the 'struct argument' array.
41
42 7. Finally, a loop fills the 'void *' array so that each item points to the
43 data contained in or pointed to by the 'struct argument' array.
44
45 8. The 'void *' argument array is what _call_function_pointer
46 expects. _call_function_pointer then has very little to do - only some
47 libffi specific stuff, then it calls ffi_call.
48
49 So, there are 4 data structures holding processed arguments:
50 - the inargs tuple (in PyCFuncPtr_call)
51 - the callargs tuple (in PyCFuncPtr_call)
52 - the 'struct arguments' array
53 - the 'void *' array
54
55 */

95/*
96 ctypes maintains thread-local storage that has space for two error numbers:
97 private copies of the system 'errno' value and, on Windows, the system error code
98 accessed by the GetLastError() and SetLastError() api functions.
99
100 Foreign functions created with CDLL(..., use_errno=True), when called, swap
101 the system 'errno' value with the private copy just before the actual
102 function call, and swapped again immediately afterwards. The 'use_errno'
103 parameter defaults to False, in this case 'ctypes_errno' is not touched.
104
105 On Windows, foreign functions created with CDLL(..., use_last_error=True) or
106 WinDLL(..., use_last_error=True) swap the system LastError value with the
107 ctypes private copy.
108
109 The values are also swapped immeditately before and after ctypes callback
110 functions are called, if the callbacks are constructed using the new
111 optional use_errno parameter set to True: CFUNCTYPE(..., use_errno=TRUE) or
112 WINFUNCTYPE(..., use_errno=True).
113
114 New ctypes functions are provided to access the ctypes private copies from
115 Python:
116
117 - ctypes.set_errno(value) and ctypes.set_last_error(value) store 'value' in
118 the private copy and returns the previous value.
119
120 - ctypes.get_errno() and ctypes.get_last_error() returns the current ctypes
121 private copies value.
122*/
123
124/*
125 This function creates and returns a thread-local Python object that has
126 space to store two integer error numbers; once created the Python object is
127 kept alive in the thread state dictionary as long as the thread itself.
128*/
129PyObject *
130_ctypes_get_errobj(int **pspace)
131{
132 PyObject *dict = PyThreadState_GetDict();
133 PyObject *errobj;
134 static PyObject *error_object_name;
135 if (dict == NULL) {
136 PyErr_SetString(PyExc_RuntimeError,
137 "cannot get thread state");
138 return NULL;
139 }
140 if (error_object_name == NULL) {
141 error_object_name = PyUnicode_InternFromString("ctypes.error_object");
142 if (error_object_name == NULL)
143 return NULL;
144 }
145 errobj = PyDict_GetItem(dict, error_object_name);
146 if (errobj) {
147 if (!PyCapsule_IsValid(errobj, CTYPES_CAPSULE_NAME_PYMEM)) {
148 PyErr_SetString(PyExc_RuntimeError,
149 "ctypes.error_object is an invalid capsule");
150 return NULL;
151 }
152 Py_INCREF(errobj);
153 }
154 else {
155 void *space = PyMem_Malloc(sizeof(int) * 2);
156 if (space == NULL)
157 return NULL;
158 memset(space, 0, sizeof(int) * 2);
159 errobj = PyCapsule_New(space, CTYPES_CAPSULE_NAME_PYMEM, pymem_destructor);
160 if (errobj == NULL) {
161 PyMem_Free(space);
162 return NULL;
163 }
164 if (-1 == PyDict_SetItem(dict, error_object_name,
165 errobj)) {
166 Py_DECREF(errobj);
167 return NULL;
168 }
169 }
170 *pspace = (int *)PyCapsule_GetPointer(errobj, CTYPES_CAPSULE_NAME_PYMEM);
171 return errobj;
172}

174static PyObject *
175get_error_internal(PyObject *self, PyObject *args, int index)
176{
177 int *space;
178 PyObject *errobj = _ctypes_get_errobj(&space);
179 PyObject *result;
180
181 if (errobj == NULL)
182 return NULL;
183 result = PyLong_FromLong(space[index]);
184 Py_DECREF(errobj);
185 return result;
186}

188static PyObject *
189set_error_internal(PyObject *self, PyObject *args, int index)
190{
191 int new_errno, old_errno;
192 PyObject *errobj;
193 int *space;
194
195 if (!PyArg_ParseTuple(args, "i", &new_errno))
196 return NULL;
197 errobj = _ctypes_get_errobj(&space);
198 if (errobj == NULL)
199 return NULL;
200 old_errno = space[index];
201 space[index] = new_errno;
202 Py_DECREF(errobj);
203 return PyLong_FromLong(old_errno);
204}

234static WCHAR *FormatError(DWORD code)
235{
236 WCHAR *lpMsgBuf;
237 DWORD n;
238 n = FormatMessageW(FORMAT_MESSAGE_ALLOCATE_BUFFER | FORMAT_MESSAGE_FROM_SYSTEM,
239 NULL,
240 code,
241 MAKELANGID(LANG_NEUTRAL, SUBLANG_DEFAULT), /* Default language */
242 (LPWSTR) &lpMsgBuf,
243 0,
244 NULL);
245 if (n) {
246 while (iswspace(lpMsgBuf[n-1]))
247 --n;
248 lpMsgBuf[n] = L'\0'; /* rstrip() */
249 }
250 return lpMsgBuf;
251}

254static void SetException(DWORD code, EXCEPTION_RECORD *pr)
255{
256 /* The 'code' is a normal win32 error code so it could be handled by
257 PyErr_SetFromWindowsErr(). However, for some errors, we have additional
258 information not included in the error code. We handle those here and
259 delegate all others to the generic function. */
260 switch (code) {
261 case EXCEPTION_ACCESS_VIOLATION:
262 /* The thread attempted to read from or write
263 to a virtual address for which it does not
264 have the appropriate access. */
265 if (pr->ExceptionInformation[0] == 0)
266 PyErr_Format(PyExc_OSError,
267 "exception: access violation reading %p",
268 pr->ExceptionInformation[1]);
269 else
270 PyErr_Format(PyExc_OSError,
271 "exception: access violation writing %p",
272 pr->ExceptionInformation[1]);
273 break;
274
275 case EXCEPTION_BREAKPOINT:
276 /* A breakpoint was encountered. */
277 PyErr_SetString(PyExc_OSError,
278 "exception: breakpoint encountered");
279 break;
280
281 case EXCEPTION_DATATYPE_MISALIGNMENT:
282 /* The thread attempted to read or write data that is
283 misaligned on hardware that does not provide
284 alignment. For example, 16-bit values must be
285 aligned on 2-byte boundaries, 32-bit values on
286 4-byte boundaries, and so on. */
287 PyErr_SetString(PyExc_OSError,
288 "exception: datatype misalignment");
289 break;
290
291 case EXCEPTION_SINGLE_STEP:
292 /* A trace trap or other single-instruction mechanism
293 signaled that one instruction has been executed. */
294 PyErr_SetString(PyExc_OSError,
295 "exception: single step");
296 break;
297
298 case EXCEPTION_ARRAY_BOUNDS_EXCEEDED:
299 /* The thread attempted to access an array element
300 that is out of bounds, and the underlying hardware
301 supports bounds checking. */
302 PyErr_SetString(PyExc_OSError,
303 "exception: array bounds exceeded");
304 break;
305
306 case EXCEPTION_FLT_DENORMAL_OPERAND:
307 /* One of the operands in a floating-point operation
308 is denormal. A denormal value is one that is too
309 small to represent as a standard floating-point
310 value. */
311 PyErr_SetString(PyExc_OSError,
312 "exception: floating-point operand denormal");
313 break;
314
315 case EXCEPTION_FLT_DIVIDE_BY_ZERO:
316 /* The thread attempted to divide a floating-point
317 value by a floating-point divisor of zero. */
318 PyErr_SetString(PyExc_OSError,
319 "exception: float divide by zero");
320 break;
321
322 case EXCEPTION_FLT_INEXACT_RESULT:
323 /* The result of a floating-point operation cannot be
324 represented exactly as a decimal fraction. */
325 PyErr_SetString(PyExc_OSError,
326 "exception: float inexact");
327 break;
328
329 case EXCEPTION_FLT_INVALID_OPERATION:
330 /* This exception represents any floating-point
331 exception not included in this list. */
332 PyErr_SetString(PyExc_OSError,
333 "exception: float invalid operation");
334 break;
335
336 case EXCEPTION_FLT_OVERFLOW:
337 /* The exponent of a floating-point operation is
338 greater than the magnitude allowed by the
339 corresponding type. */
340 PyErr_SetString(PyExc_OSError,
341 "exception: float overflow");
342 break;
343
344 case EXCEPTION_FLT_STACK_CHECK:
345 /* The stack overflowed or underflowed as the result
346 of a floating-point operation. */
347 PyErr_SetString(PyExc_OSError,
348 "exception: stack over/underflow");
349 break;
350
351 case EXCEPTION_STACK_OVERFLOW:
352 /* The stack overflowed or underflowed as the result
353 of a floating-point operation. */
354 PyErr_SetString(PyExc_OSError,
355 "exception: stack overflow");
356 break;
357
358 case EXCEPTION_FLT_UNDERFLOW:
359 /* The exponent of a floating-point operation is less
360 than the magnitude allowed by the corresponding
361 type. */
362 PyErr_SetString(PyExc_OSError,
363 "exception: float underflow");
364 break;
365
366 case EXCEPTION_INT_DIVIDE_BY_ZERO:
367 /* The thread attempted to divide an integer value by
368 an integer divisor of zero. */
369 PyErr_SetString(PyExc_OSError,
370 "exception: integer divide by zero");
371 break;
372
373 case EXCEPTION_INT_OVERFLOW:
374 /* The result of an integer operation caused a carry
375 out of the most significant bit of the result. */
376 PyErr_SetString(PyExc_OSError,
377 "exception: integer overflow");
378 break;
379
380 case EXCEPTION_PRIV_INSTRUCTION:
381 /* The thread attempted to execute an instruction
382 whose operation is not allowed in the current
383 machine mode. */
384 PyErr_SetString(PyExc_OSError,
385 "exception: privileged instruction");
386 break;
387
388 case EXCEPTION_NONCONTINUABLE_EXCEPTION:
389 /* The thread attempted to continue execution after a
390 noncontinuable exception occurred. */
391 PyErr_SetString(PyExc_OSError,
392 "exception: nocontinuable");
393 break;
394
395 default:
396 PyErr_SetFromWindowsErr(code);
397 break;
398 }
399}

401static DWORD HandleException(EXCEPTION_POINTERS *ptrs,
402 DWORD *pdw, EXCEPTION_RECORD *record)
403{
404 *pdw = ptrs->ExceptionRecord->ExceptionCode;
405 *record = *ptrs->ExceptionRecord;
406 /* We don't want to catch breakpoint exceptions, they are used to attach
407 * a debugger to the process.
408 */
409 if (*pdw == EXCEPTION_BREAKPOINT)
410 return EXCEPTION_CONTINUE_SEARCH;
411 return EXCEPTION_EXECUTE_HANDLER;
412}

451static PyObject *
452PyCArg_repr(PyCArgObject *self)
453{
454 char buffer[256];
455 switch(self->tag) {
456 case 'b':
457 case 'B':
458 sprintf(buffer, "<cparam '%c' (%d)>",
459 self->tag, self->value.b);
460 break;
461 case 'h':
462 case 'H':
463 sprintf(buffer, "<cparam '%c' (%d)>",
464 self->tag, self->value.h);
465 break;
466 case 'i':
467 case 'I':
468 sprintf(buffer, "<cparam '%c' (%d)>",
469 self->tag, self->value.i);
470 break;
471 case 'l':
472 case 'L':
473 sprintf(buffer, "<cparam '%c' (%ld)>",
474 self->tag, self->value.l);
475 break;
476
477 case 'q':
478 case 'Q':
479 sprintf(buffer,
480#ifdef MS_WIN32
481 "<cparam '%c' (%I64d)>",
482#else
483 "<cparam '%c' (%qd)>",
484#endif
485 self->tag, self->value.q);
486 break;
487 case 'd':
488 sprintf(buffer, "<cparam '%c' (%f)>",
489 self->tag, self->value.d);
490 break;
491 case 'f':
492 sprintf(buffer, "<cparam '%c' (%f)>",
493 self->tag, self->value.f);
494 break;
495
496 case 'c':
497 sprintf(buffer, "<cparam '%c' (%c)>",
498 self->tag, self->value.c);
499 break;
500
501/* Hm, are these 'z' and 'Z' codes useful at all?
502 Shouldn't they be replaced by the functionality of c_string
503 and c_wstring ?
504*/
505 case 'z':
506 case 'Z':
507 case 'P':
508 sprintf(buffer, "<cparam '%c' (%p)>",
509 self->tag, self->value.p);
510 break;
511
512 default:
513 sprintf(buffer, "<cparam '%c' at %p>",
514 self->tag, self);
515 break;
516 }
517 return PyUnicode_FromString(buffer);
518}

559/****************************************************************/
560/*
561 * Convert a PyObject * into a parameter suitable to pass to an
562 * C function call.
563 *
564 * 1. Python integers are converted to C int and passed by value.
565 * Py_None is converted to a C NULL pointer.
566 *
567 * 2. 3-tuples are expected to have a format character in the first
568 * item, which must be 'i', 'f', 'd', 'q', or 'P'.
569 * The second item will have to be an integer, float, double, long long
570 * or integer (denoting an address void *), will be converted to the
571 * corresponding C data type and passed by value.
572 *
573 * 3. Other Python objects are tested for an '_as_parameter_' attribute.
574 * The value of this attribute must be an integer which will be passed
575 * by value, or a 2-tuple or 3-tuple which will be used according
576 * to point 2 above. The third item (if any), is ignored. It is normally
577 * used to keep the object alive where this parameter refers to.
578 * XXX This convention is dangerous - you can construct arbitrary tuples
579 * in Python and pass them. Would it be safer to use a custom container
580 * datatype instead of a tuple?
581 *
582 * 4. Other Python objects cannot be passed as parameters - an exception is raised.
583 *
584 * 5. ConvParam will store the converted result in a struct containing format
585 * and value.
586 */
587
588union result {
589 char c;
590 char b;
591 short h;
592 int i;
593 long l;
594 long long q;
595 long double D;
596 double d;
597 float f;
598 void *p;
599};
600
601struct argument {
602 ffi_type *ffi_type;
603 PyObject *keep;
604 union result value;
605};

607/*
608 * Convert a single Python object into a PyCArgObject and return it.
609 */
610static int ConvParam(PyObject *obj, Py_ssize_t index, struct argument *pa)
611{
612 StgDictObject *dict;
613 pa->keep = NULL; /* so we cannot forget it later */
614
615 dict = PyObject_stgdict(obj);
616 if (dict) {
617 PyCArgObject *carg;
618 assert(dict->paramfunc);
619 /* If it has an stgdict, it is a CDataObject */
620 carg = dict->paramfunc((CDataObject *)obj);
621 if (carg == NULL)
622 return -1;
623 pa->ffi_type = carg->pffi_type;
624 memcpy(&pa->value, &carg->value, sizeof(pa->value));
625 pa->keep = (PyObject *)carg;
626 return 0;
627 }
628
629 if (PyCArg_CheckExact(obj)) {
630 PyCArgObject *carg = (PyCArgObject *)obj;
631 pa->ffi_type = carg->pffi_type;
632 Py_INCREF(obj);
633 pa->keep = obj;
634 memcpy(&pa->value, &carg->value, sizeof(pa->value));
635 return 0;
636 }
637
638 /* check for None, integer, string or unicode and use directly if successful */
639 if (obj == Py_None) {
640 pa->ffi_type = &ffi_type_pointer;
641 pa->value.p = NULL;
642 return 0;
643 }
644
645 if (PyLong_Check(obj)) {
646 pa->ffi_type = &ffi_type_sint;
647 pa->value.i = (long)PyLong_AsUnsignedLong(obj);
648 if (pa->value.i == -1 && PyErr_Occurred()) {
649 PyErr_Clear();
650 pa->value.i = PyLong_AsLong(obj);
651 if (pa->value.i == -1 && PyErr_Occurred()) {
652 PyErr_SetString(PyExc_OverflowError,
653 "int too long to convert");
654 return -1;
655 }
656 }
657 return 0;
658 }
659
660 if (PyBytes_Check(obj)) {
661 pa->ffi_type = &ffi_type_pointer;
662 pa->value.p = PyBytes_AsString(obj);
663 Py_INCREF(obj);
664 pa->keep = obj;
665 return 0;
666 }
667
668#ifdef CTYPES_UNICODE
669 if (PyUnicode_Check(obj)) {
670 pa->ffi_type = &ffi_type_pointer;
671 pa->value.p = PyUnicode_AsWideCharString(obj, NULL);
672 if (pa->value.p == NULL)
673 return -1;
674 pa->keep = PyCapsule_New(pa->value.p, CTYPES_CAPSULE_NAME_PYMEM, pymem_destructor);
675 if (!pa->keep) {
676 PyMem_Free(pa->value.p);
677 return -1;
678 }
679 return 0;
680 }
681#endif
682
683 {
684 PyObject *arg;
685 arg = PyObject_GetAttrString(obj, "_as_parameter_");
686 /* Which types should we exactly allow here?
687 integers are required for using Python classes
688 as parameters (they have to expose the '_as_parameter_'
689 attribute)
690 */
691 if (arg) {
692 int result;
693 result = ConvParam(arg, index, pa);
694 Py_DECREF(arg);
695 return result;
696 }
697 PyErr_Format(PyExc_TypeError,
698 "Don't know how to convert parameter %d",
699 Py_SAFE_DOWNCAST(index, Py_ssize_t, int));
700 return -1;
701 }
702}

704
705ffi_type *_ctypes_get_ffi_type(PyObject *obj)
706{
707 StgDictObject *dict;
708 if (obj == NULL)
709 return &ffi_type_sint;
710 dict = PyType_stgdict(obj);
711 if (dict == NULL)
712 return &ffi_type_sint;
713#if defined(MS_WIN32) && !defined(_WIN32_WCE)
714 /* This little trick works correctly with MSVC.
715 It returns small structures in registers
716 */
717 if (dict->ffi_type_pointer.type == FFI_TYPE_STRUCT) {
718 if (dict->ffi_type_pointer.size <= 4)
719 return &ffi_type_sint32;
720 else if (dict->ffi_type_pointer.size <= 8)
721 return &ffi_type_sint64;
722 }
723#endif
724 return &dict->ffi_type_pointer;
725}

728/*
729 * libffi uses:
730 *
731 * ffi_status ffi_prep_cif(ffi_cif *cif, ffi_abi abi,
732 * unsigned int nargs,
733 * ffi_type *rtype,
734 * ffi_type **atypes);
735 *
736 * and then
737 *
738 * void ffi_call(ffi_cif *cif, void *fn, void *rvalue, void **avalues);
739 */
740static int _call_function_pointer(int flags,
741 PPROC pProc,
742 void **avalues,
743 ffi_type **atypes,
744 ffi_type *restype,
745 void *resmem,
746 int argcount)
747{
748#ifdef WITH_THREAD
749 PyThreadState *_save = NULL; /* For Py_BLOCK_THREADS and Py_UNBLOCK_THREADS */
750#endif
751 PyObject *error_object = NULL;
752 int *space;
753 ffi_cif cif;
754 int cc;
755#ifdef MS_WIN32
756 int delta;
757#ifndef DONT_USE_SEH
758 DWORD dwExceptionCode = 0;
759 EXCEPTION_RECORD record;
760#endif
761#endif
762 /* XXX check before here */
763 if (restype == NULL) {
764 PyErr_SetString(PyExc_RuntimeError,
765 "No ffi_type for result");
766 return -1;
767 }
768
769 cc = FFI_DEFAULT_ABI;
770#if defined(MS_WIN32) && !defined(MS_WIN64) && !defined(_WIN32_WCE)
771 if ((flags & FUNCFLAG_CDECL) == 0)
772 cc = FFI_STDCALL;
773#endif
774 if (FFI_OK != ffi_prep_cif(&cif,
775 cc,
776 argcount,
777 restype,
778 atypes)) {
779 PyErr_SetString(PyExc_RuntimeError,
780 "ffi_prep_cif failed");
781 return -1;
782 }
783
784 if (flags & (FUNCFLAG_USE_ERRNO | FUNCFLAG_USE_LASTERROR)) {
785 error_object = _ctypes_get_errobj(&space);
786 if (error_object == NULL)
787 return -1;
788 }
789#ifdef WITH_THREAD
790 if ((flags & FUNCFLAG_PYTHONAPI) == 0)
791 Py_UNBLOCK_THREADS
792#endif
793 if (flags & FUNCFLAG_USE_ERRNO) {
794 int temp = space[0];
795 space[0] = errno;
796 errno = temp;
797 }
798#ifdef MS_WIN32
799 if (flags & FUNCFLAG_USE_LASTERROR) {
800 int temp = space[1];
801 space[1] = GetLastError();
802 SetLastError(temp);
803 }
804#ifndef DONT_USE_SEH
805 __try {
806#endif
807 delta =
808#endif
809 ffi_call(&cif, (void *)pProc, resmem, avalues);
810#ifdef MS_WIN32
811#ifndef DONT_USE_SEH
812 }
813 __except (HandleException(GetExceptionInformation(),
814 &dwExceptionCode, &record)) {
815 ;
816 }
817#endif
818 if (flags & FUNCFLAG_USE_LASTERROR) {
819 int temp = space[1];
820 space[1] = GetLastError();
821 SetLastError(temp);
822 }
823#endif
824 if (flags & FUNCFLAG_USE_ERRNO) {
825 int temp = space[0];
826 space[0] = errno;
827 errno = temp;
828 }
829#ifdef WITH_THREAD
830 if ((flags & FUNCFLAG_PYTHONAPI) == 0)
831 Py_BLOCK_THREADS
832#endif
833 Py_XDECREF(error_object);
834#ifdef MS_WIN32
835#ifndef DONT_USE_SEH
836 if (dwExceptionCode) {
837 SetException(dwExceptionCode, &record);
838 return -1;
839 }
840#endif
841#ifdef MS_WIN64
842 if (delta != 0) {
843 PyErr_Format(PyExc_RuntimeError,
844 "ffi_call failed with code %d",
845 delta);
846 return -1;
847 }
848#else
849 if (delta < 0) {
850 if (flags & FUNCFLAG_CDECL)
851 PyErr_Format(PyExc_ValueError,
852 "Procedure called with not enough "
853 "arguments (%d bytes missing) "
854 "or wrong calling convention",
855 -delta);
856 else
857 PyErr_Format(PyExc_ValueError,
858 "Procedure probably called with not enough "
859 "arguments (%d bytes missing)",
860 -delta);
861 return -1;
862 } else if (delta > 0) {
863 PyErr_Format(PyExc_ValueError,
864 "Procedure probably called with too many "
865 "arguments (%d bytes in excess)",
866 delta);
867 return -1;
868 }
869#endif
870#endif
871 if ((flags & FUNCFLAG_PYTHONAPI) && PyErr_Occurred())
872 return -1;
873 return 0;
874}

876/*
877 * Convert the C value in result into a Python object, depending on restype.
878 *
879 * - If restype is NULL, return a Python integer.
880 * - If restype is None, return None.
881 * - If restype is a simple ctypes type (c_int, c_void_p), call the type's getfunc,
882 * pass the result to checker and return the result.
883 * - If restype is another ctypes type, return an instance of that.
884 * - Otherwise, call restype and return the result.
885 */
886static PyObject *GetResult(PyObject *restype, void *result, PyObject *checker)
887{
888 StgDictObject *dict;
889 PyObject *retval, *v;
890
891 if (restype == NULL)
892 return PyLong_FromLong(*(int *)result);
893
894 if (restype == Py_None) {
895 Py_RETURN_NONE;
896 }
897
898 dict = PyType_stgdict(restype);
899 if (dict == NULL)
900 return PyObject_CallFunction(restype, "i", *(int *)result);
901
902 if (dict->getfunc && !_ctypes_simple_instance(restype)) {
903 retval = dict->getfunc(result, dict->size);
904 /* If restype is py_object (detected by comparing getfunc with
905 O_get), we have to call Py_DECREF because O_get has already
906 called Py_INCREF.
907 */
908 if (dict->getfunc == _ctypes_get_fielddesc("O")->getfunc) {
909 Py_DECREF(retval);
910 }
911 } else
912 retval = PyCData_FromBaseObj(restype, NULL, 0, result);
913
914 if (!checker || !retval)
915 return retval;
916
917 v = PyObject_CallFunctionObjArgs(checker, retval, NULL);
918 if (v == NULL)
919 _PyTraceback_Add("GetResult", "_ctypes/callproc.c", __LINE__-2);
920 Py_DECREF(retval);
921 return v;
922}

924/*
925 * Raise a new exception 'exc_class', adding additional text to the original
926 * exception string.
927 */
928void _ctypes_extend_error(PyObject *exc_class, const char *fmt, ...)
929{
930 va_list vargs;
931 PyObject *tp, *v, *tb, *s, *cls_str, *msg_str;
932
933 va_start(vargs, fmt);
934 s = PyUnicode_FromFormatV(fmt, vargs);
935 va_end(vargs);
936 if (!s)
937 return;
938
939 PyErr_Fetch(&tp, &v, &tb);
940 PyErr_NormalizeException(&tp, &v, &tb);
941 cls_str = PyObject_Str(tp);
942 if (cls_str) {
943 PyUnicode_AppendAndDel(&s, cls_str);
944 PyUnicode_AppendAndDel(&s, PyUnicode_FromString(": "));
945 if (s == NULL)
946 goto error;
947 } else
948 PyErr_Clear();
949 msg_str = PyObject_Str(v);
950 if (msg_str)
951 PyUnicode_AppendAndDel(&s, msg_str);
952 else {
953 PyErr_Clear();
954 PyUnicode_AppendAndDel(&s, PyUnicode_FromString("???"));
955 }
956 if (s == NULL)
957 goto error;
958 PyErr_SetObject(exc_class, s);
959error:
960 Py_XDECREF(tp);
961 Py_XDECREF(v);
962 Py_XDECREF(tb);
963 Py_XDECREF(s);
964}

1049/*
1050 * Requirements, must be ensured by the caller:
1051 * - argtuple is tuple of arguments
1052 * - argtypes is either NULL, or a tuple of the same size as argtuple
1053 *
1054 * - XXX various requirements for restype, not yet collected
1055 */
1056PyObject *_ctypes_callproc(PPROC pProc,
1057 PyObject *argtuple,
1058#ifdef MS_WIN32
1059 IUnknown *pIunk,
1060 GUID *iid,
1061#endif
1062 int flags,
1063 PyObject *argtypes, /* misleading name: This is a tuple of
1064 methods, not types: the .from_param
1065 class methods of the types */
1066 PyObject *restype,
1067 PyObject *checker)
1068{
1069 Py_ssize_t i, n, argcount, argtype_count;
1070 void *resbuf;
1071 struct argument *args, *pa;
1072 ffi_type **atypes;
1073 ffi_type *rtype;
1074 void **avalues;
1075 PyObject *retval = NULL;
1076
1077 n = argcount = PyTuple_GET_SIZE(argtuple);
1078#ifdef MS_WIN32
1079 /* an optional COM object this pointer */
1080 if (pIunk)
1081 ++argcount;
1082#endif
1083
1084 args = (struct argument *)alloca(sizeof(struct argument) * argcount);
1085 if (!args) {
1086 PyErr_NoMemory();
1087 return NULL;
1088 }
1089 memset(args, 0, sizeof(struct argument) * argcount);
1090 argtype_count = argtypes ? PyTuple_GET_SIZE(argtypes) : 0;
1091#ifdef MS_WIN32
1092 if (pIunk) {
1093 args[0].ffi_type = &ffi_type_pointer;
1094 args[0].value.p = pIunk;
1095 pa = &args[1];
1096 } else
1097#endif
1098 pa = &args[0];
1099
1100 /* Convert the arguments */
1101 for (i = 0; i < n; ++i, ++pa) {
1102 PyObject *converter;
1103 PyObject *arg;
1104 int err;
1105
1106 arg = PyTuple_GET_ITEM(argtuple, i); /* borrowed ref */
1107 /* For cdecl functions, we allow more actual arguments
1108 than the length of the argtypes tuple.
1109 This is checked in _ctypes::PyCFuncPtr_Call
1110 */
1111 if (argtypes && argtype_count > i) {
1112 PyObject *v;
1113 converter = PyTuple_GET_ITEM(argtypes, i);
1114 v = PyObject_CallFunctionObjArgs(converter, arg, NULL);
1115 if (v == NULL) {
1116 _ctypes_extend_error(PyExc_ArgError, "argument %d: ", i+1);
1117 goto cleanup;
1118 }
1119
1120 err = ConvParam(v, i+1, pa);
1121 Py_DECREF(v);
1122 if (-1 == err) {
1123 _ctypes_extend_error(PyExc_ArgError, "argument %d: ", i+1);
1124 goto cleanup;
1125 }
1126 } else {
1127 err = ConvParam(arg, i+1, pa);
1128 if (-1 == err) {
1129 _ctypes_extend_error(PyExc_ArgError, "argument %d: ", i+1);
1130 goto cleanup; /* leaking ? */
1131 }
1132 }
1133 }
1134
1135 rtype = _ctypes_get_ffi_type(restype);
1136 resbuf = alloca(max(rtype->size, sizeof(ffi_arg)));
1137
1138 avalues = (void **)alloca(sizeof(void *) * argcount);
1139 atypes = (ffi_type **)alloca(sizeof(ffi_type *) * argcount);
1140 if (!resbuf || !avalues || !atypes) {
1141 PyErr_NoMemory();
1142 goto cleanup;
1143 }
1144 for (i = 0; i < argcount; ++i) {
1145 atypes[i] = args[i].ffi_type;
1146#ifdef CTYPES_PASS_BY_REF_HACK
1147 size_t size = atypes[i]->size;
1148 if (IS_PASS_BY_REF(size)) {
1149 void *tmp = alloca(size);
1150 if (atypes[i]->type == FFI_TYPE_STRUCT)
1151 memcpy(tmp, args[i].value.p, size);
1152 else
1153 memcpy(tmp, (void*)&args[i].value, size);
1154
1155 avalues[i] = tmp;
1156 }
1157 else
1158#endif
1159 if (atypes[i]->type == FFI_TYPE_STRUCT)
1160 avalues[i] = (void *)args[i].value.p;
1161 else
1162 avalues[i] = (void *)&args[i].value;
1163 }
1164
1165 if (-1 == _call_function_pointer(flags, pProc, avalues, atypes,
1166 rtype, resbuf,
1167 Py_SAFE_DOWNCAST(argcount,
1168 Py_ssize_t,
1169 int)))
1170 goto cleanup;
1171
1172#ifdef WORDS_BIGENDIAN
1173 /* libffi returns the result in a buffer with sizeof(ffi_arg). This
1174 causes problems on big endian machines, since the result buffer
1175 address cannot simply be used as result pointer, instead we must
1176 adjust the pointer value:
1177 */
1178 /*
1179 XXX I should find out and clarify why this is needed at all,
1180 especially why adjusting for ffi_type_float must be avoided on
1181 64-bit platforms.
1182 */
1183 if (rtype->type != FFI_TYPE_FLOAT
1184 && rtype->type != FFI_TYPE_STRUCT
1185 && rtype->size < sizeof(ffi_arg))
1186 resbuf = (char *)resbuf + sizeof(ffi_arg) - rtype->size;
1187#endif
1188
1189#ifdef MS_WIN32
1190 if (iid && pIunk) {
1191 if (*(int *)resbuf & 0x80000000)
1192 retval = GetComError(*(HRESULT *)resbuf, iid, pIunk);
1193 else
1194 retval = PyLong_FromLong(*(int *)resbuf);
1195 } else if (flags & FUNCFLAG_HRESULT) {
1196 if (*(int *)resbuf & 0x80000000)
1197 retval = PyErr_SetFromWindowsErr(*(int *)resbuf);
1198 else
1199 retval = PyLong_FromLong(*(int *)resbuf);
1200 } else
1201#endif
1202 retval = GetResult(restype, resbuf, checker);
1203 cleanup:
1204 for (i = 0; i < argcount; ++i)
1205 Py_XDECREF(args[i].keep);
1206 return retval;
1207}

1592static PyObject *
1593resize(PyObject *self, PyObject *args)
1594{
1595 CDataObject *obj;
1596 StgDictObject *dict;
1597 Py_ssize_t size;
1598
1599 if (!PyArg_ParseTuple(args,
1600 "On:resize",
1601 &obj, &size))
1602 return NULL;
1603
1604 dict = PyObject_stgdict((PyObject *)obj);
1605 if (dict == NULL) {
1606 PyErr_SetString(PyExc_TypeError,
1607 "excepted ctypes instance");
1608 return NULL;
1609 }
1610 if (size < dict->size) {
1611 PyErr_Format(PyExc_ValueError,
1612 "minimum size is %zd",
1613 dict->size);
1614 return NULL;
1615 }
1616 if (obj->b_needsfree == 0) {
1617 PyErr_Format(PyExc_ValueError,
1618 "Memory cannot be resized because this object doesn't own it");
1619 return NULL;
1620 }
1621 if ((size_t)size <= sizeof(obj->b_value)) {
1622 /* internal default buffer is large enough */
1623 obj->b_size = size;
1624 goto done;
1625 }
1626 if (!_CDataObject_HasExternalBuffer(obj)) {
1627 /* We are currently using the objects default buffer, but it
1628 isn't large enough any more. */
1629 void *ptr = PyMem_Malloc(size);
1630 if (ptr == NULL)
1631 return PyErr_NoMemory();
1632 memset(ptr, 0, size);
1633 memmove(ptr, obj->b_ptr, obj->b_size);
1634 obj->b_ptr = ptr;
1635 obj->b_size = size;
1636 } else {
1637 void * ptr = PyMem_Realloc(obj->b_ptr, size);
1638 if (ptr == NULL)
1639 return PyErr_NoMemory();
1640 obj->b_ptr = ptr;
1641 obj->b_size = size;
1642 }
1643 done:
1644 Py_RETURN_NONE;
1645}

1647static PyObject *
1648unpickle(PyObject *self, PyObject *args)
1649{
1650 PyObject *typ, *state, *meth, *obj, *result;
1651 _Py_IDENTIFIER(__new__);
1652 _Py_IDENTIFIER(__setstate__);
1653
1654 if (!PyArg_ParseTuple(args, "OO!", &typ, &PyTuple_Type, &state))
1655 return NULL;
1656 obj = _PyObject_CallMethodIdObjArgs(typ, &PyId___new__, typ, NULL);
1657 if (obj == NULL)
1658 return NULL;
1659
1660 meth = _PyObject_GetAttrId(obj, &PyId___setstate__);
1661 if (meth == NULL) {
1662 goto error;
1663 }
1664
1665 result = PyObject_Call(meth, state, NULL);
1666 Py_DECREF(meth);
1667 if (result == NULL) {
1668 goto error;
1669 }
1670 Py_DECREF(result);
1671
1672 return obj;
1673
1674error:
1675 Py_DECREF(obj);
1676 return NULL;
1677}

1679static PyObject *
1680POINTER(PyObject *self, PyObject *cls)
1681{
1682 PyObject *result;
1683 PyTypeObject *typ;
1684 PyObject *key;
1685 char *buf;
1686
1687 result = PyDict_GetItem(_ctypes_ptrtype_cache, cls);
1688 if (result) {
1689 Py_INCREF(result);
1690 return result;
1691 }
1692 if (PyUnicode_CheckExact(cls)) {
1693 const char *name = PyUnicode_AsUTF8(cls);
1694 if (name == NULL)
1695 return NULL;
1696 buf = PyMem_Malloc(strlen(name) + 3 + 1);
1697 if (buf == NULL)
1698 return PyErr_NoMemory();
1699 sprintf(buf, "LP_%s", name);
1700 result = PyObject_CallFunction((PyObject *)Py_TYPE(&PyCPointer_Type),
1701 "s(O){}",
1702 buf,
1703 &PyCPointer_Type);
1704 PyMem_Free(buf);
1705 if (result == NULL)
1706 return result;
1707 key = PyLong_FromVoidPtr(result);
1708 if (key == NULL) {
1709 Py_DECREF(result);
1710 return NULL;
1711 }
1712 } else if (PyType_Check(cls)) {
1713 typ = (PyTypeObject *)cls;
1714 buf = PyMem_Malloc(strlen(typ->tp_name) + 3 + 1);
1715 if (buf == NULL)
1716 return PyErr_NoMemory();
1717 sprintf(buf, "LP_%s", typ->tp_name);
1718 result = PyObject_CallFunction((PyObject *)Py_TYPE(&PyCPointer_Type),
1719 "s(O){sO}",
1720 buf,
1721 &PyCPointer_Type,
1722 "_type_", cls);
1723 PyMem_Free(buf);
1724 if (result == NULL)
1725 return result;
1726 Py_INCREF(cls);
1727 key = cls;
1728 } else {
1729 PyErr_SetString(PyExc_TypeError, "must be a ctypes type");
1730 return NULL;
1731 }
1732 if (-1 == PyDict_SetItem(_ctypes_ptrtype_cache, key, result)) {
1733 Py_DECREF(result);
1734 Py_DECREF(key);
1735 return NULL;
1736 }
1737 Py_DECREF(key);
1738 return result;
1739}

1743{

1747 typ = PyDict_GetItem(_ctypes_ptrtype_cache, (PyObject *)Py_TYPE(arg));

1758static PyObject *
1759buffer_info(PyObject *self, PyObject *arg)
1760{
1761 StgDictObject *dict = PyType_stgdict(arg);
1762 PyObject *shape;
1763 Py_ssize_t i;
1764
1765 if (dict == NULL)
1766 dict = PyObject_stgdict(arg);
1767 if (dict == NULL) {
1768 PyErr_SetString(PyExc_TypeError,
1769 "not a ctypes type or object");
1770 return NULL;
1771 }
1772 shape = PyTuple_New(dict->ndim);
1773 if (shape == NULL)
1774 return NULL;
1775 for (i = 0; i < (int)dict->ndim; ++i)
1776 PyTuple_SET_ITEM(shape, i, PyLong_FromSsize_t(dict->shape[i]));
1777
1778 if (PyErr_Occurred()) {
1779 Py_DECREF(shape);
1780 return NULL;
1781 }
1782 return Py_BuildValue("siN", dict->format, dict->ndim, shape);
1783}
