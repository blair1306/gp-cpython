113static int
114usage(int exitcode, const wchar_t* program)
115{
116 FILE *f = exitcode ? stderr : stdout;
117
118 fprintf(f, usage_line, program);
119 if (exitcode)
120 fprintf(f, "Try `python -h' for more information.\n");
121 else {
122 fputs(usage_1, f);
123 fputs(usage_2, f);
124 fputs(usage_3, f);
125 fprintf(f, usage_4, (wint_t)DELIM);
126 fprintf(f, usage_5, (wint_t)DELIM, PYTHONHOMEHELP);
127 fputs(usage_6, f);
128 }
129 return exitcode;
130}

132static void RunStartupFile(PyCompilerFlags *cf)
133{
134 char *startup = Py_GETENV("PYTHONSTARTUP");
135 if (startup != NULL && startup[0] != '\0') {
136 FILE *fp = _Py_fopen(startup, "r");
137 if (fp != NULL) {
138 (void) PyRun_SimpleFileExFlags(fp, startup, 0, cf);
139 PyErr_Clear();
140 fclose(fp);
141 } else {
142 int save_errno;
143
144 save_errno = errno;
145 PySys_WriteStderr("Could not open PYTHONSTARTUP\n");
146 errno = save_errno;
147 PyErr_SetFromErrnoWithFilename(PyExc_OSError,
148 startup);
149 PyErr_Print();
150 PyErr_Clear();
151 }
152 }

155static void RunInteractiveHook(void)
156{
157 PyObject *sys, *hook, *result;
158 sys = PyImport_ImportModule("sys");
159 if (sys == NULL)
160 goto error;
161 hook = PyObject_GetAttrString(sys, "__interactivehook__");
162 Py_DECREF(sys);
163 if (hook == NULL)
164 PyErr_Clear();
165 else {
166 result = _PyObject_CallNoArg(hook);
167 Py_DECREF(hook);
168 if (result == NULL)
169 goto error;
170 else
171 Py_DECREF(result);
172 }
173 return;
174
175error:
176 PySys_WriteStderr("Failed calling sys.__interactivehook__\n");
177 PyErr_Print();
178 PyErr_Clear();
179}

182static int RunModule(wchar_t *modname, int set_argv0)
183{
184 PyObject *module, *runpy, *runmodule, *runargs, *result;
185 runpy = PyImport_ImportModule("runpy");
186 if (runpy == NULL) {
187 fprintf(stderr, "Could not import runpy module\n");
188 PyErr_Print();
189 return -1;
190 }
191 runmodule = PyObject_GetAttrString(runpy, "_run_module_as_main");
192 if (runmodule == NULL) {
193 fprintf(stderr, "Could not access runpy._run_module_as_main\n");
194 PyErr_Print();
195 Py_DECREF(runpy);
196 return -1;
197 }
198 module = PyUnicode_FromWideChar(modname, wcslen(modname));
199 if (module == NULL) {
200 fprintf(stderr, "Could not convert module name to unicode\n");
201 PyErr_Print();
202 Py_DECREF(runpy);
203 Py_DECREF(runmodule);
204 return -1;
205 }
206 runargs = Py_BuildValue("(Oi)", module, set_argv0);
207 if (runargs == NULL) {
208 fprintf(stderr,
209 "Could not create arguments for runpy._run_module_as_main\n");
210 PyErr_Print();
211 Py_DECREF(runpy);
212 Py_DECREF(runmodule);
213 Py_DECREF(module);
214 return -1;
215 }
216 result = PyObject_Call(runmodule, runargs, NULL);
217 if (result == NULL) {
218 PyErr_Print();
219 }
220 Py_DECREF(runpy);
221 Py_DECREF(runmodule);
222 Py_DECREF(module);
223 Py_DECREF(runargs);
224 if (result == NULL) {
225 return -1;
226 }
227 Py_DECREF(result);
228 return 0;
229}

231static PyObject *
232AsImportPathEntry(wchar_t *filename)
233{
234 PyObject *sys_path0 = NULL, *importer;
235
236 sys_path0 = PyUnicode_FromWideChar(filename, wcslen(filename));
237 if (sys_path0 == NULL)
238 goto error;
239
240 importer = PyImport_GetImporter(sys_path0);
241 if (importer == NULL)
242 goto error;
243
244 if (importer == Py_None) {
245 Py_DECREF(sys_path0);
246 Py_DECREF(importer);
247 return NULL;
248 }
249 Py_DECREF(importer);
250 return sys_path0;
251
252error:
253 Py_XDECREF(sys_path0);
254 PySys_WriteStderr("Failed checking if argv[0] is an import path entry\n");
255 PyErr_Print();
256 PyErr_Clear();
257 return NULL;
258}

261static int
262RunMainFromImporter(PyObject *sys_path0)
263{
264 PyObject *sys_path;
265 int sts;
266
267 /* Assume sys_path0 has already been checked by AsImportPathEntry,
268 * so put it in sys.path[0] and import __main__ */
269 sys_path = PySys_GetObject("path");
270 if (sys_path == NULL) {
271 PyErr_SetString(PyExc_RuntimeError, "unable to get sys.path");
272 goto error;
273 }
274 sts = PyList_Insert(sys_path, 0, sys_path0);
275 if (sts) {
276 sys_path0 = NULL;
277 goto error;
278 }
279
280 sts = RunModule(L"__main__", 0);
281 return sts != 0;
282
283error:
284 Py_XDECREF(sys_path0);
285 PyErr_Print();
286 return 1;
287}

289static int
290run_command(wchar_t *command, PyCompilerFlags *cf)
291{
292 PyObject *unicode, *bytes;
293 int ret;
294
295 unicode = PyUnicode_FromWideChar(command, -1);
296 if (unicode == NULL)
297 goto error;
298 bytes = PyUnicode_AsUTF8String(unicode);
299 Py_DECREF(unicode);
300 if (bytes == NULL)
301 goto error;
302 ret = PyRun_SimpleStringFlags(PyBytes_AsString(bytes), cf);
303 Py_DECREF(bytes);
304 return ret != 0;
305
306error:
307 PySys_WriteStderr("Unable to decode the command from the command line:\n");
308 PyErr_Print();
309 return 1;
310}

312static int
313run_file(FILE *fp, const wchar_t *filename, PyCompilerFlags *p_cf)
314{
315 PyObject *unicode, *bytes = NULL;
316 char *filename_str;
317 int run;
318
319 /* call pending calls like signal handlers (SIGINT) */
320 if (Py_MakePendingCalls() == -1) {
321 PyErr_Print();
322 return 1;
323 }
324
325 if (filename) {
326 unicode = PyUnicode_FromWideChar(filename, wcslen(filename));
327 if (unicode != NULL) {
328 bytes = PyUnicode_EncodeFSDefault(unicode);
329 Py_DECREF(unicode);
330 }
331 if (bytes != NULL)
332 filename_str = PyBytes_AsString(bytes);
333 else {
334 PyErr_Clear();
335 filename_str = "<encoding error>";
336 }
337 }
338 else
339 filename_str = "<stdin>";
340
341 run = PyRun_AnyFileExFlags(fp, filename_str, filename != NULL, p_cf);
342 Py_XDECREF(bytes);
343 return run != 0;
344}

379static int
380read_command_line(int argc, wchar_t **argv, _Py_CommandLineDetails *cmdline)
381{
382 PyObject *warning_option = NULL;
383 wchar_t *command = NULL;
384 wchar_t *module = NULL;
385 int c;
386
387 _PyOS_ResetGetOpt();
388
389 while ((c = _PyOS_GetOpt(argc, argv, PROGRAM_OPTS)) != EOF) {
390 if (c == 'c') {
391 size_t len;
392 /* -c is the last option; following arguments
393 that look like options are left for the
394 command to interpret. */
395
396 len = wcslen(_PyOS_optarg) + 1 + 1;
397 command = (wchar_t *)PyMem_RawMalloc(sizeof(wchar_t) * len);
398 if (command == NULL)
399 Py_FatalError(
400 "not enough memory to copy -c argument");
401 wcscpy(command, _PyOS_optarg);
402 command[len - 2] = '\n';
403 command[len - 1] = 0;
404 cmdline->command = command;
405 break;
406 }
407
408 if (c == 'm') {
409 /* -m is the last option; following arguments
410 that look like options are left for the
411 module to interpret. */
412 module = _PyOS_optarg;
413 cmdline->module = module;
414 break;
415 }
416
417 switch (c) {
418 case 'b':
419 cmdline->bytes_warning++;
420 break;
421
422 case 'd':
423 cmdline->debug++;
424 break;
425
426 case 'i':
427 cmdline->inspect++;
428 cmdline->interactive++;
429 break;
430
431 case 'I':
432 cmdline->isolated++;
433 cmdline->no_user_site_directory++;
434 break;
435
436 /* case 'J': reserved for Jython */
437
438 case 'O':
439 cmdline->optimization_level++;
440 break;
441
442 case 'B':
443 cmdline->dont_write_bytecode++;
444 break;
445
446 case 's':
447 cmdline->no_user_site_directory++;
448 break;
449
450 case 'S':
451 cmdline->no_site_import++;
452 break;
453
454 case 'E':
455 /* Handled prior to core initialization */
456 break;
457
458 case 't':
459 /* ignored for backwards compatibility */
460 break;
461
462 case 'u':
463 cmdline->use_unbuffered_io = 1;
464 break;
465
466 case 'v':
467 cmdline->verbosity++;
468 break;
469
470 case 'x':
471 cmdline->skip_first_line = 1;
472 break;
473
474 case 'h':
475 case '?':
476 cmdline->print_help++;
477 break;
478
479 case 'V':
480 cmdline->print_version++;
481 break;
482
483 case 'W':
484 if (cmdline->warning_options == NULL)
485 cmdline->warning_options = PyList_New(0);
486 if (cmdline->warning_options == NULL)
487 Py_FatalError("failure in handling of -W argument");
488 warning_option = PyUnicode_FromWideChar(_PyOS_optarg, -1);
489 if (warning_option == NULL)
490 Py_FatalError("failure in handling of -W argument");
491 if (PyList_Append(cmdline->warning_options, warning_option) == -1)
492 Py_FatalError("failure in handling of -W argument");
493 Py_DECREF(warning_option);
494 break;
495
496 case 'X':
497 /* TODO: Delay addition of X options to sys module */
498 PySys_AddXOption(_PyOS_optarg);
499 break;
500
501 case 'q':
502 cmdline->quiet_flag++;
503 break;
504
505 case 'R':
506 /* Ignored */
507 break;
508
509 /* This space reserved for other options */
510
511 default:
512 return -1;
513 /*NOTREACHED*/
514
515 }
516 }
517
518 if (command == NULL && module == NULL && _PyOS_optind < argc &&
519 wcscmp(argv[_PyOS_optind], L"-") != 0)
520 {
521 cmdline->filename = argv[_PyOS_optind];
522 }
523 return 0;
524}

526static int
527apply_command_line_and_environment(_Py_CommandLineDetails *cmdline)
528{
529 char *p;
530 Py_BytesWarningFlag = cmdline->bytes_warning;
531 Py_DebugFlag = cmdline->debug;
532 Py_InspectFlag = cmdline->inspect;
533 Py_InteractiveFlag = cmdline->interactive;
534 Py_IsolatedFlag = cmdline->isolated;
535 Py_OptimizeFlag = cmdline->optimization_level;
536 Py_DontWriteBytecodeFlag = cmdline->dont_write_bytecode;
537 Py_NoUserSiteDirectory = cmdline->no_user_site_directory;
538 Py_NoSiteFlag = cmdline->no_site_import;
539 Py_UnbufferedStdioFlag = cmdline->use_unbuffered_io;
540 Py_VerboseFlag = cmdline->verbosity;
541 Py_QuietFlag = cmdline->quiet_flag;
542
543 if (!Py_InspectFlag &&
544 (p = Py_GETENV("PYTHONINSPECT")) && *p != '\0') {
545 Py_InspectFlag = 1;
546 cmdline->inspect = 1;
547 }
548 if (!cmdline->use_unbuffered_io &&
549 (p = Py_GETENV("PYTHONUNBUFFERED")) && *p != '\0') {
550 Py_UnbufferedStdioFlag = 1;
551 cmdline->use_unbuffered_io = 1;
552 }
553
554 if (!Py_NoUserSiteDirectory &&
555 (p = Py_GETENV("PYTHONNOUSERSITE")) && *p != '\0') {
556 Py_NoUserSiteDirectory = 1;
557 cmdline->no_user_site_directory = 1;
558 }
559
560 /* TODO: Apply PYTHONWARNINGS & -W options to sys module here */
561 /* TODO: Apply -X options to sys module here */
562 return 0;
563}

565int
566Py_Main(int argc, wchar_t **argv)
567{
568 int c;
569 int sts;
570 FILE *fp = stdin;
571 char *p;
572#ifdef MS_WINDOWS
573 wchar_t *wp;
574#endif
575 int stdin_is_interactive = 0;
576 _Py_CommandLineDetails cmdline = _Py_CommandLineDetails_INIT;
577 _PyCoreConfig core_config = _PyCoreConfig_INIT;
578 PyCompilerFlags cf;
579 PyObject *main_importer_path = NULL;
580
581 cf.cf_flags = 0;
582
583 orig_argc = argc; /* For Py_GetArgcArgv() */
584 orig_argv = argv;
585
586 /* Hash randomization needed early for all string operations
587 (including -W and -X options). */
588 _PyOS_opterr = 0; /* prevent printing the error in 1st pass */
589 while ((c = _PyOS_GetOpt(argc, argv, PROGRAM_OPTS)) != EOF) {
590 if (c == 'm' || c == 'c') {
591 /* -c / -m is the last option: following arguments are
592 not interpreter options. */
593 break;
594 }
595 if (c == 'E' || c == 'I') {
596 core_config.ignore_environment++;
597 break;
598 }
599 }
600
601 char *pymalloc = Py_GETENV("PYTHONMALLOC");
602 if (_PyMem_SetupAllocators(pymalloc) < 0) {
603 fprintf(stderr,
604 "Error in PYTHONMALLOC: unknown allocator \"%s\"!\n", pymalloc);
605 exit(1);
606 }
607
608 /* Initialize the core language runtime */
609 Py_IgnoreEnvironmentFlag = core_config.ignore_environment;
610 core_config._disable_importlib = 0;
611 _Py_InitializeCore(&core_config);
612
613 /* Reprocess the command line with the language runtime available */
614 if (read_command_line(argc, argv, &cmdline)) {
615 return usage(2, argv[0]);
616 }
617
618 if (cmdline.print_help) {
619 return usage(0, argv[0]);
620 }
621
622 if (cmdline.print_version) {
623 printf("Python %s\n", cmdline.print_version >= 2 ? Py_GetVersion() : PY_VERSION);
624 return 0;
625 }
626
627 PySys_ResetWarnOptions();
628 apply_command_line_and_environment(&cmdline);
629
630#ifdef MS_WINDOWS
631 if (!Py_IgnoreEnvironmentFlag && (wp = _wgetenv(L"PYTHONWARNINGS")) &&
632 *wp != L'\0') {
633 wchar_t *buf, *warning, *context = NULL;
634
635 buf = (wchar_t *)PyMem_RawMalloc((wcslen(wp) + 1) * sizeof(wchar_t));
636 if (buf == NULL)
637 Py_FatalError(
638 "not enough memory to copy PYTHONWARNINGS");
639 wcscpy(buf, wp);
640 for (warning = wcstok_s(buf, L",", &context);
641 warning != NULL;
642 warning = wcstok_s(NULL, L",", &context)) {
643 PySys_AddWarnOption(warning);
644 }
645 PyMem_RawFree(buf);
646 }
647#else
648 if ((p = Py_GETENV("PYTHONWARNINGS")) && *p != '\0') {
649 char *buf, *oldloc;
650 PyObject *unicode;
651
652 /* settle for strtok here as there's no one standard
653 C89 wcstok */
654 buf = (char *)PyMem_RawMalloc(strlen(p) + 1);
655 if (buf == NULL)
656 Py_FatalError(
657 "not enough memory to copy PYTHONWARNINGS");
658 strcpy(buf, p);
659 oldloc = _PyMem_RawStrdup(setlocale(LC_ALL, NULL));
660 setlocale(LC_ALL, "");
661 for (p = strtok(buf, ","); p != NULL; p = strtok(NULL, ",")) {
662#ifdef __APPLE__
663 /* Use utf-8 on Mac OS X */
664 unicode = PyUnicode_FromString(p);
665#else
666 unicode = PyUnicode_DecodeLocale(p, "surrogateescape");
667#endif
668 if (unicode == NULL) {
669 /* ignore errors */
670 PyErr_Clear();
671 continue;
672 }
673 PySys_AddWarnOptionUnicode(unicode);
674 Py_DECREF(unicode);
675 }
676 setlocale(LC_ALL, oldloc);
677 PyMem_RawFree(oldloc);
678 PyMem_RawFree(buf);
679 }
680#endif
681 if (cmdline.warning_options != NULL) {
682 Py_ssize_t i;
683 for (i = 0; i < PyList_GET_SIZE(cmdline.warning_options); i++) {
684 PySys_AddWarnOptionUnicode(PyList_GET_ITEM(cmdline.warning_options, i));
685 }
686 Py_DECREF(cmdline.warning_options);
687 }
688
689 stdin_is_interactive = Py_FdIsInteractive(stdin, (char *)0);
690
691#if defined(MS_WINDOWS) || defined(__CYGWIN__)
692 /* don't translate newlines (\r\n <=> \n) */
693 _setmode(fileno(stdin), O_BINARY);
694 _setmode(fileno(stdout), O_BINARY);
695 _setmode(fileno(stderr), O_BINARY);
696#endif
697
698 if (Py_UnbufferedStdioFlag) {
699#ifdef HAVE_SETVBUF
700 setvbuf(stdin, (char *)NULL, _IONBF, BUFSIZ);
701 setvbuf(stdout, (char *)NULL, _IONBF, BUFSIZ);
702 setvbuf(stderr, (char *)NULL, _IONBF, BUFSIZ);
703#else /* !HAVE_SETVBUF */
704 setbuf(stdin, (char *)NULL);
705 setbuf(stdout, (char *)NULL);
706 setbuf(stderr, (char *)NULL);
707#endif /* !HAVE_SETVBUF */
708 }
709 else if (Py_InteractiveFlag) {
710#ifdef MS_WINDOWS
711 /* Doesn't have to have line-buffered -- use unbuffered */
712 /* Any set[v]buf(stdin, ...) screws up Tkinter :-( */
713 setvbuf(stdout, (char *)NULL, _IONBF, BUFSIZ);
714#else /* !MS_WINDOWS */
715#ifdef HAVE_SETVBUF
716 setvbuf(stdin, (char *)NULL, _IOLBF, BUFSIZ);
717 setvbuf(stdout, (char *)NULL, _IOLBF, BUFSIZ);
718#endif /* HAVE_SETVBUF */
719#endif /* !MS_WINDOWS */
720 /* Leave stderr alone - it should be unbuffered anyway. */
721 }
722
723#ifdef __APPLE__
724 /* On MacOS X, when the Python interpreter is embedded in an
725 application bundle, it gets executed by a bootstrapping script
726 that does os.execve() with an argv[0] that's different from the
727 actual Python executable. This is needed to keep the Finder happy,
728 or rather, to work around Apple's overly strict requirements of
729 the process name. However, we still need a usable sys.executable,
730 so the actual executable path is passed in an environment variable.
731 See Lib/plat-mac/bundlebuiler.py for details about the bootstrap
732 script. */
733 if ((p = Py_GETENV("PYTHONEXECUTABLE")) && *p != '\0') {
734 wchar_t* buffer;
735 size_t len = strlen(p) + 1;
736
737 buffer = PyMem_RawMalloc(len * sizeof(wchar_t));
738 if (buffer == NULL) {
739 Py_FatalError(
740 "not enough memory to copy PYTHONEXECUTABLE");
741 }
742
743 mbstowcs(buffer, p, len);
744 Py_SetProgramName(buffer);
745 /* buffer is now handed off - do not free */
746 } else {
747#ifdef WITH_NEXT_FRAMEWORK
748 char* pyvenv_launcher = getenv("__PYVENV_LAUNCHER__");
749
750 if (pyvenv_launcher && *pyvenv_launcher) {
751 /* Used by Mac/Tools/pythonw.c to forward
752 * the argv0 of the stub executable
753 */
754 wchar_t* wbuf = Py_DecodeLocale(pyvenv_launcher, NULL);
755
756 if (wbuf == NULL) {
757 Py_FatalError("Cannot decode __PYVENV_LAUNCHER__");
758 }
759 Py_SetProgramName(wbuf);
760
761 /* Don't free wbuf, the argument to Py_SetProgramName
762 * must remain valid until Py_FinalizeEx is called.
763 */
764 } else {
765 Py_SetProgramName(argv[0]);
766 }
767#else
768 Py_SetProgramName(argv[0]);
769#endif
770 }
771#else
772 Py_SetProgramName(argv[0]);
773#endif
774 /* Replaces previous call to Py_Initialize()
775 *
776 * TODO: Move environment queries (etc) into Py_ReadConfig
777 */
778 {
779 _PyMainInterpreterConfig config = _PyMainInterpreterConfig_INIT;
780
781 /* TODO: Moar config options! */
782 config.install_signal_handlers = 1;
783 /* TODO: Print any exceptions raised by these operations */
784 if (_Py_ReadMainInterpreterConfig(&config))
785 Py_FatalError("Py_Main: Py_ReadMainInterpreterConfig failed");
786 if (_Py_InitializeMainInterpreter(&config))
787 Py_FatalError("Py_Main: Py_InitializeMainInterpreter failed");
788 }
789
790 /* TODO: Move this to _PyRun_PrepareMain */
791 if (!Py_QuietFlag && (Py_VerboseFlag ||
792 (cmdline.command == NULL && cmdline.filename == NULL &&
793 cmdline.module == NULL && stdin_is_interactive))) {
794 fprintf(stderr, "Python %s on %s\n",
795 Py_GetVersion(), Py_GetPlatform());
796 if (!Py_NoSiteFlag)
797 fprintf(stderr, "%s\n", COPYRIGHT);
798 }
799
800 /* TODO: Move this to _Py_InitializeMainInterpreter */
801 if (cmdline.command != NULL) {
802 /* Backup _PyOS_optind and force sys.argv[0] = '-c' */
803 _PyOS_optind--;
804 argv[_PyOS_optind] = L"-c";
805 }
806
807 if (cmdline.module != NULL) {
808 /* Backup _PyOS_optind and force sys.argv[0] = '-m'*/
809 _PyOS_optind--;
810 argv[_PyOS_optind] = L"-m";
811 }
812
813 if (cmdline.filename != NULL) {
814 main_importer_path = AsImportPathEntry(cmdline.filename);
815 }
816
817 if (main_importer_path != NULL) {
818 /* Let RunMainFromImporter adjust sys.path[0] later */
819 PySys_SetArgvEx(argc-_PyOS_optind, argv+_PyOS_optind, 0);
820 } else {
821 /* Use config settings to decide whether or not to update sys.path[0] */
822 PySys_SetArgv(argc-_PyOS_optind, argv+_PyOS_optind);
823 }
824
825 if ((Py_InspectFlag || (cmdline.command == NULL &&
826 cmdline.filename == NULL &&
827 cmdline.module == NULL)) &&
828 isatty(fileno(stdin)) &&
829 !Py_IsolatedFlag) {
830 PyObject *v;
831 v = PyImport_ImportModule("readline");
832 if (v == NULL)
833 PyErr_Clear();
834 else
835 Py_DECREF(v);
836 }
837
838 if (cmdline.command) {
839 sts = run_command(cmdline.command, &cf);
840 PyMem_RawFree(cmdline.command);
841 } else if (cmdline.module) {
842 sts = (RunModule(cmdline.module, 1) != 0);
843 }
844 else {
845
846 if (cmdline.filename == NULL && stdin_is_interactive) {
847 Py_InspectFlag = 0; /* do exit on SystemExit */
848 RunStartupFile(&cf);
849 RunInteractiveHook();
850 }
851 /* XXX */
852
853 sts = -1; /* keep track of whether we've already run __main__ */
854
855 if (main_importer_path != NULL) {
856 sts = RunMainFromImporter(main_importer_path);
857 }
858
859 if (sts==-1 && cmdline.filename != NULL) {
860 fp = _Py_wfopen(cmdline.filename, L"r");
861 if (fp == NULL) {
862 char *cfilename_buffer;
863 const char *cfilename;
864 int err = errno;
865 cfilename_buffer = Py_EncodeLocale(cmdline.filename, NULL);
866 if (cfilename_buffer != NULL)
867 cfilename = cfilename_buffer;
868 else
869 cfilename = "<unprintable file name>";
870 fprintf(stderr, "%ls: can't open file '%s': [Errno %d] %s\n",
871 argv[0], cfilename, err, strerror(err));
872 if (cfilename_buffer)
873 PyMem_Free(cfilename_buffer);
874 return 2;
875 }
876 else if (cmdline.skip_first_line) {
877 int ch;
878 /* Push back first newline so line numbers
879 remain the same */
880 while ((ch = getc(fp)) != EOF) {
881 if (ch == '\n') {
882 (void)ungetc(ch, fp);
883 break;
884 }
885 }
886 }
887 {
888 struct _Py_stat_struct sb;
889 if (_Py_fstat_noraise(fileno(fp), &sb) == 0 &&
890 S_ISDIR(sb.st_mode)) {
891 fprintf(stderr,
892 "%ls: '%ls' is a directory, cannot continue\n",
893 argv[0], cmdline.filename);
894 fclose(fp);
895 return 1;
896 }
897 }
898 }
899
900 if (sts == -1)
901 sts = run_file(fp, cmdline.filename, &cf);
902 }
903
904 /* Check this environment variable at the end, to give programs the
905 * opportunity to set it from Python.
906 */
907 if (!Py_InspectFlag &&
908 (p = Py_GETENV("PYTHONINSPECT")) && *p != '\0')
909 {
910 Py_InspectFlag = 1;
911 }
912
913 if (Py_InspectFlag && stdin_is_interactive &&
914 (cmdline.filename != NULL || cmdline.command != NULL || cmdline.module != NULL)) {
915 Py_InspectFlag = 0;
916 RunInteractiveHook();
917 /* XXX */
918 sts = PyRun_AnyFileFlags(stdin, "<stdin>", &cf) != 0;
919 }
920
921 if (Py_FinalizeEx() < 0) {
922 /* Value unlikely to be confused with a non-error exit status or
923 other special meaning */
924 sts = 120;
925 }
926
927#ifdef __INSURE__
928 /* Insure++ is a memory analysis tool that aids in discovering
929 * memory leaks and other memory problems. On Python exit, the
930 * interned string dictionaries are flagged as being in use at exit
931 * (which it is). Under normal circumstances, this is fine because
932 * the memory will be automatically reclaimed by the system. Under
933 * memory debugging, it's a huge source of useless noise, so we
934 * trade off slower shutdown for less distraction in the memory
935 * reports. -baw
936 */
937 _Py_ReleaseInternedUnicodeStrings();
938#endif /* __INSURE__ */
939
940 return sts;
941}
