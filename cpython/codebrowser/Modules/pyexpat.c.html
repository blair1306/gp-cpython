106/* Build and set an Expat exception, including positioning
107 * information. Always returns NULL.
108 */
109static PyObject *
110set_error(xmlparseobject *self, enum XML_Error code)
111{
112 PyObject *err;
113 PyObject *buffer;
114 XML_Parser parser = self->itself;
115 int lineno = XML_GetErrorLineNumber(parser);
116 int column = XML_GetErrorColumnNumber(parser);
117
118 buffer = PyUnicode_FromFormat("%s: line %i, column %i",
119 XML_ErrorString(code), lineno, column);
120 if (buffer == NULL)
121 return NULL;
122 err = PyObject_CallFunctionObjArgs(ErrorObject, buffer, NULL);
123 Py_DECREF(buffer);
124 if ( err != NULL
125 && set_error_attr(err, "code", code)
126 && set_error_attr(err, "offset", column)
127 && set_error_attr(err, "lineno", lineno)) {
128 PyErr_SetObject(ErrorObject, err);
129 }
130 Py_XDECREF(err);
131 return NULL;
132}

232static PyObject*
233string_intern(xmlparseobject *self, const char* str)
234{
235 PyObject *result = conv_string_to_unicode(str);
236 PyObject *value;
237 /* result can be NULL if the unicode conversion failed. */
238 if (!result)
239 return result;
240 if (!self->intern)
241 return result;
242 value = PyDict_GetItem(self->intern, result);
243 if (!value) {
244 if (PyDict_SetItem(self->intern, result, result) == 0)
245 return result;
246 else
247 return NULL;
248 }
249 Py_INCREF(value);
250 Py_DECREF(result);
251 return value;
252}

254/* Return 0 on success, -1 on exception.
255 * flag_error() will be called before return if needed.
256 */
257static int
258call_character_handler(xmlparseobject *self, const XML_Char *buffer, int len)
259{
260 PyObject *args;
261 PyObject *temp;
262
263 if (!have_handler(self, CharacterData))
264 return -1;
265
266 args = PyTuple_New(1);
267 if (args == NULL)
268 return -1;
269 temp = (conv_string_len_to_unicode(buffer, len));
270 if (temp == NULL) {
271 Py_DECREF(args);
272 flag_error(self);
273 XML_SetCharacterDataHandler(self->itself,
274 noop_character_data_handler);
275 return -1;
276 }
277 PyTuple_SET_ITEM(args, 0, temp);
278 /* temp is now a borrowed reference; consider it unused. */
279 self->in_callback = 1;
280 temp = call_with_frame("CharacterData", __LINE__,
281 self->handlers[CharacterData], args, self);
282 /* temp is an owned reference again, or NULL */
283 self->in_callback = 0;
284 Py_DECREF(args);
285 if (temp == NULL) {
286 flag_error(self);
287 XML_SetCharacterDataHandler(self->itself,
288 noop_character_data_handler);
289 return -1;
290 }
291 Py_DECREF(temp);
292 return 0;
293}

306static void
307my_CharacterDataHandler(void *userData, const XML_Char *data, int len)
308{
309 xmlparseobject *self = (xmlparseobject *) userData;
310
311 if (PyErr_Occurred())
312 return;
313
314 if (self->buffer == NULL)
315 call_character_handler(self, data, len);
316 else {
317 if ((self->buffer_used + len) > self->buffer_size) {
318 if (flush_character_buffer(self) < 0)
319 return;
320 /* handler might have changed; drop the rest on the floor
321 * if there isn't a handler anymore
322 */
323 if (!have_handler(self, CharacterData))
324 return;
325 }
326 if (len > self->buffer_size) {
327 call_character_handler(self, data, len);
328 self->buffer_used = 0;
329 }
330 else {
331 memcpy(self->buffer + self->buffer_used,
332 data, len * sizeof(XML_Char));
333 self->buffer_used += len;
334 }
335 }

338static void
339my_StartElementHandler(void *userData,
340 const XML_Char *name, const XML_Char *atts[])
341{
342 xmlparseobject *self = (xmlparseobject *)userData;
343
344 if (have_handler(self, StartElement)) {
345 PyObject *container, *rv, *args;
346 int i, max;
347
348 if (PyErr_Occurred())
349 return;
350
351 if (flush_character_buffer(self) < 0)
352 return;
353 /* Set max to the number of slots filled in atts[]; max/2 is
354 * the number of attributes we need to process.
355 */
356 if (self->specified_attributes) {
357 max = XML_GetSpecifiedAttributeCount(self->itself);
358 }
359 else {
360 max = 0;
361 while (atts[max] != NULL)
362 max += 2;
363 }
364 /* Build the container. */
365 if (self->ordered_attributes)
366 container = PyList_New(max);
367 else
368 container = PyDict_New();
369 if (container == NULL) {
370 flag_error(self);
371 return;
372 }
373 for (i = 0; i < max; i += 2) {
374 PyObject *n = string_intern(self, (XML_Char *) atts[i]);
375 PyObject *v;
376 if (n == NULL) {
377 flag_error(self);
378 Py_DECREF(container);
379 return;
380 }
381 v = conv_string_to_unicode((XML_Char *) atts[i+1]);
382 if (v == NULL) {
383 flag_error(self);
384 Py_DECREF(container);
385 Py_DECREF(n);
386 return;
387 }
388 if (self->ordered_attributes) {
389 PyList_SET_ITEM(container, i, n);
390 PyList_SET_ITEM(container, i+1, v);
391 }
392 else if (PyDict_SetItem(container, n, v)) {
393 flag_error(self);
394 Py_DECREF(n);
395 Py_DECREF(v);
396 return;
397 }
398 else {
399 Py_DECREF(n);
400 Py_DECREF(v);
401 }
402 }
403 args = string_intern(self, name);
404 if (args != NULL)
405 args = Py_BuildValue("(NN)", args, container);
406 if (args == NULL) {
407 Py_DECREF(container);
408 return;
409 }
410 /* Container is now a borrowed reference; ignore it. */
411 self->in_callback = 1;
412 rv = call_with_frame("StartElement", __LINE__,
413 self->handlers[StartElement], args, self);
414 self->in_callback = 0;
415 Py_DECREF(args);
416 if (rv == NULL) {
417 flag_error(self);
418 return;
419 }
420 Py_DECREF(rv);
421 }
422}

512static PyObject *
513conv_content_model(XML_Content * const model,
514 PyObject *(*conv_string)(const XML_Char *))
515{
516 PyObject *result = NULL;
517 PyObject *children = PyTuple_New(model->numchildren);
518 int i;
519
520 if (children != NULL) {
521 assert(model->numchildren < INT_MAX);
522 for (i = 0; i < (int)model->numchildren; ++i) {
523 PyObject *child = conv_content_model(&model->children[i],
524 conv_string);
525 if (child == NULL) {
526 Py_XDECREF(children);
527 return NULL;
528 }
529 PyTuple_SET_ITEM(children, i, child);
530 }
531 result = Py_BuildValue("(iiO&N)",
532 model->type, model->quant,
533 conv_string,model->name, children);
534 }
535 return result;

704/*[clinic input]
705pyexpat.xmlparser.Parse
706
707 data: object
708 isfinal: bool(accept={int}) = False
709 /
710
711Parse XML data.
712
713`isfinal' should be true at end of input.
714[clinic start generated code]*/
715
716static PyObject *
717pyexpat_xmlparser_Parse_impl(xmlparseobject *self, PyObject *data,
718 int isfinal)
719/*[clinic end generated code: output=f4db843dd1f4ed4b input=eb616027bfa9847f]*/
720{
721 const char *s;
722 Py_ssize_t slen;
723 Py_buffer view;
724 int rc;
725
726 if (PyUnicode_Check(data)) {
727 view.buf = NULL;
728 s = PyUnicode_AsUTF8AndSize(data, &slen);
729 if (s == NULL)
730 return NULL;
731 /* Explicitly set UTF-8 encoding. Return code ignored. */
732 (void)XML_SetEncoding(self->itself, "utf-8");
733 }
734 else {
735 if (PyObject_GetBuffer(data, &view, PyBUF_SIMPLE) < 0)
736 return NULL;
737 s = view.buf;
738 slen = view.len;
739 }
740
741 while (slen > MAX_CHUNK_SIZE) {
742 rc = XML_Parse(self->itself, s, MAX_CHUNK_SIZE, 0);
743 if (!rc)
744 goto done;
745 s += MAX_CHUNK_SIZE;
746 slen -= MAX_CHUNK_SIZE;
747 }
748 Py_BUILD_ASSERT(MAX_CHUNK_SIZE <= INT_MAX);
749 assert(slen <= INT_MAX);
750 rc = XML_Parse(self->itself, s, (int)slen, isfinal);
751
752done:
753 if (view.buf != NULL)
754 PyBuffer_Release(&view);
755 return get_parse_result(self, rc);
756}

762static int
763readinst(char *buf, int buf_size, PyObject *meth)
764{
765 PyObject *str;
766 Py_ssize_t len;
767 const char *ptr;
768
769 str = PyObject_CallFunction(meth, "n", buf_size);
770 if (str == NULL)
771 goto error;
772
773 if (PyBytes_Check(str))
774 ptr = PyBytes_AS_STRING(str);
775 else if (PyByteArray_Check(str))
776 ptr = PyByteArray_AS_STRING(str);
777 else {
778 PyErr_Format(PyExc_TypeError,
779 "read() did not return a bytes object (type=%.400s)",
780 Py_TYPE(str)->tp_name);
781 goto error;
782 }
783 len = Py_SIZE(str);
784 if (len > buf_size) {
785 PyErr_Format(PyExc_ValueError,
786 "read() returned too much data: "
787 "%i bytes requested, %zd returned",
788 buf_size, len);
789 goto error;
790 }
791 memcpy(buf, ptr, len);
792 Py_DECREF(str);
793 /* len <= buf_size <= INT_MAX */
794 return (int)len;
795
796error:
797 Py_XDECREF(str);
798 return -1;
799}
