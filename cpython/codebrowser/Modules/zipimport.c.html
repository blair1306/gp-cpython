21/* zip_searchorder defines how we search for a module in the Zip
22 archive: we first search for a package __init__, then for
23 non-package .pyc, and .py entries. The .pyc entries
24 are swapped by initzipimport() if we run in optimized mode. Also,
25 '/' is replaced by SEP there. */
26static struct st_zip_searchorder zip_searchorder[] = {
27 {"/__init__.pyc", IS_PACKAGE | IS_BYTECODE},
28 {"/__init__.py", IS_PACKAGE | IS_SOURCE},
29 {".pyc", IS_BYTECODE},
30 {".py", IS_SOURCE},
31 {"", 0}
32};

97static int
98zipimport_zipimporter___init___impl(ZipImporter *self, PyObject *path)
99/*[clinic end generated code: output=141558fefdb46dc8 input=92b9ebeed1f6a704]*/
100{
101 PyObject *files, *tmp;
102 PyObject *filename = NULL;
103 Py_ssize_t len, flen;
104
105 if (PyUnicode_READY(path) == -1)
106 return -1;
107
108 len = PyUnicode_GET_LENGTH(path);
109 if (len == 0) {
110 PyErr_SetString(ZipImportError, "archive path is empty");
111 goto error;
112 }
113
114#ifdef ALTSEP
115 tmp = _PyObject_CallMethodId(path, &PyId_replace, "CC", ALTSEP, SEP);
116 if (!tmp)
117 goto error;
118 Py_DECREF(path);
119 path = tmp;
120#endif
121
122 filename = path;
123 Py_INCREF(filename);
124 flen = len;
125 for (;;) {
126 struct stat statbuf;
127 int rv;
128
129 rv = _Py_stat(filename, &statbuf);
130 if (rv == -2)
131 goto error;
132 if (rv == 0) {
133 /* it exists */
134 if (!S_ISREG(statbuf.st_mode))
135 /* it's a not file */
136 Py_CLEAR(filename);
137 break;
138 }
139 Py_CLEAR(filename);
140 /* back up one path element */
141 flen = PyUnicode_FindChar(path, SEP, 0, flen, -1);
142 if (flen == -1)
143 break;
144 filename = PyUnicode_Substring(path, 0, flen);
145 if (filename == NULL)
146 goto error;
147 }
148 if (filename == NULL) {
149 PyErr_SetString(ZipImportError, "not a Zip file");
150 goto error;
151 }
152
153 if (PyUnicode_READY(filename) < 0)
154 goto error;
155
156 files = PyDict_GetItem(zip_directory_cache, filename);
157 if (files == NULL) {
158 files = read_directory(filename);
159 if (files == NULL)
160 goto error;
161 if (PyDict_SetItem(zip_directory_cache, filename, files) != 0)
162 goto error;
163 }
164 else
165 Py_INCREF(files);
166 self->files = files;
167
168 /* Transfer reference */
169 self->archive = filename;
170 filename = NULL;
171
172 /* Check if there is a prefix directory following the filename. */
173 if (flen != len) {
174 tmp = PyUnicode_Substring(path, flen+1,
175 PyUnicode_GET_LENGTH(path));
176 if (tmp == NULL)
177 goto error;
178 self->prefix = tmp;
179 if (PyUnicode_READ_CHAR(path, len-1) != SEP) {
180 /* add trailing SEP */
181 tmp = PyUnicode_FromFormat("%U%c", self->prefix, SEP);
182 if (tmp == NULL)
183 goto error;
184 Py_SETREF(self->prefix, tmp);
185 }
186 }
187 else
188 self->prefix = PyUnicode_New(0, 0);
189 Py_DECREF(path);
190 return 0;
191
192error:
193 Py_DECREF(path);
194 Py_XDECREF(filename);
195 return -1;
196}

246/* Given a (sub)modulename, write the potential file path in the
247 archive (without extension) to the path buffer. Return the
248 length of the resulting string.
249
250 return self.prefix + name.replace('.', os.sep) */
251static PyObject*
252make_filename(PyObject *prefix, PyObject *name)
253{
254 PyObject *pathobj;
255 Py_UCS4 *p, *buf;
256 Py_ssize_t len;
257
258 len = PyUnicode_GET_LENGTH(prefix) + PyUnicode_GET_LENGTH(name) + 1;
259 p = buf = PyMem_New(Py_UCS4, len);
260 if (buf == NULL) {
261 PyErr_NoMemory();
262 return NULL;
263 }
264
265 if (!PyUnicode_AsUCS4(prefix, p, len, 0)) {
266 PyMem_Free(buf);
267 return NULL;
268 }
269 p += PyUnicode_GET_LENGTH(prefix);
270 len -= PyUnicode_GET_LENGTH(prefix);
271 if (!PyUnicode_AsUCS4(name, p, len, 1)) {
272 PyMem_Free(buf);
273 return NULL;
274 }
275 for (; *p; p++) {
276 if (*p == '.')
277 *p = SEP;
278 }
279 pathobj = PyUnicode_FromKindAndData(PyUnicode_4BYTE_KIND,
280 buf, p-buf);
281 PyMem_Free(buf);
282 return pathobj;
283}

292/* Does this path represent a directory?
293 on error, return < 0
294 if not a dir, return 0
295 if a dir, return 1
296*/
297static int
298check_is_directory(ZipImporter *self, PyObject* prefix, PyObject *path)
299{
300 PyObject *dirpath;
301 int res;
302
303 /* See if this is a "directory". If so, it's eligible to be part
304 of a namespace package. We test by seeing if the name, with an
305 appended path separator, exists. */
306 dirpath = PyUnicode_FromFormat("%U%U%c", prefix, path, SEP);
307 if (dirpath == NULL)
308 return -1;
309 /* If dirpath is present in self->files, we have a directory. */
310 res = PyDict_Contains(self->files, dirpath);
311 Py_DECREF(dirpath);
312 return res;
313}

315/* Return some information about a module. */
316static enum zi_module_info
317get_module_info(ZipImporter *self, PyObject *fullname)
318{
319 PyObject *subname;
320 PyObject *path, *fullpath, *item;
321 struct st_zip_searchorder *zso;
322
323 subname = get_subname(fullname);
324 if (subname == NULL)
325 return MI_ERROR;
326
327 path = make_filename(self->prefix, subname);
328 Py_DECREF(subname);
329 if (path == NULL)
330 return MI_ERROR;
331
332 for (zso = zip_searchorder; *zso->suffix; zso++) {
333 fullpath = PyUnicode_FromFormat("%U%s", path, zso->suffix);
334 if (fullpath == NULL) {
335 Py_DECREF(path);
336 return MI_ERROR;
337 }
338 item = PyDict_GetItem(self->files, fullpath);
339 Py_DECREF(fullpath);
340 if (item != NULL) {
341 Py_DECREF(path);
342 if (zso->type & IS_PACKAGE)
343 return MI_PACKAGE;
344 else
345 return MI_MODULE;
346 }
347 }
348 Py_DECREF(path);
349 return MI_NOT_FOUND;
350}

352typedef enum {
353 FL_ERROR = -1, /* error */
354 FL_NOT_FOUND, /* no loader or namespace portions found */
355 FL_MODULE_FOUND, /* module/package found */
356 FL_NS_FOUND /* namespace portion found: */
357 /* *namespace_portion will point to the name */
358} find_loader_result;
359
360/* The guts of "find_loader" and "find_module".
361*/
362static find_loader_result
363find_loader(ZipImporter *self, PyObject *fullname, PyObject **namespace_portion)
364{
365 enum zi_module_info mi;
366
367 *namespace_portion = NULL;
368
369 mi = get_module_info(self, fullname);
370 if (mi == MI_ERROR)
371 return FL_ERROR;
372 if (mi == MI_NOT_FOUND) {
373 /* Not a module or regular package. See if this is a directory, and
374 therefore possibly a portion of a namespace package. */
375 find_loader_result result = FL_NOT_FOUND;
376 PyObject *subname;
377 int is_dir;
378
379 /* We're only interested in the last path component of fullname;
380 earlier components are recorded in self->prefix. */
381 subname = get_subname(fullname);
382 if (subname == NULL) {
383 return FL_ERROR;
384 }
385
386 is_dir = check_is_directory(self, self->prefix, subname);
387 if (is_dir < 0)
388 result = FL_ERROR;
389 else if (is_dir) {
390 /* This is possibly a portion of a namespace
391 package. Return the string representing its path,
392 without a trailing separator. */
393 *namespace_portion = PyUnicode_FromFormat("%U%c%U%U",
394 self->archive, SEP,
395 self->prefix, subname);
396 if (*namespace_portion == NULL)
397 result = FL_ERROR;
398 else
399 result = FL_NS_FOUND;
400 }
401 Py_DECREF(subname);
402 return result;
403 }
404 /* This is a module or package. */
405 return FL_MODULE_FOUND;
406}

839/* Given a buffer, return the unsigned int that is represented by the first
840 4 bytes, encoded as little endian. This partially reimplements
841 marshal.c:r_long() */
842static unsigned int
843get_uint32(const unsigned char *buf)
844{
845 unsigned int x;
846 x = buf[0];
847 x |= (unsigned int)buf[1] << 8;
848 x |= (unsigned int)buf[2] << 16;
849 x |= (unsigned int)buf[3] << 24;
850 return x;
851}
852
853/* Given a buffer, return the unsigned int that is represented by the first
854 2 bytes, encoded as little endian. This partially reimplements
855 marshal.c:r_short() */
856static unsigned short
857get_uint16(const unsigned char *buf)
858{
859 unsigned short x;
860 x = buf[0];
861 x |= (unsigned short)buf[1] << 8;
862 return x;
863}

876/*
877 read_directory(archive) -> files dict (new reference)
878
879 Given a path to a Zip archive, build a dict, mapping file names
880 (local to the archive, using SEP as a separator) to toc entries.
881
882 A toc_entry is a tuple:
883
884 (__file__, # value to use for __file__, available for all files,
885 # encoded to the filesystem encoding
886 compress, # compression kind; 0 for uncompressed
887 data_size, # size of compressed data on disk
888 file_size, # size of decompressed data
889 file_offset, # offset of file header from start of archive
890 time, # mod time of file (in dos format)
891 date, # mod data of file (in dos format)
892 crc, # crc checksum of the data
893 )
894
895 Directories can be recognized by the trailing SEP in the name,
896 data_size and file_offset are 0.
897*/
898static PyObject *
899read_directory(PyObject *archive)
900{
901 PyObject *files = NULL;
902 FILE *fp;
903 unsigned short flags, compress, time, date, name_size;
904 unsigned int crc, data_size, file_size, header_size, header_offset;
905 unsigned long file_offset, header_position;
906 unsigned long arc_offset; /* Absolute offset to start of the zip-archive. */
907 unsigned int count, i;
908 unsigned char buffer[46];
909 char name[MAXPATHLEN + 5];
910 PyObject *nameobj = NULL;
911 PyObject *path;
912 const char *charset;
913 int bootstrap;
914 const char *errmsg = NULL;
915
916 fp = _Py_fopen_obj(archive, "rb");
917 if (fp == NULL) {
918 if (PyErr_ExceptionMatches(PyExc_OSError)) {
919 _PyErr_FormatFromCause(ZipImportError,
920 "can't open Zip file: %R", archive);
921 }
922 return NULL;
923 }
924
925 if (fseek(fp, -22, SEEK_END) == -1) {
926 goto file_error;
927 }
928 header_position = (unsigned long)ftell(fp);
929 if (header_position == (unsigned long)-1) {
930 goto file_error;
931 }
932 assert(header_position <= (unsigned long)LONG_MAX);
933 if (fread(buffer, 1, 22, fp) != 22) {
934 goto file_error;
935 }
936 if (get_uint32(buffer) != 0x06054B50u) {
937 /* Bad: End of Central Dir signature */
938 errmsg = "not a Zip file";
939 goto invalid_header;
940 }
941
942 header_size = get_uint32(buffer + 12);
943 header_offset = get_uint32(buffer + 16);
944 if (header_position < header_size) {
945 errmsg = "bad central directory size";
946 goto invalid_header;
947 }
948 if (header_position < header_offset) {
949 errmsg = "bad central directory offset";
950 goto invalid_header;
951 }
952 if (header_position - header_size < header_offset) {
953 errmsg = "bad central directory size or offset";
954 goto invalid_header;
955 }
956 header_position -= header_size;
957 arc_offset = header_position - header_offset;
958
959 files = PyDict_New();
960 if (files == NULL) {
961 goto error;
962 }
963 /* Start of Central Directory */
964 count = 0;
965 if (fseek(fp, (long)header_position, 0) == -1) {
966 goto file_error;
967 }
968 for (;;) {
969 PyObject *t;
970 size_t n;
971 int err;
972
973 n = fread(buffer, 1, 46, fp);
974 if (n < 4) {
975 goto eof_error;
976 }
977 /* Start of file header */
978 if (get_uint32(buffer) != 0x02014B50u) {
979 break; /* Bad: Central Dir File Header */
980 }
981 if (n != 46) {
982 goto eof_error;
983 }
984 flags = get_uint16(buffer + 8);
985 compress = get_uint16(buffer + 10);
986 time = get_uint16(buffer + 12);
987 date = get_uint16(buffer + 14);
988 crc = get_uint32(buffer + 16);
989 data_size = get_uint32(buffer + 20);
990 file_size = get_uint32(buffer + 24);
991 name_size = get_uint16(buffer + 28);
992 header_size = (unsigned int)name_size +
993 get_uint16(buffer + 30) /* extra field */ +
994 get_uint16(buffer + 32) /* comment */;
995
996 file_offset = get_uint32(buffer + 42);
997 if (file_offset > header_offset) {
998 errmsg = "bad local header offset";
999 goto invalid_header;
1000 }
1001 file_offset += arc_offset;
1002
1003 if (name_size > MAXPATHLEN) {
1004 name_size = MAXPATHLEN;
1005 }
1006 if (fread(name, 1, name_size, fp) != name_size) {
1007 goto file_error;
1008 }
1009 name[name_size] = '\0'; /* Add terminating null byte */
1010#if SEP != '/'
1011 for (i = 0; i < name_size; i++) {
1012 if (name[i] == '/') {
1013 name[i] = SEP;
1014 }
1015 }
1016#endif
1017 /* Skip the rest of the header.
1018 * On Windows, calling fseek to skip over the fields we don't use is
1019 * slower than reading the data because fseek flushes stdio's
1020 * internal buffers. See issue #8745. */
1021 assert(header_size <= 3*0xFFFFu);
1022 for (i = name_size; i < header_size; i++) {
1023 if (getc(fp) == EOF) {
1024 goto file_error;
1025 }
1026 }
1027
1028 bootstrap = 0;
1029 if (flags & 0x0800) {
1030 charset = "utf-8";
1031 }
1032 else if (!PyThreadState_GET()->interp->codecs_initialized) {
1033 /* During bootstrap, we may need to load the encodings
1034 package from a ZIP file. But the cp437 encoding is implemented
1035 in Python in the encodings package.
1036
1037 Break out of this dependency by assuming that the path to
1038 the encodings module is ASCII-only. */
1039 charset = "ascii";
1040 bootstrap = 1;
1041 }
1042 else {
1043 charset = "cp437";
1044 }
1045 nameobj = PyUnicode_Decode(name, name_size, charset, NULL);
1046 if (nameobj == NULL) {
1047 if (bootstrap) {
1048 PyErr_Format(PyExc_NotImplementedError,
1049 "bootstrap issue: python%i%i.zip contains non-ASCII "
1050 "filenames without the unicode flag",
1051 PY_MAJOR_VERSION, PY_MINOR_VERSION);
1052 }
1053 goto error;
1054 }
1055 if (PyUnicode_READY(nameobj) == -1) {
1056 goto error;
1057 }
1058 path = PyUnicode_FromFormat("%U%c%U", archive, SEP, nameobj);
1059 if (path == NULL) {
1060 goto error;
1061 }
1062 t = Py_BuildValue("NHIIkHHI", path, compress, data_size,
1063 file_size, file_offset, time, date, crc);
1064 if (t == NULL) {
1065 goto error;
1066 }
1067 err = PyDict_SetItem(files, nameobj, t);
1068 Py_CLEAR(nameobj);
1069 Py_DECREF(t);
1070 if (err != 0) {
1071 goto error;
1072 }
1073 count++;
1074 }
1075 fclose(fp);
1076 if (Py_VerboseFlag) {
1077 PySys_FormatStderr("# zipimport: found %u names in %R\n",
1078 count, archive);
1079 }
1080 return files;
1081
1082eof_error:
1083 set_file_error(archive, !ferror(fp));
1084 goto error;
1085
1086file_error:
1087 PyErr_Format(ZipImportError, "can't read Zip file: %R", archive);
1088 goto error;
1089
1090invalid_header:
1091 assert(errmsg != NULL);
1092 PyErr_Format(ZipImportError, "%s: %R", errmsg, archive);
1093 goto error;
1094
1095error:
1096 fclose(fp);
1097 Py_XDECREF(files);
1098 Py_XDECREF(nameobj);
1099 return NULL;
1100}

1102/* Return the zlib.decompress function object, or NULL if zlib couldn't
1103 be imported. The function is cached when found, so subsequent calls
1104 don't import zlib again. */
1105static PyObject *
1106get_decompress_func(void)
1107{
1108 static int importing_zlib = 0;
1109 PyObject *zlib;
1110 PyObject *decompress;
1111 _Py_IDENTIFIER(decompress);
1112
1113 if (importing_zlib != 0)
1114 /* Someone has a zlib.pyc in their Zip file;
1115 let's avoid a stack overflow. */
1116 return NULL;
1117 importing_zlib = 1;
1118 zlib = PyImport_ImportModuleNoBlock("zlib");
1119 importing_zlib = 0;
1120 if (zlib != NULL) {
1121 decompress = _PyObject_GetAttrId(zlib,
1122 &PyId_decompress);
1123 Py_DECREF(zlib);
1124 }
1125 else {
1126 PyErr_Clear();
1127 decompress = NULL;
1128 }
1129 if (Py_VerboseFlag)
1130 PySys_WriteStderr("# zipimport: zlib %s\n",
1131 zlib != NULL ? "available": "UNAVAILABLE");
1132 return decompress;
1133}

1135/* Given a path to a Zip file and a toc_entry, return the (uncompressed)
1136 data as a new reference. */
1137static PyObject *
1138get_data(PyObject *archive, PyObject *toc_entry)
1139{
1140 PyObject *raw_data = NULL, *data, *decompress;
1141 char *buf;
1142 FILE *fp;
1143 PyObject *datapath;
1144 unsigned short compress, time, date;
1145 unsigned int crc;
1146 Py_ssize_t data_size, file_size, bytes_size;
1147 long file_offset, header_size;
1148 unsigned char buffer[30];
1149 const char *errmsg = NULL;
1150
1151 if (!PyArg_ParseTuple(toc_entry, "OHnnlHHI", &datapath, &compress,
1152 &data_size, &file_size, &file_offset, &time,
1153 &date, &crc)) {
1154 return NULL;
1155 }
1156 if (data_size < 0) {
1157 PyErr_Format(ZipImportError, "negative data size");
1158 return NULL;
1159 }
1160
1161 fp = _Py_fopen_obj(archive, "rb");
1162 if (!fp) {
1163 return NULL;
1164 }
1165 /* Check to make sure the local file header is correct */
1166 if (fseek(fp, file_offset, 0) == -1) {
1167 goto file_error;
1168 }
1169 if (fread(buffer, 1, 30, fp) != 30) {
1170 goto eof_error;
1171 }
1172 if (get_uint32(buffer) != 0x04034B50u) {
1173 /* Bad: Local File Header */
1174 errmsg = "bad local file header";
1175 goto invalid_header;
1176 }
1177
1178 header_size = (unsigned int)30 +
1179 get_uint16(buffer + 26) /* file name */ +
1180 get_uint16(buffer + 28) /* extra field */;
1181 if (file_offset > LONG_MAX - header_size) {
1182 errmsg = "bad local file header size";
1183 goto invalid_header;
1184 }
1185 file_offset += header_size; /* Start of file data */
1186
1187 if (data_size > LONG_MAX - 1) {
1188 fclose(fp);
1189 PyErr_NoMemory();
1190 return NULL;
1191 }
1192 bytes_size = compress == 0 ? data_size : data_size + 1;
1193 if (bytes_size == 0) {
1194 bytes_size++;
1195 }
1196 raw_data = PyBytes_FromStringAndSize((char *)NULL, bytes_size);
1197 if (raw_data == NULL) {
1198 goto error;
1199 }
1200 buf = PyBytes_AsString(raw_data);
1201
1202 if (fseek(fp, file_offset, 0) == -1) {
1203 goto file_error;
1204 }
1205 if (fread(buf, 1, data_size, fp) != (size_t)data_size) {
1206 PyErr_SetString(PyExc_OSError,
1207 "zipimport: can't read data");
1208 goto error;
1209 }
1210
1211 fclose(fp);
1212 fp = NULL;
1213
1214 if (compress != 0) {
1215 buf[data_size] = 'Z'; /* saw this in zipfile.py */
1216 data_size++;
1217 }
1218 buf[data_size] = '\0';
1219
1220 if (compress == 0) { /* data is not compressed */
1221 data = PyBytes_FromStringAndSize(buf, data_size);
1222 Py_DECREF(raw_data);
1223 return data;
1224 }
1225
1226 /* Decompress with zlib */
1227 decompress = get_decompress_func();
1228 if (decompress == NULL) {
1229 PyErr_SetString(ZipImportError,
1230 "can't decompress data; "
1231 "zlib not available");
1232 goto error;
1233 }
1234 data = PyObject_CallFunction(decompress, "Oi", raw_data, -15);
1235 Py_DECREF(decompress);
1236 Py_DECREF(raw_data);
1237 return data;
1238
1239eof_error:
1240 set_file_error(archive, !ferror(fp));
1241 goto error;
1242
1243file_error:
1244 PyErr_Format(ZipImportError, "can't read Zip file: %R", archive);
1245 goto error;
1246
1247invalid_header:
1248 assert(errmsg != NULL);
1249 PyErr_Format(ZipImportError, "%s: %R", errmsg, archive);
1250 goto error;
1251
1252error:
1253 if (fp != NULL) {
1254 fclose(fp);
1255 }
1256 Py_XDECREF(raw_data);
1257 return NULL;
1258}

1260/* Lenient date/time comparison function. The precision of the mtime
1261 in the archive is lower than the mtime stored in a .pyc: we
1262 must allow a difference of at most one second. */
1263static int
1264eq_mtime(time_t t1, time_t t2)
1265{
1266 time_t d = t1 - t2;
1267 if (d < 0)
1268 d = -d;
1269 /* dostime only stores even seconds, so be lenient */
1270 return d <= 1;
1271}

1273/* Given the contents of a .pyc file in a buffer, unmarshal the data
1274 and return the code object. Return None if it the magic word doesn't
1275 match (we do this instead of raising an exception as we fall back
1276 to .py if available and we don't want to mask other errors).
1277 Returns a new reference. */
1278static PyObject *
1279unmarshal_code(PyObject *pathname, PyObject *data, time_t mtime)
1280{
1281 PyObject *code;
1282 unsigned char *buf = (unsigned char *)PyBytes_AsString(data);
1283 Py_ssize_t size = PyBytes_Size(data);
1284
1285 if (size < 12) {
1286 PyErr_SetString(ZipImportError,
1287 "bad pyc data");
1288 return NULL;
1289 }
1290
1291 if (get_uint32(buf) != (unsigned int)PyImport_GetMagicNumber()) {
1292 if (Py_VerboseFlag) {
1293 PySys_FormatStderr("# %R has bad magic\n",
1294 pathname);
1295 }
1296 Py_RETURN_NONE; /* signal caller to try alternative */
1297 }
1298
1299 if (mtime != 0 && !eq_mtime(get_uint32(buf + 4), mtime)) {
1300 if (Py_VerboseFlag) {
1301 PySys_FormatStderr("# %R has bad mtime\n",
1302 pathname);
1303 }
1304 Py_RETURN_NONE; /* signal caller to try alternative */
1305 }
1306
1307 /* XXX the pyc's size field is ignored; timestamp collisions are probably
1308 unimportant with zip files. */
1309 code = PyMarshal_ReadObjectFromString((char *)buf + 12, size - 12);
1310 if (code == NULL) {
1311 return NULL;
1312 }
1313 if (!PyCode_Check(code)) {
1314 Py_DECREF(code);
1315 PyErr_Format(PyExc_TypeError,
1316 "compiled module %R is not a code object",
1317 pathname);
1318 return NULL;
1319 }
1320 return code;
1321}

1323/* Replace any occurrences of "\r\n?" in the input string with "\n".
1324 This converts DOS and Mac line endings to Unix line endings.
1325 Also append a trailing "\n" to be compatible with
1326 PyParser_SimpleParseFile(). Returns a new reference. */
1327static PyObject *
1328normalize_line_endings(PyObject *source)
1329{
1330 char *buf, *q, *p;
1331 PyObject *fixed_source;
1332 int len = 0;
1333
1334 p = PyBytes_AsString(source);
1335 if (p == NULL) {
1336 return PyBytes_FromStringAndSize("\n\0", 2);
1337 }
1338
1339 /* one char extra for trailing \n and one for terminating \0 */
1340 buf = (char *)PyMem_Malloc(PyBytes_Size(source) + 2);
1341 if (buf == NULL) {
1342 PyErr_SetString(PyExc_MemoryError,
1343 "zipimport: no memory to allocate "
1344 "source buffer");
1345 return NULL;
1346 }
1347 /* replace "\r\n?" by "\n" */
1348 for (q = buf; *p != '\0'; p++) {
1349 if (*p == '\r') {
1350 *q++ = '\n';
1351 if (*(p + 1) == '\n')
1352 p++;
1353 }
1354 else
1355 *q++ = *p;
1356 len++;
1357 }
1358 *q++ = '\n'; /* add trailing \n */
1359 *q = '\0';
1360 fixed_source = PyBytes_FromStringAndSize(buf, len + 2);
1361 PyMem_Free(buf);
1362 return fixed_source;
1363}

1365/* Given a string buffer containing Python source code, compile it
1366 and return a code object as a new reference. */
1367static PyObject *
1368compile_source(PyObject *pathname, PyObject *source)
1369{
1370 PyObject *code, *fixed_source;
1371
1372 fixed_source = normalize_line_endings(source);
1373 if (fixed_source == NULL) {
1374 return NULL;
1375 }
1376
1377 code = Py_CompileStringObject(PyBytes_AsString(fixed_source),
1378 pathname, Py_file_input, NULL, -1);
1379
1380 Py_DECREF(fixed_source);
1381 return code;
1382}

1384/* Convert the date/time values found in the Zip archive to a value
1385 that's compatible with the time stamp stored in .pyc files. */
1386static time_t
1387parse_dostime(int dostime, int dosdate)
1388{
1389 struct tm stm;
1390
1391 memset((void *) &stm, '\0', sizeof(stm));
1392
1393 stm.tm_sec = (dostime & 0x1f) * 2;
1394 stm.tm_min = (dostime >> 5) & 0x3f;
1395 stm.tm_hour = (dostime >> 11) & 0x1f;
1396 stm.tm_mday = dosdate & 0x1f;
1397 stm.tm_mon = ((dosdate >> 5) & 0x0f) - 1;
1398 stm.tm_year = ((dosdate >> 9) & 0x7f) + 80;
1399 stm.tm_isdst = -1; /* wday/yday is ignored */
1400
1401 return mktime(&stm);
1402}

1404/* Given a path to a .pyc file in the archive, return the
1405 modification time of the matching .py file, or 0 if no source
1406 is available. */
1407static time_t
1408get_mtime_of_source(ZipImporter *self, PyObject *path)
1409{
1410 PyObject *toc_entry, *stripped;
1411 time_t mtime;
1412
1413 /* strip 'c' from *.pyc */
1414 if (PyUnicode_READY(path) == -1)
1415 return (time_t)-1;
1416 stripped = PyUnicode_FromKindAndData(PyUnicode_KIND(path),
1417 PyUnicode_DATA(path),
1418 PyUnicode_GET_LENGTH(path) - 1);
1419 if (stripped == NULL)
1420 return (time_t)-1;
1421
1422 toc_entry = PyDict_GetItem(self->files, stripped);
1423 Py_DECREF(stripped);
1424 if (toc_entry != NULL && PyTuple_Check(toc_entry) &&
1425 PyTuple_Size(toc_entry) == 8) {
1426 /* fetch the time stamp of the .py file for comparison
1427 with an embedded pyc time stamp */
1428 int time, date;
1429 time = PyLong_AsLong(PyTuple_GetItem(toc_entry, 5));
1430 date = PyLong_AsLong(PyTuple_GetItem(toc_entry, 6));
1431 mtime = parse_dostime(time, date);
1432 } else
1433 mtime = 0;
1434 return mtime;
1435}

1437/* Return the code object for the module named by 'fullname' from the

1439static PyObject *
1440get_code_from_data(ZipImporter *self, int ispackage, int isbytecode,

1458/* Get the code object associated with the module specified by
1459 'fullname'. */
1460static PyObject *
1461get_module_code(ZipImporter *self, PyObject *fullname,
1462 int *p_ispackage, PyObject **p_modpath)
1463{
1464 PyObject *code = NULL, *toc_entry, *subname;
1465 PyObject *path, *fullpath = NULL;
1466 struct st_zip_searchorder *zso;
1467
1468 subname = get_subname(fullname);
1469 if (subname == NULL)
1470 return NULL;
1471
1472 path = make_filename(self->prefix, subname);
1473 Py_DECREF(subname);
1474 if (path == NULL)
1475 return NULL;
1476
1477 for (zso = zip_searchorder; *zso->suffix; zso++) {
1478 code = NULL;
1479
1480 fullpath = PyUnicode_FromFormat("%U%s", path, zso->suffix);
1481 if (fullpath == NULL)
1482 goto exit;
1483
1484 if (Py_VerboseFlag > 1)
1485 PySys_FormatStderr("# trying %U%c%U\n",
1486 self->archive, (int)SEP, fullpath);
1487 toc_entry = PyDict_GetItem(self->files, fullpath);
1488 if (toc_entry != NULL) {
1489 time_t mtime = 0;
1490 int ispackage = zso->type & IS_PACKAGE;
1491 int isbytecode = zso->type & IS_BYTECODE;
1492
1493 if (isbytecode) {
1494 mtime = get_mtime_of_source(self, fullpath);
1495 if (mtime == (time_t)-1 && PyErr_Occurred()) {
1496 goto exit;
1497 }
1498 }
1499 Py_CLEAR(fullpath);
1500 if (p_ispackage != NULL)
1501 *p_ispackage = ispackage;
1502 code = get_code_from_data(self, ispackage,
1503 isbytecode, mtime,
1504 toc_entry);
1505 if (code == Py_None) {
1506 /* bad magic number or non-matching mtime
1507 in byte code, try next */
1508 Py_DECREF(code);
1509 continue;
1510 }
1511 if (code != NULL && p_modpath != NULL) {
1512 *p_modpath = PyTuple_GetItem(toc_entry, 0);
1513 Py_INCREF(*p_modpath);
1514 }
1515 goto exit;
1516 }
1517 else
1518 Py_CLEAR(fullpath);
1519 }
1520 PyErr_Format(ZipImportError, "can't find module %R", fullname);
1521exit:
1522 Py_DECREF(path);
1523 Py_XDECREF(fullpath);
1524 return code;
1525}
