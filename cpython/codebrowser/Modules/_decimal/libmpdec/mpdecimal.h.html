241enum {
242 MPD_ROUND_UP, /* round away from 0 */
243 MPD_ROUND_DOWN, /* round toward 0 (truncate) */
244 MPD_ROUND_CEILING, /* round toward +infinity */

249 MPD_ROUND_05UP, /* round zero or five away from 0 */
