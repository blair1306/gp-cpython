68static PyObject *
69internal_error_callback(const char *errors)
70{
71 if (errors == NULL || strcmp(errors, "strict") == 0)
72 return ERROR_STRICT;
73 else if (strcmp(errors, "ignore") == 0)
74 return ERROR_IGNORE;
75 else if (strcmp(errors, "replace") == 0)
76 return ERROR_REPLACE;
77 else
78 return PyUnicode_FromString(errors);
79}

81static PyObject *
82call_error_callback(PyObject *errors, PyObject *exc)
83{
84 PyObject *args, *cb, *r;
85 const char *str;
86
87 assert(PyUnicode_Check(errors));
88 str = PyUnicode_AsUTF8(errors);
89 if (str == NULL)
90 return NULL;
91 cb = PyCodec_LookupError(str);
92 if (cb == NULL)
93 return NULL;
94
95 args = PyTuple_New(1);
96 if (args == NULL) {
97 Py_DECREF(cb);
98 return NULL;
99 }
100
101 PyTuple_SET_ITEM(args, 0, exc);
102 Py_INCREF(exc);
103
104 r = PyObject_CallObject(cb, args);
105 Py_DECREF(args);
106 Py_DECREF(cb);
107 return r;
108}

162static int
163expand_encodebuffer(MultibyteEncodeBuffer *buf, Py_ssize_t esize)
164{
165 Py_ssize_t orgpos, orgsize, incsize;
166
167 orgpos = (Py_ssize_t)((char *)buf->outbuf -
168 PyBytes_AS_STRING(buf->outobj));
169 orgsize = PyBytes_GET_SIZE(buf->outobj);
170 incsize = (esize < (orgsize >> 1) ? (orgsize >> 1) | 1 : esize);
171
172 if (orgsize > PY_SSIZE_T_MAX - incsize) {
173 PyErr_NoMemory();
174 return -1;
175 }
176
177 if (_PyBytes_Resize(&buf->outobj, orgsize + incsize) == -1)
178 return -1;
179
180 buf->outbuf = (unsigned char *)PyBytes_AS_STRING(buf->outobj) +orgpos;
181 buf->outbuf_end = (unsigned char *)PyBytes_AS_STRING(buf->outobj)
182 + PyBytes_GET_SIZE(buf->outobj);
183
184 return 0;
185}
186#define REQUIRE_ENCODEBUFFER(buf, s) do { \
187 if ((s) < 0 || (s) > (buf)->outbuf_end - (buf)->outbuf) \
188 if (expand_encodebuffer(buf, s) == -1) \
189 goto errorexit; \
190} while(0)

197static int
198multibytecodec_encerror(MultibyteCodec *codec,
199 MultibyteCodec_State *state,
200 MultibyteEncodeBuffer *buf,
201 PyObject *errors, Py_ssize_t e)
202{
203 PyObject *retobj = NULL, *retstr = NULL, *tobj;
204 Py_ssize_t retstrsize, newpos;
205 Py_ssize_t esize, start, end;
206 const char *reason;
207
208 if (e > 0) {
209 reason = "illegal multibyte sequence";
210 esize = e;
211 }
212 else {
213 switch (e) {
214 case MBERR_TOOSMALL:
215 REQUIRE_ENCODEBUFFER(buf, -1);
216 return 0; /* retry it */
217 case MBERR_TOOFEW:
218 reason = "incomplete multibyte sequence";
219 esize = (Py_ssize_t)buf->inpos;
220 break;
221 case MBERR_INTERNAL:
222 PyErr_SetString(PyExc_RuntimeError,
223 "internal codec error");
224 return -1;
225 default:
226 PyErr_SetString(PyExc_RuntimeError,
227 "unknown runtime error");
228 return -1;
229 }
230 }
231
232 if (errors == ERROR_REPLACE) {
233 PyObject *replchar;
234 Py_ssize_t r;
235 Py_ssize_t inpos;
236 int kind;
237 void *data;
238
239 replchar = PyUnicode_FromOrdinal('?');
240 if (replchar == NULL)
241 goto errorexit;
242 kind = PyUnicode_KIND(replchar);
243 data = PyUnicode_DATA(replchar);
244
245 inpos = 0;
246 for (;;) {
247 Py_ssize_t outleft = (Py_ssize_t)(buf->outbuf_end - buf->outbuf);
248
249 r = codec->encode(state, codec->config,
250 kind, data, &inpos, 1,
251 &buf->outbuf, outleft, 0);
252 if (r == MBERR_TOOSMALL) {
253 REQUIRE_ENCODEBUFFER(buf, -1);
254 continue;
255 }
256 else
257 break;
258 }
259
260 Py_DECREF(replchar);
261
262 if (r != 0) {
263 REQUIRE_ENCODEBUFFER(buf, 1);
264 *buf->outbuf++ = '?';
265 }
266 }
267 if (errors == ERROR_IGNORE || errors == ERROR_REPLACE) {
268 buf->inpos += esize;
269 return 0;
270 }
271
272 start = (Py_ssize_t)buf->inpos;
273 end = start + esize;
274
275 /* use cached exception object if available */
276 if (buf->excobj == NULL) {
277 buf->excobj = PyObject_CallFunction(PyExc_UnicodeEncodeError,
278 "sOnns",
279 codec->encoding, buf->inobj,
280 start, end, reason);
281 if (buf->excobj == NULL)
282 goto errorexit;
283 }
284 else
285 if (PyUnicodeEncodeError_SetStart(buf->excobj, start) != 0 ||
286 PyUnicodeEncodeError_SetEnd(buf->excobj, end) != 0 ||
287 PyUnicodeEncodeError_SetReason(buf->excobj, reason) != 0)
288 goto errorexit;
289
290 if (errors == ERROR_STRICT) {
291 PyCodec_StrictErrors(buf->excobj);
292 goto errorexit;
293 }
294
295 retobj = call_error_callback(errors, buf->excobj);
296 if (retobj == NULL)
297 goto errorexit;
298
299 if (!PyTuple_Check(retobj) || PyTuple_GET_SIZE(retobj) != 2 ||
300 (!PyUnicode_Check((tobj = PyTuple_GET_ITEM(retobj, 0))) && !PyBytes_Check(tobj)) ||
301 !PyLong_Check(PyTuple_GET_ITEM(retobj, 1))) {
302 PyErr_SetString(PyExc_TypeError,
303 "encoding error handler must return "
304 "(str, int) tuple");
305 goto errorexit;
306 }
307
308 if (PyUnicode_Check(tobj)) {
309 Py_ssize_t inpos;
310
311 retstr = multibytecodec_encode(codec, state, tobj,
312 &inpos, ERROR_STRICT,
313 MBENC_FLUSH);
314 if (retstr == NULL)
315 goto errorexit;
316 }
317 else {
318 Py_INCREF(tobj);
319 retstr = tobj;
320 }
321
322 assert(PyBytes_Check(retstr));
323 retstrsize = PyBytes_GET_SIZE(retstr);
324 if (retstrsize > 0) {
325 REQUIRE_ENCODEBUFFER(buf, retstrsize);
326 memcpy(buf->outbuf, PyBytes_AS_STRING(retstr), retstrsize);
327 buf->outbuf += retstrsize;
328 }
329
330 newpos = PyLong_AsSsize_t(PyTuple_GET_ITEM(retobj, 1));
331 if (newpos < 0 && !PyErr_Occurred())
332 newpos += (Py_ssize_t)buf->inlen;
333 if (newpos < 0 || newpos > buf->inlen) {
334 PyErr_Clear();
335 PyErr_Format(PyExc_IndexError,
336 "position %zd from error handler out of bounds",
337 newpos);
338 goto errorexit;
339 }
340 buf->inpos = newpos;
341
342 Py_DECREF(retobj);
343 Py_DECREF(retstr);
344 return 0;
345
346errorexit:
347 Py_XDECREF(retobj);
348 Py_XDECREF(retstr);
349 return -1;
350}

352static int
353multibytecodec_decerror(MultibyteCodec *codec,
354 MultibyteCodec_State *state,
355 MultibyteDecodeBuffer *buf,
356 PyObject *errors, Py_ssize_t e)
357{
358 PyObject *retobj = NULL, *retuni = NULL;
359 Py_ssize_t newpos;
360 const char *reason;
361 Py_ssize_t esize, start, end;
362
363 if (e > 0) {
364 reason = "illegal multibyte sequence";
365 esize = e;
366 }
367 else {
368 switch (e) {
369 case MBERR_TOOSMALL:
370 return 0; /* retry it */
371 case MBERR_TOOFEW:
372 reason = "incomplete multibyte sequence";
373 esize = (Py_ssize_t)(buf->inbuf_end - buf->inbuf);
374 break;
375 case MBERR_INTERNAL:
376 PyErr_SetString(PyExc_RuntimeError,
377 "internal codec error");
378 return -1;
379 case MBERR_EXCEPTION:
380 return -1;
381 default:
382 PyErr_SetString(PyExc_RuntimeError,
383 "unknown runtime error");
384 return -1;
385 }
386 }
387
388 if (errors == ERROR_REPLACE) {
389 if (_PyUnicodeWriter_WriteChar(&buf->writer,
390 Py_UNICODE_REPLACEMENT_CHARACTER) < 0)
391 goto errorexit;
392 }
393 if (errors == ERROR_IGNORE || errors == ERROR_REPLACE) {
394 buf->inbuf += esize;
395 return 0;
396 }
397
398 start = (Py_ssize_t)(buf->inbuf - buf->inbuf_top);
399 end = start + esize;
400
401 /* use cached exception object if available */
402 if (buf->excobj == NULL) {
403 buf->excobj = PyUnicodeDecodeError_Create(codec->encoding,
404 (const char *)buf->inbuf_top,
405 (Py_ssize_t)(buf->inbuf_end - buf->inbuf_top),
406 start, end, reason);
407 if (buf->excobj == NULL)
408 goto errorexit;
409 }
410 else
411 if (PyUnicodeDecodeError_SetStart(buf->excobj, start) ||
412 PyUnicodeDecodeError_SetEnd(buf->excobj, end) ||
413 PyUnicodeDecodeError_SetReason(buf->excobj, reason))
414 goto errorexit;
415
416 if (errors == ERROR_STRICT) {
417 PyCodec_StrictErrors(buf->excobj);
418 goto errorexit;
419 }
420
421 retobj = call_error_callback(errors, buf->excobj);
422 if (retobj == NULL)
423 goto errorexit;
424
425 if (!PyTuple_Check(retobj) || PyTuple_GET_SIZE(retobj) != 2 ||
426 !PyUnicode_Check((retuni = PyTuple_GET_ITEM(retobj, 0))) ||
427 !PyLong_Check(PyTuple_GET_ITEM(retobj, 1))) {
428 PyErr_SetString(PyExc_TypeError,
429 "decoding error handler must return "
430 "(str, int) tuple");
431 goto errorexit;
432 }
433
434 if (_PyUnicodeWriter_WriteStr(&buf->writer, retuni) < 0)
435 goto errorexit;
436
437 newpos = PyLong_AsSsize_t(PyTuple_GET_ITEM(retobj, 1));
438 if (newpos < 0 && !PyErr_Occurred())
439 newpos += (Py_ssize_t)(buf->inbuf_end - buf->inbuf_top);
440 if (newpos < 0 || buf->inbuf_top + newpos > buf->inbuf_end) {
441 PyErr_Clear();
442 PyErr_Format(PyExc_IndexError,
443 "position %zd from error handler out of bounds",
444 newpos);
445 goto errorexit;
446 }
447 buf->inbuf = buf->inbuf_top + newpos;
448 Py_DECREF(retobj);
449 return 0;
450
451errorexit:
452 Py_XDECREF(retobj);
453 return -1;
454}

456static PyObject *
457multibytecodec_encode(MultibyteCodec *codec,
458 MultibyteCodec_State *state,
459 PyObject *text, Py_ssize_t *inpos_t,
460 PyObject *errors, int flags)
461{
462 MultibyteEncodeBuffer buf;
463 Py_ssize_t finalsize, r = 0;
464 Py_ssize_t datalen;
465 int kind;
466 void *data;
467
468 if (PyUnicode_READY(text) < 0)
469 return NULL;
470 datalen = PyUnicode_GET_LENGTH(text);
471
472 if (datalen == 0 && !(flags & MBENC_RESET))
473 return PyBytes_FromStringAndSize(NULL, 0);
474
475 buf.excobj = NULL;
476 buf.outobj = NULL;
477 buf.inobj = text; /* borrowed reference */
478 buf.inpos = 0;
479 buf.inlen = datalen;
480 kind = PyUnicode_KIND(buf.inobj);
481 data = PyUnicode_DATA(buf.inobj);
482
483 if (datalen > (PY_SSIZE_T_MAX - 16) / 2) {
484 PyErr_NoMemory();
485 goto errorexit;
486 }
487
488 buf.outobj = PyBytes_FromStringAndSize(NULL, datalen * 2 + 16);
489 if (buf.outobj == NULL)
490 goto errorexit;
491 buf.outbuf = (unsigned char *)PyBytes_AS_STRING(buf.outobj);
492 buf.outbuf_end = buf.outbuf + PyBytes_GET_SIZE(buf.outobj);
493
494 while (buf.inpos < buf.inlen) {
495 /* we don't reuse inleft and outleft here.
496 * error callbacks can relocate the cursor anywhere on buffer*/
497 Py_ssize_t outleft = (Py_ssize_t)(buf.outbuf_end - buf.outbuf);
498
499 r = codec->encode(state, codec->config,
500 kind, data,
501 &buf.inpos, buf.inlen,
502 &buf.outbuf, outleft, flags);
503 if ((r == 0) || (r == MBERR_TOOFEW && !(flags & MBENC_FLUSH)))
504 break;
505 else if (multibytecodec_encerror(codec, state, &buf, errors,r))
506 goto errorexit;
507 else if (r == MBERR_TOOFEW)
508 break;
509 }
510
511 if (codec->encreset != NULL && (flags & MBENC_RESET))
512 for (;;) {
513 Py_ssize_t outleft;
514
515 outleft = (Py_ssize_t)(buf.outbuf_end - buf.outbuf);
516 r = codec->encreset(state, codec->config, &buf.outbuf,
517 outleft);
518 if (r == 0)
519 break;
520 else if (multibytecodec_encerror(codec, state,
521 &buf, errors, r))
522 goto errorexit;
523 }
524
525 finalsize = (Py_ssize_t)((char *)buf.outbuf -
526 PyBytes_AS_STRING(buf.outobj));
527
528 if (finalsize != PyBytes_GET_SIZE(buf.outobj))
529 if (_PyBytes_Resize(&buf.outobj, finalsize) == -1)
530 goto errorexit;
531
532 if (inpos_t)
533 *inpos_t = buf.inpos;
534 Py_XDECREF(buf.excobj);
535 return buf.outobj;
536
537errorexit:
538 Py_XDECREF(buf.excobj);
539 Py_XDECREF(buf.outobj);
540 return NULL;
541}
