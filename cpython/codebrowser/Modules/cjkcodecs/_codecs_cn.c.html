18/* GBK and GB2312 map differently in few code points that are listed below:
19 *
20 * gb2312 gbk
21 * A1A4 U+30FB KATAKANA MIDDLE DOT U+00B7 MIDDLE DOT
22 * A1AA U+2015 HORIZONTAL BAR U+2014 EM DASH
23 * A844 undefined U+2015 HORIZONTAL BAR
24 */
25
26#define GBK_DECODE(dc1, dc2, writer) \
27 if ((dc1) == 0xa1 && (dc2) == 0xaa) { \
28 OUTCHAR(0x2014); \
29 } \
30 else if ((dc1) == 0xa8 && (dc2) == 0x44) { \
31 OUTCHAR(0x2015); \
32 } \
33 else if ((dc1) == 0xa1 && (dc2) == 0xa4) { \
34 OUTCHAR(0x00b7); \
35 } \
36 else if (TRYMAP_DEC(gb2312, decoded, dc1 ^ 0x80, dc2 ^ 0x80)) { \
37 OUTCHAR(decoded); \
38 } \
39 else if (TRYMAP_DEC(gbkext, decoded, dc1, dc2)) { \
40 OUTCHAR(decoded); \
41 }
42
43#define GBK_ENCODE(code, assi) \
44 if ((code) == 0x2014) { \
45 (assi) = 0xa1aa; \
46 } else if ((code) == 0x2015) { \
47 (assi) = 0xa844; \
48 } else if ((code) == 0x00b7) { \
49 (assi) = 0xa1a4; \
50 } else if ((code) != 0x30fb && TRYMAP_ENC(gbcommon, assi, code)) { \
51 ; \
52 }

54/*
55 * GB2312 codec
56 */
57
58ENCODER(gb2312)
59{
60 while (*inpos < inlen) {
61 Py_UCS4 c = INCHAR1;
62 DBCHAR code;
63
64 if (c < 0x80) {
65 WRITEBYTE1((unsigned char)c);
66 NEXT(1, 1);
67 continue;
68 }
69
70 if (c > 0xFFFF)
71 return 1;
72
73 REQUIRE_OUTBUF(2);
74 if (TRYMAP_ENC(gbcommon, code, c))
75 ;
76 else
77 return 1;
78
79 if (code & 0x8000) /* MSB set: GBK */
80 return 1;
81
82 OUTBYTE1((code >> 8) | 0x80);
83 OUTBYTE2((code & 0xFF) | 0x80);
84 NEXT(1, 2);
85 }
86
87 return 0;
88}
89
90DECODER(gb2312)
91{
92 while (inleft > 0) {
93 unsigned char c = **inbuf;
94 Py_UCS4 decoded;
95
96 if (c < 0x80) {
97 OUTCHAR(c);
98 NEXT_IN(1);
99 continue;
100 }
101
102 REQUIRE_INBUF(2);
103 if (TRYMAP_DEC(gb2312, decoded, c ^ 0x80, INBYTE2 ^ 0x80)) {
104 OUTCHAR(decoded);
105 NEXT_IN(2);
106 }
107 else
108 return 1;
109 }
110
111 return 0;
112}

115/*
116 * GBK codec
117 */
118
119ENCODER(gbk)
120{
121 while (*inpos < inlen) {
122 Py_UCS4 c = INCHAR1;
123 DBCHAR code;
124
125 if (c < 0x80) {
126 WRITEBYTE1((unsigned char)c);
127 NEXT(1, 1);
128 continue;
129 }
130
131 if (c > 0xFFFF)
132 return 1;
133
134 REQUIRE_OUTBUF(2);
135
136 GBK_ENCODE(c, code)
137 else
138 return 1;
139
140 OUTBYTE1((code >> 8) | 0x80);
141 if (code & 0x8000)
142 OUTBYTE2((code & 0xFF)); /* MSB set: GBK */
143 else
144 OUTBYTE2((code & 0xFF) | 0x80); /* MSB unset: GB2312 */
145 NEXT(1, 2);
146 }
147
148 return 0;
149}
150
151DECODER(gbk)
152{
153 while (inleft > 0) {
154 unsigned char c = INBYTE1;
155 Py_UCS4 decoded;
156
157 if (c < 0x80) {
158 OUTCHAR(c);
159 NEXT_IN(1);
160 continue;
161 }
162
163 REQUIRE_INBUF(2);
164
165 GBK_DECODE(c, INBYTE2, writer)
166 else
167 return 1;
168
169 NEXT_IN(2);
170 }
171
172 return 0;
173}

176/*
177 * GB18030 codec
178 */
179
180ENCODER(gb18030)
181{
182 while (*inpos < inlen) {
183 Py_UCS4 c = INCHAR1;
184 DBCHAR code;
185
186 if (c < 0x80) {
187 WRITEBYTE1(c);
188 NEXT(1, 1);
189 continue;
190 }
191
192 if (c >= 0x10000) {
193 Py_UCS4 tc = c - 0x10000;
194 assert (c <= 0x10FFFF);
195
196 REQUIRE_OUTBUF(4);
197
198 OUTBYTE4((unsigned char)(tc % 10) + 0x30);
199 tc /= 10;
200 OUTBYTE3((unsigned char)(tc % 126) + 0x81);
201 tc /= 126;
202 OUTBYTE2((unsigned char)(tc % 10) + 0x30);
203 tc /= 10;
204 OUTBYTE1((unsigned char)(tc + 0x90));
205
206 NEXT(1, 4);
207 continue;
208 }
209
210 REQUIRE_OUTBUF(2);
211
212 GBK_ENCODE(c, code)
213 else if (TRYMAP_ENC(gb18030ext, code, c))
214 ;
215 else {
216 const struct _gb18030_to_unibmp_ranges *utrrange;
217
218 REQUIRE_OUTBUF(4);
219
220 for (utrrange = gb18030_to_unibmp_ranges;
221 utrrange->first != 0;
222 utrrange++)
223 if (utrrange->first <= c &&
224 c <= utrrange->last) {
225 Py_UCS4 tc;
226
227 tc = c - utrrange->first +
228 utrrange->base;
229
230 OUTBYTE4((unsigned char)(tc % 10) + 0x30);
231 tc /= 10;
232 OUTBYTE3((unsigned char)(tc % 126) + 0x81);
233 tc /= 126;
234 OUTBYTE2((unsigned char)(tc % 10) + 0x30);
235 tc /= 10;
236 OUTBYTE1((unsigned char)tc + 0x81);
237
238 NEXT(1, 4);
239 break;
240 }
241
242 if (utrrange->first == 0)
243 return 1;
244 continue;
245 }
246
247 OUTBYTE1((code >> 8) | 0x80);
248 if (code & 0x8000)
249 OUTBYTE2((code & 0xFF)); /* MSB set: GBK or GB18030ext */
250 else
251 OUTBYTE2((code & 0xFF) | 0x80); /* MSB unset: GB2312 */
252
253 NEXT(1, 2);
254 }
255
256 return 0;
257}

259DECODER(gb18030)
260{
261 while (inleft > 0) {
262 unsigned char c = INBYTE1, c2;
263 Py_UCS4 decoded;
264
265 if (c < 0x80) {
266 OUTCHAR(c);
267 NEXT_IN(1);
268 continue;
269 }
270
271 REQUIRE_INBUF(2);
272
273 c2 = INBYTE2;
274 if (c2 >= 0x30 && c2 <= 0x39) { /* 4 bytes seq */
275 const struct _gb18030_to_unibmp_ranges *utr;
276 unsigned char c3, c4;
277 Py_UCS4 lseq;
278
279 REQUIRE_INBUF(4);
280 c3 = INBYTE3;
281 c4 = INBYTE4;
282 if (c < 0x81 || c > 0xFE ||
283 c3 < 0x81 || c3 > 0xFE ||
284 c4 < 0x30 || c4 > 0x39)
285 return 1;
286 c -= 0x81; c2 -= 0x30;
287 c3 -= 0x81; c4 -= 0x30;
288
289 if (c < 4) { /* U+0080 - U+FFFF */
290 lseq = ((Py_UCS4)c * 10 + c2) * 1260 +
291 (Py_UCS4)c3 * 10 + c4;
292 if (lseq < 39420) {
293 for (utr = gb18030_to_unibmp_ranges;
294 lseq >= (utr + 1)->base;
295 utr++) ;
296 OUTCHAR(utr->first - utr->base + lseq);
297 NEXT_IN(4);
298 continue;
299 }
300 }
301 else if (c >= 15) { /* U+10000 - U+10FFFF */
302 lseq = 0x10000 + (((Py_UCS4)c-15) * 10 + c2)
303 * 1260 + (Py_UCS4)c3 * 10 + c4;
304 if (lseq <= 0x10FFFF) {
305 OUTCHAR(lseq);
306 NEXT_IN(4);
307 continue;
308 }
309 }
310 return 1;
311 }
312
313 GBK_DECODE(c, c2, writer)
314 else if (TRYMAP_DEC(gb18030ext, decoded, c, c2))
315 OUTCHAR(decoded);
316 else
317 return 1;
318
319 NEXT_IN(2);
320 }
321
322 return 0;
323}

330ENCODER_INIT(hz)
331{
332 state->i = 0;
333 return 0;
334}
335
336ENCODER_RESET(hz)
337{
338 if (state->i != 0) {
339 WRITEBYTE2('~', '}');
340 state->i = 0;
341 NEXT_OUT(2);
342 }
343 return 0;
344}
345
346ENCODER(hz)
347{
348 while (*inpos < inlen) {
349 Py_UCS4 c = INCHAR1;
350 DBCHAR code;
351
352 if (c < 0x80) {
353 if (state->i) {
354 WRITEBYTE2('~', '}');
355 NEXT_OUT(2);
356 state->i = 0;
357 }
358 WRITEBYTE1((unsigned char)c);
359 NEXT(1, 1);
360 if (c == '~') {
361 WRITEBYTE1('~');
362 NEXT_OUT(1);
363 }
364 continue;
365 }
366
367 if (c > 0xFFFF)
368 return 1;
369
370 if (TRYMAP_ENC(gbcommon, code, c))
371 ;
372 else
373 return 1;
374
375 if (code & 0x8000) /* MSB set: GBK */
376 return 1;
377
378 if (state->i == 0) {
379 WRITEBYTE4('~', '{', code >> 8, code & 0xff);
380 NEXT(1, 4);
381 state->i = 1;
382 }
383 else {
384 WRITEBYTE2(code >> 8, code & 0xff);
385 NEXT(1, 2);
386 }
387 }
388
389 return 0;
390}
391
392DECODER_INIT(hz)
393{
394 state->i = 0;
395 return 0;
396}
397
398DECODER_RESET(hz)
399{
400 state->i = 0;
401 return 0;
402}
403
404DECODER(hz)
405{
406 while (inleft > 0) {
407 unsigned char c = INBYTE1;
408 Py_UCS4 decoded;
409
410 if (c == '~') {
411 unsigned char c2 = INBYTE2;
412
413 REQUIRE_INBUF(2);
414 if (c2 == '~' && state->i == 0)
415 OUTCHAR('~');
416 else if (c2 == '{' && state->i == 0)
417 state->i = 1; /* set GB */
418 else if (c2 == '\n' && state->i == 0)
419 ; /* line-continuation */
420 else if (c2 == '}' && state->i == 1)
421 state->i = 0; /* set ASCII */
422 else
423 return 1;
424 NEXT_IN(2);
425 continue;
426 }
427
428 if (c & 0x80)
429 return 1;
430
431 if (state->i == 0) { /* ASCII mode */
432 OUTCHAR(c);
433 NEXT_IN(1);
434 }
435 else { /* GB mode */
436 REQUIRE_INBUF(2);
437 if (TRYMAP_DEC(gb2312, decoded, c, INBYTE2)) {
438 OUTCHAR(decoded);
439 NEXT_IN(2);
440 }
441 else
442 return 1;
443 }
444 }
445
446 return 0;
447}
