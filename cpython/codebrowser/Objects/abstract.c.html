78/* The length hint function returns a non-negative value from o.__len__()
79 or o.__length_hint__(). If those methods aren't found the defaultvalue is
80 returned. If one of the calls fails with an exception other than TypeError
81 this function returns -1.
82*/
83
84Py_ssize_t
85PyObject_LengthHint(PyObject *o, Py_ssize_t defaultvalue)
86{
87 PyObject *hint, *result;
88 Py_ssize_t res;
89 _Py_IDENTIFIER(__length_hint__);
90 if (_PyObject_HasLen(o)) {
91 res = PyObject_Length(o);
92 if (res < 0) {
93 assert(PyErr_Occurred());
94 if (!PyErr_ExceptionMatches(PyExc_TypeError)) {
95 return -1;
96 }
97 PyErr_Clear();
98 }
99 else {
100 return res;
101 }
102 }
103 hint = _PyObject_LookupSpecial(o, &PyId___length_hint__);
104 if (hint == NULL) {
105 if (PyErr_Occurred()) {
106 return -1;
107 }
108 return defaultvalue;
109 }
110 result = _PyObject_CallNoArg(hint);
111 Py_DECREF(hint);
112 if (result == NULL) {
113 if (PyErr_ExceptionMatches(PyExc_TypeError)) {
114 PyErr_Clear();
115 return defaultvalue;
116 }
117 return -1;
118 }
119 else if (result == Py_NotImplemented) {
120 Py_DECREF(result);
121 return defaultvalue;
122 }
123 if (!PyLong_Check(result)) {
124 PyErr_Format(PyExc_TypeError, "__length_hint__ must be an integer, not %.100s",
125 Py_TYPE(result)->tp_name);
126 Py_DECREF(result);
127 return -1;
128 }
129 res = PyLong_AsSsize_t(result);
130 Py_DECREF(result);
131 if (res < 0 && PyErr_Occurred()) {
132 return -1;
133 }
134 if (res < 0) {
135 PyErr_Format(PyExc_ValueError, "__length_hint__() should return >= 0");
136 return -1;
137 }
138 return res;
139}

141PyObject *
142PyObject_GetItem(PyObject *o, PyObject *key)
143{
144 PyMappingMethods *m;
145
146 if (o == NULL || key == NULL) {
147 return null_error();
148 }
149
150 m = o->ob_type->tp_as_mapping;
151 if (m && m->mp_subscript) {
152 PyObject *item = m->mp_subscript(o, key);
153 assert((item != NULL) ^ (PyErr_Occurred() != NULL));
154 return item;
155 }
156
157 if (o->ob_type->tp_as_sequence) {
158 if (PyIndex_Check(key)) {
159 Py_ssize_t key_value;
160 key_value = PyNumber_AsSsize_t(key, PyExc_IndexError);
161 if (key_value == -1 && PyErr_Occurred())
162 return NULL;
163 return PySequence_GetItem(o, key_value);
164 }
165 else if (o->ob_type->tp_as_sequence->sq_item)
166 return type_error("sequence index must "
167 "be integer, not '%.200s'", key);
168 }
169
170 return type_error("'%.200s' object is not subscriptable", o);
171}

195 type_error("sequence index must be "
196 "integer, not '%.200s'", key);
197 return -1;
198 }
199 }
200
201 type_error("'%.200s' object does not support item assignment", o);
202 return -1;

350static int
351_IsFortranContiguous(const Py_buffer *view)
352{
353 Py_ssize_t sd, dim;
354 int i;
355
356 /* 1) len = product(shape) * itemsize
357 2) itemsize > 0
358 3) len = 0 <==> exists i: shape[i] = 0 */
359 if (view->len == 0) return 1;
360 if (view->strides == NULL) { /* C-contiguous by definition */
361 /* Trivially F-contiguous */
362 if (view->ndim <= 1) return 1;
363
364 /* ndim > 1 implies shape != NULL */
365 assert(view->shape != NULL);
366
367 /* Effectively 1-d */
368 sd = 0;
369 for (i=0; i<view->ndim; i++) {
370 if (view->shape[i] > 1) sd += 1;
371 }
372 return sd <= 1;
373 }
374
375 /* strides != NULL implies both of these */
376 assert(view->ndim > 0);
377 assert(view->shape != NULL);
378
379 sd = view->itemsize;
380 for (i=0; i<view->ndim; i++) {
381 dim = view->shape[i];
382 if (dim > 1 && view->strides[i] != sd) {
383 return 0;
384 }
385 sd *= dim;
386 }
387 return 1;
388}

390static int
391_IsCContiguous(const Py_buffer *view)
392{
393 Py_ssize_t sd, dim;
394 int i;
395
396 /* 1) len = product(shape) * itemsize
397 2) itemsize > 0
398 3) len = 0 <==> exists i: shape[i] = 0 */
399 if (view->len == 0) return 1;
400 if (view->strides == NULL) return 1; /* C-contiguous by definition */
401
402 /* strides != NULL implies both of these */
403 assert(view->ndim > 0);
404 assert(view->shape != NULL);
405
406 sd = view->itemsize;
407 for (i=view->ndim-1; i>=0; i--) {
408 dim = view->shape[i];
409 if (dim > 1 && view->strides[i] != sd) {
410 return 0;
411 }
412 sd *= dim;
413 }
414 return 1;
415}

417int
418PyBuffer_IsContiguous(const Py_buffer *view, char order)
419{
420
421 if (view->suboffsets != NULL) return 0;
422
423 if (order == 'C')
424 return _IsCContiguous(view);
425 else if (order == 'F')
426 return _IsFortranContiguous(view);
427 else if (order == 'A')
428 return (_IsCContiguous(view) || _IsFortranContiguous(view));
429 return 0;
430}

433void*
434PyBuffer_GetPointer(Py_buffer *view, Py_ssize_t *indices)
435{
436 char* pointer;
437 int i;
438 pointer = (char *)view->buf;
439 for (i = 0; i < view->ndim; i++) {
440 pointer += view->strides[i]*indices[i];
441 if ((view->suboffsets != NULL) && (view->suboffsets[i] >= 0)) {
442 pointer = *((char**)pointer) + view->suboffsets[i];
443 }
444 }
445 return (void*)pointer;
446}

481int
482PyBuffer_FromContiguous(Py_buffer *view, void *buf, Py_ssize_t len, char fort)
483{
484 int k;
485 void (*addone)(int, Py_ssize_t *, const Py_ssize_t *);
486 Py_ssize_t *indices, elements;
487 char *src, *ptr;
488
489 if (len > view->len) {
490 len = view->len;
491 }
492
493 if (PyBuffer_IsContiguous(view, fort)) {
494 /* simplest copy is all that is needed */
495 memcpy(view->buf, buf, len);
496 return 0;
497 }
498
499 /* Otherwise a more elaborate scheme is needed */
500
501 /* view->ndim <= 64 */
502 indices = (Py_ssize_t *)PyMem_Malloc(sizeof(Py_ssize_t)*(view->ndim));
503 if (indices == NULL) {
504 PyErr_NoMemory();
505 return -1;
506 }
507 for (k=0; k<view->ndim;k++) {
508 indices[k] = 0;
509 }
510
511 if (fort == 'F') {
512 addone = _Py_add_one_to_index_F;
513 }
514 else {
515 addone = _Py_add_one_to_index_C;
516 }
517 src = buf;
518 /* XXX : This is not going to be the fastest code in the world
519 several optimizations are possible.
520 */
521 elements = len / view->itemsize;
522 while (elements--) {
523 ptr = PyBuffer_GetPointer(view, indices);
524 memcpy(ptr, src, view->itemsize);
525 src += view->itemsize;
526 addone(view->ndim, indices, view->shape);
527 }
528
529 PyMem_Free(indices);
530 return 0;
531}

533int PyObject_CopyData(PyObject *dest, PyObject *src)
534{
535 Py_buffer view_dest, view_src;
536 int k;
537 Py_ssize_t *indices, elements;
538 char *dptr, *sptr;
539
540 if (!PyObject_CheckBuffer(dest) ||
541 !PyObject_CheckBuffer(src)) {
542 PyErr_SetString(PyExc_TypeError,
543 "both destination and source must be "\
544 "bytes-like objects");
545 return -1;
546 }
547
548 if (PyObject_GetBuffer(dest, &view_dest, PyBUF_FULL) != 0) return -1;
549 if (PyObject_GetBuffer(src, &view_src, PyBUF_FULL_RO) != 0) {
550 PyBuffer_Release(&view_dest);
551 return -1;
552 }
553
554 if (view_dest.len < view_src.len) {
555 PyErr_SetString(PyExc_BufferError,
556 "destination is too small to receive data from source");
557 PyBuffer_Release(&view_dest);
558 PyBuffer_Release(&view_src);
559 return -1;
560 }
561
562 if ((PyBuffer_IsContiguous(&view_dest, 'C') &&
563 PyBuffer_IsContiguous(&view_src, 'C')) ||
564 (PyBuffer_IsContiguous(&view_dest, 'F') &&
565 PyBuffer_IsContiguous(&view_src, 'F'))) {
566 /* simplest copy is all that is needed */
567 memcpy(view_dest.buf, view_src.buf, view_src.len);
568 PyBuffer_Release(&view_dest);
569 PyBuffer_Release(&view_src);
570 return 0;
571 }
572
573 /* Otherwise a more elaborate copy scheme is needed */
574
575 /* XXX(nnorwitz): need to check for overflow! */
576 indices = (Py_ssize_t *)PyMem_Malloc(sizeof(Py_ssize_t)*view_src.ndim);
577 if (indices == NULL) {
578 PyErr_NoMemory();
579 PyBuffer_Release(&view_dest);
580 PyBuffer_Release(&view_src);
581 return -1;
582 }
583 for (k=0; k<view_src.ndim;k++) {
584 indices[k] = 0;
585 }
586 elements = 1;
587 for (k=0; k<view_src.ndim; k++) {
588 /* XXX(nnorwitz): can this overflow? */
589 elements *= view_src.shape[k];
590 }
591 while (elements--) {
592 _Py_add_one_to_index_C(view_src.ndim, indices, view_src.shape);
593 dptr = PyBuffer_GetPointer(&view_dest, indices);
594 sptr = PyBuffer_GetPointer(&view_src, indices);
595 memcpy(dptr, sptr, view_src.itemsize);
596 }
597 PyMem_Free(indices);
598 PyBuffer_Release(&view_dest);
599 PyBuffer_Release(&view_src);
600 return 0;
601}

603void
604PyBuffer_FillContiguousStrides(int nd, Py_ssize_t *shape,
605 Py_ssize_t *strides, int itemsize,
606 char fort)
607{
608 int k;
609 Py_ssize_t sd;
610
611 sd = itemsize;
612 if (fort == 'F') {
613 for (k=0; k<nd; k++) {
614 strides[k] = sd;
615 sd *= shape[k];
616 }
617 }
618 else {
619 for (k=nd-1; k>=0; k--) {
620 strides[k] = sd;
621 sd *= shape[k];
622 }
623 }
624 return;
625}

627int
628PyBuffer_FillInfo(Py_buffer *view, PyObject *obj, void *buf, Py_ssize_t len,
629 int readonly, int flags)
630{
631 if (view == NULL) {
632 PyErr_SetString(PyExc_BufferError,
633 "PyBuffer_FillInfo: view==NULL argument is obsolete");
634 return -1;
635 }
636
637 if (((flags & PyBUF_WRITABLE) == PyBUF_WRITABLE) &&
638 (readonly == 1)) {
639 PyErr_SetString(PyExc_BufferError,
640 "Object is not writable.");
641 return -1;
642 }
643
644 view->obj = obj;
645 if (obj)
646 Py_INCREF(obj);
647 view->buf = buf;
648 view->len = len;
649 view->readonly = readonly;
650 view->itemsize = 1;
651 view->format = NULL;
652 if ((flags & PyBUF_FORMAT) == PyBUF_FORMAT)
653 view->format = "B";
654 view->ndim = 1;
655 view->shape = NULL;
656 if ((flags & PyBUF_ND) == PyBUF_ND)
657 view->shape = &(view->len);
658 view->strides = NULL;
659 if ((flags & PyBUF_STRIDES) == PyBUF_STRIDES)
660 view->strides = &(view->itemsize);
661 view->suboffsets = NULL;
662 view->internal = NULL;
663 return 0;
664}

680PyObject *
681PyObject_Format(PyObject *obj, PyObject *format_spec)
682{
683 PyObject *meth;
684 PyObject *empty = NULL;
685 PyObject *result = NULL;
686 _Py_IDENTIFIER(__format__);
687
688 if (format_spec != NULL && !PyUnicode_Check(format_spec)) {
689 PyErr_Format(PyExc_SystemError,
690 "Format specifier must be a string, not %.200s",
691 Py_TYPE(format_spec)->tp_name);
692 return NULL;
693 }
694
695 /* Fast path for common types. */
696 if (format_spec == NULL || PyUnicode_GET_LENGTH(format_spec) == 0) {
697 if (PyUnicode_CheckExact(obj)) {
698 Py_INCREF(obj);
699 return obj;
700 }
701 if (PyLong_CheckExact(obj)) {
702 return PyObject_Str(obj);
703 }
704 }
705
706 /* If no format_spec is provided, use an empty string */
707 if (format_spec == NULL) {
708 empty = PyUnicode_New(0, 0);
709 format_spec = empty;
710 }
711
712 /* Find the (unbound!) __format__ method */
713 meth = _PyObject_LookupSpecial(obj, &PyId___format__);
714 if (meth == NULL) {
715 if (!PyErr_Occurred())
716 PyErr_Format(PyExc_TypeError,
717 "Type %.100s doesn't define __format__",
718 Py_TYPE(obj)->tp_name);
719 goto done;
720 }
721
722 /* And call it. */
723 result = PyObject_CallFunctionObjArgs(meth, format_spec, NULL);
724 Py_DECREF(meth);
725
726 if (result && !PyUnicode_Check(result)) {
727 PyErr_Format(PyExc_TypeError,
728 "__format__ must return a str, not %.200s",
729 Py_TYPE(result)->tp_name);
730 Py_DECREF(result);
731 result = NULL;
732 goto done;
733 }
734
735done:
736 Py_XDECREF(empty);
737 return result;
738}

749/* Binary operators */
750
751#define NB_SLOT(x) offsetof(PyNumberMethods, x)
752#define NB_BINOP(nb_methods, slot) \
753 (*(binaryfunc*)(& ((char*)nb_methods)[slot]))
754#define NB_TERNOP(nb_methods, slot) \
755 (*(ternaryfunc*)(& ((char*)nb_methods)[slot]))
756
757/*
758 Calling scheme used for binary operations:
759
760 Order operations are tried until either a valid result or error:
761 w.op(v,w)[*], v.op(v,w), w.op(v,w)
762
763 [*] only when v->ob_type != w->ob_type && w->ob_type is a subclass of
764 v->ob_type
765 */
766
767static PyObject *
768binary_op1(PyObject *v, PyObject *w, const int op_slot)
769{
770 PyObject *x;
771 binaryfunc slotv = NULL;
772 binaryfunc slotw = NULL;
773
774 if (v->ob_type->tp_as_number != NULL)
775 slotv = NB_BINOP(v->ob_type->tp_as_number, op_slot);
776 if (w->ob_type != v->ob_type &&
777 w->ob_type->tp_as_number != NULL) {
778 slotw = NB_BINOP(w->ob_type->tp_as_number, op_slot);
779 if (slotw == slotv)
780 slotw = NULL;
781 }
782 if (slotv) {
783 if (slotw && PyType_IsSubtype(w->ob_type, v->ob_type)) {
784 x = slotw(v, w);
785 if (x != Py_NotImplemented)
786 return x;
787 Py_DECREF(x); /* can't do it */
788 slotw = NULL;
789 }
790 x = slotv(v, w);
791 if (x != Py_NotImplemented)
792 return x;
793 Py_DECREF(x); /* can't do it */
794 }
795 if (slotw) {
796 x = slotw(v, w);
797 if (x != Py_NotImplemented)
798 return x;
799 Py_DECREF(x); /* can't do it */
800 }

843/*
844 Calling scheme used for ternary operations:
845
846 Order operations are tried until either a valid result or error:
847 v.op(v,w,z), w.op(v,w,z), z.op(v,w,z)
848 */
849
850static PyObject *
851ternary_op(PyObject *v,
852 PyObject *w,
853 PyObject *z,
854 const int op_slot,
855 const char *op_name)
856{
857 PyNumberMethods *mv, *mw, *mz;
858 PyObject *x = NULL;
859 ternaryfunc slotv = NULL;
860 ternaryfunc slotw = NULL;
861 ternaryfunc slotz = NULL;
862
863 mv = v->ob_type->tp_as_number;
864 mw = w->ob_type->tp_as_number;
865 if (mv != NULL)
866 slotv = NB_TERNOP(mv, op_slot);
867 if (w->ob_type != v->ob_type &&
868 mw != NULL) {
869 slotw = NB_TERNOP(mw, op_slot);
870 if (slotw == slotv)
871 slotw = NULL;
872 }
873 if (slotv) {
874 if (slotw && PyType_IsSubtype(w->ob_type, v->ob_type)) {
875 x = slotw(v, w, z);
876 if (x != Py_NotImplemented)
877 return x;
878 Py_DECREF(x); /* can't do it */
879 slotw = NULL;
880 }
881 x = slotv(v, w, z);
882 if (x != Py_NotImplemented)
883 return x;
884 Py_DECREF(x); /* can't do it */
885 }
886 if (slotw) {
887 x = slotw(v, w, z);
888 if (x != Py_NotImplemented)
889 return x;
890 Py_DECREF(x); /* can't do it */
891 }
892 mz = z->ob_type->tp_as_number;
893 if (mz != NULL) {
894 slotz = NB_TERNOP(mz, op_slot);
895 if (slotz == slotv || slotz == slotw)
896 slotz = NULL;
897 if (slotz) {
898 x = slotz(v, w, z);
899 if (x != Py_NotImplemented)
900 return x;
901 Py_DECREF(x); /* can't do it */
902 }
903 }
904
905 if (z == Py_None)
906 PyErr_Format(
907 PyExc_TypeError,
908 "unsupported operand type(s) for ** or pow(): "
909 "'%.100s' and '%.100s'",
910 v->ob_type->tp_name,
911 w->ob_type->tp_name);
912 else
913 PyErr_Format(
914 PyExc_TypeError,
915 "unsupported operand type(s) for pow(): "
916 "'%.100s', '%.100s', '%.100s'",
917 v->ob_type->tp_name,
918 w->ob_type->tp_name,
919 z->ob_type->tp_name);
920 return NULL;
921}

1017/* Binary in-place operators */
1018
1019/* The in-place operators are defined to fall back to the 'normal',
1020 non in-place operations, if the in-place methods are not in place.
1021
1022 - If the left hand object has the appropriate struct members, and
1023 they are filled, call the appropriate function and return the
1024 result. No coercion is done on the arguments; the left-hand object
1025 is the one the operation is performed on, and it's up to the
1026 function to deal with the right-hand object.
1027
1028 - Otherwise, in-place modification is not supported. Handle it exactly as
1029 a non in-place operation of the same kind.
1030
1031 */
1032
1033static PyObject *
1034binary_iop1(PyObject *v, PyObject *w, const int iop_slot, const int op_slot)
1035{
1036 PyNumberMethods *mv = v->ob_type->tp_as_number;
1037 if (mv != NULL) {
1038 binaryfunc slot = NB_BINOP(mv, iop_slot);
1039 if (slot) {
1040 PyObject *x = (slot)(v, w);
1041 if (x != Py_NotImplemented) {
1042 return x;
1043 }
1044 Py_DECREF(x);
1045 }
1046 }
1047 return binary_op1(v, w, op_slot);
1048}

1059 return result;

1233/* Return a Python int from the object item.
1234 Raise TypeError if the result is not an int
1235 or if the object cannot be interpreted as an index.
1236*/
1237PyObject *
1238PyNumber_Index(PyObject *item)
1239{
1240 PyObject *result = NULL;
1241 if (item == NULL) {
1242 return null_error();
1243 }
1244
1245 if (PyLong_Check(item)) {
1246 Py_INCREF(item);
1247 return item;
1248 }
1249 if (!PyIndex_Check(item)) {
1250 PyErr_Format(PyExc_TypeError,
1251 "'%.200s' object cannot be interpreted "
1252 "as an integer", item->ob_type->tp_name);
1253 return NULL;
1254 }
1255 result = item->ob_type->tp_as_number->nb_index(item);
1256 if (!result || PyLong_CheckExact(result))
1257 return result;
1258 if (!PyLong_Check(result)) {
1259 PyErr_Format(PyExc_TypeError,
1260 "__index__ returned non-int (type %.200s)",
1261 result->ob_type->tp_name);
1262 Py_DECREF(result);
1263 return NULL;
1264 }
1265 /* Issue #17576: warn if 'result' not of exact type int. */
1266 if (PyErr_WarnFormat(PyExc_DeprecationWarning, 1,
1267 "__index__ returned non-int (type %.200s). "
1268 "The ability to return an instance of a strict subclass of int "
1269 "is deprecated, and may be removed in a future version of Python.",
1270 result->ob_type->tp_name)) {
1271 Py_DECREF(result);
1272 return NULL;
1273 }
1274 return result;
1275}

1277/* Return an error on Overflow only if err is not NULL*/
1278
1279Py_ssize_t
1280PyNumber_AsSsize_t(PyObject *item, PyObject *err)
1281{
1282 Py_ssize_t result;
1283 PyObject *runerr;
1284 PyObject *value = PyNumber_Index(item);
1285 if (value == NULL)
1286 return -1;
1287
1288 /* We're done if PyLong_AsSsize_t() returns without error. */
1289 result = PyLong_AsSsize_t(value);
1290 if (result != -1 || !(runerr = PyErr_Occurred()))
1291 goto finish;
1292
1293 /* Error handling code -- only manage OverflowError differently */
1294 if (!PyErr_GivenExceptionMatches(runerr, PyExc_OverflowError))
1295 goto finish;
1296
1297 PyErr_Clear();
1298 /* If no error-handling desired then the default clipping
1299 is sufficient.
1300 */
1301 if (!err) {
1302 assert(PyLong_Check(value));
1303 /* Whether or not it is less than or equal to
1304 zero is determined by the sign of ob_size
1305 */
1306 if (_PyLong_Sign(value) < 0)
1307 result = PY_SSIZE_T_MIN;
1308 else
1309 result = PY_SSIZE_T_MAX;
1310 }
1311 else {
1312 /* Otherwise replace the error with caller's error object. */
1313 PyErr_Format(err,
1314 "cannot fit '%.200s' into an index-sized integer",
1315 item->ob_type->tp_name);
1316 }
1317
1318 finish:
1319 Py_DECREF(value);
1320 return result;
1321}

1324PyObject *
1325PyNumber_Long(PyObject *o)
1326{
1327 PyObject *result;
1328 PyNumberMethods *m;
1329 PyObject *trunc_func;
1330 Py_buffer view;
1331 _Py_IDENTIFIER(__trunc__);
1332
1333 if (o == NULL) {
1334 return null_error();
1335 }
1336
1337 if (PyLong_CheckExact(o)) {
1338 Py_INCREF(o);
1339 return o;
1340 }
1341 m = o->ob_type->tp_as_number;
1342 if (m && m->nb_int) { /* This should include subclasses of int */
1343 result = (PyObject *)_PyLong_FromNbInt(o);
1344 if (result != NULL && !PyLong_CheckExact(result)) {
1345 Py_SETREF(result, _PyLong_Copy((PyLongObject *)result));
1346 }
1347 return result;
1348 }
1349 trunc_func = _PyObject_LookupSpecial(o, &PyId___trunc__);
1350 if (trunc_func) {
1351 result = _PyObject_CallNoArg(trunc_func);
1352 Py_DECREF(trunc_func);
1353 if (result == NULL || PyLong_CheckExact(result)) {
1354 return result;
1355 }
1356 if (PyLong_Check(result)) {
1357 Py_SETREF(result, _PyLong_Copy((PyLongObject *)result));
1358 return result;
1359 }
1360 /* __trunc__ is specified to return an Integral type,
1361 but int() needs to return an int. */
1362 m = result->ob_type->tp_as_number;
1363 if (m == NULL || m->nb_int == NULL) {
1364 PyErr_Format(
1365 PyExc_TypeError,
1366 "__trunc__ returned non-Integral (type %.200s)",
1367 result->ob_type->tp_name);
1368 Py_DECREF(result);
1369 return NULL;
1370 }
1371 Py_SETREF(result, (PyObject *)_PyLong_FromNbInt(result));
1372 if (result != NULL && !PyLong_CheckExact(result)) {
1373 Py_SETREF(result, _PyLong_Copy((PyLongObject *)result));
1374 }
1375 return result;
1376 }
1377 if (PyErr_Occurred())
1378 return NULL;
1379
1380 if (PyUnicode_Check(o))
1381 /* The below check is done in PyLong_FromUnicode(). */
1382 return PyLong_FromUnicodeObject(o, 10);
1383
1384 if (PyBytes_Check(o))
1385 /* need to do extra error checking that PyLong_FromString()
1386 * doesn't do. In particular int('9\x005') must raise an
1387 * exception, not truncate at the null.
1388 */
1389 return _PyLong_FromBytes(PyBytes_AS_STRING(o),
1390 PyBytes_GET_SIZE(o), 10);
1391
1392 if (PyByteArray_Check(o))
1393 return _PyLong_FromBytes(PyByteArray_AS_STRING(o),
1394 PyByteArray_GET_SIZE(o), 10);
1395
1396 if (PyObject_GetBuffer(o, &view, PyBUF_SIMPLE) == 0) {
1397 PyObject *bytes;
1398
1399 /* Copy to NUL-terminated buffer. */
1400 bytes = PyBytes_FromStringAndSize((const char *)view.buf, view.len);
1401 if (bytes == NULL) {
1402 PyBuffer_Release(&view);
1403 return NULL;
1404 }
1405 result = _PyLong_FromBytes(PyBytes_AS_STRING(bytes),
1406 PyBytes_GET_SIZE(bytes), 10);
1407 Py_DECREF(bytes);
1408 PyBuffer_Release(&view);
1409 return result;
1410 }
1411
1412 return type_error("int() argument must be a string, a bytes-like object "
1413 "or a number, not '%.200s'", o);
1414}

1790PyObject *
1791PySequence_Tuple(PyObject *v)
1792{
1793 PyObject *it; /* iter(v) */
1794 Py_ssize_t n; /* guess for result tuple size */
1795 PyObject *result = NULL;
1796 Py_ssize_t j;
1797
1798 if (v == NULL) {
1799 return null_error();
1800 }
1801
1802 /* Special-case the common tuple and list cases, for efficiency. */
1803 if (PyTuple_CheckExact(v)) {
1804 /* Note that we can't know whether it's safe to return
1805 a tuple *subclass* instance as-is, hence the restriction
1806 to exact tuples here. In contrast, lists always make
1807 a copy, so there's no need for exactness below. */
1808 Py_INCREF(v);
1809 return v;
1810 }
1811 if (PyList_CheckExact(v))
1812 return PyList_AsTuple(v);
1813
1814 /* Get iterator. */
1815 it = PyObject_GetIter(v);
1816 if (it == NULL)
1817 return NULL;
1818
1819 /* Guess result size and allocate space. */
1820 n = PyObject_LengthHint(v, 10);
1821 if (n == -1)
1822 goto Fail;
1823 result = PyTuple_New(n);
1824 if (result == NULL)
1825 goto Fail;
1826
1827 /* Fill the tuple. */
1828 for (j = 0; ; ++j) {
1829 PyObject *item = PyIter_Next(it);
1830 if (item == NULL) {
1831 if (PyErr_Occurred())
1832 goto Fail;
1833 break;
1834 }
1835 if (j >= n) {
1836 size_t newn = (size_t)n;
1837 /* The over-allocation strategy can grow a bit faster
1838 than for lists because unlike lists the
1839 over-allocation isn't permanent -- we reclaim
1840 the excess before the end of this routine.
1841 So, grow by ten and then add 25%.
1842 */
1843 newn += 10u;
1844 newn += newn >> 2;
1845 if (newn > PY_SSIZE_T_MAX) {
1846 /* Check for overflow */
1847 PyErr_NoMemory();
1848 Py_DECREF(item);
1849 goto Fail;
1850 }
1851 n = (Py_ssize_t)newn;
1852 if (_PyTuple_Resize(&result, n) != 0) {
1853 Py_DECREF(item);
1854 goto Fail;
1855 }
1856 }
1857 PyTuple_SET_ITEM(result, j, item);
1858 }
1859
1860 /* Cut tuple back if guess was too large. */
1861 if (j < n &&
1862 _PyTuple_Resize(&result, j) != 0)
1863 goto Fail;
1864
1865 Py_DECREF(it);
1866 return result;
1867
1868Fail:
1869 Py_XDECREF(result);
1870 Py_DECREF(it);
1871 return NULL;
1872}

1924/* Iterate over seq. Result depends on the operation:
1925 PY_ITERSEARCH_COUNT: -1 if error, else # of times obj appears in seq.
1926 PY_ITERSEARCH_INDEX: 0-based index of first occurrence of obj in seq;
1927 set ValueError and return -1 if none found; also return -1 on error.
1928 Py_ITERSEARCH_CONTAINS: return 1 if obj in seq, else 0; -1 on error.
1929*/
1930Py_ssize_t
1931_PySequence_IterSearch(PyObject *seq, PyObject *obj, int operation)
1932{
1933 Py_ssize_t n;
1934 int wrapped; /* for PY_ITERSEARCH_INDEX, true iff n wrapped around */
1935 PyObject *it; /* iter(seq) */
1936
1937 if (seq == NULL || obj == NULL) {
1938 null_error();
1939 return -1;
1940 }
1941
1942 it = PyObject_GetIter(seq);
1943 if (it == NULL) {
1944 type_error("argument of type '%.200s' is not iterable", seq);
1945 return -1;
1946 }
1947
1948 n = wrapped = 0;
1949 for (;;) {
1950 int cmp;
1951 PyObject *item = PyIter_Next(it);
1952 if (item == NULL) {
1953 if (PyErr_Occurred())
1954 goto Fail;
1955 break;
1956 }
1957
1958 cmp = PyObject_RichCompareBool(obj, item, Py_EQ);
1959 Py_DECREF(item);
1960 if (cmp < 0)
1961 goto Fail;
1962 if (cmp > 0) {
1963 switch (operation) {
1964 case PY_ITERSEARCH_COUNT:
1965 if (n == PY_SSIZE_T_MAX) {
1966 PyErr_SetString(PyExc_OverflowError,
1967 "count exceeds C integer size");
1968 goto Fail;
1969 }
1970 ++n;
1971 break;
1972
1973 case PY_ITERSEARCH_INDEX:
1974 if (wrapped) {
1975 PyErr_SetString(PyExc_OverflowError,
1976 "index exceeds C integer size");
1977 goto Fail;
1978 }
1979 goto Done;
1980
1981 case PY_ITERSEARCH_CONTAINS:
1982 n = 1;
1983 goto Done;
1984
1985 default:
1986 assert(!"unknown operation");
1987 }
1988 }
1989
1990 if (operation == PY_ITERSEARCH_INDEX) {
1991 if (n == PY_SSIZE_T_MAX)
1992 wrapped = 1;
1993 ++n;
1994 }
1995 }
1996
1997 if (operation != PY_ITERSEARCH_INDEX)
1998 goto Done;
1999
2000 PyErr_SetString(PyExc_ValueError,
2001 "sequence.index(x): x not in sequence");
2002 /* fall into failure code */
2003Fail:
2004 n = -1;
2005 /* fall through */
2006Done:
2007 Py_DECREF(it);
2008 return n;
2009
2010}

2202/* abstract_get_bases() has logically 4 return states:
2203 *
2204 * 1. getattr(cls, '__bases__') could raise an AttributeError
2205 * 2. getattr(cls, '__bases__') could raise some other exception
2206 * 3. getattr(cls, '__bases__') could return a tuple
2207 * 4. getattr(cls, '__bases__') could return something other than a tuple
2208 *
2209 * Only state #3 is a non-error state and only it returns a non-NULL object
2210 * (it returns the retrieved tuple).
2211 *
2212 * Any raised AttributeErrors are masked by clearing the exception and
2213 * returning NULL. If an object other than a tuple comes out of __bases__,
2214 * then again, the return value is NULL. So yes, these two situations
2215 * produce exactly the same results: NULL is returned and no error is set.
2216 *
2217 * If some exception other than AttributeError is raised, then NULL is also
2218 * returned, but the exception is not cleared. That's because we want the
2219 * exception to be propagated along.
2220 *
2221 * Callers are expected to test for PyErr_Occurred() when the return value
2222 * is NULL to decide whether a valid exception should be propagated or not.
2223 * When there's no exception to propagate, it's customary for the caller to
2224 * set a TypeError.
2225 */
2226static PyObject *
2227abstract_get_bases(PyObject *cls)
2228{
2229 _Py_IDENTIFIER(__bases__);
2230 PyObject *bases;
2231
2232 Py_ALLOW_RECURSION
2233 bases = _PyObject_GetAttrId(cls, &PyId___bases__);
2234 Py_END_ALLOW_RECURSION
2235 if (bases == NULL) {
2236 if (PyErr_ExceptionMatches(PyExc_AttributeError))
2237 PyErr_Clear();
2238 return NULL;
2239 }
2240 if (!PyTuple_Check(bases)) {
2241 Py_DECREF(bases);
2242 return NULL;
2243 }
2244 return bases;
2245}

2265 if (n == 0) {
2266 Py_DECREF(bases);
2267 return 0;
2268 }
2269 /* Avoid recursivity in the single inheritance case */
2270 if (n == 1) {
2271 derived = PyTuple_GET_ITEM(bases, 0);
2272 Py_DECREF(bases);
2273 continue;
2274 }
2275 for (i = 0; i < n; i++) {
2276 r = abstract_issubclass(PyTuple_GET_ITEM(bases, i), cls);
2277 if (r != 0)
2278 break;
2279 }
2280 Py_DECREF(bases);
2281 return r;
2282 }

2348{

2536/*
2537 * Flatten a sequence of bytes() objects into a C array of
2538 * NULL terminated string pointers with a NULL char* terminating the array.
2539 * (ie: an argv or env list)
2540 *
2541 * Memory allocated for the returned list is allocated using PyMem_Malloc()
2542 * and MUST be freed by _Py_FreeCharPArray().
2543 */
2544char *const *
2545_PySequence_BytesToCharpArray(PyObject* self)
2546{
2547 char **array;
2548 Py_ssize_t i, argc;
2549 PyObject *item = NULL;
2550 Py_ssize_t size;
2551
2552 argc = PySequence_Size(self);
2553 if (argc == -1)
2554 return NULL;
2555
2556 assert(argc >= 0);
2557
2558 if ((size_t)argc > (PY_SSIZE_T_MAX-sizeof(char *)) / sizeof(char *)) {
2559 PyErr_NoMemory();
2560 return NULL;
2561 }
2562
2563 array = PyMem_Malloc((argc + 1) * sizeof(char *));
2564 if (array == NULL) {
2565 PyErr_NoMemory();
2566 return NULL;
2567 }
2568 for (i = 0; i < argc; ++i) {
2569 char *data;
2570 item = PySequence_GetItem(self, i);
2571 if (item == NULL) {
2572 /* NULL terminate before freeing. */
2573 array[i] = NULL;
2574 goto fail;
2575 }
2576 /* check for embedded null bytes */
2577 if (PyBytes_AsStringAndSize(item, &data, NULL) < 0) {
2578 /* NULL terminate before freeing. */
2579 array[i] = NULL;
2580 goto fail;
2581 }
2582 size = PyBytes_GET_SIZE(item) + 1;
2583 array[i] = PyMem_Malloc(size);
2584 if (!array[i]) {
2585 PyErr_NoMemory();
2586 goto fail;
2587 }
2588 memcpy(array[i], data, size);
2589 Py_DECREF(item);
2590 }
2591 array[argc] = NULL;
2592
2593 return array;
2594
2595fail:
2596 Py_XDECREF(item);
2597 _Py_FreeCharPArray(array);
2598 return NULL;
2599}
