7/* Free list for method objects to safe malloc/free overhead
8 * The m_self element is used to chain the objects.
9 */
10static PyCFunctionObject *free_list = NULL;
11static int numfree = 0;
12#ifndef PyCFunction_MAXFREELIST
13#define PyCFunction_MAXFREELIST 256
14#endif
15
16/* undefine macro trampoline to PyCFunction_NewEx */
17#undef PyCFunction_New

19PyAPI_FUNC(PyObject *)
20PyCFunction_New(PyMethodDef *ml, PyObject *self)
21{
22 return PyCFunction_NewEx(ml, self, NULL);
23}
24
25PyObject *
26PyCFunction_NewEx(PyMethodDef *ml, PyObject *self, PyObject *module)
27{
28 PyCFunctionObject *op;
29 op = free_list;
30 if (op != NULL) {
31 free_list = (PyCFunctionObject *)(op->m_self);
32 (void)PyObject_INIT(op, &PyCFunction_Type);
33 numfree--;
34 }
35 else {
36 op = PyObject_GC_New(PyCFunctionObject, &PyCFunction_Type);
37 if (op == NULL)
38 return NULL;
39 }
40 op->m_weakreflist = NULL;
41 op->m_ml = ml;
42 Py_XINCREF(self);
43 op->m_self = self;
44 Py_XINCREF(module);
45 op->m_module = module;
46 _PyObject_GC_TRACK(op);
47 return (PyObject *)op;
48}

82static void
83meth_dealloc(PyCFunctionObject *m)
84{
85 _PyObject_GC_UNTRACK(m);
86 if (m->m_weakreflist != NULL) {
87 PyObject_ClearWeakRefs((PyObject*) m);
88 }
89 Py_XDECREF(m->m_self);
90 Py_XDECREF(m->m_module);
91 if (numfree < PyCFunction_MAXFREELIST) {
92 m->m_self = (PyObject *)free_list;
93 free_list = m;
94 numfree++;
95 }
96 else {
97 PyObject_GC_Del(m);
98 }
99}

101static PyObject *
102meth_reduce(PyCFunctionObject *m)
103{
104 PyObject *builtins;
105 PyObject *getattr;
106 _Py_IDENTIFIER(getattr);
107
108 if (m->m_self == NULL || PyModule_Check(m->m_self))
109 return PyUnicode_FromString(m->m_ml->ml_name);
110
111 builtins = PyEval_GetBuiltins();
112 getattr = _PyDict_GetItemId(builtins, &PyId_getattr);
113 return Py_BuildValue("O(Os)", getattr, m->m_self, m->m_ml->ml_name);
114}

139static PyObject *
140meth_get__qualname__(PyCFunctionObject *m, void *closure)
141{
142 /* If __self__ is a module or NULL, return m.__name__
143 (e.g. len.__qualname__ == 'len')
144
145 If __self__ is a type, return m.__self__.__qualname__ + '.' + m.__name__
146 (e.g. dict.fromkeys.__qualname__ == 'dict.fromkeys')
147
148 Otherwise return type(m.__self__).__qualname__ + '.' + m.__name__
149 (e.g. [].append.__qualname__ == 'list.append') */
150 PyObject *type, *type_qualname, *res;
151 _Py_IDENTIFIER(__qualname__);
152
153 if (m->m_self == NULL || PyModule_Check(m->m_self))
154 return PyUnicode_FromString(m->m_ml->ml_name);
155
156 type = PyType_Check(m->m_self) ? m->m_self : (PyObject*)Py_TYPE(m->m_self);
157
158 type_qualname = _PyObject_GetAttrId(type, &PyId___qualname__);
159 if (type_qualname == NULL)
160 return NULL;
161
162 if (!PyUnicode_Check(type_qualname)) {
163 PyErr_SetString(PyExc_TypeError, "<method>.__class__."
164 "__qualname__ is not a unicode object");
165 Py_XDECREF(type_qualname);
166 return NULL;
167 }
168
169 res = PyUnicode_FromFormat("%S.%s", type_qualname, m->m_ml->ml_name);
170 Py_DECREF(type_qualname);
171 return res;
172}

183meth_get__self__(PyCFunctionObject *m, void *closure)
184{

186

210static PyObject *
211meth_repr(PyCFunctionObject *m)
212{
213 if (m->m_self == NULL || PyModule_Check(m->m_self))
214 return PyUnicode_FromFormat("<built-in function %s>",
215 m->m_ml->ml_name);
216 return PyUnicode_FromFormat("<built-in method %s of %s object at %p>",
217 m->m_ml->ml_name,
218 m->m_self->ob_type->tp_name,
219 m->m_self);

222static PyObject *
223meth_richcompare(PyObject *self, PyObject *other, int op)
224{
225 PyCFunctionObject *a, *b;
226 PyObject *res;
227 int eq;
228
229 if ((op != Py_EQ && op != Py_NE) ||
230 !PyCFunction_Check(self) ||
231 !PyCFunction_Check(other))
232 {
233 Py_RETURN_NOTIMPLEMENTED;
234 }
235 a = (PyCFunctionObject *)self;
236 b = (PyCFunctionObject *)other;
237 eq = a->m_self == b->m_self;
238 if (eq)
239 eq = a->m_ml->ml_meth == b->m_ml->ml_meth;
240 if (op == Py_EQ)
241 res = eq ? Py_True : Py_False;
242 else
243 res = eq ? Py_False : Py_True;
244 Py_INCREF(res);
245 return res;
246}

248static Py_hash_t
249meth_hash(PyCFunctionObject *a)
250{
251 Py_hash_t x, y;
252 if (a->m_self == NULL)
253 x = 0;
254 else {
255 x = PyObject_Hash(a->m_self);
256 if (x == -1)
257 return -1;
258 }
259 y = _Py_HashPointer((void*)(a->m_ml->ml_meth));
260 if (y == -1)
261 return -1;
262 x ^= y;
263 if (x == -1)
264 x = -2;
265 return x;
266}

304/* Clear out the free list */
305
306int
307PyCFunction_ClearFreeList(void)
308{
309 int freelist_size = numfree;
310
311 while (free_list) {
312 PyCFunctionObject *v = free_list;
313 free_list = (PyCFunctionObject *)(v->m_self);
314 PyObject_GC_Del(v);
315 numfree--;
316 }
317 assert(numfree == 0);
318 return freelist_size;
319}
