19/* Ensure ob_item has room for at least newsize elements, and set
20 * ob_size to newsize. If newsize > ob_size on entry, the content
21 * of the new slots at exit is undefined heap trash; it's the caller's
22 * responsibility to overwrite them with sane values.
23 * The number of allocated elements may grow, shrink, or stay the same.
24 * Failure is impossible if newsize <= self.allocated on entry, although
25 * that partly relies on an assumption that the system realloc() never
26 * fails when passed a number of bytes <= the number of bytes last
27 * allocated (the C standard doesn't guarantee this, but it's hard to
28 * imagine a realloc implementation where it wouldn't be true).
29 * Note that self->ob_item may change, and even if newsize is less
30 * than ob_size on entry.
31 */
32static int
33list_resize(PyListObject *self, Py_ssize_t newsize)
34{
35 PyObject **items;
36 size_t new_allocated, num_allocated_bytes;
37 Py_ssize_t allocated = self->allocated;
38
39 /* Bypass realloc() when a previous overallocation is large enough
40 to accommodate the newsize. If the newsize falls lower than half
41 the allocated size, then proceed with the realloc() to shrink the list.
42 */
43 if (allocated >= newsize && newsize >= (allocated >> 1)) {
44 assert(self->ob_item != NULL || newsize == 0);
45 Py_SIZE(self) = newsize;
46 return 0;
47 }
48
49 /* This over-allocates proportional to the list size, making room
50 * for additional growth. The over-allocation is mild, but is
51 * enough to give linear-time amortized behavior over a long
52 * sequence of appends() in the presence of a poorly-performing
53 * system realloc().
54 * The growth pattern is: 0, 4, 8, 16, 25, 35, 46, 58, 72, 88, ...
55 * Note: new_allocated won't overflow because the largest possible value
56 * is PY_SSIZE_T_MAX * (9 / 8) + 6 which always fits in a size_t.
57 */
58 new_allocated = (size_t)newsize + (newsize >> 3) + (newsize < 9 ? 3 : 6);
59 if (new_allocated > (size_t)PY_SSIZE_T_MAX / sizeof(PyObject *)) {
60 PyErr_NoMemory();
61 return -1;
62 }
63
64 if (newsize == 0)
65 new_allocated = 0;
66 num_allocated_bytes = new_allocated * sizeof(PyObject *);
67 items = (PyObject **)PyMem_Realloc(self->ob_item, num_allocated_bytes);
68 if (items == NULL) {
69 PyErr_NoMemory();
70 return -1;
71 }
72 self->ob_item = items;
73 Py_SIZE(self) = newsize;
74 self->allocated = new_allocated;
75 return 0;
76}

241static int
242ins1(PyListObject *self, Py_ssize_t where, PyObject *v)
243{
244 Py_ssize_t i, n = Py_SIZE(self);
245 PyObject **items;
246 if (v == NULL) {
247 PyErr_BadInternalCall();
248 return -1;
249 }
250 if (n == PY_SSIZE_T_MAX) {
251 PyErr_SetString(PyExc_OverflowError,
252 "cannot add more objects to list");
253 return -1;
254 }
255
256 if (list_resize(self, n+1) < 0)
257 return -1;
258
259 if (where < 0) {
260 where += n;
261 if (where < 0)
262 where = 0;
263 }
264 if (where > n)
265 where = n;
266 items = self->ob_item;
267 for (i = n; --i >= where; )
268 items[i+1] = items[i];
269 Py_INCREF(v);
270 items[where] = v;
271 return 0;

315static void
316list_dealloc(PyListObject *op)
317{
318 Py_ssize_t i;
319 PyObject_GC_UnTrack(op);
320 Py_TRASHCAN_SAFE_BEGIN(op)
321 if (op->ob_item != NULL) {
322 /* Do it backwards, for Christian Tismer.
323 There's a simple test case where somehow this reduces
324 thrashing when a *very* large list is created and
325 immediately deleted. */
326 i = Py_SIZE(op);
327 while (--i >= 0) {
328 Py_XDECREF(op->ob_item[i]);
329 }
330 PyMem_FREE(op->ob_item);
331 }
332 if (numfree < PyList_MAXFREELIST && PyList_CheckExact(op))
333 free_list[numfree++] = op;
334 else
335 Py_TYPE(op)->tp_free((PyObject *)op);
336 Py_TRASHCAN_SAFE_END(op)
337}

339static PyObject *
340list_repr(PyListObject *v)
341{
342 Py_ssize_t i;
343 PyObject *s;
344 _PyUnicodeWriter writer;
345
346 if (Py_SIZE(v) == 0) {
347 return PyUnicode_FromString("[]");
348 }
349
350 i = Py_ReprEnter((PyObject*)v);
351 if (i != 0) {
352 return i > 0 ? PyUnicode_FromString("[...]") : NULL;
353 }
354
355 _PyUnicodeWriter_Init(&writer);
356 writer.overallocate = 1;
357 /* "[" + "1" + ", 2" * (len - 1) + "]" */
358 writer.min_length = 1 + 1 + (2 + 1) * (Py_SIZE(v) - 1) + 1;
359
360 if (_PyUnicodeWriter_WriteChar(&writer, '[') < 0)
361 goto error;
362
363 /* Do repr() on each element. Note that this may mutate the list,
364 so must refetch the list size on each iteration. */
365 for (i = 0; i < Py_SIZE(v); ++i) {
366 if (i > 0) {
367 if (_PyUnicodeWriter_WriteASCIIString(&writer, ", ", 2) < 0)
368 goto error;
369 }
370
371 if (Py_EnterRecursiveCall(" while getting the repr of a list"))
372 goto error;
373 s = PyObject_Repr(v->ob_item[i]);
374 Py_LeaveRecursiveCall();
375 if (s == NULL)
376 goto error;
377
378 if (_PyUnicodeWriter_WriteStr(&writer, s) < 0) {
379 Py_DECREF(s);
380 goto error;
381 }
382 Py_DECREF(s);
383 }
384
385 writer.overallocate = 0;
386 if (_PyUnicodeWriter_WriteChar(&writer, ']') < 0)
387 goto error;
388
389 Py_ReprLeave((PyObject *)v);
390 return _PyUnicodeWriter_Finish(&writer);
391
392error:
393 _PyUnicodeWriter_Dealloc(&writer);
394 Py_ReprLeave((PyObject *)v);
395 return NULL;
396}

551static int
552_list_clear(PyListObject *a)
553{
554 Py_ssize_t i;
555 PyObject **item = a->ob_item;
556 if (item != NULL) {
557 /* Because XDECREF can recursively invoke operations on
558 this list, we make it empty first. */
559 i = Py_SIZE(a);
560 Py_SIZE(a) = 0;
561 a->ob_item = NULL;
562 a->allocated = 0;
563 while (--i >= 0) {
564 Py_XDECREF(item[i]);
565 }
566 PyMem_FREE(item);
567 }
568 /* Never fails; the return value can be ignored.
569 Note that there is no guarantee that the list is actually empty
570 at this point, because XDECREF may have populated it again! */
571 return 0;
572}

574/* a[ilow:ihigh] = v if v != NULL.
575 * del a[ilow:ihigh] if v == NULL.
576 *
577 * Special speed gimmick: when v is NULL and ihigh - ilow <= 8, it's
578 * guaranteed the call cannot fail.
579 */
580static int
581list_ass_slice(PyListObject *a, Py_ssize_t ilow, Py_ssize_t ihigh, PyObject *v)
582{
583 /* Because [X]DECREF can recursively invoke list operations on
584 this list, we must postpone all [X]DECREF activity until
585 after the list is back in its canonical shape. Therefore
586 we must allocate an additional array, 'recycle', into which
587 we temporarily copy the items that are deleted from the
588 list. :-( */
589 PyObject *recycle_on_stack[8];
590 PyObject **recycle = recycle_on_stack; /* will allocate more if needed */
591 PyObject **item;
592 PyObject **vitem = NULL;
593 PyObject *v_as_SF = NULL; /* PySequence_Fast(v) */
594 Py_ssize_t n; /* # of elements in replacement list */
595 Py_ssize_t norig; /* # of elements in list getting replaced */
596 Py_ssize_t d; /* Change in size */
597 Py_ssize_t k;
598 size_t s;
599 int result = -1; /* guilty until proved innocent */
600#define b ((PyListObject *)v)
601 if (v == NULL)
602 n = 0;
603 else {
604 if (a == b) {
605 /* Special case "a[i:j] = a" -- copy b first */
606 v = list_slice(b, 0, Py_SIZE(b));
607 if (v == NULL)
608 return result;
609 result = list_ass_slice(a, ilow, ihigh, v);
610 Py_DECREF(v);
611 return result;
612 }
613 v_as_SF = PySequence_Fast(v, "can only assign an iterable");
614 if(v_as_SF == NULL)
615 goto Error;
616 n = PySequence_Fast_GET_SIZE(v_as_SF);
617 vitem = PySequence_Fast_ITEMS(v_as_SF);
618 }
619 if (ilow < 0)
620 ilow = 0;
621 else if (ilow > Py_SIZE(a))
622 ilow = Py_SIZE(a);
623
624 if (ihigh < ilow)
625 ihigh = ilow;
626 else if (ihigh > Py_SIZE(a))
627 ihigh = Py_SIZE(a);
628
629 norig = ihigh - ilow;
630 assert(norig >= 0);
631 d = n - norig;
632 if (Py_SIZE(a) + d == 0) {
633 Py_XDECREF(v_as_SF);
634 return _list_clear(a);
635 }
636 item = a->ob_item;
637 /* recycle the items that we are about to remove */
638 s = norig * sizeof(PyObject *);
639 /* If norig == 0, item might be NULL, in which case we may not memcpy from it. */
640 if (s) {
641 if (s > sizeof(recycle_on_stack)) {
642 recycle = (PyObject **)PyMem_MALLOC(s);
643 if (recycle == NULL) {
644 PyErr_NoMemory();
645 goto Error;
646 }
647 }
648 memcpy(recycle, &item[ilow], s);
649 }
650
651 if (d < 0) { /* Delete -d items */
652 Py_ssize_t tail;
653 tail = (Py_SIZE(a) - ihigh) * sizeof(PyObject *);
654 memmove(&item[ihigh+d], &item[ihigh], tail);
655 if (list_resize(a, Py_SIZE(a) + d) < 0) {
656 memmove(&item[ihigh], &item[ihigh+d], tail);
657 memcpy(&item[ilow], recycle, s);
658 goto Error;
659 }
660 item = a->ob_item;
661 }
662 else if (d > 0) { /* Insert d items */
663 k = Py_SIZE(a);
664 if (list_resize(a, k+d) < 0)
665 goto Error;
666 item = a->ob_item;
667 memmove(&item[ihigh+d], &item[ihigh],
668 (k - ihigh)*sizeof(PyObject *));
669 }
670 for (k = 0; k < n; k++, ilow++) {
671 PyObject *w = vitem[k];
672 Py_XINCREF(w);
673 item[ilow] = w;
674 }
675 for (k = norig - 1; k >= 0; --k)
676 Py_XDECREF(recycle[k]);
677 result = 0;
678 Error:
679 if (recycle != recycle_on_stack)
680 PyMem_FREE(recycle);
681 Py_XDECREF(v_as_SF);
682 return result;
683#undef b
684}

696static PyObject *
697list_inplace_repeat(PyListObject *self, Py_ssize_t n)
698{
699 PyObject **items;
700 Py_ssize_t size, i, j, p;
701
702
703 size = PyList_GET_SIZE(self);
704 if (size == 0 || n == 1) {
705 Py_INCREF(self);
706 return (PyObject *)self;
707 }
708
709 if (n < 1) {
710 (void)_list_clear(self);
711 Py_INCREF(self);
712 return (PyObject *)self;
713 }
714
715 if (size > PY_SSIZE_T_MAX / n) {
716 return PyErr_NoMemory();
717 }
718
719 if (list_resize(self, size*n) < 0)
720 return NULL;
721
722 p = size;
723 items = self->ob_item;
724 for (i = 1; i < n; i++) { /* Start counting at 1, not 0 */
725 for (j = 0; j < size; j++) {
726 PyObject *o = items[j];
727 Py_INCREF(o);
728 items[p++] = o;
729 }
730 }
731 Py_INCREF(self);
732 return (PyObject *)self;
733}

1104/* binarysort is the best method for sorting small arrays: it does
1105 few compares, but can do data movement quadratic in the number of
1106 elements.

1108 binary insertion. This sort is stable.
1109 On entry, must have lo <= start <= hi, and that [lo, start) is already
1110 sorted (pass start == lo if you don't know!).
1111 If islt() complains return -1, else 0.
1112 Even in case of error, the output slice will be some permutation of
1113 the input (nothing is lost or duplicated).

1115static int
1116binarysort(sortslice lo, PyObject **hi, PyObject **start)
1117{

1122 assert(lo.keys <= start && start <= hi);
1123 /* assert [lo, start) is sorted */

1125 ++start;
1126 for (; start < hi; ++start) {
1127 /* set l to where *start belongs */
1128 l = lo.keys;

1130 pivot = *r;
1131 /* Invariants:
1132 * pivot >= all in [lo, l).
1133 * pivot < all in [r, start).
1134 * The second is vacuously true at the start.
1135 */

1148 first slot after them -- that's why this sort is stable.

1151 we're not usually moving many slots. */

1476 ms->alloced = (list_size + 1) / 2;

1479 ms->temparray, but we also need to store the values. Hence,
1480 ms->alloced is capped at half of MERGESTATE_TEMP_SIZE. */
1481 if (MERGESTATE_TEMP_SIZE / 2 < ms->alloced)
1482 ms->alloced = MERGESTATE_TEMP_SIZE / 2;
1483 ms->a.values = &ms->temparray[ms->alloced];
1484 }
1485 else {
1486 ms->alloced = MERGESTATE_TEMP_SIZE;
1487 ms->a.values = NULL;
1488 }
1489 ms->a.keys = ms->temparray;
1490 ms->n = 0;
1491 ms->min_gallop = MIN_GALLOP;
1492}

1506/* Ensure enough temp memory for 'need' array slots is available.
1507 * Returns 0 on success and -1 if the memory can't be gotten.
1508 */
1509static int
1510merge_getmem(MergeState *ms, Py_ssize_t need)
1511{
1512 int multiplier;
1513
1514 assert(ms != NULL);
1515 if (need <= ms->alloced)
1516 return 0;
1517
1518 multiplier = ms->a.values != NULL ? 2 : 1;
1519
1520 /* Don't realloc! That can cost cycles to copy the old data, but
1521 * we don't care what's in the block.
1522 */
1523 merge_freemem(ms);
1524 if ((size_t)need > PY_SSIZE_T_MAX / sizeof(PyObject*) / multiplier) {
1525 PyErr_NoMemory();
1526 return -1;
1527 }
1528 ms->a.keys = (PyObject**)PyMem_Malloc(multiplier * need
1529 * sizeof(PyObject *));

1531 ms->alloced = need;
1532 if (ms->a.values != NULL)
1533 ms->a.values = &ms->a.keys[need];
1534 return 0;
1535 }
1536 PyErr_NoMemory();
1537 return -1;
1538}

1540 merge_getmem(MS, NEED))

1542/* Merge the na elements starting at ssa with the nb elements starting at
1543 * ssb.keys = ssa.keys + na in a stable way, in-place. na and nb must be > 0.
1544 * Must also have that ssa.keys[na-1] belongs at the end of the merge, and
1545 * should have na <= nb. See listsort.txt for more info. Return 0 if
1546 * successful, -1 if error.
1547 */
1548static Py_ssize_t
1549merge_lo(MergeState *ms, sortslice ssa, Py_ssize_t na,
1550 sortslice ssb, Py_ssize_t nb)
1551{
1552 Py_ssize_t k;
1553 sortslice dest;
1554 int result = -1; /* guilty until proved innocent */
1555 Py_ssize_t min_gallop;
1556
1557 assert(ms && ssa.keys && ssb.keys && na > 0 && nb > 0);
1558 assert(ssa.keys + na == ssb.keys);
1559 if (MERGE_GETMEM(ms, na) < 0)
1560 return -1;
1561 sortslice_memcpy(&ms->a, 0, &ssa, 0, na);
1562 dest = ssa;
1563 ssa = ms->a;
1564
1565 sortslice_copy_incr(&dest, &ssb);
1566 --nb;
1567 if (nb == 0)
1568 goto Succeed;
1569 if (na == 1)
1570 goto CopyB;
1571
1572 min_gallop = ms->min_gallop;
1573 for (;;) {
1574 Py_ssize_t acount = 0; /* # of times A won in a row */
1575 Py_ssize_t bcount = 0; /* # of times B won in a row */
1576
1577 /* Do the straightforward thing until (if ever) one run
1578 * appears to win consistently.
1579 */
1580 for (;;) {
1581 assert(na > 1 && nb > 0);
1582 k = ISLT(ssb.keys[0], ssa.keys[0]);
1583 if (k) {
1584 if (k < 0)
1585 goto Fail;
1586 sortslice_copy_incr(&dest, &ssb);
1587 ++bcount;
1588 acount = 0;
1589 --nb;
1590 if (nb == 0)
1591 goto Succeed;
1592 if (bcount >= min_gallop)
1593 break;
1594 }
1595 else {
1596 sortslice_copy_incr(&dest, &ssa);
1597 ++acount;
1598 bcount = 0;
1599 --na;
1600 if (na == 1)
1601 goto CopyB;
1602 if (acount >= min_gallop)
1603 break;
1604 }
1605 }
1606
1607 /* One run is winning so consistently that galloping may
1608 * be a huge win. So try that, and continue galloping until
1609 * (if ever) neither run appears to be winning consistently
1610 * anymore.
1611 */
1612 ++min_gallop;
1613 do {
1614 assert(na > 1 && nb > 0);
1615 min_gallop -= min_gallop > 1;
1616 ms->min_gallop = min_gallop;
1617 k = gallop_right(ssb.keys[0], ssa.keys, na, 0);
1618 acount = k;
1619 if (k) {
1620 if (k < 0)
1621 goto Fail;
1622 sortslice_memcpy(&dest, 0, &ssa, 0, k);
1623 sortslice_advance(&dest, k);
1624 sortslice_advance(&ssa, k);
1625 na -= k;
1626 if (na == 1)
1627 goto CopyB;
1628 /* na==0 is impossible now if the comparison
1629 * function is consistent, but we can't assume
1630 * that it is.
1631 */
1632 if (na == 0)
1633 goto Succeed;
1634 }
1635 sortslice_copy_incr(&dest, &ssb);
1636 --nb;
1637 if (nb == 0)
1638 goto Succeed;
1639
1640 k = gallop_left(ssa.keys[0], ssb.keys, nb, 0);
1641 bcount = k;
1642 if (k) {
1643 if (k < 0)
1644 goto Fail;
1645 sortslice_memmove(&dest, 0, &ssb, 0, k);
1646 sortslice_advance(&dest, k);
1647 sortslice_advance(&ssb, k);
1648 nb -= k;
1649 if (nb == 0)
1650 goto Succeed;
1651 }
1652 sortslice_copy_incr(&dest, &ssa);
1653 --na;
1654 if (na == 1)
1655 goto CopyB;
1656 } while (acount >= MIN_GALLOP || bcount >= MIN_GALLOP);
1657 ++min_gallop; /* penalize it for leaving galloping mode */
1658 ms->min_gallop = min_gallop;
1659 }
1660Succeed:
1661 result = 0;
1662Fail:
1663 if (na)
1664 sortslice_memcpy(&dest, 0, &ssa, 0, na);
1665 return result;
1666CopyB:
1667 assert(na == 1 && nb > 0);
1668 /* The last element of ssa belongs at the end of the merge. */
1669 sortslice_memmove(&dest, 0, &ssb, 0, nb);
1670 sortslice_copy(&dest, nb, &ssa, 0);
1671 return 0;
1672}

1674/* Merge the na elements starting at pa with the nb elements starting at
1675 * ssb.keys = ssa.keys + na in a stable way, in-place. na and nb must be > 0.
1676 * Must also have that ssa.keys[na-1] belongs at the end of the merge, and
1677 * should have na >= nb. See listsort.txt for more info. Return 0 if
1678 * successful, -1 if error.
1679 */
1680static Py_ssize_t
1681merge_hi(MergeState *ms, sortslice ssa, Py_ssize_t na,
1682 sortslice ssb, Py_ssize_t nb)
1683{
1684 Py_ssize_t k;
1685 sortslice dest, basea, baseb;
1686 int result = -1; /* guilty until proved innocent */
1687 Py_ssize_t min_gallop;
1688
1689 assert(ms && ssa.keys && ssb.keys && na > 0 && nb > 0);
1690 assert(ssa.keys + na == ssb.keys);
1691 if (MERGE_GETMEM(ms, nb) < 0)
1692 return -1;
1693 dest = ssb;
1694 sortslice_advance(&dest, nb-1);
1695 sortslice_memcpy(&ms->a, 0, &ssb, 0, nb);
1696 basea = ssa;
1697 baseb = ms->a;
1698 ssb.keys = ms->a.keys + nb - 1;
1699 if (ssb.values != NULL)
1700 ssb.values = ms->a.values + nb - 1;
1701 sortslice_advance(&ssa, na - 1);
1702
1703 sortslice_copy_decr(&dest, &ssa);
1704 --na;
1705 if (na == 0)
1706 goto Succeed;
1707 if (nb == 1)
1708 goto CopyA;
1709
1710 min_gallop = ms->min_gallop;
1711 for (;;) {
1712 Py_ssize_t acount = 0; /* # of times A won in a row */
1713 Py_ssize_t bcount = 0; /* # of times B won in a row */
1714
1715 /* Do the straightforward thing until (if ever) one run
1716 * appears to win consistently.
1717 */
1718 for (;;) {
1719 assert(na > 0 && nb > 1);
1720 k = ISLT(ssb.keys[0], ssa.keys[0]);
1721 if (k) {
1722 if (k < 0)
1723 goto Fail;
1724 sortslice_copy_decr(&dest, &ssa);
1725 ++acount;
1726 bcount = 0;
1727 --na;
1728 if (na == 0)
1729 goto Succeed;
1730 if (acount >= min_gallop)
1731 break;
1732 }
1733 else {
1734 sortslice_copy_decr(&dest, &ssb);
1735 ++bcount;
1736 acount = 0;
1737 --nb;
1738 if (nb == 1)
1739 goto CopyA;
1740 if (bcount >= min_gallop)
1741 break;
1742 }
1743 }
1744
1745 /* One run is winning so consistently that galloping may
1746 * be a huge win. So try that, and continue galloping until
1747 * (if ever) neither run appears to be winning consistently
1748 * anymore.
1749 */
1750 ++min_gallop;
1751 do {
1752 assert(na > 0 && nb > 1);
1753 min_gallop -= min_gallop > 1;
1754 ms->min_gallop = min_gallop;
1755 k = gallop_right(ssb.keys[0], basea.keys, na, na-1);
1756 if (k < 0)
1757 goto Fail;
1758 k = na - k;
1759 acount = k;
1760 if (k) {
1761 sortslice_advance(&dest, -k);
1762 sortslice_advance(&ssa, -k);
1763 sortslice_memmove(&dest, 1, &ssa, 1, k);
1764 na -= k;
1765 if (na == 0)
1766 goto Succeed;
1767 }
1768 sortslice_copy_decr(&dest, &ssb);
1769 --nb;
1770 if (nb == 1)
1771 goto CopyA;
1772
1773 k = gallop_left(ssa.keys[0], baseb.keys, nb, nb-1);
1774 if (k < 0)
1775 goto Fail;
1776 k = nb - k;
1777 bcount = k;
1778 if (k) {
1779 sortslice_advance(&dest, -k);
1780 sortslice_advance(&ssb, -k);
1781 sortslice_memcpy(&dest, 1, &ssb, 1, k);
1782 nb -= k;
1783 if (nb == 1)
1784 goto CopyA;
1785 /* nb==0 is impossible now if the comparison
1786 * function is consistent, but we can't assume
1787 * that it is.
1788 */
1789 if (nb == 0)
1790 goto Succeed;
1791 }
1792 sortslice_copy_decr(&dest, &ssa);
1793 --na;
1794 if (na == 0)
1795 goto Succeed;
1796 } while (acount >= MIN_GALLOP || bcount >= MIN_GALLOP);
1797 ++min_gallop; /* penalize it for leaving galloping mode */
1798 ms->min_gallop = min_gallop;
1799 }
1800Succeed:
1801 result = 0;
1802Fail:
1803 if (nb)
1804 sortslice_memcpy(&dest, -(nb-1), &baseb, 0, nb);
1805 return result;
1806CopyA:
1807 assert(nb == 1 && na > 0);
1808 /* The first element of ssb belongs at the front of the merge. */
1809 sortslice_memmove(&dest, 1-na, &ssa, 1-na, na);
1810 sortslice_advance(&dest, -na);
1811 sortslice_advance(&ssa, -na);
1812 sortslice_copy(&dest, 0, &ssb, 0);
1813 return 0;
1814}

1816/* Merge the two runs at stack indices i and i+1.
1817 * Returns 0 on success, -1 on error.
1818 */
1819static Py_ssize_t
1820merge_at(MergeState *ms, Py_ssize_t i)
1821{
1822 sortslice ssa, ssb;
1823 Py_ssize_t na, nb;
1824 Py_ssize_t k;
1825
1826 assert(ms != NULL);
1827 assert(ms->n >= 2);
1828 assert(i >= 0);
1829 assert(i == ms->n - 2 || i == ms->n - 3);
1830
1831 ssa = ms->pending[i].base;
1832 na = ms->pending[i].len;
1833 ssb = ms->pending[i+1].base;
1834 nb = ms->pending[i+1].len;
1835 assert(na > 0 && nb > 0);
1836 assert(ssa.keys + na == ssb.keys);
1837
1838 /* Record the length of the combined runs; if i is the 3rd-last
1839 * run now, also slide over the last run (which isn't involved
1840 * in this merge). The current run i+1 goes away in any case.
1841 */
1842 ms->pending[i].len = na + nb;
1843 if (i == ms->n - 3)
1844 ms->pending[i+1] = ms->pending[i+2];
1845 --ms->n;
1846
1847 /* Where does b start in a? Elements in a before that can be
1848 * ignored (already in place).
1849 */
1850 k = gallop_right(*ssb.keys, ssa.keys, na, 0);
1851 if (k < 0)
1852 return -1;
1853 sortslice_advance(&ssa, k);
1854 na -= k;
1855 if (na == 0)
1856 return 0;
1857
1858 /* Where does a end in b? Elements in b after that can be
1859 * ignored (already in place).
1860 */
1861 nb = gallop_left(ssa.keys[na-1], ssb.keys, nb, nb-1);
1862 if (nb <= 0)
1863 return nb;
1864
1865 /* Merge what remains of the runs, using a temp array with
1866 * min(na, nb) elements.
1867 */
1868 if (na <= nb)
1869 return merge_lo(ms, ssa, na, ssb, nb);
1870 else
1871 return merge_hi(ms, ssa, na, ssb, nb);
1872}

1874/* Examine the stack of runs waiting to be merged, merging adjacent runs
1875 * until the stack invariants are re-established:
1876 *
1877 * 1. len[-3] > len[-2] + len[-1]
1878 * 2. len[-2] > len[-1]
1879 *
1880 * See listsort.txt for more info.
1881 *
1882 * Returns 0 on success, -1 on error.
1883 */
1884static int
1885merge_collapse(MergeState *ms)
1886{
1887 struct s_slice *p = ms->pending;
1888
1889 assert(ms);
1890 while (ms->n > 1) {
1891 Py_ssize_t n = ms->n - 2;
1892 if ((n > 0 && p[n-1].len <= p[n].len + p[n+1].len) ||
1893 (n > 1 && p[n-2].len <= p[n-1].len + p[n].len)) {
1894 if (p[n-1].len < p[n+1].len)
1895 --n;
1896 if (merge_at(ms, n) < 0)
1897 return -1;
1898 }
1899 else if (p[n].len <= p[n+1].len) {
1900 if (merge_at(ms, n) < 0)
1901 return -1;
1902 }
1903 else
1904 break;
1905 }
1906 return 0;
1907}

1909/* Regardless of invariants, merge all runs on the stack until only one
1910 * remains. This is used at the end of the mergesort.
1911 *
1912 * Returns 0 on success, -1 on error.
1913 */
1914static int
1915merge_force_collapse(MergeState *ms)
1916{
1917 struct s_slice *p = ms->pending;
1918
1919 assert(ms);
1920 while (ms->n > 1) {
1921 Py_ssize_t n = ms->n - 2;
1922 if (n > 0 && p[n-1].len < p[n+1].len)
1923 --n;
1924 if (merge_at(ms, n) < 0)
1925 return -1;
1926 }
1927 return 0;
1928}

1930/* Compute a good value for the minimum run length; natural runs shorter
1931 * than this are boosted artificially via binary insertion.
1932 *
1933 * If n < 64, return n (it's too small to bother with fancy stuff).
1934 * Else if n is an exact power of 2, return 32.
1935 * Else return an int k, 32 <= k <= 64, such that n/k is close to, but
1936 * strictly less than, an exact power of 2.
1937 *
1938 * See listsort.txt for more info.
1939 */
1940static Py_ssize_t
1941merge_compute_minrun(Py_ssize_t n)
1942{
1943 Py_ssize_t r = 0; /* becomes 1 if any 1 bits are shifted off */
1944
1945 assert(n >= 0);
1946 while (n >= 64) {
1947 r |= n & 1;
1948 n >>= 1;
1949 }
1950 return n + r;
1951}
