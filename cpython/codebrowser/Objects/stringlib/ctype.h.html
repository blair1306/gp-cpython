52static PyObject*
53stringlib_lower(PyObject *self)
54{
55 PyObject* newobj;
56 newobj = STRINGLIB_NEW(NULL, STRINGLIB_LEN(self));
57 if (!newobj)
58 return NULL;
59 _Py_bytes_lower(STRINGLIB_STR(newobj), STRINGLIB_STR(self),
60 STRINGLIB_LEN(self));
61 return newobj;
62}
