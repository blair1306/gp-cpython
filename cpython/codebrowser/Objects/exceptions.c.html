31static PyObject *
32BaseException_new(PyTypeObject *type, PyObject *args, PyObject *kwds)
33{
34 PyBaseExceptionObject *self;
35
36 self = (PyBaseExceptionObject *)type->tp_alloc(type, 0);
37 if (!self)
38 return NULL;
39 /* the dict is created on the fly in PyObject_GenericSetAttr */
40 self->dict = NULL;
41 self->traceback = self->cause = self->context = NULL;
42 self->suppress_context = 0;
43
44 if (args) {
45 self->args = args;
46 Py_INCREF(args);
47 return (PyObject *)self;
48 }
49
50 self->args = PyTuple_New(0);
51 if (!self->args) {
52 Py_DECREF(self);
53 return NULL;
54 }
55
56 return (PyObject *)self;
57}

116{
117 const char *name;

137/*
138 * Needed for backward compatibility, since exceptions used to store
139 * all their attributes in the __dict__. Code is taken from cPickle's
140 * load_build function.
141 */
142static PyObject *
143BaseException_setstate(PyObject *self, PyObject *state)
144{
145 PyObject *d_key, *d_value;
146 Py_ssize_t i = 0;
147
148 if (state != Py_None) {
149 if (!PyDict_Check(state)) {
150 PyErr_SetString(PyExc_TypeError, "state is not a dictionary");
151 return NULL;
152 }
153 while (PyDict_Next(state, &i, &d_key, &d_value)) {
154 if (PyObject_SetAttr(self, d_key, d_value) < 0)
155 return NULL;
156 }
157 }
158 Py_RETURN_NONE;
159}

610static int
611ImportError_init(PyImportErrorObject *self, PyObject *args, PyObject *kwds)
612{
613 static char *kwlist[] = {"name", "path", 0};
614 PyObject *empty_tuple;
615 PyObject *msg = NULL;
616 PyObject *name = NULL;
617 PyObject *path = NULL;
618
619 if (BaseException_init((PyBaseExceptionObject *)self, args, NULL) == -1)
620 return -1;
621
622 empty_tuple = PyTuple_New(0);
623 if (!empty_tuple)
624 return -1;
625 if (!PyArg_ParseTupleAndKeywords(empty_tuple, kwds, "|$OO:ImportError", kwlist,
626 &name, &path)) {
627 Py_DECREF(empty_tuple);
628 return -1;
629 }
630 Py_DECREF(empty_tuple);
631
632 Py_XINCREF(name);
633 Py_XSETREF(self->name, name);
634
635 Py_XINCREF(path);
636 Py_XSETREF(self->path, path);
637
638 if (PyTuple_GET_SIZE(args) == 1) {
639 msg = PyTuple_GET_ITEM(args, 0);
640 Py_INCREF(msg);
641 }
642 Py_XSETREF(self->msg, msg);
643
644 return 0;
645}

685static PyObject *
686ImportError_getstate(PyImportErrorObject *self)
687{
688 PyObject *dict = ((PyBaseExceptionObject *)self)->dict;
689 if (self->name || self->path) {
690 _Py_IDENTIFIER(name);
691 _Py_IDENTIFIER(path);
692 dict = dict ? PyDict_Copy(dict) : PyDict_New();
693 if (dict == NULL)
694 return NULL;
695 if (self->name && _PyDict_SetItemId(dict, &PyId_name, self->name) < 0) {
696 Py_DECREF(dict);
697 return NULL;
698 }
699 if (self->path && _PyDict_SetItemId(dict, &PyId_path, self->path) < 0) {
700 Py_DECREF(dict);
701 return NULL;
702 }
703 return dict;
704 }
705 else if (dict) {
706 Py_INCREF(dict);
707 return dict;
708 }
709 else {
710 Py_RETURN_NONE;
711 }
712}

715static PyObject *
716ImportError_reduce(PyImportErrorObject *self)
717{
718 PyObject *res;
719 PyObject *args;
720 PyObject *state = ImportError_getstate(self);
721 if (state == NULL)
722 return NULL;
723 args = ((PyBaseExceptionObject *)self)->args;
724 if (state == Py_None)
725 res = PyTuple_Pack(2, Py_TYPE(self), args);
726 else
727 res = PyTuple_Pack(3, Py_TYPE(self), args, state);
728 Py_DECREF(state);
729 return res;
730}

769/* Where a function has a single filename, such as open() or some
770 * of the os module functions, PyErr_SetFromErrnoWithFilename() is
771 * called, giving a third argument which is the filename. But, so
772 * that old code using in-place unpacking doesn't break, e.g.:
773 *
774 * except OSError, (errno, strerror):
775 *
776 * we hack args so that it only contains two items. This also
777 * means we need our own __str__() which prints out the filename
778 * when it was supplied.
779 *
780 * (If a function has two filenames, such as rename(), symlink(),
781 * or copy(), PyErr_SetFromErrnoWithFilenameObjects() is called,
782 * which allows passing in a second filename.)
783 */
784
785/* This function doesn't cleanup on error, the caller should */
786static int
787oserror_parse_args(PyObject **p_args,
788 PyObject **myerrno, PyObject **strerror,
789 PyObject **filename, PyObject **filename2
790#ifdef MS_WINDOWS
791 , PyObject **winerror
792#endif
793 )
794{
795 Py_ssize_t nargs;
796 PyObject *args = *p_args;
797#ifndef MS_WINDOWS
798 /*
799 * ignored on non-Windows platforms,
800 * but parsed so OSError has a consistent signature
801 */
802 PyObject *_winerror = NULL;
803 PyObject **winerror = &_winerror;
804#endif /* MS_WINDOWS */
805
806 nargs = PyTuple_GET_SIZE(args);
807
808 if (nargs >= 2 && nargs <= 5) {
809 if (!PyArg_UnpackTuple(args, "OSError", 2, 5,
810 myerrno, strerror,
811 filename, winerror, filename2))
812 return -1;
813#ifdef MS_WINDOWS
814 if (*winerror && PyLong_Check(*winerror)) {
815 long errcode, winerrcode;
816 PyObject *newargs;
817 Py_ssize_t i;
818
819 winerrcode = PyLong_AsLong(*winerror);
820 if (winerrcode == -1 && PyErr_Occurred())
821 return -1;
822 /* Set errno to the corresponding POSIX errno (overriding
823 first argument). Windows Socket error codes (>= 10000)
824 have the same value as their POSIX counterparts.
825 */
826 if (winerrcode < 10000)
827 errcode = winerror_to_errno(winerrcode);
828 else
829 errcode = winerrcode;
830 *myerrno = PyLong_FromLong(errcode);
831 if (!*myerrno)
832 return -1;
833 newargs = PyTuple_New(nargs);
834 if (!newargs)
835 return -1;
836 PyTuple_SET_ITEM(newargs, 0, *myerrno);
837 for (i = 1; i < nargs; i++) {
838 PyObject *val = PyTuple_GET_ITEM(args, i);
839 Py_INCREF(val);
840 PyTuple_SET_ITEM(newargs, i, val);
841 }
842 Py_DECREF(args);
843 args = *p_args = newargs;
844 }
845#endif /* MS_WINDOWS */
846 }
847
848 return 0;
849}

851static int
852oserror_init(PyOSErrorObject *self, PyObject **p_args,
853 PyObject *myerrno, PyObject *strerror,
854 PyObject *filename, PyObject *filename2
855#ifdef MS_WINDOWS
856 , PyObject *winerror
857#endif
858 )
859{
860 PyObject *args = *p_args;
861 Py_ssize_t nargs = PyTuple_GET_SIZE(args);
862
863 /* self->filename will remain Py_None otherwise */
864 if (filename && filename != Py_None) {
865 if (Py_TYPE(self) == (PyTypeObject *) PyExc_BlockingIOError &&
866 PyNumber_Check(filename)) {
867 /* BlockingIOError's 3rd argument can be the number of
868 * characters written.
869 */
870 self->written = PyNumber_AsSsize_t(filename, PyExc_ValueError);
871 if (self->written == -1 && PyErr_Occurred())
872 return -1;
873 }
874 else {
875 Py_INCREF(filename);
876 self->filename = filename;
877
878 if (filename2 && filename2 != Py_None) {
879 Py_INCREF(filename2);
880 self->filename2 = filename2;
881 }
882
883 if (nargs >= 2 && nargs <= 5) {
884 /* filename, filename2, and winerror are removed from the args tuple
885 (for compatibility purposes, see test_exceptions.py) */
886 PyObject *subslice = PyTuple_GetSlice(args, 0, 2);
887 if (!subslice)
888 return -1;
889
890 Py_DECREF(args); /* replacing args */
891 *p_args = args = subslice;
892 }
893 }
894 }
895 Py_XINCREF(myerrno);
896 self->myerrno = myerrno;
897
898 Py_XINCREF(strerror);
899 self->strerror = strerror;
900
901#ifdef MS_WINDOWS
902 Py_XINCREF(winerror);
903 self->winerror = winerror;
904#endif
905
906 /* Steals the reference to args */
907 Py_XSETREF(self->args, args);
908 *p_args = args = NULL;
909
910 return 0;
911}

918static int
919oserror_use_init(PyTypeObject *type)
920{
921 /* When __init__ is defined in an OSError subclass, we want any
922 extraneous argument to __new__ to be ignored. The only reasonable
923 solution, given __new__ takes a variable number of arguments,
924 is to defer arg parsing and initialization to __init__.
925
926 But when __new__ is overridden as well, it should call our __new__
927 with the right arguments.
928
929 (see http://bugs.python.org/issue12555#msg148829 )
930 */
931 if (type->tp_init != (initproc) OSError_init &&
932 type->tp_new == (newfunc) OSError_new) {
933 assert((PyObject *) type != PyExc_OSError);
934 return 1;
935 }
936 return 0;
937}

940OSError_new(PyTypeObject *type, PyObject *args, PyObject *kwds)
941{
942 PyOSErrorObject *self = NULL;

944 PyObject *filename = NULL, *filename2 = NULL;
945#ifdef MS_WINDOWS
946 PyObject *winerror = NULL;
947#endif
948
949 Py_INCREF(args);
950

952 if (!_PyArg_NoKeywords(type->tp_name, kwds))
953 goto error;
954
955 if (oserror_parse_args(&args, &myerrno, &strerror,
956 &filename, &filename2
957#ifdef MS_WINDOWS
958 , &winerror
959#endif

964 errnomap && (PyObject *) type == PyExc_OSError) {
965 PyObject *newtype;
966 newtype = PyDict_GetItem(errnomap, myerrno);

968 assert(PyType_Check(newtype));
969 type = (PyTypeObject *) newtype;

972 goto error;
973 }
974 }

976 self = (PyOSErrorObject *) type->tp_alloc(type, 0);
977 if (!self)
978 goto error;
979
980 self->dict = NULL;
981 self->traceback = self->cause = self->context = NULL;
982 self->written = -1;
983
984 if (!oserror_use_init(type)) {
985 if (oserror_init(self, &args, myerrno, strerror, filename, filename2
986#ifdef MS_WINDOWS
987 , winerror
988#endif
989 ))
990 goto error;
991 }
992 else {
993 self->args = PyTuple_New(0);
994 if (self->args == NULL)
995 goto error;
996 }

998 Py_XDECREF(args);
999 return (PyObject *) self;
1000
1001error:
1002 Py_XDECREF(args);
1003 Py_XDECREF(self);
1004 return NULL;

1125static PyObject *
1126OSError_reduce(PyOSErrorObject *self)
1127{
1128 PyObject *args = self->args;
1129 PyObject *res = NULL, *tmp;
1130
1131 /* self->args is only the first two real arguments if there was a
1132 * file name given to OSError. */
1133 if (PyTuple_GET_SIZE(args) == 2 && self->filename) {
1134 Py_ssize_t size = self->filename2 ? 5 : 3;
1135 args = PyTuple_New(size);
1136 if (!args)
1137 return NULL;
1138
1139 tmp = PyTuple_GET_ITEM(self->args, 0);
1140 Py_INCREF(tmp);
1141 PyTuple_SET_ITEM(args, 0, tmp);
1142
1143 tmp = PyTuple_GET_ITEM(self->args, 1);
1144 Py_INCREF(tmp);
1145 PyTuple_SET_ITEM(args, 1, tmp);
1146
1147 Py_INCREF(self->filename);
1148 PyTuple_SET_ITEM(args, 2, self->filename);
1149
1150 if (self->filename2) {
1151 /*
1152 * This tuple is essentially used as OSError(*args).
1153 * So, to recreate filename2, we need to pass in
1154 * winerror as well.
1155 */
1156 Py_INCREF(Py_None);
1157 PyTuple_SET_ITEM(args, 3, Py_None);
1158
1159 /* filename2 */
1160 Py_INCREF(self->filename2);
1161 PyTuple_SET_ITEM(args, 4, self->filename2);
1162 }
1163 } else
1164 Py_INCREF(args);
1165
1166 if (self->dict)
1167 res = PyTuple_Pack(3, Py_TYPE(self), args, self->dict);
1168 else
1169 res = PyTuple_Pack(2, Py_TYPE(self), args);
1170 Py_DECREF(args);
1171 return res;
1172}

1402/* This is called "my_basename" instead of just "basename" to avoid name
1403 conflicts with glibc; basename is already prototyped if _GNU_SOURCE is
1404 defined, and Python does define that. */
1405static PyObject*
1406my_basename(PyObject *name)
1407{
1408 Py_ssize_t i, size, offset;
1409 int kind;
1410 void *data;
1411
1412 if (PyUnicode_READY(name))
1413 return NULL;
1414 kind = PyUnicode_KIND(name);
1415 data = PyUnicode_DATA(name);
1416 size = PyUnicode_GET_LENGTH(name);
1417 offset = 0;
1418 for(i=0; i < size; i++) {
1419 if (PyUnicode_READ(kind, data, i) == SEP)
1420 offset = i + 1;
1421 }
1422 if (offset != 0)
1423 return PyUnicode_Substring(name, offset, size);
1424 else {
1425 Py_INCREF(name);
1426 return name;
1427 }
1428}

1431static PyObject *
1432SyntaxError_str(PySyntaxErrorObject *self)
1433{
1434 int have_lineno = 0;
1435 PyObject *filename;
1436 PyObject *result;
1437 /* Below, we always ignore overflow errors, just printing -1.
1438 Still, we cannot allow an OverflowError to be raised, so
1439 we need to call PyLong_AsLongAndOverflow. */
1440 int overflow;
1441
1442 /* XXX -- do all the additional formatting with filename and
1443 lineno here */
1444
1445 if (self->filename && PyUnicode_Check(self->filename)) {
1446 filename = my_basename(self->filename);
1447 if (filename == NULL)
1448 return NULL;
1449 } else {
1450 filename = NULL;
1451 }
1452 have_lineno = (self->lineno != NULL) && PyLong_CheckExact(self->lineno);
1453
1454 if (!filename && !have_lineno)
1455 return PyObject_Str(self->msg ? self->msg : Py_None);
1456
1457 if (filename && have_lineno)
1458 result = PyUnicode_FromFormat("%S (%U, line %ld)",
1459 self->msg ? self->msg : Py_None,
1460 filename,
1461 PyLong_AsLongAndOverflow(self->lineno, &overflow));
1462 else if (filename)
1463 result = PyUnicode_FromFormat("%S (%U)",
1464 self->msg ? self->msg : Py_None,
1465 filename);
1466 else /* only have_lineno */
1467 result = PyUnicode_FromFormat("%S (line %ld)",
1468 self->msg ? self->msg : Py_None,
1469 PyLong_AsLongAndOverflow(self->lineno, &overflow));
1470 Py_XDECREF(filename);
1471 return result;
1472}

2716/* Helper to do the equivalent of "raise X from Y" in C, but always using
2717 * the current exception rather than passing one in.
2718 *
2719 * We currently limit this to *only* exceptions that use the BaseException
2720 * tp_init and tp_new methods, since we can be reasonably sure we can wrap
2721 * those correctly without losing data and without losing backwards
2722 * compatibility.
2723 *
2724 * We also aim to rule out *all* exceptions that might be storing additional
2725 * state, whether by having a size difference relative to BaseException,
2726 * additional arguments passed in during construction or by having a
2727 * non-empty instance dict.
2728 *
2729 * We need to be very careful with what we wrap, since changing types to
2730 * a broader exception type would be backwards incompatible for
2731 * existing codecs, and with different init or new method implementations
2732 * may either not support instantiation with PyErr_Format or lose
2733 * information when instantiated that way.
2734 *
2735 * XXX (ncoghlan): This could be made more comprehensive by exploiting the
2736 * fact that exceptions are expected to support pickling. If more builtin
2737 * exceptions (e.g. AttributeError) start to be converted to rich
2738 * exceptions with additional attributes, that's probably a better approach
2739 * to pursue over adding special cases for particular stateful subclasses.
2740 *
2741 * Returns a borrowed reference to the new exception (if any), NULL if the
2742 * existing exception was left in place.
2743 */
2744PyObject *
2745_PyErr_TrySetFromCause(const char *format, ...)
2746{
2747 PyObject* msg_prefix;
2748 PyObject *exc, *val, *tb;
2749 PyTypeObject *caught_type;
2750 PyObject **dictptr;
2751 PyObject *instance_args;
2752 Py_ssize_t num_args, caught_type_size, base_exc_size;
2753 PyObject *new_exc, *new_val, *new_tb;
2754 va_list vargs;
2755 int same_basic_size;
2756
2757 PyErr_Fetch(&exc, &val, &tb);
2758 caught_type = (PyTypeObject *)exc;
2759 /* Ensure type info indicates no extra state is stored at the C level
2760 * and that the type can be reinstantiated using PyErr_Format
2761 */
2762 caught_type_size = caught_type->tp_basicsize;
2763 base_exc_size = _PyExc_BaseException.tp_basicsize;
2764 same_basic_size = (
2765 caught_type_size == base_exc_size ||
2766 (PyType_SUPPORTS_WEAKREFS(caught_type) &&
2767 (caught_type_size == base_exc_size + (Py_ssize_t)sizeof(PyObject *))
2768 )
2769 );
2770 if (caught_type->tp_init != (initproc)BaseException_init ||
2771 caught_type->tp_new != BaseException_new ||
2772 !same_basic_size ||
2773 caught_type->tp_itemsize != _PyExc_BaseException.tp_itemsize) {
2774 /* We can't be sure we can wrap this safely, since it may contain
2775 * more state than just the exception type. Accordingly, we just
2776 * leave it alone.
2777 */
2778 PyErr_Restore(exc, val, tb);
2779 return NULL;
2780 }
2781
2782 /* Check the args are empty or contain a single string */
2783 PyErr_NormalizeException(&exc, &val, &tb);
2784 instance_args = ((PyBaseExceptionObject *)val)->args;
2785 num_args = PyTuple_GET_SIZE(instance_args);
2786 if (num_args > 1 ||
2787 (num_args == 1 &&
2788 !PyUnicode_CheckExact(PyTuple_GET_ITEM(instance_args, 0)))) {
2789 /* More than 1 arg, or the one arg we do have isn't a string
2790 */
2791 PyErr_Restore(exc, val, tb);
2792 return NULL;
2793 }
2794
2795 /* Ensure the instance dict is also empty */
2796 dictptr = _PyObject_GetDictPtr(val);
2797 if (dictptr != NULL && *dictptr != NULL &&
2798 PyDict_GET_SIZE(*dictptr) > 0) {
2799 /* While we could potentially copy a non-empty instance dictionary
2800 * to the replacement exception, for now we take the more
2801 * conservative path of leaving exceptions with attributes set
2802 * alone.
2803 */
2804 PyErr_Restore(exc, val, tb);
2805 return NULL;
2806 }
2807
2808 /* For exceptions that we can wrap safely, we chain the original
2809 * exception to a new one of the exact same type with an
2810 * error message that mentions the additional details and the
2811 * original exception.
2812 *
2813 * It would be nice to wrap OSError and various other exception
2814 * types as well, but that's quite a bit trickier due to the extra
2815 * state potentially stored on OSError instances.
2816 */
2817 /* Ensure the traceback is set correctly on the existing exception */
2818 if (tb != NULL) {
2819 PyException_SetTraceback(val, tb);
2820 Py_DECREF(tb);
2821 }
2822
2823#ifdef HAVE_STDARG_PROTOTYPES
2824 va_start(vargs, format);
2825#else
2826 va_start(vargs);
2827#endif
2828 msg_prefix = PyUnicode_FromFormatV(format, vargs);
2829 va_end(vargs);
2830 if (msg_prefix == NULL) {
2831 Py_DECREF(exc);
2832 Py_DECREF(val);
2833 return NULL;
2834 }
2835
2836 PyErr_Format(exc, "%U (%s: %S)",
2837 msg_prefix, Py_TYPE(val)->tp_name, val);
2838 Py_DECREF(exc);
2839 Py_DECREF(msg_prefix);
2840 PyErr_Fetch(&new_exc, &new_val, &new_tb);
2841 PyErr_NormalizeException(&new_exc, &new_val, &new_tb);
2842 PyException_SetCause(new_val, val);
2843 PyErr_Restore(new_exc, new_val, new_tb);
2844 return new_val;
2845}
