14{

68static int
69descr_check(PyDescrObject *descr, PyObject *obj, PyObject **pres)
70{
71 if (obj == NULL) {
72 Py_INCREF(descr);
73 *pres = (PyObject *)descr;
74 return 1;
75 }
76 if (!PyObject_TypeCheck(obj, descr->d_type)) {
77 PyErr_Format(PyExc_TypeError,
78 "descriptor '%V' for '%s' objects "
79 "doesn't apply to '%s' object",
80 descr_name((PyDescrObject *)descr), "?",
81 descr->d_type->tp_name,
82 obj->ob_type->tp_name);
83 *pres = NULL;
84 return 1;
85 }
86 return 0;
87}

89static PyObject *
90classmethod_get(PyMethodDescrObject *descr, PyObject *obj, PyObject *type)
91{
92 /* Ensure a valid type. Class methods ignore obj. */
93 if (type == NULL) {
94 if (obj != NULL)
95 type = (PyObject *)obj->ob_type;
96 else {
97 /* Wot - no type?! */
98 PyErr_Format(PyExc_TypeError,
99 "descriptor '%V' for type '%s' "
100 "needs either an object or a type",
101 descr_name((PyDescrObject *)descr), "?",
102 PyDescr_TYPE(descr)->tp_name);
103 return NULL;
104 }
105 }
106 if (!PyType_Check(type)) {
107 PyErr_Format(PyExc_TypeError,
108 "descriptor '%V' for type '%s' "
109 "needs a type, not a '%s' as arg 2",
110 descr_name((PyDescrObject *)descr), "?",
111 PyDescr_TYPE(descr)->tp_name,
112 type->ob_type->tp_name);
113 return NULL;
114 }
115 if (!PyType_IsSubtype((PyTypeObject *)type, PyDescr_TYPE(descr))) {
116 PyErr_Format(PyExc_TypeError,
117 "descriptor '%V' for type '%s' "
118 "doesn't apply to type '%s'",
119 descr_name((PyDescrObject *)descr), "?",
120 PyDescr_TYPE(descr)->tp_name,
121 ((PyTypeObject *)type)->tp_name);
122 return NULL;
123 }
124 return PyCFunction_NewEx(descr->d_method, type, NULL);
125}

218static PyObject *
219methoddescr_call(PyMethodDescrObject *descr, PyObject *args, PyObject *kwargs)
220{
221 Py_ssize_t nargs;
222 PyObject *self, *result;
223
224 /* Make sure that the first argument is acceptable as 'self' */
225 assert(PyTuple_Check(args));
226 nargs = PyTuple_GET_SIZE(args);
227 if (nargs < 1) {
228 PyErr_Format(PyExc_TypeError,
229 "descriptor '%V' of '%.100s' "
230 "object needs an argument",
231 descr_name((PyDescrObject *)descr), "?",
232 PyDescr_TYPE(descr)->tp_name);
233 return NULL;
234 }
235 self = PyTuple_GET_ITEM(args, 0);
236 if (!_PyObject_RealIsSubclass((PyObject *)Py_TYPE(self),
237 (PyObject *)PyDescr_TYPE(descr))) {
238 PyErr_Format(PyExc_TypeError,
239 "descriptor '%V' "
240 "requires a '%.100s' object "
241 "but received a '%.100s'",
242 descr_name((PyDescrObject *)descr), "?",
243 PyDescr_TYPE(descr)->tp_name,
244 self->ob_type->tp_name);
245 return NULL;
246 }
247
248 result = _PyMethodDef_RawFastCallDict(descr->d_method, self,
249 &PyTuple_GET_ITEM(args, 1), nargs - 1,
250 kwargs);
251 result = _Py_CheckFunctionResult((PyObject *)descr, result, NULL);
252 return result;
253}

256PyObject *
257_PyMethodDescr_FastCallKeywords(PyObject *descrobj,

260{
261 assert(Py_TYPE(descrobj) == &PyMethodDescr_Type);
262 PyMethodDescrObject *descr = (PyMethodDescrObject *)descrobj;
263 PyObject *self, *result;

266 if (nargs < 1) {
267 PyErr_Format(PyExc_TypeError,
268 "descriptor '%V' of '%.100s' "
269 "object needs an argument",
270 descr_name((PyDescrObject *)descr), "?",
271 PyDescr_TYPE(descr)->tp_name);
272 return NULL;
273 }
274 self = args[0];
275 if (!_PyObject_RealIsSubclass((PyObject *)Py_TYPE(self),
276 (PyObject *)PyDescr_TYPE(descr))) {
277 PyErr_Format(PyExc_TypeError,

293static PyObject *
294classmethoddescr_call(PyMethodDescrObject *descr, PyObject *args,
295 PyObject *kwds)
296{
297 Py_ssize_t argc;
298 PyObject *self, *func, *result, **stack;
299
300 /* Make sure that the first argument is acceptable as 'self' */
301 assert(PyTuple_Check(args));
302 argc = PyTuple_GET_SIZE(args);
303 if (argc < 1) {
304 PyErr_Format(PyExc_TypeError,
305 "descriptor '%V' of '%.100s' "
306 "object needs an argument",
307 descr_name((PyDescrObject *)descr), "?",
308 PyDescr_TYPE(descr)->tp_name);
309 return NULL;
310 }
311 self = PyTuple_GET_ITEM(args, 0);
312 if (!PyType_Check(self)) {
313 PyErr_Format(PyExc_TypeError,
314 "descriptor '%V' requires a type "
315 "but received a '%.100s'",
316 descr_name((PyDescrObject *)descr), "?",
317 PyDescr_TYPE(descr)->tp_name,
318 self->ob_type->tp_name);
319 return NULL;
320 }
321 if (!PyType_IsSubtype((PyTypeObject *)self, PyDescr_TYPE(descr))) {
322 PyErr_Format(PyExc_TypeError,
323 "descriptor '%V' "
324 "requires a subtype of '%.100s' "
325 "but received '%.100s",
326 descr_name((PyDescrObject *)descr), "?",
327 PyDescr_TYPE(descr)->tp_name,
328 self->ob_type->tp_name);
329 return NULL;
330 }
331
332 func = PyCFunction_NewEx(descr->d_method, self, NULL);
333 if (func == NULL)
334 return NULL;
335 stack = &PyTuple_GET_ITEM(args, 1);
336 result = _PyObject_FastCallDict(func, stack, argc - 1, kwds);
337 Py_DECREF(func);
338 return result;
339}

393static PyObject *
394calculate_qualname(PyDescrObject *descr)
395{
396 PyObject *type_qualname, *res;
397 _Py_IDENTIFIER(__qualname__);
398
399 if (descr->d_name == NULL || !PyUnicode_Check(descr->d_name)) {
400 PyErr_SetString(PyExc_TypeError,
401 "<descriptor>.__name__ is not a unicode object");
402 return NULL;
403 }
404
405 type_qualname = _PyObject_GetAttrId((PyObject *)descr->d_type,
406 &PyId___qualname__);
407 if (type_qualname == NULL)
408 return NULL;
409
410 if (!PyUnicode_Check(type_qualname)) {
411 PyErr_SetString(PyExc_TypeError, "<descriptor>.__objclass__."
412 "__qualname__ is not a unicode object");
413 Py_XDECREF(type_qualname);
414 return NULL;
415 }
416
417 res = PyUnicode_FromFormat("%S.%S", type_qualname, descr->d_name);
418 Py_DECREF(type_qualname);
419 return res;
420}

431static PyObject *
432descr_reduce(PyDescrObject *descr)
433{
434 PyObject *builtins;
435 PyObject *getattr;
436 _Py_IDENTIFIER(getattr);
437
438 builtins = PyEval_GetBuiltins();
439 getattr = _PyDict_GetItemId(builtins, &PyId_getattr);
440 return Py_BuildValue("O(OO)", getattr, PyDescr_TYPE(descr),
441 PyDescr_NAME(descr));
442}

1008static void
1009wrapper_dealloc(wrapperobject *wp)
1010{
1011 PyObject_GC_UnTrack(wp);
1012 Py_TRASHCAN_SAFE_BEGIN(wp)
1013 Py_XDECREF(wp->descr);

1019#define TEST_COND(cond) ((cond) ? Py_True : Py_False)
1020
1021static PyObject *
1022wrapper_richcompare(PyObject *a, PyObject *b, int op)
1023{
1024 intptr_t result;
1025 PyObject *v;
1026 PyWrapperDescrObject *a_descr, *b_descr;
1027
1028 assert(a != NULL && b != NULL);
1029
1030 /* both arguments should be wrapperobjects */
1031 if (!Wrapper_Check(a) || !Wrapper_Check(b)) {
1032 v = Py_NotImplemented;
1033 Py_INCREF(v);
1034 return v;
1035 }
1036
1037 /* compare by descriptor address; if the descriptors are the same,
1038 compare by the objects they're bound to */
1039 a_descr = ((wrapperobject *)a)->descr;
1040 b_descr = ((wrapperobject *)b)->descr;
1041 if (a_descr == b_descr) {
1042 a = ((wrapperobject *)a)->self;
1043 b = ((wrapperobject *)b)->self;
1044 return PyObject_RichCompare(a, b, op);
1045 }
1046
1047 result = a_descr - b_descr;
1048 switch (op) {
1049 case Py_EQ:
1050 v = TEST_COND(result == 0);
1051 break;
1052 case Py_NE:
1053 v = TEST_COND(result != 0);
1054 break;
1055 case Py_LE:
1056 v = TEST_COND(result <= 0);
1057 break;
1058 case Py_GE:
1059 v = TEST_COND(result >= 0);
1060 break;
1061 case Py_LT:
1062 v = TEST_COND(result < 0);
1063 break;
1064 case Py_GT:
1065 v = TEST_COND(result > 0);
1066 break;
1067 default:
1068 PyErr_BadArgument();
1069 return NULL;
1070 }
1071 Py_INCREF(v);
1072 return v;
1073}

1077{
1078 Py_hash_t x, y;
1079 x = _Py_HashPointer(wp->descr);
1080 if (x == -1)
1081 return -1;
1082 y = PyObject_Hash(wp->self);
1083 if (y == -1)

1362{

1364 PyObject *args;
1365 PyObject *ret;
1366 propertyobject *gs = (propertyobject *)self;
1367
1368 if (obj == NULL || obj == Py_None) {
1369 Py_INCREF(self);
1370 return self;
1371 }
1372 if (gs->prop_get == NULL) {
1373 PyErr_SetString(PyExc_AttributeError, "unreadable attribute");
1374 return NULL;
1375 }
1376 args = cached_args;
1377 cached_args = NULL;
1378 if (!args) {
1379 args = PyTuple_New(1);
1380 if (!args)
1381 return NULL;
1382 _PyObject_GC_UNTRACK(args);
1383 }
1384 Py_INCREF(obj);
1385 PyTuple_SET_ITEM(args, 0, obj);
1386 ret = PyObject_Call(gs->prop_get, args, NULL);
1387 if (cached_args == NULL && Py_REFCNT(args) == 1) {
1388 assert(PyTuple_GET_SIZE(args) == 1);
1389 assert(PyTuple_GET_ITEM(args, 0) == obj);
1390 cached_args = args;
1391 Py_DECREF(obj);
1392 }
1393 else {
1394 assert(Py_REFCNT(args) >= 1);
1395 _PyObject_GC_TRACK(args);
1396 Py_DECREF(args);
1397 }
1398 return ret;
1399}

1409 else
1410 func = gs->prop_set;
1411 if (func == NULL) {
1412 PyErr_SetString(PyExc_AttributeError,
