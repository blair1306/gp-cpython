60/* Protection against deeply nested recursive calls
61
62 In Python 3.0, this protection has two levels:
63 * normal anti-recursion protection is triggered when the recursion level
64 exceeds the current recursion limit. It raises a RecursionError, and sets
65 the "overflowed" flag in the thread state structure. This flag
66 temporarily *disables* the normal protection; this allows cleanup code
67 to potentially outgrow the recursion limit while processing the
68 RecursionError.
69 * "last chance" anti-recursion protection is triggered when the recursion
70 level exceeds "current recursion limit + 50". By construction, this
71 protection can only be triggered when the "overflowed" flag is set. It
72 means the cleanup code has itself gone into an infinite loop, or the
73 RecursionError has been mistakingly ignored. When this protection is
74 triggered, the interpreter aborts with a Fatal Error.
75
76 In addition, the "overflowed" flag is automatically reset when the
77 recursion level drops below "current recursion limit - 50". This heuristic
78 is meant to ensure that the normal anti-recursion protection doesn't get
79 disabled too long.
80
81 Please note: this scheme has its own limitations. See:
82 http://mail.python.org/pipermail/python-dev/2008-August/082106.html
83 for some observations.
84*/
85PyAPI_FUNC(void) Py_SetRecursionLimit(int);
86PyAPI_FUNC(int) Py_GetRecursionLimit(void);

88#define Py_EnterRecursiveCall(where) \
89 (_Py_MakeRecCheck(PyThreadState_GET()->recursion_depth) && \
90 _Py_CheckRecursiveCall(where))
91#define Py_LeaveRecursiveCall() \
92 do{ if(_Py_MakeEndRecCheck(PyThreadState_GET()->recursion_depth)) \
93 PyThreadState_GET()->overflowed = 0; \
94 } while(0)
95PyAPI_FUNC(int) _Py_CheckRecursiveCall(const char *where);
96PyAPI_DATA(int) _Py_CheckRecursionLimit;
97
98#ifdef USE_STACKCHECK
99/* With USE_STACKCHECK, we artificially decrement the recursion limit in order
100 to trigger regular stack checks in _Py_CheckRecursiveCall(), except if
101 the "overflowed" flag is set, in which case we need the true value
102 of _Py_CheckRecursionLimit for _Py_MakeEndRecCheck() to function properly.
103*/
104# define _Py_MakeRecCheck(x) \
105 (++(x) > (_Py_CheckRecursionLimit += PyThreadState_GET()->overflowed - 1))
106#else
107# define _Py_MakeRecCheck(x) (++(x) > _Py_CheckRecursionLimit)
108#endif
109
110/* Compute the "lower-water mark" for a recursion limit. When
111 * Py_LeaveRecursiveCall() is called with a recursion depth below this mark,
112 * the overflowed flag is reset to 0. */
113#define _Py_RecursionLimitLowerWaterMark(limit) \
114 (((limit) > 200) \
115 ? ((limit) - 50) \
116 : (3 * ((limit) >> 2)))
117
118#define _Py_MakeEndRecCheck(x) \
119 (--(x) < _Py_RecursionLimitLowerWaterMark(_Py_CheckRecursionLimit))
120
121#define Py_ALLOW_RECURSION \
122 do { unsigned char _old = PyThreadState_GET()->recursion_critical;\
123 PyThreadState_GET()->recursion_critical = 1;
124
125#define Py_END_ALLOW_RECURSION \
126 PyThreadState_GET()->recursion_critical = _old; \
127 } while(0);

138/* Interface for threads.
139
140 A module that plans to do a blocking system call (or something else
141 that lasts a long time and doesn't touch Python data) can allow other
142 threads to run as follows:
143
144 ...preparations here...
145 Py_BEGIN_ALLOW_THREADS
146 ...blocking system call here...
147 Py_END_ALLOW_THREADS
148 ...interpret result here...
149
150 The Py_BEGIN_ALLOW_THREADS/Py_END_ALLOW_THREADS pair expands to a
151 {}-surrounded block.
152 To leave the block in the middle (e.g., with return), you must insert
153 a line containing Py_BLOCK_THREADS before the return, e.g.
154
155 if (...premature_exit...) {
156 Py_BLOCK_THREADS
157 PyErr_SetFromErrno(PyExc_OSError);
158 return NULL;
159 }
160
161 An alternative is:
162
163 Py_BLOCK_THREADS
164 if (...premature_exit...) {
165 PyErr_SetFromErrno(PyExc_OSError);
166 return NULL;
167 }
168 Py_UNBLOCK_THREADS
169
170 For convenience, that the value of 'errno' is restored across
171 Py_END_ALLOW_THREADS and Py_BLOCK_THREADS.
172
173 WARNING: NEVER NEST CALLS TO Py_BEGIN_ALLOW_THREADS AND
174 Py_END_ALLOW_THREADS!!!
175
176 The function PyEval_InitThreads() should be called only from
177 init_thread() in "_threadmodule.c".
178
179 Note that not yet all candidates have been converted to use this
180 mechanism!
181*/
