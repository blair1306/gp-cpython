62/* BEWARE:
63
64 Each interface exports both functions and macros. Extension modules should
65 use the functions, to ensure binary compatibility across Python versions.
66 Because the Python implementation is free to change internal details, and
67 the macros may (or may not) expose details for speed, if you do use the
68 macros you must recompile your extensions with each Python release.
69
70 Never mix calls to PyMem_ with calls to the platform malloc/realloc/
71 calloc/free. For example, on Windows different DLLs may end up using
72 different heaps, and if you use PyMem_Malloc you'll get the memory from the
73 heap used by the Python DLL; it could be a disaster if you free()'ed that
74 directly in your own extension. Using PyMem_Free instead ensures Python
75 can return the memory to the proper heap. As another example, in
76 PYMALLOC_DEBUG mode, Python wraps all calls to all PyMem_ and PyObject_
77 memory functions in special debugging wrappers that add additional
78 debugging info to dynamic memory blocks. The system routines have no idea
79 what to do with that stuff, and the Python wrappers have no idea what to do
80 with raw blocks obtained directly by the system routines then.
81
82 The GIL must be held when using these APIs.
83*/
84

90/* Functions
91
92 Functions supplying platform-independent semantics for malloc/realloc/
93 free. These functions make sure that allocating 0 bytes returns a distinct
94 non-NULL pointer (whenever possible -- if we're flat out of memory, NULL
95 may be returned), even if the platform malloc and realloc don't.
96 Returned pointers must be checked for NULL explicitly. No action is
97 performed on failure (no exception is set, no warning is printed, etc).
98*/

112/* Macros. */

114/* PyMem_MALLOC(0) means malloc(1). Some systems would return NULL
115 for malloc(0), which would be treated as an error. Some platforms
116 would return a pointer with no memory behind it, which would break
117 pymalloc. To solve these problems, allocate an extra byte. */
118/* Returns NULL to indicate error if a negative size or size larger than
119 Py_ssize_t can represent is supplied. Helps prevents security holes. */
120#define PyMem_MALLOC(n) PyMem_Malloc(n)
121#define PyMem_REALLOC(p, n) PyMem_Realloc(p, n)
122#define PyMem_FREE(p) PyMem_Free(p)

124/*
125 * Type-oriented memory interface
126 * ==============================
127 *
128 * Allocate memory for n objects of the given type. Returns a new pointer
129 * or NULL if the request was too large or memory allocation failed. Use
130 * these macros rather than doing the multiplication yourself so that proper
131 * overflow checking is always done.
132 */
133
134#define PyMem_New(type, n) \
135 ( ((size_t)(n) > PY_SSIZE_T_MAX / sizeof(type)) ? NULL : \
136 ( (type *) PyMem_Malloc((n) * sizeof(type)) ) )
137#define PyMem_NEW(type, n) \
138 ( ((size_t)(n) > PY_SSIZE_T_MAX / sizeof(type)) ? NULL : \
139 ( (type *) PyMem_MALLOC((n) * sizeof(type)) ) )
140
141/*
142 * The value of (p) is always clobbered by this macro regardless of success.
143 * The caller MUST check if (p) is NULL afterwards and deal with the memory
144 * error if so. This means the original value of (p) MUST be saved for the
145 * caller's memory error handler to not lose track of it.
146 */
147#define PyMem_Resize(p, type, n) \
148 ( (p) = ((size_t)(n) > PY_SSIZE_T_MAX / sizeof(type)) ? NULL : \
149 (type *) PyMem_Realloc((p), (n) * sizeof(type)) )
150#define PyMem_RESIZE(p, type, n) \
151 ( (p) = ((size_t)(n) > PY_SSIZE_T_MAX / sizeof(type)) ? NULL : \
152 (type *) PyMem_REALLOC((p), (n) * sizeof(type)) )
153

161typedef enum {
162 /* PyMem_RawMalloc(), PyMem_RawRealloc() and PyMem_RawFree() */
163 PYMEM_DOMAIN_RAW,
164
165 /* PyMem_Malloc(), PyMem_Realloc() and PyMem_Free() */

168 /* PyObject_Malloc(), PyObject_Realloc() and PyObject_Free() */
169 PYMEM_DOMAIN_OBJ
170} PyMemAllocatorDomain;
171
172typedef struct {
173 /* user context passed as the first argument to the 4 functions */
174 void *ctx;
175
176 /* allocate a memory block */
177 void* (*malloc) (void *ctx, size_t size);
178
179 /* allocate a memory block initialized by zeros */
180 void* (*calloc) (void *ctx, size_t nelem, size_t elsize);
181
182 /* allocate or resize a memory block */
183 void* (*realloc) (void *ctx, void *ptr, size_t new_size);
184
185 /* release a memory block */
186 void (*free) (void *ctx, void *ptr);
187} PyMemAllocatorEx;

193/* Set the memory block allocator of the specified domain.
194
195 The new allocator must return a distinct non-NULL pointer when requesting
196 zero bytes.
197
198 For the PYMEM_DOMAIN_RAW domain, the allocator must be thread-safe: the GIL
199 is not held when the allocator is called.
200
201 If the new allocator is not a hook (don't call the previous allocator), the
202 PyMem_SetupDebugHooks() function must be called to reinstall the debug hooks
203 on top on the new allocator. */
204PyAPI_FUNC(void) PyMem_SetAllocator(PyMemAllocatorDomain domain,
205 PyMemAllocatorEx *allocator);

207/* Setup hooks to detect bugs in the following Python memory allocator
208 functions:
209
210 - PyMem_RawMalloc(), PyMem_RawRealloc(), PyMem_RawFree()
211 - PyMem_Malloc(), PyMem_Realloc(), PyMem_Free()
212 - PyObject_Malloc(), PyObject_Realloc() and PyObject_Free()
213
214 Newly allocated memory is filled with the byte 0xCB, freed memory is filled
215 with the byte 0xDB. Additionnal checks:
216
217 - detect API violations, ex: PyObject_Free() called on a buffer allocated
218 by PyMem_Malloc()
219 - detect write before the start of the buffer (buffer underflow)
220 - detect write after the end of the buffer (buffer overflow)
221
222 The function does nothing if Python is not compiled is debug mode. */
223PyAPI_FUNC(void) PyMem_SetupDebugHooks(void);
