231static int
232cleanreturn(int retval, freelist_t *freelist)
233{
234 int index;
235
236 if (retval == 0) {
237 /* A failure occurred, therefore execute all of the cleanup
238 functions.
239 */
240 for (index = 0; index < freelist->first_available; ++index) {
241 freelist->entries[index].destructor(NULL,
242 freelist->entries[index].item);
243 }
244 }
245 if (freelist->entries_malloced)
246 PyMem_FREE(freelist->entries);
247 return retval;
248}

251static int
252vgetargs1_impl(PyObject *compat_args, PyObject **stack, Py_ssize_t nargs, const char *format,
253 va_list *p_va, int flags)
254{
255 char msgbuf[256];
256 int levels[32];
257 const char *fname = NULL;
258 const char *message = NULL;
259 int min = -1;
260 int max = 0;
261 int level = 0;
262 int endfmt = 0;
263 const char *formatsave = format;
264 Py_ssize_t i;
265 const char *msg;
266 int compat = flags & FLAG_COMPAT;
267 freelistentry_t static_entries[STATIC_FREELIST_ENTRIES];
268 freelist_t freelist;
269
270 assert(nargs == 0 || stack != NULL);
271
272 freelist.entries = static_entries;
273 freelist.first_available = 0;
274 freelist.entries_malloced = 0;
275
276 flags = flags & ~FLAG_COMPAT;
277
278 while (endfmt == 0) {
279 int c = *format++;
280 switch (c) {
281 case '(':
282 if (level == 0)
283 max++;
284 level++;
285 if (level >= 30)
286 Py_FatalError("too many tuple nesting levels "
287 "in argument format string");
288 break;
289 case ')':
290 if (level == 0)
291 Py_FatalError("excess ')' in getargs format");
292 else
293 level--;
294 break;
295 case '\0':
296 endfmt = 1;
297 break;
298 case ':':
299 fname = format;
300 endfmt = 1;
301 break;
302 case ';':
303 message = format;
304 endfmt = 1;
305 break;
306 case '|':
307 if (level == 0)
308 min = max;
309 break;
310 default:
311 if (level == 0) {
312 if (Py_ISALPHA(Py_CHARMASK(c)))
313 if (c != 'e') /* skip encoded */
314 max++;
315 }
316 break;
317 }
318 }
319
320 if (level != 0)
321 Py_FatalError(/* '(' */ "missing ')' in getargs format");
322
323 if (min < 0)
324 min = max;
325
326 format = formatsave;
327
328 if (max > STATIC_FREELIST_ENTRIES) {
329 freelist.entries = PyMem_NEW(freelistentry_t, max);
330 if (freelist.entries == NULL) {
331 PyErr_NoMemory();
332 return 0;
333 }
334 freelist.entries_malloced = 1;
335 }
336
337 if (compat) {
338 if (max == 0) {
339 if (compat_args == NULL)
340 return 1;
341 PyErr_Format(PyExc_TypeError,
342 "%.200s%s takes no arguments",
343 fname==NULL ? "function" : fname,
344 fname==NULL ? "" : "()");
345 return cleanreturn(0, &freelist);
346 }
347 else if (min == 1 && max == 1) {
348 if (compat_args == NULL) {
349 PyErr_Format(PyExc_TypeError,
350 "%.200s%s takes at least one argument",
351 fname==NULL ? "function" : fname,
352 fname==NULL ? "" : "()");
353 return cleanreturn(0, &freelist);
354 }
355 msg = convertitem(compat_args, &format, p_va, flags, levels,
356 msgbuf, sizeof(msgbuf), &freelist);
357 if (msg == NULL)
358 return cleanreturn(1, &freelist);
359 seterror(levels[0], msg, levels+1, fname, message);
360 return cleanreturn(0, &freelist);
361 }
362 else {
363 PyErr_SetString(PyExc_SystemError,
364 "old style getargs format uses new features");
365 return cleanreturn(0, &freelist);
366 }
367 }
368
369 if (nargs < min || max < nargs) {
370 if (message == NULL)
371 PyErr_Format(PyExc_TypeError,
372 "%.150s%s takes %s %d argument%s (%ld given)",
373 fname==NULL ? "function" : fname,
374 fname==NULL ? "" : "()",
375 min==max ? "exactly"
376 : nargs < min ? "at least" : "at most",
377 nargs < min ? min : max,
378 (nargs < min ? min : max) == 1 ? "" : "s",
379 Py_SAFE_DOWNCAST(nargs, Py_ssize_t, long));
380 else
381 PyErr_SetString(PyExc_TypeError, message);
382 return cleanreturn(0, &freelist);
383 }
384
385 for (i = 0; i < nargs; i++) {
386 if (*format == '|')
387 format++;
388 msg = convertitem(stack[i], &format, p_va,
389 flags, levels, msgbuf,
390 sizeof(msgbuf), &freelist);
391 if (msg) {
392 seterror(i+1, msg, levels, fname, message);
393 return cleanreturn(0, &freelist);
394 }
395 }
396
397 if (*format != '\0' && !Py_ISALPHA(Py_CHARMASK(*format)) &&
398 *format != '(' &&
399 *format != '|' && *format != ':' && *format != ';') {
400 PyErr_Format(PyExc_SystemError,
401 "bad format string: %.200s", formatsave);
402 return cleanreturn(0, &freelist);
403 }
404
405 return cleanreturn(1, &freelist);
406}

408static int
409vgetargs1(PyObject *args, const char *format, va_list *p_va, int flags)
410{
411 PyObject **stack;
412 Py_ssize_t nargs;
413
414 if (!(flags & FLAG_COMPAT)) {
415 assert(args != NULL);
416
417 if (!PyTuple_Check(args)) {
418 PyErr_SetString(PyExc_SystemError,
419 "new style getargs format but argument is not a tuple");
420 return 0;
421 }
422
423 stack = &PyTuple_GET_ITEM(args, 0);
424 nargs = PyTuple_GET_SIZE(args);
425 }
426 else {
427 stack = NULL;
428 nargs = 0;
429 }
430
431 return vgetargs1_impl(args, stack, nargs, format, p_va, flags);
432}

435static void
436seterror(Py_ssize_t iarg, const char *msg, int *levels, const char *fname,
437 const char *message)
438{
439 char buf[512];
440 int i;
441 char *p = buf;
442
443 if (PyErr_Occurred())
444 return;
445 else if (message == NULL) {
446 if (fname != NULL) {
447 PyOS_snprintf(p, sizeof(buf), "%.200s() ", fname);
448 p += strlen(p);
449 }
450 if (iarg != 0) {
451 PyOS_snprintf(p, sizeof(buf) - (p - buf),
452 "argument %" PY_FORMAT_SIZE_T "d", iarg);
453 i = 0;
454 p += strlen(p);
455 while (i < 32 && levels[i] > 0 && (int)(p-buf) < 220) {
456 PyOS_snprintf(p, sizeof(buf) - (p - buf),
457 ", item %d", levels[i]-1);
458 p += strlen(p);
459 i++;
460 }
461 }
462 else {
463 PyOS_snprintf(p, sizeof(buf) - (p - buf), "argument");
464 p += strlen(p);
465 }
466 PyOS_snprintf(p, sizeof(buf) - (p - buf), " %.256s", msg);
467 message = buf;
468 }
469 if (msg[0] == '(') {
470 PyErr_SetString(PyExc_SystemError, message);
471 }
472 else {
473 PyErr_SetString(PyExc_TypeError, message);
474 }
475}

478/* Convert a tuple argument.
479 On entry, *p_format points to the character _after_ the opening '('.
480 On successful exit, *p_format points to the closing ')'.
481 If successful:
482 *p_format and *p_va are updated,
483 *levels and *msgbuf are untouched,
484 and NULL is returned.
485 If the argument is invalid:
486 *p_format is unchanged,
487 *p_va is undefined,
488 *levels is a 0-terminated list of item numbers,
489 *msgbuf contains an error message, whose format is:
490 "must be <typename1>, not <typename2>", where:
491 <typename1> is the name of the expected type, and
492 <typename2> is the name of the actual type,
493 and msgbuf is returned.
494*/
495
496static const char *
497converttuple(PyObject *arg, const char **p_format, va_list *p_va, int flags,
498 int *levels, char *msgbuf, size_t bufsize, int toplevel,
499 freelist_t *freelist)
500{
501 int level = 0;
502 int n = 0;
503 const char *format = *p_format;
504 int i;
505 Py_ssize_t len;
506
507 for (;;) {
508 int c = *format++;
509 if (c == '(') {
510 if (level == 0)
511 n++;
512 level++;
513 }
514 else if (c == ')') {
515 if (level == 0)
516 break;
517 level--;
518 }
519 else if (c == ':' || c == ';' || c == '\0')
520 break;
521 else if (level == 0 && Py_ISALPHA(Py_CHARMASK(c)))
522 n++;
523 }
524
525 if (!PySequence_Check(arg) || PyBytes_Check(arg)) {
526 levels[0] = 0;
527 PyOS_snprintf(msgbuf, bufsize,
528 toplevel ? "expected %d arguments, not %.50s" :
529 "must be %d-item sequence, not %.50s",
530 n,
531 arg == Py_None ? "None" : arg->ob_type->tp_name);
532 return msgbuf;
533 }
534
535 len = PySequence_Size(arg);
536 if (len != n) {
537 levels[0] = 0;
538 if (toplevel) {
539 PyOS_snprintf(msgbuf, bufsize,
540 "expected %d arguments, not %" PY_FORMAT_SIZE_T "d",
541 n, len);
542 }
543 else {
544 PyOS_snprintf(msgbuf, bufsize,
545 "must be sequence of length %d, "
546 "not %" PY_FORMAT_SIZE_T "d",
547 n, len);
548 }
549 return msgbuf;
550 }
551
552 format = *p_format;
553 for (i = 0; i < n; i++) {
554 const char *msg;
555 PyObject *item;
556 item = PySequence_GetItem(arg, i);
557 if (item == NULL) {
558 PyErr_Clear();
559 levels[0] = i+1;
560 levels[1] = 0;
561 strncpy(msgbuf, "is not retrievable", bufsize);
562 return msgbuf;
563 }
564 msg = convertitem(item, &format, p_va, flags, levels+1,
565 msgbuf, bufsize, freelist);
566 /* PySequence_GetItem calls tp->sq_item, which INCREFs */
567 Py_XDECREF(item);
568 if (msg != NULL) {
569 levels[0] = i+1;
570 return msg;
571 }
572 }
573
574 *p_format = format;
575 return NULL;
576}

580
581static const char *
582convertitem(PyObject *arg, const char **p_format, va_list *p_va, int flags,
583 int *levels, char *msgbuf, size_t bufsize, freelist_t *freelist)
584{
585 const char *msg;
586 const char *format = *p_format;
587
588 if (*format == '(' /* ')' */) {
589 format++;
590 msg = converttuple(arg, &format, p_va, flags, levels, msgbuf,
591 bufsize, 0, freelist);
592 if (msg == NULL)
593 format++;
594 }
595 else {
596 msg = convertsimple(arg, &format, p_va, flags,
597 msgbuf, bufsize, freelist);
598 if (msg != NULL)
599 levels[0] = 0;
600 }
601 if (msg == NULL)
602 *p_format = format;
603 return msg;
604}

643/* Convert a non-tuple argument. Return NULL if conversion went OK,
644 or a string with a message describing the failure. The message is
645 formatted as "must be <desired type>, not <actual type>".
646 When failing, an exception may or may not have been raised.
647 Don't call if a tuple is expected.
648
649 When you add new format codes, please don't forget poor skipitem() below.
650*/
651
652static const char *
653convertsimple(PyObject *arg, const char **p_format, va_list *p_va, int flags,
654 char *msgbuf, size_t bufsize, freelist_t *freelist)
655{
656 /* For # codes */
657#define FETCH_SIZE int *q=NULL;Py_ssize_t *q2=NULL;\
658 if (flags & FLAG_SIZE_T) q2=va_arg(*p_va, Py_ssize_t*); \
659 else q=va_arg(*p_va, int*);
660#define STORE_SIZE(s) \
661 if (flags & FLAG_SIZE_T) \
662 *q2=s; \
663 else { \
664 if (INT_MAX < s) { \
665 PyErr_SetString(PyExc_OverflowError, \
666 "size does not fit in an int"); \
667 return converterr("", arg, msgbuf, bufsize); \
668 } \
669 *q = (int)s; \
670 }
671#define BUFFER_LEN ((flags & FLAG_SIZE_T) ? *q2:*q)
672#define RETURN_ERR_OCCURRED return msgbuf
673
674 const char *format = *p_format;
675 char c = *format++;
676 const char *sarg;
677
678 switch (c) {
679
680 case 'b': { /* unsigned byte -- very short int */
681 char *p = va_arg(*p_va, char *);
682 long ival;
683 if (float_argument_error(arg))
684 RETURN_ERR_OCCURRED;
685 ival = PyLong_AsLong(arg);
686 if (ival == -1 && PyErr_Occurred())
687 RETURN_ERR_OCCURRED;
688 else if (ival < 0) {
689 PyErr_SetString(PyExc_OverflowError,
690 "unsigned byte integer is less than minimum");
691 RETURN_ERR_OCCURRED;
692 }
693 else if (ival > UCHAR_MAX) {
694 PyErr_SetString(PyExc_OverflowError,
695 "unsigned byte integer is greater than maximum");
696 RETURN_ERR_OCCURRED;
697 }
698 else
699 *p = (unsigned char) ival;
700 break;
701 }
702
703 case 'B': {/* byte sized bitfield - both signed and unsigned
704 values allowed */
705 char *p = va_arg(*p_va, char *);
706 long ival;
707 if (float_argument_error(arg))
708 RETURN_ERR_OCCURRED;
709 ival = PyLong_AsUnsignedLongMask(arg);
710 if (ival == -1 && PyErr_Occurred())
711 RETURN_ERR_OCCURRED;
712 else
713 *p = (unsigned char) ival;
714 break;
715 }
716
717 case 'h': {/* signed short int */
718 short *p = va_arg(*p_va, short *);
719 long ival;
720 if (float_argument_error(arg))
721 RETURN_ERR_OCCURRED;
722 ival = PyLong_AsLong(arg);
723 if (ival == -1 && PyErr_Occurred())
724 RETURN_ERR_OCCURRED;
725 else if (ival < SHRT_MIN) {
726 PyErr_SetString(PyExc_OverflowError,
727 "signed short integer is less than minimum");
728 RETURN_ERR_OCCURRED;
729 }
730 else if (ival > SHRT_MAX) {
731 PyErr_SetString(PyExc_OverflowError,
732 "signed short integer is greater than maximum");
733 RETURN_ERR_OCCURRED;
734 }
735 else
736 *p = (short) ival;
737 break;
738 }
739
740 case 'H': { /* short int sized bitfield, both signed and
741 unsigned allowed */
742 unsigned short *p = va_arg(*p_va, unsigned short *);
743 long ival;
744 if (float_argument_error(arg))
745 RETURN_ERR_OCCURRED;
746 ival = PyLong_AsUnsignedLongMask(arg);
747 if (ival == -1 && PyErr_Occurred())
748 RETURN_ERR_OCCURRED;
749 else
750 *p = (unsigned short) ival;
751 break;
752 }
753
754 case 'i': {/* signed int */
755 int *p = va_arg(*p_va, int *);
756 long ival;
757 if (float_argument_error(arg))
758 RETURN_ERR_OCCURRED;
759 ival = PyLong_AsLong(arg);
760 if (ival == -1 && PyErr_Occurred())
761 RETURN_ERR_OCCURRED;
762 else if (ival > INT_MAX) {
763 PyErr_SetString(PyExc_OverflowError,
764 "signed integer is greater than maximum");
765 RETURN_ERR_OCCURRED;
766 }
767 else if (ival < INT_MIN) {
768 PyErr_SetString(PyExc_OverflowError,
769 "signed integer is less than minimum");
770 RETURN_ERR_OCCURRED;
771 }
772 else
773 *p = ival;
774 break;
775 }
776
777 case 'I': { /* int sized bitfield, both signed and
778 unsigned allowed */
779 unsigned int *p = va_arg(*p_va, unsigned int *);

1162 responsible for PyMem_Free()ing it after

1183 "(buffer_len is NULL)",
1184 arg, msgbuf, bufsize);

1189 Py_DECREF(s);

1213
1214 - the encoded string has to be 0-terminated

1219 allocated and the data copied into it;

1221 buffer; the caller is responsible for
1222 PyMem_Free()ing it after usage
1223
1224 */

1226 Py_DECREF(s);
1227 return converterr(
1228 "encoded string without null bytes",
1229 arg, msgbuf, bufsize);

1233 Py_DECREF(s);

1235 RETURN_ERR_OCCURRED;

1238 Py_DECREF(s);

1240 arg, msgbuf, bufsize);

1243 }
1244 Py_DECREF(s);
1245 break;
1246 }

1303 arg, msgbuf, bufsize);

1305 else {
1306 p = va_arg(*p_va, PyObject **);

1308 }

1310 }
1311
1312
1313 case 'w': { /* "w*": memory buffer, read-write access */

1318 "(invalid use of 'w' format character)",

1323 supports it directly. */

1334 return converterr(
1335 "(cleanup problem)",
1336 arg, msgbuf, bufsize);
1337 }
1338 break;
1339 }

1341 default:
1342 return converterr("(impossible<bad format char>)", arg, msgbuf, bufsize);
1343
1344 }
1345
1346 *p_format = format;

1350#undef STORE_SIZE
1351#undef BUFFER_LEN
1352#undef RETURN_ERR_OCCURRED
1353}

1355static Py_ssize_t
1356convertbuffer(PyObject *arg, const void **p, const char **errmsg)
1357{
1358 PyBufferProcs *pb = Py_TYPE(arg)->tp_as_buffer;
1359 Py_ssize_t count;
1360 Py_buffer view;
1361
1362 *errmsg = NULL;
1363 *p = NULL;
1364 if (pb != NULL && pb->bf_releasebuffer != NULL) {
1365 *errmsg = "read-only bytes-like object";
1366 return -1;
1367 }
1368
1369 if (getbuffer(arg, &view, errmsg) < 0)
1370 return -1;
1371 count = view.len;
1372 *p = view.buf;
1373 PyBuffer_Release(&view);
1374 return count;
1375}

1378getbuffer(PyObject *arg, Py_buffer *view, const char **errmsg)
1379{

1385 PyBuffer_Release(view);
1386 *errmsg = "contiguous buffer";
1387 return -1;
1388 }
1389 return 0;
1390}

1595static int
1596vgetargskeywords(PyObject *args, PyObject *kwargs, const char *format,
1597 char **kwlist, va_list *p_va, int flags)
1598{
1599 char msgbuf[512];
1600 int levels[32];
1601 const char *fname, *msg, *custom_msg;
1602 int min = INT_MAX;
1603 int max = INT_MAX;
1604 int i, pos, len;
1605 int skip = 0;
1606 Py_ssize_t nargs, nkwargs;
1607 PyObject *current_arg;
1608 freelistentry_t static_entries[STATIC_FREELIST_ENTRIES];
1609 freelist_t freelist;
1610
1611 freelist.entries = static_entries;
1612 freelist.first_available = 0;
1613 freelist.entries_malloced = 0;
1614
1615 assert(args != NULL && PyTuple_Check(args));
1616 assert(kwargs == NULL || PyDict_Check(kwargs));
1617 assert(format != NULL);
1618 assert(kwlist != NULL);
1619 assert(p_va != NULL);
1620
1621 /* grab the function name or custom error msg first (mutually exclusive) */
1622 fname = strchr(format, ':');
1623 if (fname) {
1624 fname++;
1625 custom_msg = NULL;
1626 }
1627 else {
1628 custom_msg = strchr(format,';');
1629 if (custom_msg)
1630 custom_msg++;
1631 }
1632
1633 /* scan kwlist and count the number of positional-only parameters */
1634 for (pos = 0; kwlist[pos] && !*kwlist[pos]; pos++) {
1635 }
1636 /* scan kwlist and get greatest possible nbr of args */
1637 for (len = pos; kwlist[len]; len++) {
1638 if (!*kwlist[len]) {
1639 PyErr_SetString(PyExc_SystemError,
1640 "Empty keyword parameter name");
1641 return cleanreturn(0, &freelist);
1642 }
1643 }
1644
1645 if (len > STATIC_FREELIST_ENTRIES) {
1646 freelist.entries = PyMem_NEW(freelistentry_t, len);
1647 if (freelist.entries == NULL) {
1648 PyErr_NoMemory();
1649 return 0;
1650 }
1651 freelist.entries_malloced = 1;
1652 }
1653
1654 nargs = PyTuple_GET_SIZE(args);
1655 nkwargs = (kwargs == NULL) ? 0 : PyDict_GET_SIZE(kwargs);
1656 if (nargs + nkwargs > len) {
1657 /* Adding "keyword" (when nargs == 0) prevents producing wrong error
1658 messages in some special cases (see bpo-31229). */
1659 PyErr_Format(PyExc_TypeError,
1660 "%.200s%s takes at most %d %sargument%s (%zd given)",
1661 (fname == NULL) ? "function" : fname,
1662 (fname == NULL) ? "" : "()",
1663 len,
1664 (nargs == 0) ? "keyword " : "",
1665 (len == 1) ? "" : "s",
1666 nargs + nkwargs);
1667 return cleanreturn(0, &freelist);
1668 }
1669
1670 /* convert tuple args and keyword args in same loop, using kwlist to drive process */
1671 for (i = 0; i < len; i++) {
1672 if (*format == '|') {
1673 if (min != INT_MAX) {
1674 PyErr_SetString(PyExc_SystemError,
1675 "Invalid format string (| specified twice)");
1676 return cleanreturn(0, &freelist);
1677 }
1678
1679 min = i;
1680 format++;
1681
1682 if (max != INT_MAX) {
1683 PyErr_SetString(PyExc_SystemError,
1684 "Invalid format string ($ before |)");
1685 return cleanreturn(0, &freelist);
1686 }
1687 }
1688 if (*format == '$') {
1689 if (max != INT_MAX) {
1690 PyErr_SetString(PyExc_SystemError,
1691 "Invalid format string ($ specified twice)");
1692 return cleanreturn(0, &freelist);
1693 }
1694
1695 max = i;
1696 format++;
1697
1698 if (max < pos) {
1699 PyErr_SetString(PyExc_SystemError,
1700 "Empty parameter name after $");
1701 return cleanreturn(0, &freelist);
1702 }
1703 if (skip) {
1704 /* Now we know the minimal and the maximal numbers of
1705 * positional arguments and can raise an exception with
1706 * informative message (see below). */
1707 break;
1708 }
1709 if (max < nargs) {
1710 if (max == 0) {
1711 PyErr_Format(PyExc_TypeError,
1712 "%.200s%s takes no positional arguments",
1713 (fname == NULL) ? "function" : fname,
1714 (fname == NULL) ? "" : "()");
1715 }
1716 else {
1717 PyErr_Format(PyExc_TypeError,
1718 "%.200s%s takes %s %d positional arguments"
1719 " (%d given)",
1720 (fname == NULL) ? "function" : fname,
1721 (fname == NULL) ? "" : "()",
1722 (min != INT_MAX) ? "at most" : "exactly",
1723 max, nargs);
1724 }
1725 return cleanreturn(0, &freelist);
1726 }
1727 }
1728 if (IS_END_OF_FORMAT(*format)) {
1729 PyErr_Format(PyExc_SystemError,
1730 "More keyword list entries (%d) than "
1731 "format specifiers (%d)", len, i);
1732 return cleanreturn(0, &freelist);
1733 }
1734 if (!skip) {
1735 if (i < nargs) {
1736 current_arg = PyTuple_GET_ITEM(args, i);
1737 }
1738 else if (nkwargs && i >= pos) {
1739 current_arg = PyDict_GetItemString(kwargs, kwlist[i]);
1740 if (current_arg)
1741 --nkwargs;
1742 }
1743 else {
1744 current_arg = NULL;
1745 }
1746
1747 if (current_arg) {
1748 msg = convertitem(current_arg, &format, p_va, flags,
1749 levels, msgbuf, sizeof(msgbuf), &freelist);
1750 if (msg) {
1751 seterror(i+1, msg, levels, fname, custom_msg);
1752 return cleanreturn(0, &freelist);
1753 }
1754 continue;
1755 }
1756
1757 if (i < min) {
1758 if (i < pos) {
1759 assert (min == INT_MAX);
1760 assert (max == INT_MAX);
1761 skip = 1;
1762 /* At that moment we still don't know the minimal and
1763 * the maximal numbers of positional arguments. Raising
1764 * an exception is deferred until we encounter | and $
1765 * or the end of the format. */
1766 }
1767 else {
1768 PyErr_Format(PyExc_TypeError, "%.200s%s missing required "
1769 "argument '%s' (pos %d)",
1770 (fname == NULL) ? "function" : fname,
1771 (fname == NULL) ? "" : "()",
1772 kwlist[i], i+1);
1773 return cleanreturn(0, &freelist);
1774 }
1775 }
1776 /* current code reports success when all required args
1777 * fulfilled and no keyword args left, with no further
1778 * validation. XXX Maybe skip this in debug build ?
1779 */
1780 if (!nkwargs && !skip) {
1781 return cleanreturn(1, &freelist);
1782 }
1783 }
1784
1785 /* We are into optional args, skip thru to any remaining
1786 * keyword args */
1787 msg = skipitem(&format, p_va, flags);
1788 if (msg) {
1789 PyErr_Format(PyExc_SystemError, "%s: '%s'", msg,
1790 format);
1791 return cleanreturn(0, &freelist);
1792 }
1793 }
1794
1795 if (skip) {
1796 PyErr_Format(PyExc_TypeError,
1797 "%.200s%s takes %s %d positional arguments"
1798 " (%d given)",
1799 (fname == NULL) ? "function" : fname,
1800 (fname == NULL) ? "" : "()",
1801 (Py_MIN(pos, min) < i) ? "at least" : "exactly",
1802 Py_MIN(pos, min), nargs);
1803 return cleanreturn(0, &freelist);
1804 }
1805
1806 if (!IS_END_OF_FORMAT(*format) && (*format != '|') && (*format != '$')) {
1807 PyErr_Format(PyExc_SystemError,
1808 "more argument specifiers than keyword list entries "
1809 "(remaining format:'%s')", format);
1810 return cleanreturn(0, &freelist);
1811 }
1812
1813 if (nkwargs > 0) {
1814 PyObject *key;
1815 Py_ssize_t j;
1816 /* make sure there are no arguments given by name and position */
1817 for (i = pos; i < nargs; i++) {
1818 current_arg = PyDict_GetItemString(kwargs, kwlist[i]);
1819 if (current_arg) {
1820 /* arg present in tuple and in dict */
1821 PyErr_Format(PyExc_TypeError,
1822 "argument for %.200s%s given by name ('%s') "
1823 "and position (%d)",
1824 (fname == NULL) ? "function" : fname,
1825 (fname == NULL) ? "" : "()",
1826 kwlist[i], i+1);
1827 return cleanreturn(0, &freelist);
1828 }
1829 }
1830 /* make sure there are no extraneous keyword arguments */
1831 j = 0;
1832 while (PyDict_Next(kwargs, &j, &key, NULL)) {
1833 int match = 0;
1834 if (!PyUnicode_Check(key)) {
1835 PyErr_SetString(PyExc_TypeError,
1836 "keywords must be strings");
1837 return cleanreturn(0, &freelist);
1838 }
1839 for (i = pos; i < len; i++) {
1840 if (_PyUnicode_EqualToASCIIString(key, kwlist[i])) {
1841 match = 1;
1842 break;
1843 }
1844 }
1845 if (!match) {
1846 PyErr_Format(PyExc_TypeError,
1847 "'%U' is an invalid keyword "
1848 "argument for %.200s%s",
1849 key,
1850 (fname == NULL) ? "this function" : fname,
1851 (fname == NULL) ? "" : "()");
1852 return cleanreturn(0, &freelist);
1853 }
1854 }
1855 }
1856
1857 return cleanreturn(1, &freelist);
1858}

1991{
1992 Py_ssize_t i, nkwargs;
1993
1994 if (kwargs != NULL) {
1995 return PyDict_GetItem(kwargs, key);

2013 }
2014 }

2264{
2265 const char *format = *p_format;

2268 switch (c) {
2269
2270 /*
2271 * codes that take a single data pointer as an argument
2272 * (the type of the pointer is irrelevant)
2273 */
2274
2275 case 'b': /* byte -- very short int */
2276 case 'B': /* byte as bitfield */
2277 case 'h': /* short int */
2278 case 'H': /* short int as bitfield */
2279 case 'i': /* int */
2280 case 'I': /* int sized bitfield */
2281 case 'l': /* long int */
2282 case 'k': /* long int sized bitfield */
2283 case 'L': /* long long */
2284 case 'K': /* long long sized bitfield */
2285 case 'n': /* Py_ssize_t */
2286 case 'f': /* float */
2287 case 'd': /* double */
2288 case 'D': /* complex double */
2289 case 'c': /* char */
2290 case 'C': /* unicode char */
2291 case 'p': /* boolean predicate */
2292 case 'S': /* string object */
2293 case 'Y': /* string object */
2294 case 'U': /* unicode string object */
2295 {
2296 if (p_va != NULL) {
2297 (void) va_arg(*p_va, void *);
2298 }
2299 break;
2300 }
2301
2302 /* string codes */
2303
2304 case 'e': /* string with encoding */
2305 {
2306 if (p_va != NULL) {
2307 (void) va_arg(*p_va, const char *);
2308 }
2309 if (!(*format == 's' || *format == 't'))
2310 /* after 'e', only 's' and 't' is allowed */
2311 goto err;
2312 format++;
2313 }
2314 /* fall through */
2315
2316 case 's': /* string */
2317 case 'z': /* string or None */
2318 case 'y': /* bytes */
2319 case 'u': /* unicode string */
2320 case 'Z': /* unicode string or None */
2321 case 'w': /* buffer, read-write */
2322 {
2323 if (p_va != NULL) {
2324 (void) va_arg(*p_va, char **);
2325 }
2326 if (*format == '#') {
2327 if (p_va != NULL) {
2328 if (flags & FLAG_SIZE_T)
2329 (void) va_arg(*p_va, Py_ssize_t *);
2330 else
2331 (void) va_arg(*p_va, int *);
2332 }
2333 format++;
2334 } else if ((c == 's' || c == 'z' || c == 'y') && *format == '*') {
2335 format++;
2336 }
2337 break;
2338 }
2339
2340 case 'O': /* object */
2341 {
2342 if (*format == '!') {
2343 format++;
2344 if (p_va != NULL) {
2345 (void) va_arg(*p_va, PyTypeObject*);
2346 (void) va_arg(*p_va, PyObject **);

2349 else if (*format == '&') {

2351 if (p_va != NULL) {
2352 (void) va_arg(*p_va, converter);
2353 (void) va_arg(*p_va, void *);
2354 }
2355 format++;
2356 }
2357 else {
2358 if (p_va != NULL) {
2359 (void) va_arg(*p_va, PyObject **);
2360 }
2361 }
2362 break;
2363 }
2364
2365 case '(': /* bypass tuple, not handled at all previously */

2383 return "Unmatched right paren in format string";

2385 default:
2386err:

2388
2389 }
2390
2391 *p_format = format;

2397unpack_stack(PyObject **args, Py_ssize_t nargs, const char *name,
2398 Py_ssize_t min, Py_ssize_t max, va_list vargs)
2399{
2400 Py_ssize_t i;
2401 PyObject **o;
2402
2403 assert(min >= 0);
2404 assert(min <= max);
2405
2406 if (nargs < min) {
2407 if (name != NULL)
2408 PyErr_Format(
2409 PyExc_TypeError,
2410 "%.200s expected %s%zd arguments, got %zd",
2411 name, (min == max ? "" : "at least "), min, nargs);
2412 else
2413 PyErr_Format(
2414 PyExc_TypeError,
2415 "unpacked tuple should have %s%zd elements,"
2416 " but has %zd",
2417 (min == max ? "" : "at least "), min, nargs);
2418 return 0;
2419 }
2420
2421 if (nargs == 0) {
2422 return 1;
2423 }
2424
2425 if (nargs > max) {
2426 if (name != NULL)
2427 PyErr_Format(
2428 PyExc_TypeError,
2429 "%.200s expected %s%zd arguments, got %zd",
2430 name, (min == max ? "" : "at most "), max, nargs);
2431 else
2432 PyErr_Format(
2433 PyExc_TypeError,
2434 "unpacked tuple should have %s%zd elements,"
2435 " but has %zd",
2436 (min == max ? "" : "at most "), max, nargs);
2437 return 0;
2438 }
2439
2440 for (i = 0; i < nargs; i++) {
2441 o = va_arg(vargs, PyObject **);
2442 *o = args[i];
2443 }
2444 return 1;
2445}
