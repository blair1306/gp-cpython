110#define MICROSECONDS_TO_TIMESPEC(microseconds, ts) \
111do { \
112 struct timeval tv; \
113 GETTIMEOFDAY(&tv); \
114 tv.tv_usec += microseconds % 1000000; \
115 tv.tv_sec += microseconds / 1000000; \
116 tv.tv_sec += tv.tv_usec / 1000000; \
117 tv.tv_usec %= 1000000; \
118 ts.tv_sec = tv.tv_sec; \
119 ts.tv_nsec = tv.tv_usec * 1000; \
120} while(0)

123/* A pthread mutex isn't sufficient to model the Python lock type
124 * because, according to Draft 5 of the docs (P1003.4a/D5), both of the
125 * following are undefined:
126 * -> a thread tries to lock a mutex it already has locked
127 * -> a thread tries to unlock a mutex locked by a different thread
128 * pthread mutexes are designed for serializing threads over short pieces
129 * of code anyway, so wouldn't be an appropriate implementation of
130 * Python's locks regardless.
131 *
132 * The pthread_lock struct implements a Python lock as a "locked?" bit
133 * and a <condition, mutex> pair. In general, if the bit can be acquired
134 * instantly, it is, else the pair is used to block the thread until the
135 * bit is cleared. 9 May 1994 tim@ksr.com
136 */
137
138typedef struct {
139 char locked; /* 0=unlocked, 1=locked */
140 /* a <cond, mutex> pair to handle an acquire of a locked lock */
141 pthread_cond_t lock_released;
142 pthread_mutex_t mut;
143} pthread_lock;

187unsigned long
188PyThread_start_new_thread(void (*func)(void *), void *arg)
189{
190 pthread_t th;
191 int status;
192#if defined(THREAD_STACK_SIZE) || defined(PTHREAD_SYSTEM_SCHED_SUPPORTED)
193 pthread_attr_t attrs;
194#endif
195#if defined(THREAD_STACK_SIZE)
196 size_t tss;
197#endif
198
199 dprintf(("PyThread_start_new_thread called\n"));
200 if (!initialized)
201 PyThread_init_thread();
202
203#if defined(THREAD_STACK_SIZE) || defined(PTHREAD_SYSTEM_SCHED_SUPPORTED)
204 if (pthread_attr_init(&attrs) != 0)
205 return PYTHREAD_INVALID_THREAD_ID;
206#endif
207#if defined(THREAD_STACK_SIZE)
208 tss = (_pythread_stacksize != 0) ? _pythread_stacksize
209 : THREAD_STACK_SIZE;
210 if (tss != 0) {
211 if (pthread_attr_setstacksize(&attrs, tss) != 0) {
212 pthread_attr_destroy(&attrs);
213 return PYTHREAD_INVALID_THREAD_ID;
214 }
215 }
216#endif
217#if defined(PTHREAD_SYSTEM_SCHED_SUPPORTED)
218 pthread_attr_setscope(&attrs, PTHREAD_SCOPE_SYSTEM);
219#endif
220
221 status = pthread_create(&th,
222#if defined(THREAD_STACK_SIZE) || defined(PTHREAD_SYSTEM_SCHED_SUPPORTED)
223 &attrs,
224#else
225 (pthread_attr_t*)NULL,
226#endif
227 (void* (*)(void *))func,
228 (void *)arg
229 );
230
231#if defined(THREAD_STACK_SIZE) || defined(PTHREAD_SYSTEM_SCHED_SUPPORTED)
232 pthread_attr_destroy(&attrs);
233#endif
234 if (status != 0)
235 return PYTHREAD_INVALID_THREAD_ID;
236
237 pthread_detach(th);
238
239#if SIZEOF_PTHREAD_T <= SIZEOF_LONG
240 return (unsigned long) th;
241#else
242 return (unsigned long) *(unsigned long *) &th;
243#endif
244}

333PyLockStatus
334PyThread_acquire_lock_timed(PyThread_type_lock lock, PY_TIMEOUT_T microseconds,
335 int intr_flag)
336{
337 PyLockStatus success;
338 sem_t *thelock = (sem_t *)lock;
339 int status, error = 0;
340 struct timespec ts;
341
342 (void) error; /* silence unused-but-set-variable warning */
343 dprintf(("PyThread_acquire_lock_timed(%p, %lld, %d) called\n",
344 lock, microseconds, intr_flag));
345
346 if (microseconds > 0)
347 MICROSECONDS_TO_TIMESPEC(microseconds, ts);
348 do {
349 if (microseconds > 0)
350 status = fix_status(sem_timedwait(thelock, &ts));
351 else if (microseconds == 0)
352 status = fix_status(sem_trywait(thelock));
353 else
354 status = fix_status(sem_wait(thelock));
355 /* Retry if interrupted by a signal, unless the caller wants to be
356 notified. */
357 } while (!intr_flag && status == EINTR);
358
359 /* Don't check the status if we're stopping because of an interrupt. */
360 if (!(intr_flag && status == EINTR)) {
361 if (microseconds > 0) {
362 if (status != ETIMEDOUT)
363 CHECK_STATUS("sem_timedwait");
364 }
365 else if (microseconds == 0) {
366 if (status != EAGAIN)
367 CHECK_STATUS("sem_trywait");
368 }
369 else {
370 CHECK_STATUS("sem_wait");
371 }
372 }
373
374 if (status == 0) {
375 success = PY_LOCK_ACQUIRED;
376 } else if (intr_flag && status == EINTR) {
377 success = PY_LOCK_INTR;
378 } else {
379 success = PY_LOCK_FAILURE;
380 }
381
382 dprintf(("PyThread_acquire_lock_timed(%p, %lld, %d) -> %d\n",
383 lock, microseconds, intr_flag, success));
384 return success;
385}

403 * Lock support.

405PyThread_type_lock
406PyThread_allocate_lock(void)
407{
408 pthread_lock *lock;
409 int status, error = 0;
410
411 dprintf(("PyThread_allocate_lock called\n"));
412 if (!initialized)
413 PyThread_init_thread();
414
415 lock = (pthread_lock *) PyMem_RawMalloc(sizeof(pthread_lock));
416 if (lock) {
417 memset((void *)lock, '\0', sizeof(pthread_lock));
418 lock->locked = 0;
419
420 status = pthread_mutex_init(&lock->mut,
421 pthread_mutexattr_default);
422 CHECK_STATUS_PTHREAD("pthread_mutex_init");
423 /* Mark the pthread mutex underlying a Python mutex as
424 pure happens-before. We can't simply mark the
425 Python-level mutex as a mutex because it can be
426 acquired and released in different threads, which
427 will cause errors. */
428 _Py_ANNOTATE_PURE_HAPPENS_BEFORE_MUTEX(&lock->mut);
429
430 status = pthread_cond_init(&lock->lock_released,
431 pthread_condattr_default);
432 CHECK_STATUS_PTHREAD("pthread_cond_init");
433
434 if (error) {
435 PyMem_RawFree((void *)lock);
436 lock = 0;
437 }
438 }
439
440 dprintf(("PyThread_allocate_lock() -> %p\n", lock));
441 return (PyThread_type_lock) lock;
442}

444void
445PyThread_free_lock(PyThread_type_lock lock)
446{
447 pthread_lock *thelock = (pthread_lock *)lock;
448 int status, error = 0;
449
450 (void) error; /* silence unused-but-set-variable warning */
451 dprintf(("PyThread_free_lock(%p) called\n", lock));
452
453 /* some pthread-like implementations tie the mutex to the cond
454 * and must have the cond destroyed first.
455 */
456 status = pthread_cond_destroy( &thelock->lock_released );
457 CHECK_STATUS_PTHREAD("pthread_cond_destroy");
458
459 status = pthread_mutex_destroy( &thelock->mut );
460 CHECK_STATUS_PTHREAD("pthread_mutex_destroy");
461
462 PyMem_RawFree((void *)thelock);
463}

465PyLockStatus
466PyThread_acquire_lock_timed(PyThread_type_lock lock, PY_TIMEOUT_T microseconds,
467 int intr_flag)
468{
469 PyLockStatus success = PY_LOCK_FAILURE;
470 pthread_lock *thelock = (pthread_lock *)lock;
471 int status, error = 0;
472
473 dprintf(("PyThread_acquire_lock_timed(%p, %lld, %d) called\n",
474 lock, microseconds, intr_flag));
475
476 if (microseconds == 0) {
477 status = pthread_mutex_trylock( &thelock->mut );
478 if (status != EBUSY)
479 CHECK_STATUS_PTHREAD("pthread_mutex_trylock[1]");
480 }
481 else {
482 status = pthread_mutex_lock( &thelock->mut );
483 CHECK_STATUS_PTHREAD("pthread_mutex_lock[1]");
484 }
485 if (status == 0) {
486 if (thelock->locked == 0) {
487 success = PY_LOCK_ACQUIRED;
488 }
489 else if (microseconds != 0) {
490 struct timespec ts;
491 if (microseconds > 0)
492 MICROSECONDS_TO_TIMESPEC(microseconds, ts);
493 /* continue trying until we get the lock */
494
495 /* mut must be locked by me -- part of the condition
496 * protocol */
497 while (success == PY_LOCK_FAILURE) {
498 if (microseconds > 0) {
499 status = pthread_cond_timedwait(
500 &thelock->lock_released,
501 &thelock->mut, &ts);
502 if (status == ETIMEDOUT)
503 break;
504 CHECK_STATUS_PTHREAD("pthread_cond_timed_wait");
505 }
506 else {
507 status = pthread_cond_wait(
508 &thelock->lock_released,
509 &thelock->mut);
510 CHECK_STATUS_PTHREAD("pthread_cond_wait");
511 }
512
513 if (intr_flag && status == 0 && thelock->locked) {
514 /* We were woken up, but didn't get the lock. We probably received
515 * a signal. Return PY_LOCK_INTR to allow the caller to handle
516 * it and retry. */
517 success = PY_LOCK_INTR;
518 break;
519 }
520 else if (status == 0 && !thelock->locked) {
521 success = PY_LOCK_ACQUIRED;
522 }
523 }
524 }
525 if (success == PY_LOCK_ACQUIRED) thelock->locked = 1;
526 status = pthread_mutex_unlock( &thelock->mut );
527 CHECK_STATUS_PTHREAD("pthread_mutex_unlock[1]");
528 }
529
530 if (error) success = PY_LOCK_FAILURE;
531 dprintf(("PyThread_acquire_lock_timed(%p, %lld, %d) -> %d\n",
532 lock, microseconds, intr_flag, success));
533 return success;
534}

536void
537PyThread_release_lock(PyThread_type_lock lock)
538{
539 pthread_lock *thelock = (pthread_lock *)lock;
540 int status, error = 0;
541
542 (void) error; /* silence unused-but-set-variable warning */
543 dprintf(("PyThread_release_lock(%p) called\n", lock));
544
545 status = pthread_mutex_lock( &thelock->mut );
546 CHECK_STATUS_PTHREAD("pthread_mutex_lock[3]");
547
548 thelock->locked = 0;
549
550 /* wake up someone (anyone, if any) waiting on the lock */
551 status = pthread_cond_signal( &thelock->lock_released );
552 CHECK_STATUS_PTHREAD("pthread_cond_signal");
553
554 status = pthread_mutex_unlock( &thelock->mut );
555 CHECK_STATUS_PTHREAD("pthread_mutex_unlock[3]");
556}

566/* set the thread stack size.
567 * Return 0 if size is valid, -1 if size is invalid,
568 * -2 if setting stack size is not supported.
569 */
570static int
571_pythread_pthread_set_stacksize(size_t size)
572{
573#if defined(THREAD_STACK_SIZE)
574 pthread_attr_t attrs;
575 size_t tss_min;
576 int rc = 0;
577#endif
578
579 /* set to default */
580 if (size == 0) {
581 _pythread_stacksize = 0;
582 return 0;
583 }
584
585#if defined(THREAD_STACK_SIZE)
586#if defined(PTHREAD_STACK_MIN)
587 tss_min = PTHREAD_STACK_MIN > THREAD_STACK_MIN ? PTHREAD_STACK_MIN
588 : THREAD_STACK_MIN;
589#else
590 tss_min = THREAD_STACK_MIN;
591#endif
592 if (size >= tss_min) {
593 /* validate stack size by setting thread attribute */
594 if (pthread_attr_init(&attrs) == 0) {
595 rc = pthread_attr_setstacksize(&attrs, size);
596 pthread_attr_destroy(&attrs);
597 if (rc == 0) {
598 _pythread_stacksize = size;
599 return 0;
600 }
601 }
602 }
603 return -1;
604#else
605 return -2;
606#endif
607}
