96/* Write repr(o) to sys.stdout using sys.stdout.encoding and 'backslashreplace'
97 error handler. If sys.stdout has a buffer attribute, use
98 sys.stdout.buffer.write(encoded), otherwise redecode the string and use
99 sys.stdout.write(redecoded).
100
101 Helper function for sys_displayhook(). */
102static int
103sys_displayhook_unencodable(PyObject *outf, PyObject *o)
104{
105 PyObject *stdout_encoding = NULL;
106 PyObject *encoded, *escaped_str, *repr_str, *buffer, *result;
107 const char *stdout_encoding_str;
108 int ret;
109
110 stdout_encoding = _PyObject_GetAttrId(outf, &PyId_encoding);
111 if (stdout_encoding == NULL)
112 goto error;
113 stdout_encoding_str = PyUnicode_AsUTF8(stdout_encoding);
114 if (stdout_encoding_str == NULL)
115 goto error;
116
117 repr_str = PyObject_Repr(o);
118 if (repr_str == NULL)
119 goto error;
120 encoded = PyUnicode_AsEncodedString(repr_str,
121 stdout_encoding_str,
122 "backslashreplace");
123 Py_DECREF(repr_str);
124 if (encoded == NULL)
125 goto error;
126
127 buffer = _PyObject_GetAttrId(outf, &PyId_buffer);
128 if (buffer) {
129 result = _PyObject_CallMethodIdObjArgs(buffer, &PyId_write, encoded, NULL);
130 Py_DECREF(buffer);
131 Py_DECREF(encoded);
132 if (result == NULL)
133 goto error;
134 Py_DECREF(result);
135 }
136 else {
137 PyErr_Clear();
138 escaped_str = PyUnicode_FromEncodedObject(encoded,
139 stdout_encoding_str,
140 "strict");
141 Py_DECREF(encoded);
142 if (PyFile_WriteObject(escaped_str, outf, Py_PRINT_RAW) != 0) {
143 Py_DECREF(escaped_str);
144 goto error;
145 }
146 Py_DECREF(escaped_str);
147 }
148 ret = 0;
149 goto finally;
150
151error:
152 ret = -1;
153finally:
154 Py_XDECREF(stdout_encoding);
155 return ret;
156}

158static PyObject *
159sys_displayhook(PyObject *self, PyObject *o)
160{
161 PyObject *outf;
162 PyInterpreterState *interp = PyThreadState_GET()->interp;
163 PyObject *modules = interp->modules;
164 PyObject *builtins;
165 static PyObject *newline = NULL;
166 int err;
167
168 builtins = _PyDict_GetItemId(modules, &PyId_builtins);
169 if (builtins == NULL) {
170 PyErr_SetString(PyExc_RuntimeError, "lost builtins module");
171 return NULL;
172 }
173
174 /* Print value except if None */
175 /* After printing, also assign to '_' */
176 /* Before, set '_' to None to avoid recursion */
177 if (o == Py_None) {
178 Py_RETURN_NONE;
179 }
180 if (_PyObject_SetAttrId(builtins, &PyId__, Py_None) != 0)
181 return NULL;
182 outf = _PySys_GetObjectId(&PyId_stdout);
183 if (outf == NULL || outf == Py_None) {
184 PyErr_SetString(PyExc_RuntimeError, "lost sys.stdout");
185 return NULL;
186 }
187 if (PyFile_WriteObject(o, outf, 0) != 0) {
188 if (PyErr_ExceptionMatches(PyExc_UnicodeEncodeError)) {
189 /* repr(o) is not encodable to sys.stdout.encoding with
190 * sys.stdout.errors error handler (which is probably 'strict') */
191 PyErr_Clear();
192 err = sys_displayhook_unencodable(outf, o);
193 if (err)
194 return NULL;
195 }
196 else {
197 return NULL;
198 }
199 }
200 if (newline == NULL) {
201 newline = PyUnicode_FromString("\n");
202 if (newline == NULL)
203 return NULL;
204 }
205 if (PyFile_WriteObject(newline, outf, Py_PRINT_RAW) != 0)
206 return NULL;
207 if (_PyObject_SetAttrId(builtins, &PyId__, o) != 0)
208 return NULL;
209 Py_RETURN_NONE;
210}

356static int
357trace_init(void)
358{
359 static const char * const whatnames[7] = {
360 "call", "exception", "line", "return",
361 "c_call", "c_exception", "c_return"
362 };
363 PyObject *name;
364 int i;
365 for (i = 0; i < 7; ++i) {
366 if (whatstrings[i] == NULL) {
367 name = PyUnicode_InternFromString(whatnames[i]);
368 if (name == NULL)
369 return -1;
370 whatstrings[i] = name;
371 }
372 }
373 return 0;
374}

377static PyObject *
378call_trampoline(PyObject* callback,
379 PyFrameObject *frame, int what, PyObject *arg)
380{
381 PyObject *result;
382 PyObject *stack[3];
383
384 if (PyFrame_FastToLocalsWithError(frame) < 0) {
385 return NULL;
386 }
387
388 stack[0] = (PyObject *)frame;
389 stack[1] = whatstrings[what];
390 stack[2] = (arg != NULL) ? arg : Py_None;
391
392 /* call the Python-level function */
393 result = _PyObject_FastCall(callback, stack, 3);
394
395 PyFrame_LocalsToFast(frame, 1);
396 if (result == NULL) {
397 PyTraceBack_Here(frame);
398 }
399
400 return result;
401}

403static int
404profile_trampoline(PyObject *self, PyFrameObject *frame,
405 int what, PyObject *arg)
406{
407 PyObject *result;
408
409 if (arg == NULL)
410 arg = Py_None;
411 result = call_trampoline(self, frame, what, arg);
412 if (result == NULL) {
413 PyEval_SetProfile(NULL, NULL);
414 return -1;
415 }
416 Py_DECREF(result);
417 return 0;

420static int
421trace_trampoline(PyObject *self, PyFrameObject *frame,
422 int what, PyObject *arg)
423{
424 PyObject *callback;
425 PyObject *result;
426
427 if (what == PyTrace_CALL)
428 callback = self;
429 else
430 callback = frame->f_trace;
431 if (callback == NULL)
432 return 0;
433 result = call_trampoline(callback, frame, what, arg);
434 if (result == NULL) {
435 PyEval_SetTrace(NULL, NULL);
436 Py_CLEAR(frame->f_trace);
437 return -1;
438 }
439 if (result != Py_None) {
440 Py_XSETREF(frame->f_trace, result);
441 }
442 else {
443 Py_DECREF(result);
444 }
445 return 0;
446}

601static PyObject *
602sys_setrecursionlimit(PyObject *self, PyObject *args)
603{
604 int new_limit, mark;
605 PyThreadState *tstate;
606
607 if (!PyArg_ParseTuple(args, "i:setrecursionlimit", &new_limit))
608 return NULL;
609
610 if (new_limit < 1) {
611 PyErr_SetString(PyExc_ValueError,
612 "recursion limit must be greater or equal than 1");
613 return NULL;
614 }
615
616 /* Issue #25274: When the recursion depth hits the recursion limit in
617 _Py_CheckRecursiveCall(), the overflowed flag of the thread state is
618 set to 1 and a RecursionError is raised. The overflowed flag is reset
619 to 0 when the recursion depth goes below the low-water mark: see
620 Py_LeaveRecursiveCall().
621
622 Reject too low new limit if the current recursion depth is higher than
623 the new low-water mark. Otherwise it may not be possible anymore to
624 reset the overflowed flag to 0. */
625 mark = _Py_RecursionLimitLowerWaterMark(new_limit);
626 tstate = PyThreadState_GET();
627 if (tstate->recursion_depth >= mark) {
628 PyErr_Format(PyExc_RecursionError,
629 "cannot set the recursion limit to %i at "
630 "the recursion depth %i: the limit is too low",
631 new_limit, tstate->recursion_depth);
632 return NULL;
633 }
634
635 Py_SetRecursionLimit(new_limit);
636 Py_RETURN_NONE;
637}

918 being passed straight through to the caller, who is responsible for
919 using it correctly. */
920#pragma warning(push)
921#pragma warning(disable:4996)

923static PyObject *
924sys_getwindowsversion(PyObject *self)
925{
926 PyObject *version;
927 int pos = 0;
928 OSVERSIONINFOEX ver;
929 DWORD realMajor, realMinor, realBuild;
930 HANDLE hKernel32;
931 wchar_t kernel32_path[MAX_PATH];
932 LPVOID verblock;
933 DWORD verblock_size;
934
935 ver.dwOSVersionInfoSize = sizeof(ver);
936 if (!GetVersionEx((OSVERSIONINFO*) &ver))
937 return PyErr_SetFromWindowsErr(0);
938
939 version = PyStructSequence_New(&WindowsVersionType);
940 if (version == NULL)
941 return NULL;
942
943 PyStructSequence_SET_ITEM(version, pos++, PyLong_FromLong(ver.dwMajorVersion));
944 PyStructSequence_SET_ITEM(version, pos++, PyLong_FromLong(ver.dwMinorVersion));
945 PyStructSequence_SET_ITEM(version, pos++, PyLong_FromLong(ver.dwBuildNumber));
946 PyStructSequence_SET_ITEM(version, pos++, PyLong_FromLong(ver.dwPlatformId));
947 PyStructSequence_SET_ITEM(version, pos++, PyUnicode_FromString(ver.szCSDVersion));
948 PyStructSequence_SET_ITEM(version, pos++, PyLong_FromLong(ver.wServicePackMajor));
949 PyStructSequence_SET_ITEM(version, pos++, PyLong_FromLong(ver.wServicePackMinor));
950 PyStructSequence_SET_ITEM(version, pos++, PyLong_FromLong(ver.wSuiteMask));
951 PyStructSequence_SET_ITEM(version, pos++, PyLong_FromLong(ver.wProductType));
952
953 realMajor = ver.dwMajorVersion;
954 realMinor = ver.dwMinorVersion;
955 realBuild = ver.dwBuildNumber;
956
957 // GetVersion will lie if we are running in a compatibility mode.
958 // We need to read the version info from a system file resource
959 // to accurately identify the OS version. If we fail for any reason,
960 // just return whatever GetVersion said.
961 hKernel32 = GetModuleHandleW(L"kernel32.dll");
962 if (hKernel32 && GetModuleFileNameW(hKernel32, kernel32_path, MAX_PATH) &&
963 (verblock_size = GetFileVersionInfoSizeW(kernel32_path, NULL)) &&
964 (verblock = PyMem_RawMalloc(verblock_size))) {
965 VS_FIXEDFILEINFO *ffi;
966 UINT ffi_len;
967
968 if (GetFileVersionInfoW(kernel32_path, 0, verblock_size, verblock) &&
969 VerQueryValueW(verblock, L"", (LPVOID)&ffi, &ffi_len)) {
970 realMajor = HIWORD(ffi->dwProductVersionMS);
971 realMinor = LOWORD(ffi->dwProductVersionMS);
972 realBuild = HIWORD(ffi->dwProductVersionLS);
973 }
974 PyMem_RawFree(verblock);
975 }
976 PyStructSequence_SET_ITEM(version, pos++, Py_BuildValue("(kkk)",
977 realMajor,
978 realMinor,
979 realBuild
980 ));
981
982 if (PyErr_Occurred()) {
983 Py_DECREF(version);
984 return NULL;
985 }
986
987 return version;
988}

1068size_t
1069_PySys_GetSizeOf(PyObject *o)
1070{
1071 PyObject *res = NULL;
1072 PyObject *method;
1073 Py_ssize_t size;
1074
1075 /* Make sure the type is initialized. float gets initialized late */
1076 if (PyType_Ready(Py_TYPE(o)) < 0)
1077 return (size_t)-1;
1078
1079 method = _PyObject_LookupSpecial(o, &PyId___sizeof__);
1080 if (method == NULL) {
1081 if (!PyErr_Occurred())
1082 PyErr_Format(PyExc_TypeError,
1083 "Type %.100s doesn't define __sizeof__",
1084 Py_TYPE(o)->tp_name);
1085 }
1086 else {
1087 res = _PyObject_CallNoArg(method);
1088 Py_DECREF(method);
1089 }
1090
1091 if (res == NULL)
1092 return (size_t)-1;
1093
1094 size = PyLong_AsSsize_t(res);
1095 Py_DECREF(res);
1096 if (size == -1 && PyErr_Occurred())
1097 return (size_t)-1;
1098
1099 if (size < 0) {
1100 PyErr_SetString(PyExc_ValueError, "__sizeof__() should return >= 0");
1101 return (size_t)-1;
1102 }
1103
1104 /* add gc_head size */
1105 if (PyObject_IS_GC(o))
1106 return ((size_t)size) + sizeof(PyGC_Head);
1107 return (size_t)size;
1108}

2398/* APIs to write to sys.stdout or sys.stderr using a printf-like interface.
2399 Adapted from code submitted by Just van Rossum.
2400
2401 PySys_WriteStdout(format, ...)
2402 PySys_WriteStderr(format, ...)
2403
2404 The first function writes to sys.stdout; the second to sys.stderr. When
2405 there is a problem, they write to the real (C level) stdout or stderr;
2406 no exceptions are raised.
2407
2408 PyErr_CheckSignals() is not called to avoid the execution of the Python
2409 signal handlers: they may raise a new exception whereas sys_write()
2410 ignores all exceptions.
2411
2412 Both take a printf-style format string as their first argument followed
2413 by a variable length argument list determined by the format string.
2414
2415 *** WARNING ***
2416
2417 The format should limit the total size of the formatted output string to
2418 1000 bytes. In particular, this means that no unrestricted "%s" formats
2419 should occur; these should be limited using "%.<N>s where <N> is a
2420 decimal number calculated so that <N> plus the maximum size of other
2421 formatted text does not exceed 1000 bytes. Also watch out for "%f",
2422 which can print hundreds of digits for very large numbers.
2423
2424 */
