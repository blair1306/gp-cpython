6int
7PyOS_mystrnicmp(const char *s1, const char *s2, Py_ssize_t size)
8{
9 if (size == 0)
10 return 0;
11 while ((--size > 0) &&
12 (tolower((unsigned)*s1) == tolower((unsigned)*s2))) {
13 if (!*s1++ || !*s2++)
14 break;
15 }
16 return tolower((unsigned)*s1) - tolower((unsigned)*s2);
17}
