68/* Parse input from a file and execute it */
69int
70PyRun_AnyFileExFlags(FILE *fp, const char *filename, int closeit,
71 PyCompilerFlags *flags)
72{
73 if (filename == NULL)
74 filename = "???";
75 if (Py_FdIsInteractive(fp, filename)) {
76 int err = PyRun_InteractiveLoopFlags(fp, filename, flags);
77 if (closeit)
78 fclose(fp);
79 return err;
80 }
81 else
82 return PyRun_SimpleFileExFlags(fp, filename, closeit, flags);
83}

85int
86PyRun_InteractiveLoopFlags(FILE *fp, const char *filename_str, PyCompilerFlags *flags)
87{
88 PyObject *filename, *v;
89 int ret, err;
90 PyCompilerFlags local_flags;
91
92 filename = PyUnicode_DecodeFSDefault(filename_str);
93 if (filename == NULL) {
94 PyErr_Print();
95 return -1;
96 }
97
98 if (flags == NULL) {
99 flags = &local_flags;
100 local_flags.cf_flags = 0;
101 }
102 v = _PySys_GetObjectId(&PyId_ps1);
103 if (v == NULL) {
104 _PySys_SetObjectId(&PyId_ps1, v = PyUnicode_FromString(">>> "));
105 Py_XDECREF(v);
106 }
107 v = _PySys_GetObjectId(&PyId_ps2);
108 if (v == NULL) {
109 _PySys_SetObjectId(&PyId_ps2, v = PyUnicode_FromString("... "));
110 Py_XDECREF(v);
111 }
112 err = -1;
113 for (;;) {
114 ret = PyRun_InteractiveOneObject(fp, filename, flags);
115 _PY_DEBUG_PRINT_TOTAL_REFS();
116 if (ret == E_EOF) {
117 err = 0;
118 break;
119 }
120 /*
121 if (ret == E_NOMEM)
122 break;
123 */
124 }
125 Py_DECREF(filename);
126 return err;
127}

153int
154PyRun_InteractiveOneObject(FILE *fp, PyObject *filename, PyCompilerFlags *flags)
155{
156 PyObject *m, *d, *v, *w, *oenc = NULL, *mod_name;
157 mod_ty mod;
158 PyArena *arena;
159 const char *ps1 = "", *ps2 = "", *enc = NULL;
160 int errcode = 0;
161 _Py_IDENTIFIER(encoding);
162 _Py_IDENTIFIER(__main__);
163
164 mod_name = _PyUnicode_FromId(&PyId___main__); /* borrowed */
165 if (mod_name == NULL) {
166 PyErr_Print();
167 return -1;
168 }
169
170 if (fp == stdin) {
171 /* Fetch encoding from sys.stdin if possible. */
172 v = _PySys_GetObjectId(&PyId_stdin);
173 if (v && v != Py_None) {
174 oenc = _PyObject_GetAttrId(v, &PyId_encoding);
175 if (oenc)
176 enc = PyUnicode_AsUTF8(oenc);
177 if (!enc)
178 PyErr_Clear();
179 }
180 }
181 v = _PySys_GetObjectId(&PyId_ps1);
182 if (v != NULL) {
183 v = PyObject_Str(v);
184 if (v == NULL)
185 PyErr_Clear();
186 else if (PyUnicode_Check(v)) {
187 ps1 = PyUnicode_AsUTF8(v);
188 if (ps1 == NULL) {
189 PyErr_Clear();
190 ps1 = "";
191 }
192 }
193 }
194 w = _PySys_GetObjectId(&PyId_ps2);
195 if (w != NULL) {
196 w = PyObject_Str(w);
197 if (w == NULL)
198 PyErr_Clear();
199 else if (PyUnicode_Check(w)) {
200 ps2 = PyUnicode_AsUTF8(w);
201 if (ps2 == NULL) {
202 PyErr_Clear();
203 ps2 = "";
204 }
205 }
206 }
207 arena = PyArena_New();
208 if (arena == NULL) {
209 Py_XDECREF(v);
210 Py_XDECREF(w);
211 Py_XDECREF(oenc);
212 return -1;
213 }
214 mod = PyParser_ASTFromFileObject(fp, filename, enc,
215 Py_single_input, ps1, ps2,
216 flags, &errcode, arena);
217 Py_XDECREF(v);
218 Py_XDECREF(w);
219 Py_XDECREF(oenc);
220 if (mod == NULL) {
221 PyArena_Free(arena);
222 if (errcode == E_EOF) {
223 PyErr_Clear();
224 return E_EOF;
225 }
226 PyErr_Print();
227 return -1;
228 }
229 m = PyImport_AddModuleObject(mod_name);
230 if (m == NULL) {
231 PyArena_Free(arena);
232 return -1;
233 }
234 d = PyModule_GetDict(m);
235 v = run_mod(mod, filename, d, d, flags, arena);
236 PyArena_Free(arena);
237 if (v == NULL) {
238 PyErr_Print();
239 flush_io();
240 return -1;
241 }
242 Py_DECREF(v);
243 flush_io();
244 return 0;
245}

264/* Check whether a file maybe a pyc file: Look at the extension,
265 the file type, and, if we may close it, at the first few bytes. */
266
267static int
268maybe_pyc_file(FILE *fp, const char* filename, const char* ext, int closeit)
269{
270 if (strcmp(ext, ".pyc") == 0)
271 return 1;
272
273 /* Only look into the file if we are allowed to close it, since
274 it then should also be seekable. */
275 if (closeit) {
276 /* Read only two bytes of the magic. If the file was opened in
277 text mode, the bytes 3 and 4 of the magic (\r\n) might not
278 be read as they are on disk. */
279 unsigned int halfmagic = PyImport_GetMagicNumber() & 0xFFFF;
280 unsigned char buf[2];
281 /* Mess: In case of -x, the stream is NOT at its start now,
282 and ungetc() was used to push back the first newline,
283 which makes the current stream position formally undefined,
284 and a x-platform nightmare.
285 Unfortunately, we have no direct way to know whether -x
286 was specified. So we use a terrible hack: if the current
287 stream position is not 0, we assume -x was specified, and
288 give up. Bug 132850 on SourceForge spells out the
289 hopelessness of trying anything else (fseek and ftell
290 don't work predictably x-platform for text-mode files).
291 */
292 int ispyc = 0;
293 if (ftell(fp) == 0) {
294 if (fread(buf, 1, 2, fp) == 2 &&
295 ((unsigned int)buf[1]<<8 | buf[0]) == halfmagic)
296 ispyc = 1;
297 rewind(fp);
298 }
299 return ispyc;
300 }
301 return 0;
302}

304static int
305set_main_loader(PyObject *d, const char *filename, const char *loader_name)
306{
307 PyInterpreterState *interp;
308 PyThreadState *tstate;
309 PyObject *filename_obj, *bootstrap, *loader_type = NULL, *loader;
310 int result = 0;
311
312 filename_obj = PyUnicode_DecodeFSDefault(filename);
313 if (filename_obj == NULL)
314 return -1;
315 /* Get current thread state and interpreter pointer */
316 tstate = PyThreadState_GET();
317 interp = tstate->interp;
318 bootstrap = PyObject_GetAttrString(interp->importlib,
319 "_bootstrap_external");
320 if (bootstrap != NULL) {
321 loader_type = PyObject_GetAttrString(bootstrap, loader_name);
322 Py_DECREF(bootstrap);
323 }
324 if (loader_type == NULL) {
325 Py_DECREF(filename_obj);
326 return -1;
327 }
328 loader = PyObject_CallFunction(loader_type, "sN", "__main__", filename_obj);
329 Py_DECREF(loader_type);
330 if (loader == NULL) {
331 return -1;
332 }
333 if (PyDict_SetItemString(d, "__loader__", loader) < 0) {
334 result = -1;
335 }
336 Py_DECREF(loader);
337 return result;
338}

340int
341PyRun_SimpleFileExFlags(FILE *fp, const char *filename, int closeit,
342 PyCompilerFlags *flags)
343{
344 PyObject *m, *d, *v;
345 const char *ext;
346 int set_file_name = 0, ret = -1;
347 size_t len;
348
349 m = PyImport_AddModule("__main__");
350 if (m == NULL)
351 return -1;
352 Py_INCREF(m);
353 d = PyModule_GetDict(m);
354 if (PyDict_GetItemString(d, "__file__") == NULL) {
355 PyObject *f;
356 f = PyUnicode_DecodeFSDefault(filename);
357 if (f == NULL)
358 goto done;
359 if (PyDict_SetItemString(d, "__file__", f) < 0) {
360 Py_DECREF(f);
361 goto done;
362 }
363 if (PyDict_SetItemString(d, "__cached__", Py_None) < 0) {
364 Py_DECREF(f);
365 goto done;
366 }
367 set_file_name = 1;
368 Py_DECREF(f);
369 }
370 len = strlen(filename);
371 ext = filename + len - (len > 4 ? 4 : 0);
372 if (maybe_pyc_file(fp, filename, ext, closeit)) {
373 FILE *pyc_fp;
374 /* Try to run a pyc file. First, re-open in binary */
375 if (closeit)
376 fclose(fp);
377 if ((pyc_fp = _Py_fopen(filename, "rb")) == NULL) {
378 fprintf(stderr, "python: Can't reopen .pyc file\n");
379 goto done;
380 }
381
382 if (set_main_loader(d, filename, "SourcelessFileLoader") < 0) {
383 fprintf(stderr, "python: failed to set __main__.__loader__\n");
384 ret = -1;
385 fclose(pyc_fp);
386 goto done;
387 }
388 v = run_pyc_file(pyc_fp, filename, d, d, flags);
389 fclose(pyc_fp);
390 } else {
391 /* When running from stdin, leave __main__.__loader__ alone */
392 if (strcmp(filename, "<stdin>") != 0 &&
393 set_main_loader(d, filename, "SourceFileLoader") < 0) {
394 fprintf(stderr, "python: failed to set __main__.__loader__\n");
395 ret = -1;
396 goto done;
397 }
398 v = PyRun_FileExFlags(fp, filename, Py_file_input, d, d,
399 closeit, flags);
400 }
401 flush_io();
402 if (v == NULL) {
403 PyErr_Print();
404 goto done;
405 }
406 Py_DECREF(v);
407 ret = 0;
408 done:
409 if (set_file_name && PyDict_DelItemString(d, "__file__"))
410 PyErr_Clear();
411 Py_DECREF(m);
412 return ret;
413}

432static int
433parse_syntax_error(PyObject *err, PyObject **message, PyObject **filename,
434 int *lineno, int *offset, PyObject **text)
435{
436 int hold;
437 PyObject *v;
438 _Py_IDENTIFIER(msg);
439 _Py_IDENTIFIER(filename);
440 _Py_IDENTIFIER(lineno);
441 _Py_IDENTIFIER(offset);
442 _Py_IDENTIFIER(text);
443
444 *message = NULL;
445 *filename = NULL;
446
447 /* new style errors. `err' is an instance */
448 *message = _PyObject_GetAttrId(err, &PyId_msg);
449 if (!*message)
450 goto finally;
451
452 v = _PyObject_GetAttrId(err, &PyId_filename);
453 if (!v)
454 goto finally;
455 if (v == Py_None) {
456 Py_DECREF(v);
457 *filename = _PyUnicode_FromId(&PyId_string);
458 if (*filename == NULL)
459 goto finally;
460 Py_INCREF(*filename);
461 }
462 else {
463 *filename = v;
464 }
465
466 v = _PyObject_GetAttrId(err, &PyId_lineno);
467 if (!v)
468 goto finally;
469 hold = _PyLong_AsInt(v);
470 Py_DECREF(v);
471 if (hold < 0 && PyErr_Occurred())
472 goto finally;
473 *lineno = hold;
474
475 v = _PyObject_GetAttrId(err, &PyId_offset);
476 if (!v)
477 goto finally;
478 if (v == Py_None) {
479 *offset = -1;
480 Py_DECREF(v);
481 } else {
482 hold = _PyLong_AsInt(v);
483 Py_DECREF(v);
484 if (hold < 0 && PyErr_Occurred())
485 goto finally;
486 *offset = hold;
487 }
488
489 v = _PyObject_GetAttrId(err, &PyId_text);
490 if (!v)
491 goto finally;
492 if (v == Py_None) {
493 Py_DECREF(v);
494 *text = NULL;
495 }
496 else {
497 *text = v;
498 }
499 return 1;
500
501finally:
502 Py_XDECREF(*message);
503 Py_XDECREF(*filename);
504 return 0;
505}

513static void
514print_error_text(PyObject *f, int offset, PyObject *text_obj)
515{
516 const char *text;
517 const char *nl;
518
519 text = PyUnicode_AsUTF8(text_obj);
520 if (text == NULL)
521 return;
522
523 if (offset >= 0) {
524 if (offset > 0 && (size_t)offset == strlen(text) && text[offset - 1] == '\n')
525 offset--;
526 for (;;) {
527 nl = strchr(text, '\n');
528 if (nl == NULL || nl-text >= offset)
529 break;
530 offset -= (int)(nl+1-text);
531 text = nl+1;
532 }
533 while (*text == ' ' || *text == '\t' || *text == '\f') {
534 text++;
535 offset--;
536 }
537 }
538 PyFile_WriteString(" ", f);
539 PyFile_WriteString(text, f);
540 if (*text == '\0' || text[strlen(text)-1] != '\n')
541 PyFile_WriteString("\n", f);
542 if (offset == -1)
543 return;
544 PyFile_WriteString(" ", f);
545 while (--offset > 0)
546 PyFile_WriteString(" ", f);
547 PyFile_WriteString("^\n", f);
548}

550static void
551handle_system_exit(void)
552{
553 PyObject *exception, *value, *tb;
554 int exitcode = 0;
555
556 if (Py_InspectFlag)
557 /* Don't exit if -i flag was given. This flag is set to 0
558 * when entering interactive mode for inspecting. */
559 return;
560
561 PyErr_Fetch(&exception, &value, &tb);
562 fflush(stdout);
563 if (value == NULL || value == Py_None)
564 goto done;
565 if (PyExceptionInstance_Check(value)) {
566 /* The error code should be in the `code' attribute. */
567 _Py_IDENTIFIER(code);
568 PyObject *code = _PyObject_GetAttrId(value, &PyId_code);
569 if (code) {
570 Py_DECREF(value);
571 value = code;
572 if (value == Py_None)
573 goto done;
574 }
575 /* If we failed to dig out the 'code' attribute,
576 just let the else clause below print the error. */
577 }
578 if (PyLong_Check(value))
579 exitcode = (int)PyLong_AsLong(value);
580 else {
581 PyObject *sys_stderr = _PySys_GetObjectId(&PyId_stderr);
582 /* We clear the exception here to avoid triggering the assertion
583 * in PyObject_Str that ensures it won't silently lose exception
584 * details.
585 */
586 PyErr_Clear();
587 if (sys_stderr != NULL && sys_stderr != Py_None) {
588 PyFile_WriteObject(value, sys_stderr, Py_PRINT_RAW);
589 } else {
590 PyObject_Print(value, stderr, Py_PRINT_RAW);
591 fflush(stderr);
592 }
593 PySys_WriteStderr("\n");
594 exitcode = 1;
595 }
596 done:
597 /* Restore and clear the exception info, in order to properly decref
598 * the exception, value, and traceback. If we just exit instead,
599 * these leak, which confuses PYTHONDUMPREFS output, and may prevent
600 * some finalizers from running.
601 */
602 PyErr_Restore(exception, value, tb);
603 PyErr_Clear();
604 Py_Exit(exitcode);
605 /* NOTREACHED */
606}

608void
609PyErr_PrintEx(int set_sys_last_vars)
610{
611 PyObject *exception, *v, *tb, *hook;
612
613 if (PyErr_ExceptionMatches(PyExc_SystemExit)) {
614 handle_system_exit();
615 }
616 PyErr_Fetch(&exception, &v, &tb);
617 if (exception == NULL)
618 return;
619 PyErr_NormalizeException(&exception, &v, &tb);
620 if (tb == NULL) {
621 tb = Py_None;
622 Py_INCREF(tb);
623 }
624 PyException_SetTraceback(v, tb);
625 if (exception == NULL)
626 return;
627 /* Now we know v != NULL too */
628 if (set_sys_last_vars) {
629 _PySys_SetObjectId(&PyId_last_type, exception);
630 _PySys_SetObjectId(&PyId_last_value, v);
631 _PySys_SetObjectId(&PyId_last_traceback, tb);
632 }
633 hook = _PySys_GetObjectId(&PyId_excepthook);
634 if (hook) {
635 PyObject* stack[3];
636 PyObject *result;
637
638 stack[0] = exception;
639 stack[1] = v;
640 stack[2] = tb;
641 result = _PyObject_FastCall(hook, stack, 3);
642 if (result == NULL) {
643 PyObject *exception2, *v2, *tb2;
644 if (PyErr_ExceptionMatches(PyExc_SystemExit)) {
645 handle_system_exit();
646 }
647 PyErr_Fetch(&exception2, &v2, &tb2);
648 PyErr_NormalizeException(&exception2, &v2, &tb2);
649 /* It should not be possible for exception2 or v2
650 to be NULL. However PyErr_Display() can't
651 tolerate NULLs, so just be safe. */
652 if (exception2 == NULL) {
653 exception2 = Py_None;
654 Py_INCREF(exception2);
655 }
656 if (v2 == NULL) {
657 v2 = Py_None;
658 Py_INCREF(v2);
659 }
660 fflush(stdout);
661 PySys_WriteStderr("Error in sys.excepthook:\n");
662 PyErr_Display(exception2, v2, tb2);
663 PySys_WriteStderr("\nOriginal exception was:\n");
664 PyErr_Display(exception, v, tb);
665 Py_DECREF(exception2);
666 Py_DECREF(v2);
667 Py_XDECREF(tb2);
668 }
669 Py_XDECREF(result);
670 } else {
671 PySys_WriteStderr("sys.excepthook is missing\n");
672 PyErr_Display(exception, v, tb);
673 }
674 Py_XDECREF(exception);
675 Py_XDECREF(v);
676 Py_XDECREF(tb);
677}

679static void
680print_exception(PyObject *f, PyObject *value)
681{
682 int err = 0;
683 PyObject *type, *tb;
684 _Py_IDENTIFIER(print_file_and_line);
685
686 if (!PyExceptionInstance_Check(value)) {
687 err = PyFile_WriteString("TypeError: print_exception(): Exception expected for value, ", f);
688 err += PyFile_WriteString(Py_TYPE(value)->tp_name, f);
689 err += PyFile_WriteString(" found\n", f);
690 if (err)
691 PyErr_Clear();
692 return;
693 }
694
695 Py_INCREF(value);
696 fflush(stdout);
697 type = (PyObject *) Py_TYPE(value);
698 tb = PyException_GetTraceback(value);
699 if (tb && tb != Py_None)
700 err = PyTraceBack_Print(tb, f);
701 if (err == 0 &&
702 _PyObject_HasAttrId(value, &PyId_print_file_and_line))
703 {
704 PyObject *message, *filename, *text;
705 int lineno, offset;
706 if (!parse_syntax_error(value, &message, &filename,
707 &lineno, &offset, &text))
708 PyErr_Clear();
709 else {
710 PyObject *line;
711
712 Py_DECREF(value);
713 value = message;
714
715 line = PyUnicode_FromFormat(" File \"%U\", line %d\n",
716 filename, lineno);
717 Py_DECREF(filename);
718 if (line != NULL) {
719 PyFile_WriteObject(line, f, Py_PRINT_RAW);
720 Py_DECREF(line);
721 }
722
723 if (text != NULL) {
724 print_error_text(f, offset, text);
725 Py_DECREF(text);
726 }
727
728 /* Can't be bothered to check all those
729 PyFile_WriteString() calls */
730 if (PyErr_Occurred())
731 err = -1;
732 }
733 }
734 if (err) {
735 /* Don't do anything else */
736 }
737 else {
738 PyObject* moduleName;
739 char* className;
740 _Py_IDENTIFIER(__module__);
741 assert(PyExceptionClass_Check(type));
742 className = PyExceptionClass_Name(type);
743 if (className != NULL) {
744 char *dot = strrchr(className, '.');
745 if (dot != NULL)
746 className = dot+1;
747 }
748
749 moduleName = _PyObject_GetAttrId(type, &PyId___module__);
750 if (moduleName == NULL || !PyUnicode_Check(moduleName))
751 {
752 Py_XDECREF(moduleName);
753 err = PyFile_WriteString("<unknown>", f);
754 }
755 else {
756 if (!_PyUnicode_EqualToASCIIId(moduleName, &PyId_builtins))
757 {
758 err = PyFile_WriteObject(moduleName, f, Py_PRINT_RAW);
759 err += PyFile_WriteString(".", f);
760 }
761 Py_DECREF(moduleName);
762 }
763 if (err == 0) {
764 if (className == NULL)
765 err = PyFile_WriteString("<unknown>", f);
766 else
767 err = PyFile_WriteString(className, f);
768 }
769 }
770 if (err == 0 && (value != Py_None)) {
771 PyObject *s = PyObject_Str(value);
772 /* only print colon if the str() of the
773 object is not the empty string
774 */
775 if (s == NULL) {
776 PyErr_Clear();
777 err = -1;
778 PyFile_WriteString(": <exception str() failed>", f);
779 }
780 else if (!PyUnicode_Check(s) ||
781 PyUnicode_GetLength(s) != 0)
782 err = PyFile_WriteString(": ", f);
783 if (err == 0)
784 err = PyFile_WriteObject(s, f, Py_PRINT_RAW);
785 Py_XDECREF(s);
786 }
787 /* try to write a newline in any case */
788 if (err < 0) {
789 PyErr_Clear();
790 }
791 err += PyFile_WriteString("\n", f);
792 Py_XDECREF(tb);
793 Py_DECREF(value);
794 /* If an error happened here, don't show it.
795 XXX This is wrong, but too many callers rely on this behavior. */
796 if (err != 0)
797 PyErr_Clear();
798}

800static const char cause_message[] =
801 "\nThe above exception was the direct cause "
802 "of the following exception:\n\n";
803
804static const char context_message[] =
805 "\nDuring handling of the above exception, "
806 "another exception occurred:\n\n";
807
808static void
809print_exception_recursive(PyObject *f, PyObject *value, PyObject *seen)
810{
811 int err = 0, res;
812 PyObject *cause, *context;
813
814 if (seen != NULL) {
815 /* Exception chaining */
816 if (PySet_Add(seen, value) == -1)
817 PyErr_Clear();
818 else if (PyExceptionInstance_Check(value)) {
819 cause = PyException_GetCause(value);
820 context = PyException_GetContext(value);
821 if (cause) {
822 res = PySet_Contains(seen, cause);
823 if (res == -1)
824 PyErr_Clear();
825 if (res == 0) {
826 print_exception_recursive(
827 f, cause, seen);
828 err |= PyFile_WriteString(
829 cause_message, f);
830 }
831 }
832 else if (context &&
833 !((PyBaseExceptionObject *)value)->suppress_context) {
834 res = PySet_Contains(seen, context);
835 if (res == -1)
836 PyErr_Clear();
837 if (res == 0) {
838 print_exception_recursive(
839 f, context, seen);
840 err |= PyFile_WriteString(
841 context_message, f);
842 }
843 }
844 Py_XDECREF(context);
845 Py_XDECREF(cause);
846 }
847 }
848 print_exception(f, value);
849 if (err != 0)
850 PyErr_Clear();
851}

853void
854PyErr_Display(PyObject *exception, PyObject *value, PyObject *tb)
855{
856 PyObject *seen;
857 PyObject *f = _PySys_GetObjectId(&PyId_stderr);
858 if (PyExceptionInstance_Check(value)
859 && tb != NULL && PyTraceBack_Check(tb)) {
860 /* Put the traceback on the exception, otherwise it won't get
861 displayed. See issue #18776. */
862 PyObject *cur_tb = PyException_GetTraceback(value);
863 if (cur_tb == NULL)
864 PyException_SetTraceback(value, tb);
865 else
866 Py_DECREF(cur_tb);
867 }
868 if (f == Py_None) {
869 /* pass */
870 }
871 else if (f == NULL) {
872 _PyObject_Dump(value);
873 fprintf(stderr, "lost sys.stderr\n");
874 }
875 else {
876 /* We choose to ignore seen being possibly NULL, and report
877 at least the main exception (it could be a MemoryError).
878 */
879 seen = PySet_New(NULL);
880 if (seen == NULL)
881 PyErr_Clear();
882 print_exception_recursive(f, value, seen);
883 Py_XDECREF(seen);
884 }
885}

944static void
945flush_io(void)
946{
947 PyObject *f, *r;
948 PyObject *type, *value, *traceback;
949
950 /* Save the current exception */
951 PyErr_Fetch(&type, &value, &traceback);
952
953 f = _PySys_GetObjectId(&PyId_stderr);
954 if (f != NULL) {
955 r = _PyObject_CallMethodId(f, &PyId_flush, NULL);
956 if (r)
957 Py_DECREF(r);
958 else
959 PyErr_Clear();
960 }
961 f = _PySys_GetObjectId(&PyId_stdout);
962 if (f != NULL) {
963 r = _PyObject_CallMethodId(f, &PyId_flush, NULL);
964 if (r)
965 Py_DECREF(r);
966 else
967 PyErr_Clear();
968 }
969
970 PyErr_Restore(type, value, traceback);
971}

987static PyObject *
988run_pyc_file(FILE *fp, const char *filename, PyObject *globals,
989 PyObject *locals, PyCompilerFlags *flags)
990{
991 PyCodeObject *co;
992 PyObject *v;
993 long magic;
994 long PyImport_GetMagicNumber(void);
995
996 magic = PyMarshal_ReadLongFromFile(fp);
997 if (magic != PyImport_GetMagicNumber()) {
998 if (!PyErr_Occurred())
999 PyErr_SetString(PyExc_RuntimeError,
1000 "Bad magic number in .pyc file");
1001 return NULL;
1002 }
1003 /* Skip mtime and size */
1004 (void) PyMarshal_ReadLongFromFile(fp);
1005 (void) PyMarshal_ReadLongFromFile(fp);
1006 if (PyErr_Occurred())
1007 return NULL;
1008
1009 v = PyMarshal_ReadLastObjectFromFile(fp);
1010 if (v == NULL || !PyCode_Check(v)) {
1011 Py_XDECREF(v);
1012 PyErr_SetString(PyExc_RuntimeError,
1013 "Bad code object in .pyc file");
1014 return NULL;
1015 }
1016 co = (PyCodeObject *)v;
1017 v = PyEval_EvalCode((PyObject*)co, globals, locals);
1018 if (v && flags)
1019 flags->cf_flags |= (co->co_flags & PyCF_MASK);
1020 Py_DECREF(co);
1021 return v;
1022}

1271/* Set the error appropriate to the given input error code (see errcode.h) */
1272
1273static void
1274err_input(perrdetail *err)
1275{
1276 PyObject *v, *w, *errtype, *errtext;
1277 PyObject *msg_obj = NULL;
1278 char *msg = NULL;
1279 int offset = err->offset;
1280
1281 errtype = PyExc_SyntaxError;
1282 switch (err->error) {
1283 case E_ERROR:
1284 return;
1285 case E_SYNTAX:
1286 errtype = PyExc_IndentationError;
1287 if (err->expected == INDENT)
1288 msg = "expected an indented block";
1289 else if (err->token == INDENT)
1290 msg = "unexpected indent";
1291 else if (err->token == DEDENT)
1292 msg = "unexpected unindent";
1293 else {
1294 errtype = PyExc_SyntaxError;
1295 msg = "invalid syntax";
1296 }
1297 break;
1298 case E_TOKEN:
1299 msg = "invalid token";
1300 break;
1301 case E_EOFS:
1302 msg = "EOF while scanning triple-quoted string literal";
1303 break;
1304 case E_EOLS:
1305 msg = "EOL while scanning string literal";
1306 break;
1307 case E_INTR:
1308 if (!PyErr_Occurred())
1309 PyErr_SetNone(PyExc_KeyboardInterrupt);
1310 goto cleanup;
1311 case E_NOMEM:
1312 PyErr_NoMemory();
1313 goto cleanup;
1314 case E_EOF:
1315 msg = "unexpected EOF while parsing";
1316 break;
1317 case E_TABSPACE:
1318 errtype = PyExc_TabError;
1319 msg = "inconsistent use of tabs and spaces in indentation";
1320 break;
1321 case E_OVERFLOW:
1322 msg = "expression too long";
1323 break;
1324 case E_DEDENT:
1325 errtype = PyExc_IndentationError;
1326 msg = "unindent does not match any outer indentation level";
1327 break;
1328 case E_TOODEEP:
1329 errtype = PyExc_IndentationError;
1330 msg = "too many levels of indentation";
1331 break;
1332 case E_DECODE: {
1333 PyObject *type, *value, *tb;
1334 PyErr_Fetch(&type, &value, &tb);
1335 msg = "unknown decode error";
1336 if (value != NULL)
1337 msg_obj = PyObject_Str(value);
1338 Py_XDECREF(type);
1339 Py_XDECREF(value);
1340 Py_XDECREF(tb);
1341 break;
1342 }
1343 case E_LINECONT:
1344 msg = "unexpected character after line continuation character";
1345 break;
1346
1347 case E_IDENTIFIER:
1348 msg = "invalid character in identifier";
1349 break;
1350 case E_BADSINGLE:
1351 msg = "multiple statements found while compiling a single statement";
1352 break;
1353 default:
1354 fprintf(stderr, "error=%d\n", err->error);
1355 msg = "unknown parsing error";
1356 break;
1357 }
1358 /* err->text may not be UTF-8 in case of decoding errors.
1359 Explicitly convert to an object. */
1360 if (!err->text) {
1361 errtext = Py_None;
1362 Py_INCREF(Py_None);
1363 } else {
1364 errtext = PyUnicode_DecodeUTF8(err->text, err->offset,
1365 "replace");
1366 if (errtext != NULL) {
1367 Py_ssize_t len = strlen(err->text);
1368 offset = (int)PyUnicode_GET_LENGTH(errtext);
1369 if (len != err->offset) {
1370 Py_DECREF(errtext);
1371 errtext = PyUnicode_DecodeUTF8(err->text, len,
1372 "replace");
1373 }
1374 }
1375 }
1376 v = Py_BuildValue("(OiiN)", err->filename,
1377 err->lineno, offset, errtext);
1378 if (v != NULL) {
1379 if (msg_obj)
1380 w = Py_BuildValue("(OO)", msg_obj, v);
1381 else
1382 w = Py_BuildValue("(sO)", msg, v);
1383 } else
1384 w = NULL;
1385 Py_XDECREF(v);
1386 PyErr_SetObject(errtype, w);
1387 Py_XDECREF(w);
1388cleanup:
1389 Py_XDECREF(msg_obj);
1390 if (err->text != NULL) {
1391 PyObject_FREE(err->text);
1392 err->text = NULL;
1393 }
1394}
