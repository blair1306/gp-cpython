50static PyObject *
51builtin___build_class__(PyObject *self, PyObject **args, Py_ssize_t nargs,
52 PyObject *kwnames)
53{
54 PyObject *func, *name, *bases, *mkw, *meta, *winner, *prep, *ns;
55 PyObject *cls = NULL, *cell = NULL;
56 int isclass = 0; /* initialize to prevent gcc warning */
57
58 if (nargs < 2) {
59 PyErr_SetString(PyExc_TypeError,
60 "__build_class__: not enough arguments");
61 return NULL;
62 }
63 func = args[0]; /* Better be callable */
64 if (!PyFunction_Check(func)) {
65 PyErr_SetString(PyExc_TypeError,
66 "__build_class__: func must be a function");
67 return NULL;
68 }
69 name = args[1];
70 if (!PyUnicode_Check(name)) {
71 PyErr_SetString(PyExc_TypeError,
72 "__build_class__: name is not a string");
73 return NULL;
74 }
75 bases = _PyStack_AsTupleSlice(args, nargs, 2, nargs);
76 if (bases == NULL)
77 return NULL;
78
79 if (kwnames == NULL) {
80 meta = NULL;
81 mkw = NULL;
82 }
83 else {
84 mkw = _PyStack_AsDict(args + nargs, kwnames);
85 if (mkw == NULL) {
86 Py_DECREF(bases);
87 return NULL;
88 }
89
90 meta = _PyDict_GetItemId(mkw, &PyId_metaclass);
91 if (meta != NULL) {
92 Py_INCREF(meta);
93 if (_PyDict_DelItemId(mkw, &PyId_metaclass) < 0) {
94 Py_DECREF(meta);
95 Py_DECREF(mkw);
96 Py_DECREF(bases);
97 return NULL;
98 }
99 /* metaclass is explicitly given, check if it's indeed a class */
100 isclass = PyType_Check(meta);
101 }
102 }
103 if (meta == NULL) {
104 /* if there are no bases, use type: */
105 if (PyTuple_GET_SIZE(bases) == 0) {
106 meta = (PyObject *) (&PyType_Type);
107 }
108 /* else get the type of the first base */
109 else {
110 PyObject *base0 = PyTuple_GET_ITEM(bases, 0);
111 meta = (PyObject *) (base0->ob_type);
112 }
113 Py_INCREF(meta);
114 isclass = 1; /* meta is really a class */
115 }
116
117 if (isclass) {
118 /* meta is really a class, so check for a more derived
119 metaclass, or possible metaclass conflicts: */
120 winner = (PyObject *)_PyType_CalculateMetaclass((PyTypeObject *)meta,
121 bases);
122 if (winner == NULL) {
123 Py_DECREF(meta);
124 Py_XDECREF(mkw);
125 Py_DECREF(bases);
126 return NULL;
127 }
128 if (winner != meta) {
129 Py_DECREF(meta);
130 meta = winner;
131 Py_INCREF(meta);
132 }
133 }
134 /* else: meta is not a class, so we cannot do the metaclass
135 calculation, so we will use the explicitly given object as it is */
136 prep = _PyObject_GetAttrId(meta, &PyId___prepare__);
137 if (prep == NULL) {
138 if (PyErr_ExceptionMatches(PyExc_AttributeError)) {
139 PyErr_Clear();
140 ns = PyDict_New();
141 }
142 else {
143 Py_DECREF(meta);
144 Py_XDECREF(mkw);
145 Py_DECREF(bases);
146 return NULL;
147 }
148 }
149 else {
150 PyObject *pargs[2] = {name, bases};
151 ns = _PyObject_FastCallDict(prep, pargs, 2, mkw);
152 Py_DECREF(prep);
153 }
154 if (ns == NULL) {
155 Py_DECREF(meta);
156 Py_XDECREF(mkw);
157 Py_DECREF(bases);
158 return NULL;
159 }
160 cell = PyEval_EvalCodeEx(PyFunction_GET_CODE(func), PyFunction_GET_GLOBALS(func), ns,
161 NULL, 0, NULL, 0, NULL, 0, NULL,
162 PyFunction_GET_CLOSURE(func));
163 if (cell != NULL) {
164 PyObject *margs[3] = {name, bases, ns};
165 cls = _PyObject_FastCallDict(meta, margs, 3, mkw);
166 if (cls != NULL && PyType_Check(cls) && PyCell_Check(cell)) {
167 PyObject *cell_cls = PyCell_GET(cell);
168 if (cell_cls != cls) {
169 /* TODO: In 3.7, DeprecationWarning will become RuntimeError.
170 * At that point, cell_error won't be needed.
171 */
172 int cell_error;
173 if (cell_cls == NULL) {
174 const char *msg =
175 "__class__ not set defining %.200R as %.200R. "
176 "Was __classcell__ propagated to type.__new__?";
177 cell_error = PyErr_WarnFormat(
178 PyExc_DeprecationWarning, 1, msg, name, cls);
179 } else {
180 const char *msg =
181 "__class__ set to %.200R defining %.200R as %.200R";
182 PyErr_Format(PyExc_TypeError, msg, cell_cls, name, cls);
183 cell_error = 1;
184 }
185 if (cell_error) {
186 Py_DECREF(cls);
187 cls = NULL;
188 goto error;
189 } else {
190 /* Fill in the cell, since type.__new__ didn't do it */
191 PyCell_Set(cell, cls);
192 }
193 }
194 }
195 }
196error:
197 Py_XDECREF(cell);
198 Py_DECREF(ns);
199 Py_DECREF(meta);
200 Py_XDECREF(mkw);
201 Py_DECREF(bases);
202 return cls;
203}

610static const char *
611source_as_string(PyObject *cmd, const char *funcname, const char *what, PyCompilerFlags *cf, PyObject **cmd_copy)
612{
613 const char *str;
614 Py_ssize_t size;
615 Py_buffer view;
616
617 *cmd_copy = NULL;
618 if (PyUnicode_Check(cmd)) {
619 cf->cf_flags |= PyCF_IGNORE_COOKIE;
620 str = PyUnicode_AsUTF8AndSize(cmd, &size);
621 if (str == NULL)
622 return NULL;
623 }
624 else if (PyBytes_Check(cmd)) {
625 str = PyBytes_AS_STRING(cmd);
626 size = PyBytes_GET_SIZE(cmd);
627 }
628 else if (PyByteArray_Check(cmd)) {
629 str = PyByteArray_AS_STRING(cmd);
630 size = PyByteArray_GET_SIZE(cmd);
631 }
632 else if (PyObject_GetBuffer(cmd, &view, PyBUF_SIMPLE) == 0) {
633 /* Copy to NUL-terminated buffer. */
634 *cmd_copy = PyBytes_FromStringAndSize(
635 (const char *)view.buf, view.len);
636 PyBuffer_Release(&view);
637 if (*cmd_copy == NULL) {
638 return NULL;
639 }
640 str = PyBytes_AS_STRING(*cmd_copy);
641 size = PyBytes_GET_SIZE(*cmd_copy);
642 }
643 else {
644 PyErr_Format(PyExc_TypeError,
645 "%s() arg 1 must be a %s object",
646 funcname, what);
647 return NULL;
648 }
649
650 if (strlen(str) != (size_t)size) {
651 PyErr_SetString(PyExc_ValueError,
652 "source code string cannot contain null bytes");
653 Py_CLEAR(*cmd_copy);
654 return NULL;
655 }
656 return str;
657}
